1 require.def('antie/lib/intervalformat', function() {
  2 	var _unitOrder = ['years', 'months', 'days', 'hours', 'minutes', 'seconds'];
  3 	var _divisors = {
  4 		'years': 60 * 60 * 24 * 365,
  5 		'months': 60 * 60 * 24 * 30.436875,
  6 		'weeks': 60 * 60 * 24 * 7,
  7 		'days': 60 * 60 * 24,
  8 		'hours': 60 * 60,
  9 		'minutes': 60,

 14 	 * Method to wrap Array.prototype.indexOf and add support for pre-JS1.5 browsers
 15 	 * @private
 16 	 */
 17 	function _indexOf(arr, obj, start) {
 18 		if(typeof(Array.prototype.indexOf) === 'function') {
 19 			return arr.indexOf(obj, start);
 20 		} else {
 21 			for (var i = (start || 0); i < arr.length; i++) {
 22 				if (arr[i] == obj) {
 23 					return i;
 24 				}
 25 			}
 26 			return -1;
 27 		}
 28 	}

 30 	/**
 31 	 * Formats a time interval, e.g. 3850 seconds as "1 hour, 4 minutes and 10 seconds",
 32 	 * or "1 hour and 4 minutes" or "64 minutes and 10 seconds", etc.
 33 	 *
 34 	 * @param {Integer} seconds Number of seconds to format.
 35 	 * @param {Array} [units] An array of which may be used in the output. One of: years,
 36 	 *			  months, weeks, days, hours, minutes or seconds.
 37 	 * @param {Integer} [maxParts] The maximum number of units to output.
 38 	 * @param {Object} [unitMinimums] An object specifying the minimum number of each unit
 39 	 *				 required before those units are included in the output.
 40 	 * @param {String} [andString] The string to include between the final 2 units. If not
 41 	 * 				specified " and " is used.	
 42 	 */
 43 	return function(seconds, units, maxParts, unitMinimums, andString) {
 44 		if(!units) {
 45 			units = ['years', 'months', 'days', 'hours', 'minutes', 'seconds'];
 46 		}
 47 		if(!andString) {
 48 			andString = " and ";
 49 		}
 50 
 51 		var stringParts = [];
 52 		for(var i = 0; i < _unitOrder.length; i++) {
 53 			var unit = _unitOrder[i];
 54 			if(_indexOf(units, unit) !== -1) {
 55 				var count = Math.floor(seconds / _divisors[unit]);
 56 				if(!unitMinimums || !unitMinimums[unit] || count >= unitMinimums[unit]) {
 57 					seconds %= _divisors[unit];
 58 					if(count > 1) {
 59 						stringParts.push(count + " " + unit);
 60 					} else if(count == 1) {
 61 						stringParts.push(count + " " + unit.substring(0, unit.length - 1));
 62 					}
 63 					if(maxParts && (stringParts.length === maxParts)) {
 64 						break;
 65 					}
 66 				}
 67 			}
 68 		}
 69 
 70 		if(stringParts.length > 1) {
 71 			var	smallestUnit = stringParts.pop(),
 72 				secondSmallestUnit = stringParts.pop();
 73 
 74 			stringParts.splice(stringParts.length - 2, 2, [secondSmallestUnit + andString + smallestUnit]);
 75 		}
 76 
 77 		return stringParts.join(', ');
 78 	};
