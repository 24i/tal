 27 require.def('antie/widgets/container',
 28 	[
 29         'antie/widgets/widget',
 30         'antie/events/focusevent',
 31         'antie/events/blurevent'
 32 	],
 33 	function(Widget, FocusEvent, BlurEvent) {
 34 		'use strict';
 35 
 36 		/**
 37 		 * The abstract Container widget class represents any Widget that may contain children widgets.
 38 		 * @name antie.widgets.Container
 39 		 * @class
 40 		 * @abstract
 41 		 * @extends antie.widgets.Widget
 42 		 * @requires antie.events.FocusEvent
 43 		 * @requires antie.events.BlurEvent
 44 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 45 		 */
 46 		var Container;
 47 		Container = Widget.extend(/** @lends antie.widgets.Container.prototype */ {
 48 			/**
 49 			 * @constructor
 50 			 * @ignore
 51 			 */
 52 			init: function(id) {
 53 				/*
 54                 * Performance consideration - do we need to store 2 references to each child widget? 
 55                 * One keyed by ID, one in an array to maintain order?
 56                 */
 57 				this._childWidgets = {};
 58 				this._childWidgetOrder = [];
 59 				this._activeChildWidget = null;
 60 				this._autoRenderChildren = true;
 61 
 62 				this._super(id);
 63 
 64 				this.addClass('container');
 65 			},
 66 			/**
 67 			 * Renders the widget and any child widgets to device-specific output.
 68 			 * @param {antie.devices.Device} device The device to render to.
 69 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 70 			 */
 71 			render: function(device) {
 72 			    var i;
 73 				if(!this.outputElement) {
 74 					this.outputElement = device.createContainer(this.id, this.getClasses());
 75 				} else {
 76 					device.clearElement(this.outputElement);
 77 				}
 78 				for(i=0; i<this._childWidgetOrder.length; i++) {
 79 					device.appendChildElement(this.outputElement, this._childWidgetOrder[i].render(device));
 80 				}
 81 				return this.outputElement;
 82 			},
 83 			/**
 84 			 * Appends a child widget to this widget.
 85 			 * @param {antie.widgets.Widget} widget The child widget to add.
 86 			 */
 87 			appendChildWidget: function(widget) {
 88 				if(!this.hasChildWidget(widget.id)) {
 89 					this._childWidgets[widget.id] = widget;
 90 					this._childWidgetOrder.push(widget);
 91 					widget.parentWidget = this;
 92 
 93 					// If there's no active child widget set, try and set it to this
 94 					// (Will only have an affect if it's focusable (i.e. contains a button))
 95 					if(this._activeChildWidget == null) {
 96 						this.setActiveChildWidget(widget);
 97 					}
 98 
 99 					if(this.outputElement && this._autoRenderChildren) {
100 						var device = this.getCurrentApplication().getDevice();
101 						device.appendChildElement(this.outputElement,  widget.render(device));
102 					}
103 
104 					return widget;
105 				}
106 			},
107 
108 			/**
109 			 * Inserts a widget before the current one.
110 			 * @param {antie.widgets.Widget} widget The child widget to
111 			 * add.
112 			 */
113 			prependWidget: function(widget) {
114 				// Find the current widget's order.
115                 if(this.parentWidget instanceof Container) {
116                     var widgetOrder, insertionIndex;
117                     widgetOrder = this.parentWidget._childWidgetOrder;
118                     
119                     for (insertionIndex = 0; insertionIndex !== widgetOrder.length; insertionIndex +=1) {
120                     if (widgetOrder[insertionIndex] === this) {
121                         break;
122                         }
123                     }
124                 
125                     // Insert the new widget in the current one's position.
126                     this.parentWidget.insertChildWidget(insertionIndex, widget);
127                 }
128 				return widget;
129 			},
130 			/**
131 			 * Inserts a child widget at the specified index.
132 			 * @param {Integer} index The index where to insert the child widget.
133 			 * @param {antie.widgets.Widget} widget The child widget to add.
134 			 */
135 			insertChildWidget: function(index, widget) {
136 				if(!this.hasChildWidget(widget.id)) {
137 					if(index >= this._childWidgetOrder.length) {
138 						return this.appendChildWidget(widget);
139 					}
140 
141 					this._childWidgets[widget.id] = widget;
142 					this._childWidgetOrder.splice(index, 0, widget);
143 					widget.parentWidget = this;
144 
145 					// If there's no active child widget set, try and set it to this
146 					// (Will only have an affect if it's focusable (i.e. contains a button))
147 					if(this._activeChildWidget == null) {
148 						this.setActiveChildWidget(widget);
149 					}
150 
151 					if(this.outputElement && this._autoRenderChildren) {
152 						var device = this.getCurrentApplication().getDevice();
153 
154 						if(!widget.outputElement) {
155 							widget.render(device);
156 						}
157 
158 						device.insertChildElementBefore(this.outputElement, widget.outputElement, this._childWidgetOrder[index+1].outputElement);
159 					}
160 
161 					return widget;
162 				}
163 			},
164 			/**
165 			 * Remove all child widgets from this widget.
166 			 */
167 			removeChildWidgets: function() {
168 				if(this._isFocussed && this._activeChildWidget) {
169 					var logger = this.getCurrentApplication().getDevice().getLogger();
170 					logger.warn('Removing widget that currently has focus: ' + this._activeChildWidget.id);
171 				}
172 
173 				if(this.outputElement) {
174 					var device = this.getCurrentApplication().getDevice();
175 					device.clearElement(this.outputElement);
176 				}
177 				for(var i=0; i<this._childWidgetOrder.length; i++) {
178 					this._childWidgetOrder[i].parentWidget = null;
179 				}
180 
181 				this._childWidgets = {};
182 				this._childWidgetOrder = [];
183 				this._activeChildWidget = null;
184 			},
185 			/**
186 			 * Removes a specific child widget from this widget.
187 			 * @param {antie.widgets.Widget} widget The child widget to remove.
188 			 * @param {Boolean} [retainElement] Pass <code>true</code> to retain the child output element of the given widget
189 			 */
190 			removeChildWidget: function(widget, retainElement) {
191 				if(!widget) {
192 					return;
193 				}
194 				
195 				var widget_index = this.getIndexOfChildWidget(widget);
196 				if (widget_index < 0) {
197 					return;
198 				}
199 
200 				if(widget._isFocussed) {
201 					var logger = this.getCurrentApplication().getDevice().getLogger();
202 					logger.warn('Removing widget that currently has focus: ' + widget.id);
203 				}
204 
205 				if(!retainElement && widget.outputElement) {
206 					var device = this.getCurrentApplication().getDevice();
207 					device.removeElement(widget.outputElement);
208 				}
209 
210 				this._childWidgetOrder.splice(widget_index, 1);
211 				delete(this._childWidgets[widget.id]);
212 				
213 				widget.parentWidget = null;
214 
215 			},
216 			/**
217 			 * Checks to see if a specific widget is a direct child of this widget.
218 			 * @param {String} id The widget id of the widget to check to see if it is a direct child of this widget.
219 			 */
220 			hasChildWidget: function(id) {
221 				return this._childWidgets[id] != null;
222 			},
223 			/**
224 			 * Get a child widget from its unique ID.
225 			 * @param {String} id The id of the child widget to return.
226 			 * @returns antie.widgets.Widget of the widget with the given ID, otherwise undefined if the child does not exist.
227 			 */
228 			getChildWidget: function(id) {
229 				return this._childWidgets[id];
230 			},
231 			/**
232 			 * Get an array of all this widget's children.
233 			 * @returns An array of all this widget's children.
234 			 */
235 			getChildWidgets: function() {
236 				return this._childWidgetOrder;
237 			},
238 			getIndexOfChildWidget: function(widget) {
239 				var device = this.getCurrentApplication().getDevice();
240 				return device.arrayIndexOf(this._childWidgetOrder, widget);
241 			},
242 			/**
243 			 * Attempt to set focus to the given child widget.
244 			 *
245 			 * Note: You can only set focus to a focusable widget. A focusable widget is one that
246 			 * contains an enabled antie.widgets.Button as either a direct or indirect child.
247 			 *
248 			 * Note: Widgets have 2 independent states: active and focussed. A focussed widget is
249 			 * either the Button with focus, or any parent of that Button. An active widget is
250 			 * one which is the active child of its parent Container. When the parent widget
251 			 * receives focus, focus will be placed on the active child.
252 			 *
253 			 * Classes 'active' and 'focus' are appended to widgets with these states.
254 			 * 
255 			 * @param {antie.widgets.Widget} widget The child widget to set focus to.
256 			 * @returns Boolean true if the child widget was focusable, otherwise boolean false.
257 			 */
258 			setActiveChildWidget: function(widget) {
259 				if (!widget) {
260 					return false;
261 				}
262 				if(this.hasChildWidget(widget.id) && widget.isFocusable()) {
263 					if(this._activeChildWidget && this._activeChildWidget !== widget) {
264 						this._activeChildWidget.removeClass('active');
265 						this._setActiveChildFocussed(false);
266 					}
267 					widget.addClass('active');
268 					this._activeChildWidget = widget;
269 
270 					if(this.getCurrentApplication().getFocussedWidget() == null) {
271 						var widgetIterator = this;
272 						while(widgetIterator.parentWidget != null) {
273 							widgetIterator.parentWidget._activeChildWidget = widgetIterator;
274 							widgetIterator._isFocussed = true;
275 
276 							widgetIterator = widgetIterator.parentWidget;
277 						}
278 					}
279 					if(this._isFocussed) {
280 						this._setActiveChildFocussed(true);
281 					}
282 					return true;
283 				}
284 				return false;
285 			},
286 			/**
287 			 * Attempts to set focus to the child widget at the given index.
288 			 * @see #setActiveChildWidget
289 			 * @param {Integer} index Index of the child widget to set focus to.
290 			 * @returns Boolean true if the child widget was focusable, otherwise boolean false.
291 			 */
292 			setActiveChildIndex: function(index) {
293 				if(index < 0 || index >= this._childWidgetOrder.length) {
294 					throw new Error("Widget::setActiveChildIndex Index out of bounds. " + this.id + " contains " + this._childWidgetOrder.length + " children, but an index of " + index + " was specified.");
295 				}
296 				return this.setActiveChildWidget(this._childWidgetOrder[index]);
297 			},
298 			/**
299 			 * Get the current active widget.
300 			 * @returns The current active widget
301 			 */
302 			getActiveChildWidget: function() {
303 				return this._activeChildWidget; 
304 			},
305 			/**
306 			 * Flags the active child as focussed or blurred.
307 			 * @param {Boolean} focus True if the active child is to be focussed, False if the active child is to be blurred.
308 			 * @private
309 			 */
310 			_setActiveChildFocussed: function(focus) {
311 				if(this._activeChildWidget && (this._activeChildWidget._isFocussed != focus)) {
312 					this._activeChildWidget._isFocussed = focus;
313 					if(focus) {
314 						this._activeChildWidget.addClass('focus');
315 						this._activeChildWidget.bubbleEvent(new FocusEvent(this._activeChildWidget));
316 						// TODO: force focus to change in the application (rather than relying on the above
317 						// TODO: even to propagate to the application level
318 					} else {
319 						this._activeChildWidget.removeClass('focus');
320 						this._activeChildWidget.bubbleEvent(new BlurEvent(this._activeChildWidget));
321 					}
322 					this._activeChildWidget._setActiveChildFocussed(focus);
323 				}
324 			},
325 			/**
326 			 * Gets the number of direct child widgets.
327 			 * @returns The number of direct child widgets.
328 			 */
329 			getChildWidgetCount: function() {
330 				return this._childWidgetOrder.length;
331 			},
332 			/**
333 			 * Checks to see if a widget is focussable, i.e. contains an enabled button.
334 			 * @see antie.widgets.Button
335 			 */
336 			isFocusable: function() {
337 				for(var i=0; i<this._childWidgetOrder.length; i++) {
338 					if(this._childWidgetOrder[i].isFocusable()) {
339 						if(this._activeChildWidget == null) {
340 							//this._activeChildWidget = this._childWidgetOrder[i];
341 							this.setActiveChildWidget(this._childWidgetOrder[i]);
342 						}
343 						return true;
344 					}
345 				}
346 				return false;
347 			},
348 			setAutoRenderChildren: function(autoRenderChildren) {
349 				this._autoRenderChildren = autoRenderChildren;
350 			},
351 
352 			/**
353 			 * Broadcasts an event from the application level to every single
354 			 * object it contains.
355 			 */
356 			broadcastEvent: function(evt) {
357 				this.fireEvent(evt);
358 				if(!evt.isPropagationStopped()) {
359 					for(var i=0; i<this._childWidgetOrder.length; i++) {
360 						this._childWidgetOrder[i].broadcastEvent(evt);
361 					}
362 				}
363 			},
364 			/**
365 			 * Moves focus to a button within this container. Focused button will be that which follows
366 			 * the current 'active' path.
367 			 * @returns Boolean true if focus has been moved to a button. Otherwise returns false.
368 			 */
369 			focus: function() {
370 				if(this._activeChildWidget) {
371 					return this._activeChildWidget.focus();
372 				}
373 				return false;
374 			}
375 		});
376 		return Container;
377 	}
378 );
