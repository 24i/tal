 28 require.def('antie/events/componentevent',
 29 	['antie/events/event'],
 30 	function(Event) {
 31 		/**
 32 		 * Class of events raised when a {@link antie.widgets.Component} is being loaded/shown/hidden.
 33 		 * @name antie.events.ComponentEvent
 34 		 * @class
 35 		 * @extends antie.events.Event
 36 		 * @param {String} type The type of event.
 37 		 * @param {antie.widgets.ComponentContainer} container The container of the component that fired the event.
 38 		 * @param {antie.widgets.Component} component The component that fired the event.
 39 		 * @param {Object} args Any arguments that were passed into the component when loaded.
 40 		 * @param {Object} state Any state information that was stored on the component history stack for this component.
 41 		 * @param {Boolean} fromBack True if the event was raised as a result of the user navigating 'back' in the component history.
 42 		 */
 43 		return Event.extend(/** @lends antie.events.ComponentEvent.prototype */ {
 44 			/**
 45 			 * @constructor
 46 			 * @ignore
 47 			 */
 48 			init: function(type, container, component, args, state, fromBack) {
 49 				this.container = container;
 50 				this.component = component;
 51 				this.args = args;
 52 				this.state = state;
 53 				this.fromBack = fromBack;
 54 				this._super(type);
 55 			}
 56 		});
 57 	}
 58 );
