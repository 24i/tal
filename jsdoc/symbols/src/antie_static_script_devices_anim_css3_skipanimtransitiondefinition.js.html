 27 require.def(
 28     'antie/devices/anim/css3/skipanimtransitiondefinition',
 29     [
 30         'antie/devices/anim/css3/transitiondefinition'
 31     ],
 32     function(TransitionDefinition) {
 33         'use strict';
 34         return TransitionDefinition.extend(
 35             {
 36                 init: function (transDef) {
 37                     this._super();
 38                     this._copyToSelf(transDef);
 39                 },
 40                 
 41                 _copyToSelf: function(transDef) {
 42                     var props, property, i;
 43                     props = transDef.getProperties();
 44                     for(i = 0; i !== props.length; i += 1) {
 45                         property = props[i];
 46                         this.setProperty(
 47                             property, 
 48                             {
 49                                 duration: transDef.getPropertyDuration(property),
 50                                 delay: transDef.getPropertyDelay(property),
 51                                 timingFn: transDef.getPropertyTimingFn(property)
 52                             }
 53                         );
 54                     }
 55                 },
 56                 
 57                 getPropertyDuration: function () {
 58                     return 0;
 59                 }
 60             }
 61             
 62             
 63         );
 64     }
 65 );
