 27 require.def('antie/runtimecontext',
 28 	[
 29 		'antie/class'
 30 	],
 31 	/**
 32 	 * Static class for accessing the current running application
 33 	 * @name antie.RuntimeContext
 34 	 * @class
 35 	 * @static
 36 	 * @private
 37 	 */
 38 	function(Class) {
 39 		'use strict';
 40 
 41 		var applicationObject;
 42 
 43 		var RuntimeContext = Class.extend(/** @lends antie.RuntimeContext.prototype */ {
 44 			/**
 45 			 * Clears the currently executing application.
 46 			 * @private
 47 			 */
 48 			clearCurrentApplication: function() {
 49 				applicationObject = undefined;
 50 			},
 51 
 52 			/**
 53 			 * Sets the currently executing application.
 54 			 * @private
 55 			 */
 56 			setCurrentApplication: function(app) {
 57 				if (applicationObject) {
 58 					throw new Error("RuntimeContext.setCurrentApplication called for a second time. You can only have one application instance running at any time.");
 59 				} else {
 60 					applicationObject = app;
 61 				}
 62 			},
 63 
 64 			/**
 65 			 * Returns the currently executing application.
 66 			 * @private
 67 			 */
 68 			getCurrentApplication: function() {
 69 				return applicationObject;
 70 			},
 71 
 72 			/**
 73 			 * Returns the current device
 74 			 * @private
 75 			 */
 76 			getDevice: function() {
 77 				return this.getCurrentApplication().getDevice();
 78 			}
 79 		});
 80 
 81 		return new RuntimeContext();
 82 	}
 83 );
