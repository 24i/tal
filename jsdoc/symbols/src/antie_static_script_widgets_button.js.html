 27 require.def('antie/widgets/button',
 28 	[
 29 	 	'antie/widgets/container',
 30 	 	'antie/events/focusevent',
 31 	 	'antie/events/focusdelayevent',
 32 	 	'antie/events/blurevent',
 33 	 	'antie/events/keyevent',
 34 	 	'antie/events/selectevent'
 35 	],
 36 	function(Container, FocusEvent, FocusDelayEvent, BlurEvent, KeyEvent, SelectEvent) {
 37 		'use strict';
 38 
 39 		/**
 40 		 * The Button widget class represents a container widget which can receive focus and be selected/activated by the user.
 41 		 * @name antie.widgets.Button
 42 		 * @class
 43 		 * @extends antie.widgets.Container
 44 		 * @requires antie.events.FocusEvent
 45 		 * @requires antie.events.FocusDelayEvent
 46 		 * @requires antie.events.BlurEvent
 47 		 * @requires antie.events.KeyEvent
 48 		 * @requires antie.events.SelectEvent
 49 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 50 		 * @param {boolean} [animationEnabled] If true, a focus delay will be set before displaying the button
 51 		 */
 52 		return Container.extend(/** @lends antie.widgets.Button.prototype */ {
 53 			/**
 54 			 * @constructor
 55 			 * @ignore
 56 			 */
 57 			init: function(id, animationEnabled) {
 58 				//TODO refactor this to set focusDelay explicitly rather than using animationEnabled with fixed focusDelay
 59 				this._super(id);
 60 				this.addClass('button');
 61 				this.addClass('buttonBlurred');
 62 
 63 				this._focusDelayHandle = null;
 64 				this._disabled = false;
 65 				
 66 				/* Reduce the focusDelayTimeout for devices that don't have animation enabled */
 67 				if(typeof animationEnabled === "boolean" && !animationEnabled){
 68 					this._focusDelayTimeout = 500;
 69 				}
 70 				else{
 71 					this._focusDelayTimeout = 1500;
 72 				}
 73 
 74 				/* if the ENTER key is pressed, translate into into a SelectEvent on this button */
 75 				var self = this;
 76 				this.addEventListener('keydown', function(e) {
 77 					if(e.keyCode == KeyEvent.VK_ENTER) {
 78 						self.select();
 79 						e.stopPropagation();
 80 					}
 81 				});
 82 			},
 83 			/**
 84 			 * Renders the widget and any child widgets to device-specific output.
 85 			 * @param {antie.devices.Device} device The device to render to.
 86 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 87 			 */
 88 			render: function(device) {
 89 				this.outputElement = device.createButton(this.id, this.getClasses(), "#");
 90 				for(var i=0; i<this._childWidgetOrder.length; i++) {
 91 					device.appendChildElement(this.outputElement, this._childWidgetOrder[i].render(device));
 92 				}
 93 				return this.outputElement;
 94 			},
 95 			/**
 96 			 * Checks to see if a widget is focussable (as we're a button, returns true).
 97 			 * @see antie.widgets.Button
 98 			 */
 99 			isFocusable: function() {
100 				// a widget can receive focus if it or any of it's descendants are Buttons
101 				// We're a button, so we are
102 				return !this._disabled;
103 			},
104 			setDisabled: function(disabled) {
105 				this._disabled = disabled;
106 				if(disabled) {
107 					this.addClass("buttonDisabled");
108 				} else {
109 					this.removeClass("buttonDisabled");
110 				}
111 			},
112 			/**
113 			 * Gives this button focus by setting active children back up the widget tree.
114 			 * @see antie.widgets.Button
115 			 * @param {Boolean} [force] Pass <code>true</code> to force focus to a disabled button.
116 			 * @returns Boolean true if focus has been moved to the button. Otherwise returns false.
117 			 */
118 			focus: function(force) {
119 				var origDisabled = this._disabled;
120 				if(force) {
121 					this._disabled = false;
122 				}
123 
124 				var focusChanged = true;
125 				var w = this;
126 				while(w.parentWidget) {
127 					if(!w.parentWidget.setActiveChildWidget(w)) {
128 						focusChanged = false;
129 					}
130 					w = w.parentWidget;
131 				}
132 
133 				this._disabled = origDisabled;
134 				
135 				return focusChanged;
136 			},
137 			select: function() {
138 				this.bubbleEvent(new SelectEvent(this));
139 			},
140 			/**
141 			 * Flags the active child as focussed or blurred.
142 			 * @param {Boolean} focus True if the active child is to be focussed, False if the active child is to be blurred.
143 			 * @private
144 			 */
145 			_setActiveChildFocussed: function(focus) {
146 				if(this._focusDelayHandle) {
147 					clearTimeout(this._focusDelayHandle);
148 				}
149 				if(focus) {
150 					this.removeClass('buttonBlurred');
151 					this.addClass('buttonFocussed');
152 
153 					var self = this;
154 					// Fire a focus delay event if this button has had focus for more than x-seconds.
155 					this._focusDelayHandle = setTimeout(function() {
156 						self.bubbleEvent(new FocusDelayEvent(self));
157 					}, this._focusDelayTimeout);
158 
159 					this.getCurrentApplication()._setFocussedWidget(this);
160 				} else {
161 					this.removeClass('buttonFocussed');
162 					this.addClass('buttonBlurred');
163 				}
164 			},
165 			removeFocus: function() {
166 				this._super();
167 				this.removeClass('buttonFocussed');
168 				this.addClass('buttonBlurred');
169 			}
170 		});
171 	}
172 );
