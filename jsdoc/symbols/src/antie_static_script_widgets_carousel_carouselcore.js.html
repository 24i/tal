 27 require.def('antie/widgets/carousel/carouselcore',
 28 	[
 29 		'antie/widgets/container',
 30         'antie/widgets/carousel/navigators/bookendednavigator',
 31         'antie/widgets/carousel/mask',
 32         'antie/widgets/carousel/strips/widgetstrip',
 33         'antie/widgets/carousel/aligners/aligner',
 34         'antie/widgets/carousel/orientations/vertical',
 35         'antie/widgets/carousel/orientations/horizontal'
 36 	],
 37 	function (
 38         Container,
 39         BookendedNavigator,
 40         Mask,
 41         WidgetStrip,
 42         Aligner,
 43         verticalOrientation,
 44         horizontalOrientation
 45     ) {
 46 		"use strict";
 47         /**
 48          * Ordered list of widgets that can be navigated by moving the list or the selection point
 49          * Use antie.widgets.Carousel instead if you need old container methods.
 50          * @name antie.widgets.carousel.CarouselCore
 51          * @class
 52          * @extends antie.widgets.Widget
 53          * @param {string} id The id of the carousel, id_CarouselMask will be used as the id for the mask element
 54          * and id_WidgetStrip will be used as the id of the widget strip element
 55          * @param {Object} [orientation=antie.widgets.carousel.CarouselCore.orientations.VERTICAL] the orientation object of
 56          * the carousel. Vertical by default, for horizontal pass in antie.widgets.carousel.CarouselCore.orientations.HORIZONTAL
 57          */
 58 		var Carousel = Container.extend(/** @lends antie.widgets.carousel.CarouselCore.prototype */ {
 59             /**
 60              * @constructor
 61              * @ignore
 62              */
 63 			init: function (id, orientation) {
 64 				this.id = id;
 65 				this._super(id);
 66                 this._setOrientation(orientation || Carousel.orientations.VERTICAL);
 67                 this.setWidgetStrip(WidgetStrip);
 68 				this._mask = new Mask(this.id + '_CarouselMask', this._widgetStrip, this._orientation);
 69 				this._directAppend(this._mask);
 70                 this.setNavigator(BookendedNavigator);
 71                 this._aligner = new Aligner(this._mask);
 72                 this._setAlignEventsFromStripToHaveCarouselAsTarget();
 73                 this._autoCalculate = true;
 74 			},
 75 
 76             /**
 77              * Renders the widget and any child widgets to device-specific output.
 78              * @param {antie.devices.Device} device The device to render to.
 79              * @returns A device-specific object that represents the Carousel as displayed on the device
 80              * (in a browser, two nested DIVs with the inner containing child widgets and the outer acting as a mask);
 81              */
 82 			render: function (device) {
 83                 this.outputElement = this._mask.render(device);
 84 				return this.outputElement;
 85 			},
 86 
 87             /**
 88              * Adds a widget to the end of the carousel
 89              * @param {antie.widgets.Widget} widget The widget to append to the carousel
 90              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
 91              * (Height for a vertical carousel or width for horizontal.) If providied, this value will be used in
 92              * positioning calculations rather then a calculated value (can be useful when widgets change size)
 93              * Note length only currently working with non-wrapping strips.
 94              */
 95 			append: function (widget, length) {
 96                 widget.addClass('carouselItem');
 97 				return this._widgetStrip.append(widget, length);
 98 			},
 99 
100             /**
101              * Adds a widget to the end of the carousel
102              * @param {Number} index A zero based index to begin insertion at, i.e. 0 prepends.
103              * @param {antie.widgets.Widget} widget The widget to append to the carousel
104              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
105              * (Height for a vertical carousel or width for horizontal.) If provided, this value will be used in
106              * positioning calculations rather then a calculated value (can be useful when widgets change size)
107              * Note length only currently working with non-wrapping strips.
108              */
109             insert: function (index, widget, length) {
110                 widget.addClass('carouselItem');
111                 return this._widgetStrip.insert(index, widget, length);
112             },
113 
114             /**
115              * Manually sets lengths of elements for movement calculations - useful for elements which change size while moving.
116              * @param lengths {number} | {Array} If provided with a number all lengths will be set equal to this number
117              * If provided with an array, the lengths will be set with the corresponding widgets (so the first number will be used
118              * for the first widget's length, etc..)
119              * Note only currently working with non-wrapping strips.
120              */
121             setWidgetLengths: function (lengths) {
122                 this._widgetStrip.setLengths(lengths);
123             },
124 
125             /**
126              * Removes a widget from the carousel
127              * @param {antie.widgets.Widget} widget. Widget to remove from the DOM
128              */
129             remove: function (widget, retainElement) {
130                 if (this.hasChildWidget(widget.id)) {
131                     widget.removeClass('carouselItem');
132                     return this._widgetStrip.remove(widget, retainElement);
133                 }
134             },
135 
136             /**
137              * Removes all widgets from the carousel
138              */
139             removeAll: function () {
140                 this._widgetStrip.removeAll();
141             },
142 
143             /**
144              * Aligns the carousel to the next enabled widget after that currently aligned.
145              * If no alignment has been performed previously it will align to the next enabled widget after that at index 0
146              * If a wrapping strip and navigator are used the alignment will wrap to the start after the last widget is reached.
147              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
148              * @param {Object} [options] An animation options object
149              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
150              * @param {Number} [options.duration] The duration of the alignment in ms
151              * @param {String} [options.easing] The alignment easing function
152              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
153              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
154              */
155 			alignNext: function (options) {
156                 this._aligner.alignNext(this._navigator, options);
157 			},
158 
159             /**
160              * Aligns the carousel to the enabled widget before that currently aligned.
161              * If no alignment has been performed previously it will align to the first enabled widget before that at index 0
162              * If a wrapping strip and navigator are used the alignment will wrap to the end after the first widget is reached.
163              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
164              * @param {Object} [options] An animation options object
165              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
166              * @param {Number} [options.duration] The duration of the alignment in ms
167              * @param {String} [options.easing] The alignment easing function
168              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
169              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
170              */
171             alignPrevious: function (options) {
172                 this._aligner.alignPrevious(this._navigator, options);
173 			},
174 
175             /**
176              * Aligns the carousel to the widget at the specified index
177              * Will always move forward if the index is after that currently aligned and backwards if index is before
178              * that currently aligned.
179              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
180              * @param {Number} index The index of the widget to align on.
181              * @param {Object} [options] An animation options object
182              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
183              * @param {Number} [options.duration] The duration of the alignment in ms
184              * @param {String} [options.easing] The alignment easing function
185              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
186              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
187              */
188             alignToIndex: function (index, options) {
189                 this._aligner.alignToIndex(index, options);
190             },
191 
192             /**
193              * Instantly completes any in-flight alignment animations, firing any callbacks that were provided.
194              * If several alignments have been queued, all will complete in order.
195              */
196             completeAlignment: function () {
197                 this._aligner.complete();
198             },
199 
200             /**
201              * Sets the point along the Mask at which alignments are made
202              * @param pixelsFromEdgeToWidgetEdge A value in pixels from the primary edge (top or left for Vertical/Horizontal)
203              * at which widgets are aligned.
204              */
205             setAlignPoint: function (pixelsFromEdgeToWidgetEdge) {
206                 this._mask.setAlignPoint(pixelsFromEdgeToWidgetEdge);
207             },
208 
209             /**
210              * Sets the point along the Mask at which alignments are made
211              * @param fractionOfMaskLength a value between 0 and 1 specifying how far along the mask a widget should
212              * be aligned. i.e. 0.5 is the centre of the mask.
213              */
214             setNormalisedAlignPoint: function (fractionOfMaskLength) {
215                 this._mask.setNormalisedAlignPoint(fractionOfMaskLength);
216             },
217 
218             /**
219              * Sets the point along the aligned widget at which alignments are made
220              * @param fractionOfWidgetLength a value between 0 and 1 specifying the point along the widget which will be
221              * aligned with the mask alignment point. i.e. in a horizontal Carousel, 0.5 is the centre of the widget,
222              * 0 is the left edge, 1 is the right edge
223              */
224             setNormalisedWidgetAlignPoint: function (fractionOfWidgetLength) {
225                 this._mask.setNormalisedWidgetAlignPoint(fractionOfWidgetLength);
226             },
227 
228             /**
229              * Some widget strips peform calculations which require elements to be present in the document.
230              * This method manually performs those recalculations.
231              */
232             recalculate: function () {
233                 this._widgetStrip.recalculate();
234             },
235 
236             /**
237              * Some widget strips peform calculations which require elements to be present in the document.
238              * By default these calculations are performed whenever their values might be invalidated (after appending elements
239              * for instance) This method can be used to be disable/enable this behaviour for performance optimisation.
240              */
241             autoCalculate: function (on) {
242                 this._widgetStrip.autoCalculate(on);
243             },
244 
245             /**
246              * @returns the index of the currently active widget
247              */
248             getActiveIndex: function () {
249                 return this._navigator.currentIndex();
250             },
251 
252             /**
253              * Sets the currently active index
254              * @param {Number} index the index of the widget to be made active.
255              * If this is invalid or corresponds to a disabled widget the active index will not change
256              */
257             setActiveIndex: function (index) {
258                 this._navigator.setIndex(index);
259             },
260 
261             /**
262              * @returns the index first focussable index after the index of the active widget
263              */
264             nextIndex: function () {
265                 return this._navigator.nextIndex();
266             },
267 
268             /**
269              * @returns the index first focussable index before the index of the active widget
270              */
271             previousIndex: function () {
272                 return this._navigator.previousIndex();
273             },
274 
275             /**
276              * Sets the currently active widget
277              * @param {antie.widgets.Widget} widget the widget to be made active.
278              * If the widget is not in the Carousel or corresponds to a disabled widget the active widget will not change
279              */
280             setActiveWidget: function (widget) {
281                 var index;
282                 index = this._widgetStrip.getIndexOfChildWidget(widget);
283                 this._navigator.setIndex(index);
284             },
285 
286             /**
287              * Sets the navigator class used to determine focus position
288              * @param {Function} Navigator the constructor function of the type of navigator the carousel should use, e.g.
289              * antie.widgets.carousel.navigators.BookendedNavigator or antie.widgets.carousel.navigators.WrappingNavigator
290              * On construction, the carousel uses antie.widgets.carousel.navigators.BookendedNavigator by default
291              */
292             setNavigator: function (Navigator) {
293                 this._navigator = new Navigator(this._widgetStrip);
294             },
295 
296             /**
297              * Sets the widget strip class used to manage widgets and elements within the carousel
298              * @param {Function} WidgetStrip the constructor function of the type of Widget strip the carousel should use, e.g.
299              * antie.widgets.carousel.navigators.WidgetStrip, antie.widgets.carousel.navigators.WrappingStrip
300              * On construction, the carousel uses antie.widgets.carousel.navigators.WidgetStrip by default
301              */
302             setWidgetStrip: function (WidgetStrip) {
303                 this._widgetStrip = new WidgetStrip(this.id + '_WidgetStrip', this._orientation);
304                 if (this._navigator) {
305                     this._navigator.setContainer(this._widgetStrip);
306                 }
307                 if (this._mask) {
308                     this._mask.setWidgetStrip(this._widgetStrip);
309                 }
310 
311             },
312 
313             /**
314              * @returns {Array} The widgets currently in the carousel
315              */
316             items: function () {
317                 return this._widgetStrip.widgets();
318             },
319 
320             /**
321              * @returns {Object} The orientation object associated with the carousel
322              */
323             orientation: function () {
324                 return this._orientation;
325             },
326 
327             _setOrientation: function (orientation) {
328                 this._orientation = orientation;
329             },
330 
331             _setAlignEventsFromStripToHaveCarouselAsTarget: function () {
332                 this._remapWidgetStripEventToCarousel('beforealign');
333                 this._remapWidgetStripEventToCarousel('afteralign');
334             },
335 
336             _remapWidgetStripEventToCarousel: function (eventName) {
337                 var self = this;
338                 this._remapEvent = this._remapEvent || function (evt) {
339                     if (evt.target === self._widgetStrip) {
340                         evt.target = self;
341                     }
342                 };
343                 this.addEventListener(eventName, this._remapEvent);
344             },
345 
346             _directAppend: function (widget) {
347                 return this.appendChildWidget(widget);
348             }
349 		});
350 
351         Carousel.orientations = {
352             VERTICAL: verticalOrientation,
353             HORIZONTAL: horizontalOrientation
354         };
355 		return Carousel;
356 	}
357 );
