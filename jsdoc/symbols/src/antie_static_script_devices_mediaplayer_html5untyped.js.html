 28 require.def(
 29     "antie/devices/mediaplayer/html5untyped",
 30     [
 31         "antie/runtimecontext",
 32         "antie/devices/device",
 33         "antie/devices/mediaplayer/html5"
 34     ],
 35     function(RuntimeContext, Device, HTML5MediaPlayer) {
 36         "use strict";
 37 
 38         /**
 39          * Main MediaPlayer implementation for HTML5 devices specifies a 'type'
 40          * attribute in the source element.          
 41          * This device modifier implements a new version of the function to
 42          * generate the source element without setting a type attribute.
 43          * @name antie.devices.mediaplayer.HTML5Untyped
 44          * @class
 45          * @extends antie.devices.mediaplayer.HTML5
 46          */
 47         var Player = HTML5MediaPlayer.extend( /** @lends antie.devices.mediaplayer.HTML5Untyped.prototype */ {
 48             init: function() {
 49                this._super();
 50             },
 51 
 52             _generateSourceElement: function(url) {
 53                 var device = RuntimeContext.getDevice();
 54                 var sourceElement = device._createElement('source');
 55                 sourceElement.src = url;
 56                 return sourceElement;
 57             }
 58         });
 59 
 60         var instance = new Player();
 61 
 62         // Mixin this MediaPlayer implementation, so that
 63         // device.getMediaPlayer() returns the correct implementation for the
 64         // device
 65         Device.prototype.getMediaPlayer = function() {
 66             return instance;
 67         };
 68 
 69         return Player;
 70     }
 71 );
