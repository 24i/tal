 27 require.def('antie/audiosource',
 28         ['antie/mediasource'],
 29         function(MediaSource) {
 30 			/**
 31 			 * An audio media source. Provides storage of source and type information about an audio media source.
 32 			 * @name antie.AudioSource
 33 			 * @class
 34 			 * @extends antie.MediaSource
 35 			 * @param {String} src The URI of the media.
 36 			 * @param {String} type The type of the media.
 37 			 */
 38             var AudioSource = MediaSource.extend(/** @lends antie.AudioSource.prototype */ {
 39 						/**
 40 						* Get the media type of this source. In this case <code>MediaType.MEDIA_TYPE_AUDIO</code>.
 41 						* @returns The media type of this content.
 42 						*/
 43 						getMediaType: function() {
 44 							return MediaSource.MEDIA_TYPE_AUDIO;
 45 						}						
 46                     });
 47 
 48             return AudioSource;
 49         });
