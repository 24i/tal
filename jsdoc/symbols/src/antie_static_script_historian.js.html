 27 require.def("antie/historian",
 28     [
 29         "antie/class"
 30     ],
 31     function(Class) {
 32         'use strict';
 33 
 34         // History stack is built up with most recent item at the FRONT (first), oldest items at the BACK (last). 
 35         var Historian = Class.extend({
 36             /**
 37              * @constructor
 38              * @ignore
 39              */
 40             init: function(currentUrl) {
 41                 var i;
 42                 
 43                 this._historyArray = currentUrl.split(Historian.HISTORY_TOKEN);
 44                 this._currentUrl = this._historyArray.shift(); // non-history portion of the URL
 45                 for(i = 0; i !== this._historyArray.length; i += 1) {
 46                     this._historyArray[i] =  Historian.HISTORY_TOKEN + this._historyArray[i];
 47                 }
 48             },
 49 
 50             /**
 51              * Returns a URL to navigate back to the previous application
 52              * @returns {String} The first history item in currentUrl as a non-encoded URL, or the empty string if there is no history.
 53              */
 54             back: function() {
 55                 var recent, remaining, fragmentSeparator, self;
 56                 self = this;
 57                 
 58                 function findRecentAndRemaining() {
 59                     var history = self._historyArray;
 60                     if (history.length > 0) {
 61                         recent = history[0].split(Historian.HISTORY_TOKEN)[1];
 62                         remaining = history.slice(1, history.length).join("");
 63                     } else {
 64                         recent = remaining = '';
 65                     }
 66                 }
 67                 
 68                 function processRoute() {
 69                     if(recent.indexOf(Historian.ROUTE_TOKEN) !== -1) {
 70                         fragmentSeparator = '';
 71                         recent = recent.replace(Historian.ROUTE_TOKEN, '#');
 72                     }
 73                 }
 74                 
 75                 function buildBackUrl() {
 76                     if(remaining) {
 77                         return recent + fragmentSeparator + remaining;
 78                     }
 79                     return recent;
 80                 }
 81                 
 82                 fragmentSeparator = '#';
 83                 findRecentAndRemaining();
 84                 processRoute();
 85                 return buildBackUrl();
 86             },
 87             
 88             /**
 89              * Returns a URL that allows navigation to the destination url while preserving history.
 90              * @param {String} destinationUrl, The non uri-encoded destination url including route fragment if applicable.
 91              * @returns {String} A non encoded uri with history information appended, the exact format of this is subject to change and should not be depended upon.
 92              */
 93             forward: function(destinationUrl) {
 94                 var fragmentSeparator, self;
 95                 self = this;
 96                 
 97                 function isRouteInDestination() {
 98                     return (destinationUrl.indexOf('#') !== -1);
 99                 }
100                 
101                 function replaceRouteInSource() {
102                     if (self._currentUrl.indexOf('#') !== -1) {
103                         self._currentUrl = self._currentUrl.replace('#', Historian.ROUTE_TOKEN);
104                     }
105                 }
106                 
107                 function addCurrentUrlToHistory() {
108                     self._historyArray.unshift(Historian.HISTORY_TOKEN + self._currentUrl);
109                 }
110 
111                 // Some devices have a de facto URL length limit of around 1000 characters, so trim URL by dropping history elements.
112                 // Keep the oldest history entry - drop oldest items from the middle.
113                 function trimUrlHistoryToLength() {
114                     while (self._historyArray.length > 1 && self.toString().length + destinationUrl.length > 999) {
115                         self._historyArray.splice(-2, 1);
116                     }
117                 }
118 
119                 if (this._currentUrl === '') {
120                     return destinationUrl;
121                 }
122                 
123                 replaceRouteInSource();
124                 addCurrentUrlToHistory();
125                 trimUrlHistoryToLength();
126                 fragmentSeparator = isRouteInDestination() ? '' : '#';
127                 return destinationUrl + fragmentSeparator + this.toString();
128             },
129 
130             /**
131              * Returns a string representation of the current history stack. This is not useful externally.
132              * @returns {String} A string representing the current history stack, to be appended to the current route within the application.
133              */
134             toString: function() {
135                 return this._historyArray.join("");
136             },
137 
138             /**
139              * Returns a Boolean to indicate whether the history stack contains valid return URLs. This excludes the 'return to broadcast' special case.
140              * @returns {Boolean} True if the history stack contains one or more valid return URLs.
141              */
142             hasHistory: function() {
143                 var historyMinimumLength = this.hasBroadcastOrigin() ? 2 : 1;
144                 return this._historyArray.length >= historyMinimumLength;
145             },
146 
147             /**
148              * Returns a Boolean to indicate whether the first entry in the history stack is the special 'broadcast' entry.
149              * @returns {Boolean} True if the first entry in the history stack is the special 'broadcast' entry.
150              */
151             hasBroadcastOrigin: function() {
152                 return this._historyArray.length > 0 && this._historyArray[this._historyArray.length - 1] === Historian.HISTORY_TOKEN + Historian.BROADCAST_ENTRY;
153             }
154         });
155         
156         Historian.HISTORY_TOKEN = '&*history=';
157         Historian.ROUTE_TOKEN = '&*route=';
158         Historian.BROADCAST_ENTRY = 'broadcast';
159         
160         return Historian;
161     }
162 );
