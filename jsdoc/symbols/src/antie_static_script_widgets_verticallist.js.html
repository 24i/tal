 27 require.def('antie/widgets/verticallist',
 28 	[
 29 	 	'antie/widgets/list',
 30 	 	'antie/events/keyevent'
 31 	],
 32 	function(List, KeyEvent) {
 33 		'use strict';
 34 
 35 		/**
 36 		 * The VerticalList widget is a container widget that supports spatial navigation between items using {@link KeyEvent.VK_UP} and {@link KeyEvent.VK_DOWN}.
 37 		 * @name antie.widgets.VerticalList
 38 		 * @class
 39 		 * @extends antie.widgets.List
 40 		 * @requires antie.events.KeyEvent
 41 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 42 		 * @param {antie.Formatter} [itemFormatter] A formatter class used on each data item to generate the list item child widgets.
 43 		 * @param {antie.DataSource|Array} [dataSource] An array of data to be used to generate the list items, or an asynchronous data source.
 44 		 */
 45 		return List.extend(/** @lends antie.widgets.VerticalList.prototype */ {
 46 			/**
 47 			 * @constructor
 48 			 * @ignore
 49 			 */
 50 			init: function(id, itemFormatter, dataSource) {
 51 				this._super(id, itemFormatter, dataSource);
 52 				this.addClass('verticallist');
 53 
 54 				var self = this;
 55 				this.addEventListener('keydown', function(e) { self._onKeyDown(e); });
 56 			},
 57 			/**
 58 			 * Key handler for vertical lists. Processes KeyEvent.VK_UP and KeyEvent.VK_DOWN keys and stops propagation
 59 			 * if the keypress is handled. Otherwise allows the event to be bubbled up to the parent widget to allow
 60 			 * spatial navigation out of the list.
 61 			 * @param {antie.events.KeyEvent} evt The key event.
 62 			 */
 63 			_onKeyDown: function(evt) {
 64 				if(evt.keyCode != KeyEvent.VK_UP && evt.keyCode != KeyEvent.VK_DOWN) {
 65 					return;
 66 				}
 67 
 68 				var _newSelectedIndex = this._selectedIndex;
 69 				var _newSelectedWidget = null;
 70 				do {
 71 					if(evt.keyCode == KeyEvent.VK_UP) {
 72 						_newSelectedIndex--;
 73 					} else if(evt.keyCode == KeyEvent.VK_DOWN) {
 74 						_newSelectedIndex++;
 75 					}
 76 					if(_newSelectedIndex < 0 || _newSelectedIndex >= this._childWidgetOrder.length) {
 77 						break;
 78 					}
 79 					var _widget = this._childWidgetOrder[_newSelectedIndex];
 80 					if(_widget.isFocusable()) {
 81 						_newSelectedWidget = _widget;
 82 						break;
 83 					}
 84 				} while(true);
 85 
 86 				if(_newSelectedWidget) {
 87 					this.setActiveChildWidget(_newSelectedWidget);
 88 					evt.stopPropagation();
 89 
 90 				}
 91 			}
 92 		});
 93 	}
 94 );
