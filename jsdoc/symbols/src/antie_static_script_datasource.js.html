 27 require.def('antie/datasource',
 28 	['antie/class'],
 29 	function(Class) {
 30 		'use strict';
 31 
 32 		/**
 33 		 * Utility class to wrap disparate functions into a common interface for binding to lists.
 34 		 * @name antie.DataSource
 35 		 * @class
 36 		 * @extends antie.Class
 37 		 * @param {antie.widgets.Component} component Component which 'owns' this data.
 38 		 * @param {Object} obj Object on which to call 'func' method.
 39 		 * @param {String} func Name of function to call.
 40 		 * @param {Array} args Arguments to pass the function.
 41 		 */
 42 		return Class.extend(/** @lends antie.DataSource.prototype */ {
 43 			/**
 44 			 * @constructor
 45 			 * @ignore
 46 			 */
 47 			init: function(component, obj, func, args) {
 48 				this._request = null;
 49 				this._component = component;
 50 				this._obj = obj;
 51 				this._func = func;
 52 				this._args = args;
 53 
 54 
 55 	
 56 				if(component) {
 57 					var self = this;
 58 
 59 					var beforeHideListener = function() {
 60 						component.removeEventListener('beforehide', beforeHideListener);
 61 						self.abort();
 62 					};
 63 					component.addEventListener('beforehide', beforeHideListener);
 64 				}
 65 			},
 66 			/**
 67 		 	 * Performs the request for data.
 68 		 	 * @param {Object} callbacks Object containing onSuccess and onError callback functions.
 69 			 */
 70 			load: function(callbacks) {
 71 				this._request = this._obj[this._func].apply(this._obj, [callbacks].concat(this._args || []));
 72 			},
 73 			/**
 74 		 	 * Aborts a currently loading request.
 75 			 */
 76 			abort: function() {
 77 				if(this._request) {
 78 					this._request.abort();
 79 				}
 80 			}
 81 		});
 82 	}
 83 );
