 27 require.def('antie/callbackmanager',
 28     [
 29         'antie/class'
 30     ],
 31     function(Class) {
 32         'use strict';
 33 
 34         /**
 35          * Utility class to deal with adding/removing callbacks and calling all current callbacks.
 36          * @name antie.CallbackManager
 37          * @class
 38          * @extends antie.Class
 39          */
 40         return Class.extend(/** @lends antie.CallbackManager.prototype */ {
 41             /**
 42              * @constructor
 43              * @ignore
 44              */
 45             init: function() {
 46                 this._callbacks = [ ];
 47             },
 48 
 49             /**
 50              * Add a callback.
 51              *
 52              * Note that failing to remove callbacks when you are finished with them can stop garbage collection
 53              * of objects/closures containing those callbacks and so create memory leaks in your application.
 54              * @param {Object} thisArg The object to use as "this" when calling the callback.
 55              * @param {Function} callback The callback function
 56              */
 57             addCallback: function(thisArg, callback) {
 58                 if (this._getIndexOf(thisArg,callback) === undefined) {
 59                     this._callbacks.push([thisArg, callback]);
 60                 }
 61             },
 62 
 63             /**
 64              * Remove the specified callback.
 65              * @param {Object} thisArg The object that was used as "this" when adding the callback.
 66              * @param {Function} callback The callback function
 67              */
 68             removeCallback: function(thisArg, callback) {
 69                 var foundIndex = this._getIndexOf(thisArg,callback);
 70 
 71                 if (foundIndex !== undefined) {
 72                     this._callbacks.splice(foundIndex, 1);
 73                 }
 74             },
 75 
 76             _getIndexOf: function(thisArg, callback) {
 77                 var result;
 78                 for (var i = 0; i < this._callbacks.length; i++) {
 79                     if (this._callbacks[i][0] === thisArg && this._callbacks[i][1] === callback) {
 80                         result = i;
 81                         break;
 82                     }
 83                 }
 84                 return result;
 85             },
 86 
 87 
 88             /**
 89              * Remove all callbacks.
 90              */
 91             removeAllCallbacks: function() {
 92                 this._callbacks = [ ];
 93             },
 94 
 95             /**
 96              * Call all callbacks, providing all supplied arguments to each of the calls.
 97              * @example manager.callAll(1);
 98              * @example manager.callAll(1,2,3,4);
 99              */
100             callAll: function() {
101                 // Convert arguments object to args array.
102                 var args = Array.prototype.slice.call(arguments);
103 
104                 for (var i = 0; i < this._callbacks.length; i++) {
105                     var thisArg = this._callbacks[i][0];
106                     var callback = this._callbacks[i][1];
107                     callback.apply(thisArg, args);
108                 }
109             }
110         });
111     }
112 );
