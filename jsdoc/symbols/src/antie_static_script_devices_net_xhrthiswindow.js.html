 27 require.def(
 28 	'antie/devices/net/xhrthiswindow',
 29 	[
 30 		'antie/devices/browserdevice'
 31 	],
 32 	function (Device) {
 33 		'use strict';
 34 
 35 		/**
 36 		 * Loads a resource from a URL.
 37          * This for devices where the activation object (i.e. 'this') within an event listener on an XMLHttpRequest instance
 38          * is a reference to the window object, not the XMLHttpRequest instance.
 39 		 * @param {String} url The URL to load.
 40 		 * @param {Object} opts Object containing onLoad and onError callback functions.
 41 		 * @returns The request object used to load the resource.
 42 		 */
 43 		Device.prototype.loadURL = function (url, opts) {
 44 			var xhr = new XMLHttpRequest();
 45 			xhr.open(opts.method || 'GET', url, true);
 46 			xhr.onreadystatechange = function () {
 47 				if (xhr.readyState === 4) {
 48 					xhr.onreadystatechange = null;
 49 					if (xhr.status >= 200 && xhr.status < 300) {
 50 						if (opts.onLoad) {
 51 							opts.onLoad(xhr.responseText, xhr.status);
 52 						}
 53 					} else {
 54 						if (opts.onError) {
 55 							opts.onError(xhr.responseText, xhr.status);
 56 						}
 57 					}
 58 				}
 59 			};
 60 			if (opts && opts.headers) {
 61 				for (var header in opts.headers) {
 62 					if (opts.headers.hasOwnProperty(header)) {
 63 						xhr.setRequestHeader(header, opts.headers[header]);
 64 					}
 65 				}
 66 			}
 67 			try {
 68 				xhr.send(opts.data || null);
 69 			} catch (ex) {
 70 				if (opts.onError) {
 71 					opts.onError(ex);
 72 				}
 73 			}
 74 			return xhr;
 75 		};
 76 	}
 77 );
