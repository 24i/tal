 27 require.def('antie/widgets/horizontalprogress',
 28 	[
 29 	 	'antie/widgets/container',
 30 	 	'antie/widgets/label'
 31 	],
 32 	function(Container, Label) {
 33 		'use strict';
 34 
 35 		/**
 36 		 * The horizontal progress widget provides a UI control for showing progress (with associated label)
 37 		 * @name antie.widgets.HorizontalProgress
 38 		 * @class
 39 		 * @extends antie.widgets.Container
 40 		 * @requires antie.widgets.Label
 41 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 42 		 * @param {Boolean} [showLabel] Pass <code>true</code> to show a label that indicates the value of the progress shown.
 43 		 * @param {Float} [initialValue] The initial value (default 0).
 44 		 */
 45 		return Container.extend(/** @lends antie.widgets.HorizontalProgress.prototype */ {
 46 			/**
 47 			 * @constructor
 48 			 * @ignore
 49 			 */
 50 			init: function(id, showLabel, initialValue) {
 51 				this._value = initialValue ? initialValue : 0;
 52 				this._moveHandle = false;
 53 				this._lastLeft = null;
 54 
 55 				this._super(id);
 56 				this.addClass('horizontalprogress');
 57 
 58 				if(showLabel) {
 59 					this._label = new Label(id + "_label", "");
 60 					this.addClass('haslabel');
 61 				}
 62 			},
 63 			/**
 64 			 * Renders the widget and any child widgets to device-specific output.
 65 			 * @param {antie.devices.Device} device The device to render to.
 66 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 67 			 */
 68 			render: function(device) {
 69 				this.outputElement = device.createContainer(this.id, this.getClasses());
 70 				this._leftElement = device.createContainer(this.id+'_left');
 71 				this._innerElement = device.createContainer(this.id+'_inner');
 72 				device.appendChildElement(this._leftElement, this._innerElement);
 73 				device.appendChildElement(this.outputElement, this._leftElement);
 74 
 75 				if(this._label) {
 76 					device.appendChildElement(this.outputElement, this._label.render(device));
 77 				}
 78 
 79 				this._moveInner();
 80 
 81 				return this.outputElement;
 82 			},
 83 			/**
 84 			 * Moves the inner element to show the current value.
 85 			 * @private
 86 			 */
 87 			_moveInner: function() {	
 88 				var device = this.getCurrentApplication().getDevice();
 89 				var elsize = device.getElementSize(this._leftElement);
 90 				var handleSize = device.getElementSize(this._innerElement)
 91 				var left = Math.floor(this._value * (elsize.width - handleSize.width));
 92 
 93 				if(left != this._lastLeft) {
 94 					this._lastLeft = left;
 95 
 96 					if(this._moveHandle) {
 97 						device.stopAnimation(this._moveHandle);
 98 					}
 99 	
100 					var config = device.getConfig();
101 					var animate = !config.widgets || !config.widgets.horizontalprogress || (config.widgets.horizontalprogress.animate !== false);
102 					this._moveHandle = device.moveElementTo({
103 						el: this._innerElement, 
104 						to: {
105 							left: left 
106 						},
107 						skipAnim: !animate
108 					});
109 				}
110 			},
111 			/**
112 			 * Returns the current value shown by the progress indicator.
113 			 */
114 			getValue: function() {
115 				return this._value;
116 			},
117 			/**
118 			 * Sets the current value to be shown by the progress indicator.
119 			 * @param {Float} val The value to show (between 0.0 and 1.0 inclusive).
120 			 */
121 			setValue: function(val) {
122 				this._value = val;
123 				this._moveInner();
124 			},
125 			/**
126 			 * Sets the text to show in the label.
127 			 * @param {String} val The text to show.
128 			 */
129 			setText: function(val) {
130 				if(this._label) {
131 					this._label.setText(val);
132 				}
133 			}
134 		});
135 	}
136 );
