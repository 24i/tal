 25 require.def ('antie/widgets/state_view_container', [
 26     'antie/widgets/container',
 27     'antie/events/keyevent',
 28     'antie/widgets/button'
 29 ], function(Container, KeyEvent, Button){
 30     'use strict';
 31 
 32     var keyMapping = { }
 33     keyMapping[KeyEvent.VK_UP] = 'up';
 34     keyMapping[KeyEvent.VK_DOWN] = 'down';
 35     keyMapping[KeyEvent.VK_LEFT] = 'left';
 36     keyMapping[KeyEvent.VK_RIGHT] = 'right';
 37     keyMapping[KeyEvent.VK_BACK] = 'back';
 38 
 39      var StateViewContainer = Container.extend( {
 40 
 41         init:function(controller) {
 42             this._super();
 43 
 44             var self = this;
 45 
 46 
 47             this.focusButtonHack = new Button();
 48             this.appendChildWidget(this.focusButtonHack);
 49 
 50             this.addEventListener("select",  function() { controller.select(); });
 51 
 52             this.addEventListener("keydown", function(evt){
 53                var mappedFunctionName = keyMapping[evt.keyCode];
 54                 if (mappedFunctionName){
 55                    controller[mappedFunctionName]();
 56                }
 57             });
 58 
 59             this.addEventListener("afterhide", function(evt) {
 60                   self.focusHack();
 61             });
 62 
 63             this.addEventListener("beforeshow", function(evt) {
 64                    self.getCurrentApplication().setActiveComponent(evt.component.id);
 65             });
 66 
 67         },
 68         focusHack: function() {
 69            this._isFocussed = true;
 70            this.focusButtonHack.focus();
 71         }
 72     } )
 73 
 74     return StateViewContainer;
 75 
 76 });
