 24 require.def('antie/widgets/carousel/strips/utility/initstate',
 25     [
 26         'antie/widgets/carousel/strips/utility/state'
 27     ],
 28     function (State) {
 29         'use strict';
 30         var InitState;
 31         InitState = State.extend({
 32             init: function () {
 33             },
 34 
 35             append: function (context, parent, widget) {
 36                 this._render(widget);
 37                 this._attach(context, parent, widget, 'appendChildElement');
 38             },
 39 
 40             prepend: function (context, parent, widget) {
 41                 this._render(widget);
 42                 this._attach(context, parent, widget, 'prependChildElement');
 43             },
 44 
 45             detach: function (context, widget) {
 46             },
 47 
 48             hasLength: function () {
 49                 return false;
 50             },
 51 
 52             inView: function () {
 53                 return false;
 54             },
 55 
 56             _getDevice: function (widget) {
 57                 return widget.getCurrentApplication().getDevice();
 58             },
 59 
 60             _render: function (widget) {
 61                 var device = this._getDevice(widget);
 62                 widget.render(device);
 63             },
 64 
 65             _attach: function (context, parent, widget, attachMethodName) {
 66                 var device = this._getDevice(widget);
 67                 device[attachMethodName](parent.outputElement, widget.outputElement);
 68                 context.setState('ATTACHED');
 69             }
 70         });
 71 
 72         return InitState;
 73     }
 74 );
