 37 		/**
 38 		 * The HorizontalCarousel widget extends the HorizontalList widget to modify the animation behaviour to render a carousel rather than a list.
 39 		 * @name antie.widgets.HorizontalCarousel
 40 		 * @class
 41 		 * @extends antie.widgets.HorizontalList
 42 		 * @requires antie.widgets.List
 43 		 * @requires antie.events.KeyEvent
 44 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 45 		 * @param {antie.Formatter} [itemFormatter] A formatter class used on each data item to generate the list item child widgets.
 46 		 * @param {antie.DataSource|Array} [dataSource] An array of data to be used to generate the list items, or an asynchronous data source.
 47          * @deprecated This class is deprecated in favour of of antie.widgets.carousel
 48 		 */
 49 		var HorizontalCarousel = HorizontalList.extend(/** @lends antie.widgets.HorizontalCarousel.prototype */ {
 50 			/**
 51 			 * @constructor
 52 			 * @ignore
 53 			 */
 54 			init: function(id, itemFormatter, dataSource, overrideAnimation, activeWidgetAlignment, BeforeSelectedItemChangeEvent) {
 55 				this._prefixClones = 0;
 56 				this._wrapMode = HorizontalCarousel.WRAP_MODE_VISUAL;
 57 				this._viewportMode = HorizontalCarousel.VIEWPORT_MODE_NONE;
 58 				this._viewportSize = 0;
 59 				this._activateThenScroll = false;
 60 				this._scrollHandle = null;
 61 				this._keepHidden = false;
 62 				this._multiWidthItems = false;
 63 				this._overrideAnimation = overrideAnimation;
 64 				this._activeWidgetAlignment = activeWidgetAlignment || HorizontalCarousel.ALIGNMENT_CENTER;
 65 				this._activeWidgetAnimationFPS = 25;
 66 				this._activeWidgetAnimationDuration = 840;
 67 				this._activeWidgetAnimationEasing = 'easeFromTo';
 68 				this._nodeOffset = 0;
 69 				this._childWidgetsInDocument = [];
 70 				this._paddingItemsCreated = false;
 71 				this._super(id, itemFormatter, dataSource);
 72 				this.addClass('horizontalcarousel');
 73 
 74 				var self = this;
 75 				this.addEventListener('databound', function(evt) {
 76 					if(evt.target !== self) {
 77 						return;
 78 					}
 79 
 80 					// Delaying this because our mask might not have a size yet. Shouldn't be a problem for dynamic data
 81 					// source, only for static carousels (such as the menu). It's been found to need to differ on
 82 					// devices.
 83 					var config = self.getCurrentApplication().getDevice().getConfig();
 84 					var delay = 100;
 85 					if(config.widgets && config.widgets.horizontalcarousel && config.widgets.horizontalcarousel.bindDelay) {
 86 						delay = config.widgets.horizontalcarousel.bindDelay;
 87 					}
 88 					setTimeout(function() { self._onDataBound(evt); }, delay);
 89 				});

 91 			/**
 92 			 * Renders the widget and any child widgets to device-specific output.
 93 			 * @param {antie.devices.Device} device The device to render to.
 94 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 95 			 */
 96 			render: function(device) {
 97 				// keep the element hidden until data is bound and items created
 98 				if(!this._maskElement) {
 99 					this._maskElement = device.createContainer(this.id + '_mask', ['horizontallistmask', 'notscrolling']);
100 				} else {
101 					device.clearElement(this._maskElement);
102 					this._childWidgetsInDocument = [];
103 				}
104 
105 				if(this._viewportMode != HorizontalCarousel.VIEWPORT_MODE_DOM) {
106 					device.appendChildElement(this._maskElement, this._super(device));
107 				} else {
108 					if(!this._dataBound && this._dataSource && this._itemFormatter) {
109 						this._createDataBoundItems(device);
110 					}
111 					if(!this.outputElement) {
112 						if(this._renderMode == List.RENDER_MODE_LIST) {
113 							this.outputElement = device.createList(this.id, this.getClasses());
114 						} else {
115 							this.outputElement = device.createContainer(this.id, this.getClasses());
116 						}
117 					}
118 					device.appendChildElement(this._maskElement, this.outputElement);
119 				}
120 
121 				// Don't hide if we're never going to databind (or it'll never be shown);
122 				if(this._dataSource) {
123 					device.hideElement({
124 						el: this._maskElement,
125 						skipAnim: true
126 					});
127 				} else {
128 					var self = this;
129 					var config = device.getConfig();
130 					var delay = 100;
131 					if(config.widgets && config.widgets.horizontalcarousel && config.widgets.horizontalcarousel.bindDelay) {
132 						delay = config.widgets.horizontalcarousel.bindDelay;
133 					}
134 					setTimeout(function() { self._onDataBound(); }, delay);
135 				}
136 
137 				return this._maskElement;
138 			},
139 			refreshViewport: function() {
140 				var _centerWidget = this._activeChildWidget || this._childWidgetOrder[0];
141 				if(!_centerWidget) {
142 					return;
143 				}
144 
145 				if(this._viewportMode == HorizontalCarousel.VIEWPORT_MODE_DOM) {
146 					this.setAutoRenderChildren(true);
147 
148 					var device = this.getCurrentApplication().getDevice();
149 					var i, index, elpos, elsize;
150 
151 					if(!_centerWidget.outputElement) {
152 						_centerWidget.outputElement = _centerWidget.render(device);
153 					}
154 
155 					// iterate through the widgets currently in the document
156 					// removing any that are no-longer in or near the viewport
157 					for(i=0; i<this._childWidgetsInDocument.length; i++) {
158 						index = i + this._nodeOffset;
159 						if(index < this._selectedIndex - this._viewportSize || index > this._selectedIndex + this._viewportSize) {
160 							if(this._childWidgetsInDocument[i].outputElement) {
161 								device.removeElement(this._childWidgetsInDocument[i].outputElement);
162 							}
163 						}
164 					}
165 
166 					// find the elements that are in the view port and add them
167 					// to the document (and keep a record of them)
168 					this._childWidgetsInDocument = [];
169 					var start = ((start = this._selectedIndex - this._viewportSize) < 0) ? 0 : start;
170 					for(i=start; (i <= this._selectedIndex + this._viewportSize) && (i < this._childWidgetOrder.length) ; i++) {
171 						index = i - start + this._prefixClones;
172 
173 						this._childWidgetOrder[i].addClass('inviewport');
174 						if(!this._childWidgetOrder[i].outputElement) {
175 							this._childWidgetOrder[i].outputElement = this._childWidgetOrder[i].render(device);
176 						}
177 						if(!device.getElementParent(this._childWidgetOrder[i].outputElement)) {
178 							device.insertChildElementAt(this.outputElement, this._childWidgetOrder[i].outputElement, index);
179 						}
180 						this._childWidgetsInDocument.push(this._childWidgetOrder[i]);
181 					}
182 					this._nodeOffset = this._selectedIndex - this._viewportSize;
183 					if(this._nodeOffset < 0) this._nodeOffset = 0;
184 
185 					// reposition the carousel over the active item
186 					elpos = device.getElementOffset(_centerWidget.outputElement);
187 					elsize = device.getElementSize(_centerWidget.outputElement);
188 					this._alignToElement(_centerWidget.outputElement, true);
189 					//device.scrollElementToCenter(this._maskElement, elpos.left + (elsize.width / 2), null, true);
190 
191 					this.setAutoRenderChildren(false);
192 				} else if((this._viewportMode == HorizontalCarousel.VIEWPORT_MODE_CLASSES) && this.outputElement && _centerWidget.outputElement) {
193 					device = this.getCurrentApplication().getDevice();
194 					elpos = device.getElementOffset(_centerWidget.outputElement);
195 					elsize = device.getElementSize(_centerWidget.outputElement);
196 					var maskSize = device.getElementSize(this._maskElement);
197 					var nodes = device.getChildElementsByTagName(this.outputElement,
198 							this._renderMode === List.RENDER_MODE_LIST ? 'li' : 'div'
199 					);
200 
201 					var viewportLeft = (elpos.left + (elsize.width / 2)) - (maskSize.width / 2);
202 					var viewportRight = (elpos.left + (elsize.width / 2)) + (maskSize.width / 2);
203 					var nearViewportLeft = (elpos.left + (elsize.width / 2)) - (maskSize.width * 1.5);
204 					var nearViewportRight = (elpos.left + (elsize.width / 2)) + (maskSize.width * 1.5);
205 
206 					var count = this._childWidgetOrder.length;
207 
208 					for(var i=0; i<nodes.length; i++) {
209 						var node = nodes[i];
210 						if(!node.cloneOfWidget) {
211 							var w = this._childWidgetOrder[i - this._prefixClones];
212 							if(w) {
213 								w.removeClass('inviewport');
214 								w.removeClass('nearviewport');
215 							}
216 						}
217 					}
218 
219 					for(var i=0; i<nodes.length; i++) {
220 						var node = nodes[i];
221 						var nodepos = device.getElementOffset(node);
222 						var nodesize = device.getElementSize(node);
223 						var w = node.cloneOfWidget || this._childWidgetOrder[i - this._prefixClones];
224 						if(!w) { continue; }
225 
226 						if(((nodepos.left + nodesize.width) >= viewportLeft) && (nodepos.left < viewportRight)) {
227 							// work out which elements are on screen and given them a 'inviewport' class
228 							if(node.cloneOfWidget) {
229 								device.removeClassFromElement(node, 'nearviewport');
230 								device.addClassToElement(node, 'inviewport');
231 							} 
232 							w.removeClass('nearviewport');
233 							w.addClass('inviewport');
234 						} else if(((nodepos.left + nodesize.width) >= nearViewportLeft) && (nodepos.left < nearViewportRight)) {
235 							// work out which elements are near the screen, and give them a 'nearviewport' class
236 							if(node.cloneOfWidget) {
237 								device.removeClassFromElement(node, 'inviewport');
238 								device.addClassToElement(node, 'nearviewport');
239 							} 
240 							w.removeClass('inviewport');
241 							w.addClass('nearviewport');
242 						} else if(node.cloneOfWidget) {
243 							device.removeClassFromElement(node, 'inviewport');
244 							device.removeClassFromElement(node, 'nearviewport');
245 						}
246 					}
247 				}
248 			},

259 			 * Attempt to set focus to the given child widget.
260 			 * @param {antie.widgets.Widget} widget The child widget to set focus to.
261 			 * @param {Boolean} [reposition] Set to <code>true</code> if you want to scroll the carousel to the new item.
262 			 * @returns Boolean true if the child widget was focusable, otherwise boolean false.
263 			 */
264 			setActiveChildWidget: function(widget, reposition) {
265 				var moved = this._super(widget);
266 
267 				if(this._activeChildWidget && this.outputElement && reposition) {
268 					if(this._viewportMode != HorizontalCarousel.VIEWPORT_MODE_DOM) {
269 						var device = this.getCurrentApplication().getDevice();
270 						var elpos = device.getElementOffset(this._activeChildWidget.outputElement);
271 						var elsize = device.getElementSize(this._activeChildWidget.outputElement);
272 						this._alignToElement(this._activeChildWidget.outputElement, true);
273 					}
274 					this.refreshViewport();
275 				}
276 
277 				return moved;

279 			/**
280 			 * Attempts to set focus to the child widget at the given index.
281 			 * @see #setActiveChildWidget
282 			 * @param {Integer} index Index of the child widget to set focus to.
283 			 * @returns Boolean true if the child widget was focusable, otherwise boolean false.
284 			 */
285 			setActiveChildIndex: function(index, reposition) {
286 				if(index < 0 || index >= this._childWidgetOrder.length) {

289 				return this.setActiveChildWidget(this._childWidgetOrder[index], reposition);

292 				this._prefixClones = 0;

295 			rebindDataSource: function() {
296 				var device = this.getCurrentApplication().getDevice();
297 				var config = device.getConfig();
298 				var animate = !config.widgets || !config.widgets.horizontalcarousel || (config.widgets.horizontalcarousel.fade !== false);
299 
300 				var self = this;
301 				var func = this._super;
302 				device.hideElement({
303 					el: this._maskElement,
304 					skipAnim: !animate,
305 					onComplete: function() {
306 						func.call(self);
307 					}
308 				});

310 			/**
311 			 * Handle key events to scroll the carousel.
312 			 * @private
313 			 */
314 			_onKeyDown: function(evt) {
315 				// This event handler is already bound (int HorizontalList), we override it to add wrapping logic
316 
317 				// Block all movement if the carousel is scrolling
318 				if(this._scrollHandle && (
319 					evt.keyCode == KeyEvent.VK_LEFT ||
320 					evt.keyCode == KeyEvent.VK_RIGHT ||
321 					evt.keyCode == KeyEvent.VK_UP ||
322 					evt.keyCode == KeyEvent.VK_DOWN
323 				)) {
324 					evt.stopPropagation();
325 					return;
326 				}
327 
328 				switch (evt.keyCode) {
329 					case KeyEvent.VK_LEFT:
330 						if (this.selectPreviousChildWidget()) {
331 							evt.stopPropagation();
332 						}
333 						
334 						break;
335 					case KeyEvent.VK_RIGHT:
336 						if (this.selectNextChildWidget()) {
337 							evt.stopPropagation();
338 						}
339 						
340 						break;
341 				}
342 			},

344 			/**
345 			 * DataBound event handler. Clone carousel items to allow infinite scrolling.
346 			 * @private
347 			 */
348 			_onDataBound: function(evt) {
349 				var self = this;
350 				var application = this.getCurrentApplication();
351 				if(!application) {
352 					// application has been destroyed, abort
353 					return;
354 				}
355 				var device = application.getDevice();
356 
357 				if(this._childWidgetOrder.length > 0) {
358 					// TODO: tidy up
359 					var moveLastToFirst = function() {
360 						var last = self._items[self._childWidgetOrder[self._childWidgetOrder.length - 1]];
361 						var first = self._items[self._childWidgetOrder[0]];
362 						if(last) {
363 							device.removeElement(last.outputElement);
364 							device.prependElement(self.outputElement, last.outputElement);
365 							var lastWidget = self._childWidgetOrder.pop();
366 							self._childWidgetOrder.splice(0, 0, lastWidget);
367 							self._selectedIndex++;
368 						}
369 					}
370 
371 					// How this implements wrap-around infinite scrolling:
372 					// * Prepend a copy of the last page of items
373 					// * Append a copy of the first page of items
374 					// * Scroll through items normally
375 					// * When moving left and hitting index -1 flip scrollLeft to end item
376 					// * When moving right and hitting out-of-bounds index, flip scrollLeft to start item
377 					//
378 					// Adding clones means higher memory usage and more to scroll, but this approach works
379 					// with any number of items and allows for smooth transitions between start and end.
380 
381 					var maskSize = device.getElementSize(this._maskElement);
382 					var copyWidth = 0;
383 					var visibleWidth = 0;
384 					var i = 0;
385 					var prefixClones = 0;
386 				
387 					this._nodeOffset = 0;
388 					this._childWidgetsInDocument = [];
389 
390 					if(this._viewportMode == HorizontalCarousel.VIEWPORT_MODE_NONE) {
391 						for(var w=0; w<this._childWidgetOrder.length; w++) {
392 							this._childWidgetOrder[w].addClass('inviewport');
393 						}
394 					}
395 
396 					if(this._wrapMode != HorizontalCarousel.WRAP_MODE_VISUAL) {
397 						if(this._paddingItemsCreated) {
398 							var paddingFunction = (this._renderMode == List.RENDER_MODE_LIST)
399 													? device.createListItem
400 													: device.createContainer;
401 
402 							var leftPadding = paddingFunction.call(device, this.id + 'PaddingLeft', ['viewportPadding', 'viewportPaddingLeft']);
403 							device.setElementSize(leftPadding, {width: maskSize.width});
404 							device.prependChildElement(this.outputElement, leftPadding);
405 
406 							var rightPadding = paddingFunction.call(device, this.id + 'PaddingRight', ['viewportPadding', 'viewportPaddingRight']);
407 							device.setElementSize(rightPadding, {width: maskSize.width});
408 							device.appendChildElement(this.outputElement, rightPadding);
409 
410 							prefixClones = 1;
411 						}
412 						this._paddingItemsCreated = true;
413 					} else {
414 						// TODO: there's an optimisation we could do here, but it may not work with all carousels
415 						// TODO: especially those that have different sized elements.
416 						// TODO:
417 						// TODO: For carousels with items that all have the same width we can use 'maskSize.width / 2'
418 
419 						var requiredWidth = this._multiWidthItems ? maskSize.width : Math.ceil(maskSize.width / 2);
420 
421 						while(copyWidth < requiredWidth) {
422 							var w = this._childWidgetOrder[i];
423 							var clone = device.cloneElement(w.outputElement, true, "clone", "_clone");
424 							clone.cloneOfWidget = w;
425 
426 							if(w.hasClass('active')) {
427 								device.removeClassFromElement(clone, 'active', true);
428 							}
429 							if(w.hasClass('focus')) {
430 								device.removeClassFromElement(clone, 'focus', true);
431 								device.removeClassFromElement(clone, 'buttonFocussed', true);
432 							}
433 							device.appendChildElement(this.outputElement, clone);
434 							var w = device.getElementSize(w.outputElement).width;
435 							if(i === 0 && this._childWidgetOrder.length !== 1) {
436 								requiredWidth += w;
437 							}
438 							copyWidth += w;
439 							i++;
440 							if(i == this._childWidgetOrder.length) i = 0;
441 						}
442 
443 						copyWidth = 0;
444 						i = this._childWidgetOrder.length-1;
445 						while(copyWidth < requiredWidth) {
446 							var w = this._childWidgetOrder[i];
447 							var clone = device.cloneElement(w.outputElement, true, "clone", "_clone");
448 							clone.cloneOfWidget = w;
449 
450 							if(w.hasClass('active')) {
451 								device.removeClassFromElement(clone, 'active', true);
452 							}
453 							if(w.hasClass('focus')) {
454 								device.removeClassFromElement(clone, 'focus', true);
455 							}
456 
457 							device.prependChildElement(this.outputElement, clone);
458 							copyWidth += device.getElementSize(w.outputElement).width;
459 							i--;
460 							prefixClones++;
461 							if(i == -1) i = this._childWidgetOrder.length-1;
462 						}
463 					}
464 
465 					this._prefixClones = prefixClones;
466 
467 					// TODO: we shouldn't really do this here - we can't assume the preselected item is at index 0
468 					// TODO: or at least support moving to the correct item when the selected index is change
469 					// TODO: from an external class
470 					if(this._activeChildWidget && (this._viewportMode != HorizontalCarousel.VIEWPORT_MODE_DOM)) {
471 						var elpos = device.getElementOffset(this._activeChildWidget.outputElement);
472 						var elsize = device.getElementSize(this._activeChildWidget.outputElement);
473 						this._alignToElement(this._activeChildWidget.outputElement, true);
474 					}
475 
476 					this.refreshViewport();
477 				}
478 
479 				// everything is now in place, show the carousel				
480                 this.show({});
481 			},

484              * Shows the carousel.
485              * @param {Object}    options Details of the element to be shown, with optional parameters.
486              * @param {Boolean} [options.skipAnim] By default the showing of the element will be animated (faded in). Pass <code>true</code> here to prevent animation.
487              * @param {Function} [options.onComplete] Callback function to be called when the element has been shown.
488              * @param {Number}    [options.fps=25] Frames per second for fade animation.
489              * @param {Number}    [options.duration=840] Duration of fade animation, in milliseconds (ms).
490              * @param {String}    [options.easing=linear] Easing style for fade animation.
491              * @returns Boolean true if animation was called, otherwise false
492              */
493             show: function( options ){
494               //  this._super( options );
495                 var application = this.getCurrentApplication();
496                 if(!application) {
497                     return;
498                 }
499                 var device = application.getDevice();
500 
501                 if(!this._keepHidden) {
502                     var config = device.getConfig();
503                     var animate = !config.widgets || !config.widgets.horizontalcarousel || (config.widgets.horizontalcarousel.fade !== false);
504 
505                     options.el = this._maskElement;
506                     options.skipAnim = !animate;
507 
508                     return device.showElement(options);
509                 }
510                 return false;

513 			/**
514 			 * Set whether to support wrapping within the carousel.
515 			 * @param {Integer} wrapMode 	Pass <code>HorizontalCarousel.WRAP_MODE_NONE</code> for no wrapping.
516 			 * 				Pass <code>HorizontalCarousel.WRAP_MODE_NAVIGATION_ONLY</code> to allow navigation to wrap.
517 			 * 				Pass <code>HorizontalCarousel.WRAP_MODE_VISUAL</code> to visually wrap the carousel (includes navigation).
518 			 */
519 			setWrapMode: function(wrapMode) {
520 				if(this._viewportMode == HorizontalCarousel.VIEWPORT_MODE_DOM) {
521 					if(wrapMode == HorizontalCarousel.WRAP_MODE_VISUAL) {
522 						throw new Error('HorizontalCarousel::setWrapMode - VIEWPORT_MODE_DOM not supported for WRAP_MODE_VISUAL');
523 					}
524 				}
525 				this._wrapMode = wrapMode;
526 			},
527 			/**
528 			 * Set method used to control which carousel items are in this rendered DOM
529 			 * @param {Integer} viewportMode One of <code>HorizontalCarousel.VIEWPORT_MODE_NONE</code>,
530 			 *							   <code>HorizontalCarousel.VIEWPORT_MODE_DOM</code> or 
531 			 *							   <code>HorizontalCarousel.VIEWPORT_MODE_CLASSES</code>.
532 			 * @param {Integer} size		 Number of items in the viewport.
533 			 */
534 			setViewportMode: function(viewportMode, size) {
535 				if(this._wrapMode == HorizontalCarousel.WRAP_MODE_VISUAL) {
536 					if(viewportMode == HorizontalCarousel.VIEWPORT_MODE_DOM) {
537 						throw new Error('HorizontalCarousel::setViewportMode - VIEWPORT_MODE_DOM not supported for WRAP_MODE_VISUAL');
538 					}
539 				}

707 			/**
708 			 * Finds a selectable widget in the specified direction and moves
709 			 * the focus to it.
710 			 */
711 			_moveChildWidgetSelection: function(direction) {
712 				var device = this.getCurrentApplication().getDevice();
713 
714 				if (this._scrollHandle) {
715 					device.stopAnimation(this._scrollHandle);
716 				}
717 
718 				var _newIndex = this._selectedIndex;
719 				var _nodeIndex = this._selectedIndex + this._prefixClones;
720 				var _oldSelectedWidget = this._activeChildWidget;
721 				var _newSelectedWidget = null;
722 				var _centerElement = null;
723 				var _wrapped = false;
724 
725 				do {
726 					if (direction == HorizontalCarousel.SELECTION_DIRECTION_LEFT) {
727 						_nodeIndex--;
728 
729 						if (_newIndex > 0) {
730 							_newIndex--;
731 						} else if (this._wrapMode && this._childWidgetOrder.length > 3) { /* Only wrap when more than 3 items */
732 							_newIndex = this._childWidgetOrder.length - 1;
733 							_wrapped = true;
734 						} else {
735 							break;
736 						}
737 					} else if (direction == HorizontalCarousel.SELECTION_DIRECTION_RIGHT) {
738 						_nodeIndex++;
739 
740 						if (_newIndex < this._childWidgetOrder.length - 1) {
741 							_newIndex++;
742 						} else if (this._wrapMode && this._childWidgetOrder.length > 3) { /* Only wrap when more than 3 items */
743 							_newIndex = 0;
744 							_wrapped = true;
745 						} else {
746 							break;
747 						}
748 					}
749 
750 					var _widget = this._childWidgetOrder[_newIndex];
751 					if (_widget.isFocusable()) {
752 						_newSelectedWidget = _widget;
753 						break;
754 					}
755 				} while(true);
756 
757 				// Centre on a cloned carousel item if we're wrapping to the other end.
758 				// Otherwise, just go to the new selected item.
759 				if (_wrapped && this._wrapMode === HorizontalCarousel.WRAP_MODE_VISUAL)	{
760 					_centerElement = this._getWrappedElement(direction, _oldSelectedWidget.outputElement);
761 				}
762 				else if (_newSelectedWidget) {
763 					_centerElement = _newSelectedWidget.outputElement;
764 				}
765 				
766 				if (_newSelectedWidget && _centerElement) {
767 					var self = this;
768 
769 					this.bubbleEvent(new BeforeSelectedItemChangeEvent(this, _newSelectedWidget, _newIndex));
770 
771 					var scrollDone = function() {
772 						if (!self._activateThenScroll) {
773 							self.setActiveChildWidget(_newSelectedWidget);
774 							self._selectedIndex = _newIndex;
775 						}
776 
777 						// If we've just moved to the fake item off the end of the wrapped carousel,
778 						// snap to the real item at the opposite end when the animation completes.
779 						if (_wrapped && self._wrapMode === HorizontalCarousel.WRAP_MODE_VISUAL) {
780 							self._alignToElement(self._activeChildWidget.outputElement, true);
781 						}
782 
783 						// Allow the carousel to move again.
784 						self.refreshViewport();
785 						self._scrollHandle = null;
786 					}
787 
788 					if(this._activateThenScroll) {
789 						this.setActiveChildWidget(_newSelectedWidget);
790 						this._selectedIndex = _newIndex;
791 					}
792 
793 					// If the offset is zero it means the element is not in the DOM, i.e. the other end of the carousel, scroll to 1 pixel
794 					// otherwise in CSS3 the scrollDone event will never be called
795 					var elpos = device.getElementOffset(_centerElement);
796 					if(elpos.left == 0) {
797 						elpos.left = 1;
798 					}
799 
800 					var config = device.getConfig();
801 					var animate = !config.widgets || !config.widgets.horizontalcarousel || (config.widgets.horizontalcarousel.animate !== false);
802 					this._scrollHandle = this._alignToElement(_centerElement, this._isAnimationOverridden(animate), scrollDone);
803 
804 					return true;
805 				} else {
806 					return false;
807 				}
808 			},
809 
810 			_getWrappedElement : function(direction, element) {
811 				// Return the next/previous widget in the carousel - used to grab dummy widgets
812 				// used in the visual wrapping mode.
813 				do {
814 					element = (direction === HorizontalCarousel.SELECTION_DIRECTION_RIGHT ? element.nextSibling : element.previousSibling);
815 				} while (element && element.nodeType != 1);
816 				return element;
817 			},
818 
819 			_isAnimationOverridden : function(animate) {
820 				return this._overrideAnimation || !animate; 
821 			},
822 
823 			_alignToElement: function(el, skipAnimation, onAnimationCompleteHandler) {
824 				var device = this.getCurrentApplication().getDevice();
825 				var widgetpos = device.getElementOffset(el);
826 				var widgetsize = device.getElementSize(el);
827 				var masksize = device.getElementSize(this._maskElement);
828 				var offset = this._activeWidgetAlignmentOffset || 0;
829 
830 				var newLeftPosition;
831 				switch(this._activeWidgetAlignment) {
832 					case HorizontalCarousel.ALIGNMENT_CENTER:
833 						newLeftPosition = widgetpos.left - (masksize.width - widgetsize.width)/2 + offset;
834 						break;
835 					case HorizontalCarousel.ALIGNMENT_LEFT:
836 						newLeftPosition = widgetpos.left + offset;
837 						break;
838 					case HorizontalCarousel.ALIGNMENT_RIGHT:
839 						newLeftPosition = widgetpos.left - (masksize.width - widgetsize.width) - offset;
840 						break;
841 				}
842 
843 				return device.scrollElementTo({
844 					el: this._maskElement, 
845 					to: {
846 						left: newLeftPosition 
847 					},
848 					fps : this.getWidgetAnimationFPS(),
849 					duration: this.getWidgetAnimationDuration(),
850 					easing: this.getWidgetAnimationEasing(),
851 					skipAnim: skipAnimation, 
852 					onComplete: onAnimationCompleteHandler
853 				});
854 			}
855 		});
