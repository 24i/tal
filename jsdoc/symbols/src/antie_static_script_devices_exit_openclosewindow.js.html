 29 require.def(
 30 	'antie/devices/exit/openclosewindow',
 31 	['antie/devices/browserdevice'],
 32 	function(Device) {
 33         'use strict';
 34 
 35         /**
 36          * Exits the application by invoking window.open() on the current window,
 37          * then window.close().
 38          */
 39         Device.prototype.exit = function() {
 40             // Workaround to make the browser think this window was opened via script
 41             window.open('', '_self');
 42             
 43             // Close the current window
 44             window.close();
 45         };
 46     }
 47 );
