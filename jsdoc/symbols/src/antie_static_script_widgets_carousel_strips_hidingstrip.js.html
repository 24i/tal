 24 require.def('antie/widgets/carousel/strips/hidingstrip',
 25     [
 26         'antie/widgets/carousel/strips/cullingstrip',
 27         'antie/widgets/carousel/strips/utility/widgetcontext',
 28         'antie/widgets/carousel/strips/utility/visibilitystates'
 29     ],
 30     function (CullingStrip, WidgetContext, VISIBILITY_STATES) {
 31         'use strict';
 32         var HidingStrip;
 33         /**
 34          * A container for the widgets displayed within a carousel, in which widgets are only created when they
 35          * come into view, and that sets widgets to visibility: hidden and opacity: 0 when they go out of view
 36          * @name antie.widgets.carousel.strips.HidingStrip
 37          * @class
 38          * @extends antie.widgets.carousel.strips.CullingStrip
 39          * @param {String} id The unique ID of the widget.
 40          * @param {Object} orientation an object representing the strip's orientation.
 41          * One of antie.widgets.carousel.orientations.Horizontal or antie.widgets.carousel.orientations.Vertical
 42          */
 43         HidingStrip = CullingStrip.extend(/** @lends antie.widgets.carousel.strips.HidingStrip.prototype */{
 44             createContext: function (widget, parent) {
 45                 return new WidgetContext(widget, parent, VISIBILITY_STATES);
 46             }
 47         });
 48         return HidingStrip;
 49     }
 50 );
