 27 require.def('antie/widgets/scrubbar',
 28 	['antie/widgets/horizontalslider'],
 29 	function(HorizontalSlider) {
 30 		'use strict';
 31 
 32 		/**
 33 		 * The ScrubBar provides a media scrub bar showing current position and portion of the stream that is currently buffered.
 34 		 * @name antie.widgets.ScrubBar
 35 		 * @class
 36 		 * @extends antie.widgets.HorizontalSlider
 37 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 38 		 * @param {double} [initialValue] Initial value for the slider.
 39 		 * @param {double} [smallIncrement] Amount to increase/decrease the value by when VK_LEFT or VK_RIGHT is pressed.
 40 		 * @param {double} [largeIncrement] Amount to increase/decrease the value by when VK_LEFT or VK_RIGHT is held down.
 41 		 * @param {double} [largeIncrementAfter] Number of smallIncrements to perform until switching to largeIncrement when key is held down.
 42 		 */
 43 		return HorizontalSlider.extend(/** @lends antie.widgets.ScrubBar.prototype */ {
 44 			init: function(id, initialValue, smallIncrement, largeIncrement, largeIncrementAfter) {
 45 				this._super(id, initialValue, smallIncrement, largeIncrement, largeIncrementAfter);
 46 
 47 				this._bufferedRange = {start: 0, end: 0};
 48 				this._lastBufferLeft = -1;
 49 				this._lastBufferWidth = -1;
 50 			},
 51 
 52 			/**
 53 			 * Renders the widget and any child widgets to device-specific output.
 54 			 * @param {antie.devices.Device} device The device to render to.
 55 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 56 			 */
 57 			render: function(device) {
 58 				this.outputElement = this._super(device);
 59 				this._buffer = device.createButton(this.id+'_buffer');
 60 				device.addClassToElement(this._buffer, 'scrubbarbuffer');
 61 				device.prependChildElement(this.outputElement, this._buffer);
 62 				return this.outputElement;
 63 			},
 64 			_moveBuffer: function() {
 65 				if(this.outputElement) {
 66 					var device = this.getCurrentApplication().getDevice();
 67 					var elsize = device.getElementSize(this.outputElement);
 68 					var left = Math.floor(this._bufferedRange.start * elsize.width);
 69 					var width = Math.floor((this._bufferedRange.end - this._bufferedRange.start) * elsize.width);
 70 
 71 					if(this._lastBufferLeft != left) {
 72 						this._lastBufferLeft = left;
 73 						device.moveElementTo({
 74 							el: this._buffer,
 75 							to: {
 76 								left: left,
 77 								top: null
 78 							},
 79 							skipAnim: true
 80 						});
 81 					}
 82 					if(this._lastBufferWidth != width) {
 83 						this._lastBufferWidth = width;
 84 						device.setElementSize(this._buffer, {width: width});
 85 					}
 86 				}
 87 			},
 88 			getBufferedRange: function() {
 89 				return this._bufferedRange;
 90 			},
 91 			setBufferedRange: function(bufferedRange) {
 92 				this._bufferedRange = bufferedRange;
 93 				this._moveBuffer();
 94 			}
 95 		});
 96 	}
 97 );
