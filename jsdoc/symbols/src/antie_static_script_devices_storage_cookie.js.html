 27 require.def(
 28 	'antie/devices/storage/cookie',
 29 	[
 30 		'antie/devices/browserdevice',
 31 		'antie/storageprovider'
 32 	],
 33 	function(Device, StorageProvider) {
 34 		// http://www.quirksmode.org/js/cookies.html
 35 		var namespaces = {};
 36 
 37 		var default_days = 366;
 38 		var pathParts = document.location.pathname.split("/");
 39 		pathParts.pop();
 40 		var path = pathParts.join("/") + "/";
 41 
 42 		function createCookie(namespace, value, days) {
 43 			value = encodeURIComponent(value);
 44 			days = days || default_days;
 45 			var date = new Date();
 46 			date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
 47 			var expires = "; expires=" + date.toGMTString();
 48 			document.cookie = namespace + "=" + value + expires + "; path=" + path;
 49 		}
 50 
 51 		function readCookie(namespace) {
 52 			var nameEQ = namespace + "=";
 53 			var ca = document.cookie.split(';');
 54 			for (var i = 0; i < ca.length; i++) {
 55 				var c = ca[i];
 56 				while (c.charAt(0) == ' ') {
 57 					c = c.substring(1, c.length);
 58 				}
 59 				if (c.indexOf(nameEQ) === 0) {
 60 					return decodeURIComponent(c.substring(nameEQ.length, c.length));
 61 				}
 62 			}
 63 			return null;
 64 		}
 65 
 66 		function eraseCookie(namespace) {
 67 			createCookie(namespace, "", -1);
 68 		}
 69 
 70 		var CookieStorage = StorageProvider.extend({
 71 			init: function(namespace) {
 72 				this._super();
 73 
 74 				this._namespace = namespace;
 75 
 76 				var cookie = readCookie(namespace);
 77 
 78 				if(cookie) {
 79 					this._valueCache = Device.prototype.decodeJson(cookie);
 80 					if(this._valueCache) {
 81 						this._save();
 82 					} else {
 83 						this._valueCache = {};
 84 					}
 85 				}
 86 			},
 87 			setItem: function(key, value) {
 88 				this._super(key, value);
 89 				this._save();
 90 			},
 91 			removeItem: function(key) {
 92 				this._super(key);
 93 				this._save();
 94 			},
 95 			clear: function() {
 96 				this._super();
 97 				this._save();
 98 
 99 				// delete it from the stored namespaces
100 				// so it will be reloaded the next time
101 				// we get it
102 				delete namespaces[this._namespace];
103 			},
104 			_isEmpty: function() {
105 				var prop;
106 				for(prop in this._valueCache) {
107 					return false;
108 				}
109 				return true;
110 			},
111 			_save: function() {
112 				if(this._isEmpty()) {
113 					eraseCookie(this._namespace);
114 				} else {
115 					var json = Device.prototype.encodeJson(this._valueCache);
116 					createCookie(this._namespace, json);
117 				}
118 			}
119 		});
120 
121 		Device.prototype.getPersistentStorage = function(namespace) {
122 			if(!namespaces[namespace]) {
123 				namespaces[namespace] = new CookieStorage(namespace);
124 			}
125 			return namespaces[namespace];
126 		};
127 	}
128 );
