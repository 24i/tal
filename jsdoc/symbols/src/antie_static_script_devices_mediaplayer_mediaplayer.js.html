 28 require.def(
 29     "antie/devices/mediaplayer/mediaplayer",
 30     [
 31         "antie/class",
 32         "antie/callbackmanager",
 33         "antie/runtimecontext"
 34     ],
 35     function(Class, CallbackManager, RuntimeContext) {
 36         "use strict";
 37 
 38         /**
 39          * Base class for media playback device modifiers.
 40          * @name antie.devices.mediaplayer.MediaPlayer
 41          * @class
 42          * @abstract
 43          * @extends antie.Class
 44          */
 45         var MediaPlayer = Class.extend(/** @lends antie.devices.mediaplayer.MediaPlayer.prototype */ {
 46             /**
 47              * @constructor
 48              * @ignore
 49              */
 50             init: function() {
 51                 this._callbackManager = new CallbackManager();
 52             },
 53 
 54             /**
 55              * Add an event callback to receive all media player events from now on.
 56              *
 57              * Note that failing to remove callbacks when you are finished with them can stop garbage collection
 58              * of objects/closures containing those callbacks and so create memory leaks in your application.
 59              * @param {Object} thisArg The object to use as "this" when calling the callback.
 60              * @param {Function} callback Function to which events are passed (e.g. to be bubbled up the component hierarchy).
 61              */
 62             addEventCallback: function(thisArg, callback) {
 63                 this._callbackManager.addCallback(thisArg, callback);
 64             },
 65 
 66             /**
 67              * Stop receiving events with the specified callback.
 68              * @param {Object} thisArg The object specified to use as "this" when adding the callback.
 69              * @param {Function} callback Function to which events are no longer to be passed
 70              */
 71             removeEventCallback: function(thisArg, callback) {
 72                 this._callbackManager.removeCallback(thisArg, callback);
 73             },
 74 
 75             /**
 76              * Stop receiving events to any callbacks.
 77              */
 78             removeAllEventCallbacks: function() {
 79                 this._callbackManager.removeAllCallbacks();
 80             },
 81 
 82             /**
 83              * Protected method, for use by subclasses to emit events of any specified type, adding in the
 84              * standard payload used by all events.
 85              * @param {String} eventType The type of the event to be emitted.
 86              * @protected
 87              */
 88             _emitEvent: function(eventType) {
 89 
 90                 var event = {
 91                     type: eventType,
 92                     currentTime: this.getCurrentTime(),
 93                     seekableRange: this.getSeekableRange(),
 94                     duration: this.getDuration(),
 95                     url: this.getSource(),
 96                     mimeType: this.getMimeType(),
 97                     state: this.getState()
 98                 };
 99 
100                 this._callbackManager.callAll(event);
101             },
102 
103             /**
104              * Offset used when attempting to playFrom() the end of media. This allows the media to play briefly before completing.
105              * @constant {Number}
106              */
107             CLAMP_OFFSET_FROM_END_OF_RANGE: 1.1,
108 
109             /**
110              * Clamp a time value so it does not exceed the current range.
111              * Clamps to near the end instead of the end itself to allow for devices that cannot seek to the very end of the media.
112              * @param {Number} seconds The time value to clamp in seconds from the start of the media
113              * @protected
114              */
115             _getClampedTime: function(seconds) {
116                 var range = this.getSeekableRange();
117                 var offsetFromEnd = this._getClampOffsetFromConfig();
118                 var nearToEnd = Math.max(range.end - offsetFromEnd, range.start);
119                 if (seconds < range.start) {
120                     return range.start;
121                 } else if (seconds > nearToEnd) {
122                     return nearToEnd;
123                 } else {
124                     return seconds;
125                 }
126             },
127 
128             /**
129              * Time (in seconds) compared to current time within which seeking has no effect.
130              * @constant {Number}
131              */
132             CURRENT_TIME_TOLERANCE: 1,
133 
134             /**
135              * Check whether a time value is near to the current media play time.
136              * @param {Number} seconds The time value to test, in seconds from the start of the media
137              * @protected
138              */
139             _isNearToCurrentTime: function(seconds) {
140                 var currentTime = this.getCurrentTime();
141                 var targetTime = this._getClampedTime(seconds);
142                 return Math.abs(currentTime - targetTime) <= this.CURRENT_TIME_TOLERANCE;
143             },
144 
145             /**
146              * Set the media resource to be played.
147              * Calling this in any state other than EMPTY is an error.
148              * @param {antie.devices.mediaplayer.MediaPlayer.TYPE} mediaType Value from the MediaPlayer.TYPE enum; audio or video.
149              * @param {String} url location of the media resource to play
150              * @param {String} mimeType type of media resource
151              */
152             setSource: function (mediaType, url, mimeType) {
153                 throw new Error("setSource method has not been implemented");
154             },
155 
156             /**
157              * Request that the media seeks to and starts playing from Time.
158              * A media source must have been set with setSource before calling this.
159              * This may transition to the buffering state if enough media data is not yet available to play.
160              * If the media is buffering, call this to resume playback in a playing state once buffering ends.
161              * Calling this in state EMPTY or STOPPED is an error.
162              * Clamps the time to the seekable range of the media.
163              * If trying to seek to (or past) the very end of the media, this will actually seek to before the end.
164              * This allows the media playback to complete normally.
165              * @param {Number} seconds Time to play from in seconds from the start of the media
166              */
167             playFrom: function (seconds) {
168                 throw new Error("playFrom method has not been implemented");
169             },
170 
171             /**
172              * Begin playback of the media resource from wherever the device chooses.
173              * For On Demand assets, this will normally be the start, for Live stream assets it should be the live point
174              * but could be the start of the stream window on some devices.
175              * This function can only be called from the STOPPED state; calling it from any other state is an error.
176              * To begin playback from a specified time offset, use the playFrom function instead.
177              */
178             beginPlayback: function () {
179                 throw new Error("beginPlayback method has not been implemented");
180             },
181 
182             /**
183              * Begin playback of the media resource from Time.
184              * A media source must have been set with setSource before calling this.
185              * This can be used to resume media after changing source.
186              * This function can only be called from the STOPPED state; calling it from any other state is an error.
187              * Clamps the time to the seekable range of the media.
188              * If trying to play at (or past) the very end of the media, this will actually begin playback before the end.
189              * @param {Number} seconds Time to play from in seconds from the start of the media
190              */
191             beginPlaybackFrom: function (seconds) {
192                 throw new Error("beginPlaybackFrom method has not been implemented");
193             },
194 
195             /**
196              * Request that the media be paused.
197              * If the Media is playing, call this to pause it.
198              * If the media is buffering, call this to resume playback in a paused state once buffering ends.
199              * Calling this in state EMPTY or STOPPED is an error.
200              */
201             pause: function () {
202                 throw new Error("pause method has not been implemented");
203             },
204 
205             /**
206              * Request that the media resume playing after being paused.
207              * If the Media is paused, call this to resume playing it.
208              * If the media is buffering, call this to resume playback in a playing state once buffering ends.
209              * Calling this in state EMPTY or STOPPED is an error.
210              */
211             resume: function () {
212                 throw new Error("resume method has not been implemented");
213             },
214 
215             /**
216              * Request that the media be stopped.
217              * If the Media is playing, call this to stop the media.
218              * Note that the source is still set after calling stop.
219              * Call reset after stop to unset the source.
220              * Calling this in state EMPTY is an error.
221              */
222             stop: function () {
223                 throw new Error("stop method has not been implemented");
224             },
225 
226             /**
227              * Reset the Media Player.
228              * When the media is stopped, calling reset will reset the player to a clean state with no source set.
229              * Calling this in any state other than STOPPED or ERROR is an error.
230              */
231             reset: function () {
232                 throw new Error("reset method has not been implemented");
233             },
234 
235             /**
236              * Get the source URL.
237              * If no source is set (in state EMPTY for example), then this returns undefined.
238              * @return {String} The URL
239              */
240             getSource: function () {
241                 throw new Error("getSource method has not been implemented");
242             },
243 
244             /**
245             * Get the source MIME type.
246             * If no source is set (in state EMPTY for example), then this returns undefined.
247             * @return {String} The MIME type
248             */
249             getMimeType: function () {
250                 throw new Error("getMimeType method has not been implemented");
251             },
252 
253             /**
254             * Get the current play time.
255             * If no current time is available, then this returns undefined.
256             * @return {Number} The current play time in seconds from the start of the media.
257             */
258             getCurrentTime: function () {
259                 throw new Error("getCurrentTime method has not been implemented");
260             },
261 
262             /**
263             * Get the available seekable range of media.
264             * Returns a range Object with 'start' and 'end' numeric properties, giving the start and end of the available media in seconds from the start of the media.
265             * For VOD playback, 'start' is zero and 'end' is the last possible seek time (in many cases equal to duration).
266             * For Live playback, 'start' may be non-zero, reflecting the amount of 'live rewind' available before the current play position.
267             * For live playback, 'end' is the current live time.
268             * For live playback, both 'start' and 'end' may advance over time.
269             * If no range is available, this returns undefined.
270             * @return {Object} Object with 'start' and 'end' times in seconds, or undefined.
271             */
272             getSeekableRange: function () {
273                 throw new Error("getSeekableRange method has not been implemented");
274             },
275 
276             /**
277              * Get the duration of the media in seconds.
278              * For VOD playback, this is the duration of the media.
279              * For Live playback, this is positive Infinity.
280              * If no duration is available, this returns undefined.
281              * @return {Number} Duration of media in seconds, or Infinity, or undefined.
282              */
283             getDuration: function() {
284                 throw new Error("getDuration method has not been implemented");
285             },
286 
287             /**
288             * Get the current state of the Media PLayer state machine.
289             * @return {antie.devices.mediaplayer.MediaPlayer.STATE} The current state of the Media Player state machine.
290             */
291             getState: function () {
292                 throw new Error("getState method has not been implemented");
293             },
294 
295             _getClampOffsetFromConfig: function() {
296                 var clampOffsetFromEndOfRange;
297                 var config = RuntimeContext.getDevice().getConfig();
298                 if (config && config.streaming && config.streaming.overrides) {
299                     clampOffsetFromEndOfRange = config.streaming.overrides.clampOffsetFromEndOfRange;
300                 }
301 
302                 if(clampOffsetFromEndOfRange !== undefined) {
303                     return clampOffsetFromEndOfRange;
304                 } else {
305                     return this.CLAMP_OFFSET_FROM_END_OF_RANGE;
306                 }
307             }
308         });
309 
310         /**
311         * An enumeration of possible media player states.
312         * @name antie.devices.mediaplayer.MediaPlayer.STATE
313         * @enum {String}
314         */
315         MediaPlayer.STATE = {
316             EMPTY:      "EMPTY",     // No source set
317             STOPPED:    "STOPPED",   // Source set but no playback
318             BUFFERING:  "BUFFERING", // Not enough data to play, waiting to download more
319             PLAYING:    "PLAYING",   // Media is playing
320             PAUSED:     "PAUSED",    // Media is paused
321             COMPLETE:   "COMPLETE",  // Media has reached its end point
322             ERROR:      "ERROR"      // An error occurred
323         };
324 
325         /**
326         * Media Player event names
327         * @name antie.devices.mediaplayer.MediaPlayer.EVENT
328         * @enum {String}
329         */
330         MediaPlayer.EVENT = {
331             STOPPED:   "stopped",   // Event fired when playback is stopped
332             BUFFERING: "buffering", // Event fired when playback has to suspend due to buffering
333             PLAYING:   "playing",   // Event fired when starting (or resuming) playing of the media
334             PAUSED:    "paused",    // Event fired when media playback pauses
335             COMPLETE:  "complete",  // Event fired when media playback has reached the end of the media
336             ERROR:     "error",     // Event fired when an error condition occurs
337             STATUS:    "status",    // Event fired regularly during play
338             SENTINEL_ENTER_BUFFERING:  "sentinel-enter-buffering", // Event fired when a sentinel has to act because the device has started buffering but not reported it
339             SENTINEL_EXIT_BUFFERING:   "sentinel-exit-buffering",  // Event fired when a sentinel has to act because the device has finished buffering but not reported it
340             SENTINEL_PAUSE:            "sentinel-pause",           // Event fired when a sentinel has to act because the device has failed to pause when expected
341             SENTINEL_PLAY:             "sentinel-play",            // Event fired when a sentinel has to act because the device has failed to play when expected
342             SENTINEL_SEEK:             "sentinel-seek",            // Event fired when a sentinel has to act because the device has failed to seek to the correct location
343             SENTINEL_COMPLETE:         "sentinel-complete",        // Event fired when a sentinel has to act because the device has completed the media but not reported it
344             SENTINEL_PAUSE_FAILURE:    "sentinel-pause-failure",   // Event fired when the pause sentinel has failed twice, so it is giving up
345             SENTINEL_SEEK_FAILURE:     "sentinel-seek-failure"     // Event fired when the seek sentinel has failed twice, so it is giving up
346         };
347 
348         /**
349          * An enumeration of valid Media Types.
350          * @name antie.devices.mediaplayer.MediaPlayer.TYPE
351          * @enum {String}
352          */
353         MediaPlayer.TYPE = {
354             VIDEO: "video",
355             AUDIO: "audio"
356         };
357 
358         /**
359          * Levels of support for the live media player
360          * @name antie.devices.mediaplayer.MediaPlayer.LIVE_SUPPORT
361          * @enum {String}
362          */
363         MediaPlayer.LIVE_SUPPORT = {
364             NONE: "none",
365             PLAYABLE: "playable",
366             RESTARTABLE: "restartable",
367             SEEKABLE: "seekable"
368         };
369 
370         return MediaPlayer;
371     }
