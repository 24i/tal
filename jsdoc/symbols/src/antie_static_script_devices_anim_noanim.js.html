 27 require.def(
 28 	'antie/devices/anim/noanim',
 29 	['antie/devices/browserdevice',
 30     'antie/devices/anim/shared/transitionendpoints'],
 31 	function(Device, TransitionEndPoints) {
 32         'use strict';
 33 
 34         /* documented in antie.devices.Device */
 35 		Device.prototype.scrollElementTo = function(options) {
 36 			if(new RegExp("_mask$").test(options.el.id)) {
 37 				if (options.el.childNodes.length === 0) {
 38 					return null;
 39 				}
 40 				options.el.style.position = 'relative';
 41 				options.el = options.el.childNodes[0];
 42 				options.el.style.position = 'relative';
 43 			} else {
 44 				return null;
 45 			}
 46 			var startLeft = Math.abs(options.el.style.left.replace(/px/, '')) || 0;
 47 			var changeLeft = (options.to.left !== undefined) ? (options.to.left - startLeft) : 0;
 48 			var startTop = Math.abs(options.el.style.top.replace(/px/, '')) || 0;
 49 			var changeTop = (options.to.top !== undefined) ? (options.to.top - startTop) : 0;
 50 			if ((changeLeft == 0) && (changeTop == 0)) {
 51 				if (options.onComplete) {
 52 					options.onComplete();
 53 				}
 54 				return null;
 55 			}
 56 
 57 			if (options.to.left !== undefined) {
 58 				options.el.style.left = 0 - options.to.left + "px";
 59 			}
 60 			if (options.to.top !== undefined) {
 61 				options.el.style.top = 0 - options.to.top + "px";
 62 			}
 63 			if (options.onComplete) {
 64 				options.onComplete();
 65 			}
 66 
 67 			return null;
 68 		};
 69 
 70         /* documented in antie.devices.Device */
 71 		Device.prototype.moveElementTo = function(options) {
 72 			// Performance consideration: if left or top is null they are ignored to prevent the additional
 73 			// work animating them.
 74 
 75 			var startLeft = parseInt(options.el.style.left.replace(/px|em|pt/,"")) || 0;
 76 			var changeLeft = (options.to.left !== undefined) ? (options.to.left - startLeft) : 0;
 77 			var startTop = parseInt(options.el.style.top.replace(/px|em|pt/,"")) || 0;
 78 			var changeTop = (options.to.top !== undefined) ? (options.to.top - startTop) : 0;
 79 
 80 			if ((changeLeft === 0) && (changeTop === 0)) {
 81                 if (options.onComplete) {
 82                     options.onComplete();
 83                 }
 84 				return null;
 85 			}
 86 
 87 			if (options.to.left !== undefined) {
 88 				options.el.style.left = options.to.left + "px";
 89 			}
 90 			if (options.to.top !== undefined) {
 91 				options.el.style.top = options.to.top + "px";
 92 			}
 93 			if (options.onComplete) {
 94 				options.onComplete();
 95 			}
 96 		};
 97 
 98 		/* documented in antie.devices.device */
 99 		Device.prototype.hideElement = function(options) {
100 			options.el.style.visibility = "hidden";
101 			options.el.style.opacity = 0;
102 			if (typeof options.onComplete == "function") {
103 				options.onComplete();
104 			}
105 		};
106 
107 		/* documented in antie.devices.device */
108 		Device.prototype.showElement = function(options) {
109 			options.el.style.visibility = "visible";
110 			options.el.style.opacity = 1;
111 			if (typeof options.onComplete == "function") {
112 				options.onComplete();
113 			}
114 		};
115 
116         Device.prototype.tweenElementStyle = function(options) {
117             var transEndPoints, i, prop, properties, elStyle;
118             elStyle = options.el.style;
119 
120             transEndPoints = new TransitionEndPoints(
121                 {
122                     to: options.to,
123                     from: options.from,
124                     units: options.units
125                 }
126             );
127 
128             properties = transEndPoints.getProperties();
129 
130             for (i = 0; i !== properties.length; i += 1){
131                 prop = properties[i];
132                 elStyle[prop] =  transEndPoints.getPropertyDestination(prop);
133             }
134             if (typeof options.onComplete === "function") {
135                 options.onComplete();
136             }
137         };
138 
139         /* documented in antie.devices.device */
140 		Device.prototype.stopAnimation = function(anim) {
141 			// Left intentionally blank
142 		};
143 
144 		/* documented in antie.devices.device */
145 		Device.prototype.isAnimationDisabled = function(){
146 			return true;
147 		};	
148 	}
