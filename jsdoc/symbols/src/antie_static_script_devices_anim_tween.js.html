 27 require.def(
 28     'antie/devices/anim/tween',
 29     [
 30      'antie/devices/browserdevice',
 31      'antie/lib/shifty'
 32      ],
 33      function(Device, Tweenable) {
 34         'use strict';
 35 
 36         // A set of queues of DOM updates to perform. Each animation framerate gets its own queue
 37         // so they are in sync between themselves.
 38         var animQueues = {};
 39 
 40         /**
 41          * Internal function: given a new tween value for an animation, add it to a queue
 42          * of DOM manipulations to be performed at the next update. If there is no queue yet,
 43          * create it and start the update cycle after half a frame has elapsed, to give other
 44          * animation updates a chance to come in.
 45          */
 46         function addTweenToQueue(options, tweenValues) {
 47             // A separate queue exists for each framerate. Get or create the appropriate queue.
 48             var queueKey = 'fps' + options.fps;
 49             var frameIntervalMs = 1000 / options.fps;
 50             var queue = animQueues[queueKey];
 51 
 52             // Create a new queue if one doesn't already exist (implemented as an array).
 53             if (!queue) {
 54                 queue = [];
 55                 animQueues[queueKey] = queue;
 56             }
 57 
 58             // Start processing the queue periodically if we're not already.
 59             // Wait half a frame before starting the first cycle - gives other animation updates at
 60             // the same frame rate a chance to come in.
 61             if (!queue.isProcessing) {
 62                 queue.isProcessing = true;
 63                 setTimeout(function() { startIntervalTimer(queue, frameIntervalMs); }, frameIntervalMs / 2);
 64 
 65                 // First tween in a cycle should be applied immediately. It contains initial values.
 66                 step(options, tweenValues);
 67             }
 68             else {
 69                 // Queue is already being processed. Add the new entry to the queue.
 70                 queue.push({options: options, values: tweenValues});
 71             }
 72         }
 73 
 74         /**
 75          * Internal function: When we receive a callback to say an animation
 76          * has completed (either it's been cancelled or it's finished), drain
 77          * any outstanding steps from the queue. This ensures that in the case
 78          * of clients cancelling the animation, the element stops being updated
 79          * immediately.
 80          */
 81         function drainTweensFromQueue(options) {
 82             var queue, i, q;
 83             queue = animQueues['fps' + options.fps];
 84             if (queue) {
 85                 for (i = 0; i < queue.length; i++) {
 86                     q = queue[i];
 87                     if (q.options === options) {
 88                         step(q.options, q.values);
 89                         queue.splice(i, 1);
 90                         i--;
 91                     }
 92                 }
 93             }
 94         }
 95 
 96         /**
 97          * Internal function. Start a periodic interval timer for a given framerate.
 98          */
 99         function startIntervalTimer(queue, period) {
100             // Store timer ID with the queue to allow it to be stopped later.
101             queue.intervalId = setInterval(function() { processQueue(queue); }, period);
102         }
103 
104         /**
105          * Internal function. To be called periodically on a queue of operations. Apply each update to the
106          * DOM, then clear the queue ready to be refilled. Stop the periodic timer if the queue remains empty.
107          */
108         function processQueue(queue) {
109             // Is the queue still empty after it was last cleared? Stop the timer - the animations have
110             // probably finished and will not provide any further updates.
111             if (queue.length === 0) {
112                 clearInterval(queue.intervalId);
113                 queue.isProcessing = false;
114                 queue.intervalId = null;
115             }
116             else {
117                 // We have some DOM updates to do. Do each one in sequence, then clear the queue ready for the next round.
118                 try {
119                     for (var i = 0; i < queue.length; i++) {
120                         var q = queue[i];
121                         step(q.options, q.values);
122                     }
123                 }
124                 finally {
125                     // Truncating the array length to zero clears it.
126                     queue.length = 0;
127                 }
128             }
129         }
130 
131         /**
132          * Internal function. Perform the DOM updates required for the update.
133          */
134         function step(options, tweenValues) {
135             for (var p in options.to) {
136                 if (tweenValues[p] !== null && tweenValues[p] !== undefined) {
137                     if (/scroll/.test(p)) {
138                         options.el[p] = tweenValues[p];
139                     } else {
140                         options.el.style[p] = tweenValues[p];
141                     }
142                 }
143             }
144         }
145 
146         Device.prototype._tween = function (options) {
147             var anim = new Tweenable(options);
148             var self = this;
149 
150             var opts = {
151                     el: options.el,
152                     initialState: options.from || {},
153                     from: options.from || {},
154                     to: options.to || {},
155                     duration: options.duration || 840,
156                     easing: options.easing || 'easeFromTo',
157                     fps: options.fps || 25,
158                     start: function() {
159                         if (options.className) {
160                             self.removeClassFromElement(options.el, "not" + options.className);
161                             self.addClassToElement(options.el,  options.className);
162                         }
163                         self.removeClassFromElement(self.getTopLevelElement(), "notanimating");
164                         self.addClassToElement(self.getTopLevelElement(), "animating");
165                         if (options.onStart) {
166                             options.onStart();
167                         }
168                     },
169                     step: function () {
170                         addTweenToQueue(opts, this);
171                     },
172                     callback: function () {
173                         if(options.className) {
174                             self.removeClassFromElement(options.el, options.className);
175                             self.addClassToElement(options.el, "not" + options.className);
176                         }
177                         self.removeClassFromElement(self.getTopLevelElement(), "animating");
178                         self.addClassToElement(self.getTopLevelElement(), "notanimating");
179                         // Send this animation to its final state immediately.
180                         drainTweensFromQueue(opts);
181                         if (this) {
182                             step(opts, this);
183                         }
184                         // Fire client callback if it exists
185                         if (typeof options.onComplete === 'function') {
186                             options.onComplete();
187                         }
188                     }
189             };
190 
191             anim.tween(opts);
192 
193             return anim;
194         };
195     }
196 );
