 27 require.def(
 28 	'antie/devices/anim/tween',
 29 	[
 30 		'antie/devices/browserdevice',
 31 		'antie/lib/shifty'
 32 	],
 33 	function(Device, Tweenable) {
 34 		Device.prototype._tween = function (options) {
 35 			var anim = new Tweenable(options);
 36 			var self = this;
 37 
 38 			var opts = {
 39 					initialState: options.from || {},
 40 					from: options.from || {},
 41 					to: options.to || {},
 42 					duration: options.duration || 840,
 43 					easing: options.easing || 'easeFromTo',
 44 					fps: options.fps || 25,
 45 					start: function() {
 46 						if (options.className) {
 47 							self.removeClassFromElement(options.el, "not" + options.className);
 48 							self.addClassToElement(options.el,  options.className);
 49 						}
 50 						self.removeClassFromElement(self.getTopLevelElement(), "notanimating");
 51 						self.addClassToElement(self.getTopLevelElement(), "animating");
 52 						if (options.onStart) {
 53 							options.onStart();
 54 						}
 55 					},
 56 					step: function () {
 57 						for (var p in options.to) {
 58 							if (this[p] != null) {
 59 								if (/scroll/.test(p)) {
 60 									options.el[p] = this[p];
 61 								} else {
 62 									options.el.style[p] = this[p];
 63 								}
 64 							}
 65 						}
 66 					},
 67 					callback: function () {
 68 						if(options.className) {
 69 							self.removeClassFromElement(options.el, options.className);
 70 							self.addClassToElement(options.el, "not" + options.className);
 71 						}
 72 						self.removeClassFromElement(self.getTopLevelElement(), "animating");
 73 						self.addClassToElement(self.getTopLevelElement(), "notanimating");
 74 						if (options.onComplete) {
 75 							options.onComplete();
 76 						}
 77 					}
 78 				};
 79 
 80 			anim.tween(opts);
 81 
 82 			return anim;
 83 		};
 84 	}
 85 );
