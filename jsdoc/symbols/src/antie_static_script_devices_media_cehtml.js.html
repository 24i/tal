 27 require.def(
 28     'antie/devices/media/cehtml',
 29     [
 30         'antie/devices/device',
 31         'antie/widgets/media',
 32         'antie/events/mediaevent',
 33         'antie/events/mediaerrorevent',
 34         'antie/events/mediasourceerrorevent',
 35         'antie/mediasource',
 36         'antie/devices/media/seekstate'
 37     ],
 38 
 39     function(Device, Media, MediaEvent, MediaErrorEvent, MediaSourceErrorEvent, MediaSource, SeekState ) {
 40 
 41         var CEHTMLPlayer = Media.extend({
 42             init: function(id, mediaType) {
 43                 this._super(id);
 44                 this._seekState = new SeekState( this );
 45 
 46                 this._updateInterval = null;
 47                 this._loaded = false;
 48                 this._eventsBound = false;
 49 
 50                 // Create the DOM element now so the wrapped functions can modify attributes
 51                 // before it is placed in the Document during rendering.
 52                 this._mediaElement = this._createCEHTMLObjectElement((mediaType == "audio") ? "audio/mp4" : "video/mp4");
 53                 this._mediaElement.width = 1280;
 54                 this._mediaElement.height = 720;
 55 
 56                 if (mediaType == "audio") {
 57                     this._mediaType = "audio";
 58                 } else if (mediaType == "video") {
 59                     this._mediaType = "video";
 60                 } else {
 61                     throw new Error('Unrecognised media type: ' + mediaType);
 62                 }
 63             },
 64             _createCEHTMLObjectElement: function(contentType) {
 65                 var device = this.getCurrentApplication().getDevice();
 66                 var obj = device._createElement("object", this.id, this.getClasses());
 67                 obj.setAttribute("type", contentType);
 68                 var div = device.createContainer();
 69                 div.innerHTML = '<object type="' + contentType + '" id="' + this.id + '" class="' + obj.className + '" ' + 'style="width: 100%; height: 100%; position: absolute; z-index: -1"' + ' />';
 70                 return div.childNodes[0];
 71             },
 72             render: function(device) {
 73                 if (this.outputElement !== this._mediaElement) {
 74                     this.outputElement = this._mediaElement;
 75                 }
 76 
 77                 return this._mediaElement;
 78             },
 79             // (not part of HTML5 media)
 80             setWindow: function(left, top, width, height) {
 81                 if (this._mediaType == "audio") {
 82                     throw new Error('Unable to set window size for CE-HTML audio.');
 83                 }
 84                 var device = this.getCurrentApplication().getDevice();
 85                 device.setElementSize(this._mediaElement, {width:width, height:height});
 86                 device.setElementPosition(this._mediaElement, {left:left, top:top});
 87             },
 88             // readonly attribute MediaError error;
 89             getError: function() {
 90                 // TODO: CE-HTML implementation
 91             },
 92             // Similar to src attribute or 'source' child elements:
 93             // attribute DOMString src;
 94             setSources: function(sources, tags) {
 95                 this._seekState = new SeekState( this );
 96 
 97                 var currentMediaType = this._mediaElement.type;
 98                 var newMediaType = sources[0].getContentType();
 99 
100                 if (currentMediaType !== undefined && currentMediaType !== newMediaType) {
101                     if (this._requiresMediaTypeFix()) {
102                         var oldDimensions = {
103                             left: this._mediaElement.style.left,
104                             top: this._mediaElement.style.top,
105                             width: this._mediaElement.style.width,
106                             height: this._mediaElement.style.height
107                         };
108 
109                         this.destroy();
110                         this._mediaElement = this._createCEHTMLObjectElement(newMediaType);
111                         this.setWindow(oldDimensions.left, oldDimensions.top, oldDimensions.width, oldDimensions.height);
112                         this._eventsBound = false;
113                     }
114                 }
115 
116                 this._mediaElement.type = sources[0].getContentType();
117                 this._mediaElement.data = sources[0].getURL(tags);
118 
119                 if (!this._eventsBound) {
120                     var self = this;
121                     // Convert all media events into our internal representation and bubble them through
122                     // the UI widgets
123                     // Handle CE-HTML playstate change events
124                     // Note: this has to be bound after setting player.data
125                     this._mediaElement.onPlayStateChange = function() {
126                         switch (self._mediaElement.playState) {
127                             case CEHTMLPlayer.PLAY_STATE_STOPPED:
128                                 if (self._updateInterval) {
129                                     window.clearInterval(self._updateInterval);
130                                     self._updateInterval = false;
131                                 }
132                                 break;
133                             case CEHTMLPlayer.PLAY_STATE_PLAYING:
134                                 self._seekState.playing();
135                                 if (!self._loaded) {
136                                     self.bubbleEvent(new MediaEvent("loadedmetadata", self));
137                                     self.bubbleEvent(new MediaEvent("canplaythrough", self));
138                                     self._loaded = true;
139                                 }
140                                 self.bubbleEvent(new MediaEvent("play", self));
141                                 self.bubbleEvent(new MediaEvent("playing", self));
142                                 if (!self._updateInterval) {
143                                     self._updateInterval = window.setInterval(function() {
144                                         self.bubbleEvent(new MediaEvent("timeupdate", self));
145                                     }, 900);
146                                 }
147                                 break;
148                             case CEHTMLPlayer.PLAY_STATE_PAUSED:
149                                 self.bubbleEvent(new MediaEvent("pause", self));
150                                 break;
151                             case CEHTMLPlayer.PLAY_STATE_CONNECTING:
152                                 self.bubbleEvent(new MediaEvent("loadstart", self));
153                                 break;
154                             case CEHTMLPlayer.PLAY_STATE_BUFFERING:
155                                 self.bubbleEvent(new MediaEvent("waiting", self));
156                                 break;
157                             case CEHTMLPlayer.PLAY_STATE_FINISHED:
158                                 self.bubbleEvent(new MediaEvent("ended", self));
159                                 break;
160                             case CEHTMLPlayer.PLAY_STATE_ERROR:
161                                 self.bubbleEvent(new MediaErrorEvent(self, 0));
162                                 break;
163                             default:
164                                 // do nothing
165                                 break;
166                         }
167                     };
168 
169                     this._eventsBound = true;
170                 }
171                 this.bubbleEvent(new MediaEvent("canplay", this));
172             },
173             getSources: function() {
174                 return [new MediaSource(this._mediaElement.data, this._mediaElement.type)];
175             },
176             // readonly attribute DOMString currentSrc;
177             getCurrentSource: function() {
178                 return this._mediaElement.data;
179             },
180             /*
181              const unsigned short NETWORK_EMPTY = 0;
182              const unsigned short NETWORK_IDLE = 1;
183              const unsigned short NETWORK_LOADING = 2;
184              const unsigned short NETWORK_NO_SOURCE = 3;
185              readonly attribute unsigned short networkState;
186              */
187             getNetworkState: function() {
188                 // Consider using NotifySocket object if we can consider having a persistent connection to the server
189                 // or error code from playState == error (6) return true when the error code == 1 (network connection lost)
190                 // TODO: CE-HTML implementation
191             },
192             // attribute DOMString preload;
193             // @returns "none", "metadata" or "auto"
194             getPreload: function() {
195                 // TODO: CE-HTML implementation
196                 return "none";
197             },
198             setPreload: function(preload) {
199                 // TODO: CE-HTML implementation
200             },
201             // readonly attribute TimeRanges buffered;
202             getBuffered: function() {
203                 // TODO: CE-HTML implementation
204                 return [];
205             },
206             // void load();
207             load: function() {
208                 // TODO: CE-HTML implementation
209             },
210             // DOMString canPlayType(in DOMString type);
211             canPlayType: function(type) {
212                 // TODO: CE-HTML implementation
213                 return true;
214             },
215             /*
216              const unsigned short HAVE_NOTHING = 0;
217              const unsigned short HAVE_METADATA = 1;
218              const unsigned short HAVE_CURRENT_DATA = 2;
219              const unsigned short HAVE_FUTURE_DATA = 3;
220              const unsigned short HAVE_ENOUGH_DATA = 4;
221              readonly attribute unsigned short readyState;
222              */
223             getReadyState: function() {
224                 // TODO: CE-HTML implementation
225                 return 0;
226             },
227             // readonly attribute boolean seeking;
228             getSeeking: function() {
229                 // TODO: CE-HTML implementation
230                 return false;
231             },
232             // attribute double currentTime;
233             setCurrentTime: function(currentTime) {
234                 // TODO: CE-HTML implementation
235                 // to emulate HTML5 we should throw an INVALID_STATE_ERR when we can't seek to the new position
236                 // understandably this call may be blocking until the seek has completed
237                 this._seekState.seekTo( currentTime );
238                 return this._mediaElement.seek(currentTime * 1000);
239             },
240             getCurrentTime: function() {
241                 return this._mediaElement.playPosition / 1000;
242             },
243             // readonly attribute double initialTime;
244             getInitialTime: function() {
245                 // TODO: CE-HTML implementation
246                 return 0;
247             },
248             // readonly attribute double duration;
249             getDuration: function() {
250                 // TODO: CE-HTML implementation
251                 return this._mediaElement.playTime / 1000;
252             },
253             // readonly attribute Date startOffsetTime;
254             getStartOffsetTime: function() {
255                 // TODO: CE-HTML implementation
256                 return 0;
257             },
258             // readonly attribute boolean paused;
259             getPaused: function() {
260                 return this._mediaElement.playState === CEHTMLPlayer.PLAY_STATE_PAUSED;
261             },
262             // attribute double defaultPlaybackRate;
263             getDefaultPlaybackRate: function() {
264                 // TODO: CE-HTML implementation
265                 return 1;
266             },
267             // attribute double playbackRate;
268             getPlaybackRate: function() {
269                 // TODO: CE-HTML implementation
270                 return 1;
271             },
272             setPlaybackRate: function(playbackRate) {
273                 // TODO: CE-HTML implementation
274             },
275             // readonly attribute TimeRanges played;
276             getPlayed: function() {
277                 // TODO: CE-HTML implementation
278                 return [];
279             },
280             // readonly attribute TimeRanges seekable;
281             getSeekable: function() {
282                 // TODO: CE-HTML implementation
283                 return [];
284             },
285             // readonly attribute boolean ended;
286             getEnded: function() {
287                 return this._mediaElement.playState === CEHTMLPlayer.PLAY_STATE_FINISHED;
288             },
289             // attribute boolean autoplay;
290             getAutoPlay: function() {
291                 // TODO: CE-HTML implementation
292                 return false;
293             },
294             setAutoPlay: function(autoplay) {
295                 // TODO: CE-HTML implementation
296             },
297             // attribute boolean loop;
298             getLoop: function() {
299                 // TODO: CE-HTML implementation
300                 return false;
301             },
302             setLoop: function(loop) {
303                 // TODO: CE-HTML implementation
304             },
305             // void play();
306             play: function() {
307                 this._mediaElement.play(1);
308             },
309             stop: function() {
310                 this._mediaElement.stop();
311 
312                 if (this._updateInterval) {
313                     window.clearInterval(this._updateInterval);
314                     this._updateInterval = false;
315                 }
316             },
317             // void pause();
318             pause: function() {
319                 this._mediaElement.play(0);
320             },
321             // attribute boolean controls;
322             setNativeControls: function(controls) {
323                 // TODO: CE-HTML implementation
324             },
325             getNativeControls: function() {
326                 // TODO: CE-HTML implementation
327                 return false;
328             },
329             destroy: function() {
330                 this.stop();
331 
332                 var device = this.getCurrentApplication().getDevice();
333                 device.removeElement(this._mediaElement);
334             },
335             _requiresMediaTypeFix : function() {
336                 return false;
337             }
338 
339         });
340 
341         CEHTMLPlayer.PLAY_STATE_STOPPED = 0;
342         CEHTMLPlayer.PLAY_STATE_PLAYING = 1;
343         CEHTMLPlayer.PLAY_STATE_PAUSED = 2;
344         CEHTMLPlayer.PLAY_STATE_CONNECTING = 3;
345         CEHTMLPlayer.PLAY_STATE_BUFFERING = 4;
346         CEHTMLPlayer.PLAY_STATE_FINISHED = 5;
347         CEHTMLPlayer.PLAY_STATE_ERROR = 6;
348 
349         Device.prototype.createPlayer = function(id, mediaType) {
350             return new CEHTMLPlayer(id, mediaType);
351         };
352         Device.prototype.getPlayerEmbedMode = function(mediaType) {
353             return Media.EMBED_MODE_BACKGROUND;
354         };
355 
356         return CEHTMLPlayer;
357     }
358 );
