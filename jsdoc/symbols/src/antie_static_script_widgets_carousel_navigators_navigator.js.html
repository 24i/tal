 24 require.def('antie/widgets/carousel/navigators/navigator',
 25     [
 26         'antie/class',
 27         'antie/events/selecteditemchangeevent',
 28         'antie/events/beforeselecteditemchangeevent'
 29     ],
 30 
 31     /**
 32      * Abstract base Navigator class, used for operations involving widget indices,
 33      * for example setting the active widget or determining which to set active.
 34      * @abstract
 35      * @name antie.widgets.carousel.navigators.Navigator
 36      * @abstract
 37      * @class
 38      * @extends antie.widgets.Class
 39      */
 40     function (Class, SelectedItemChangeEvent, BeforeSelectedItemChangeEvent) {
 41         "use strict";
 42         var Navigator;
 43 
 44         Navigator = Class.extend(/** @lends antie.widgets.carousel.navigators.Navigator.prototype */{
 45             /**
 46              * @constructor
 47              * @ignore
 48              */
 49             init: function (container) {
 50                 this.setContainer(container);
 51             },
 52 
 53             /**
 54              * @returns {Number} the index of the currently active widget, or null if there is no active widget
 55              * (for example if there are no widgets)
 56              */
 57             currentIndex: function () {
 58                 return (this._getActiveIndex());
 59             },
 60 
 61             /**
 62              * @returns {Number} the index of the next focusable widget
 63              */
 64             nextIndex: function () {
 65                 var currentIndex = this.currentIndex();
 66                 return this.indexAfter(currentIndex);
 67             },
 68 
 69             /**
 70              * @param index
 71              * @returns {Number} the first focussable index after that supplied
 72              */
 73             indexAfter: function (index) {
 74                 return this._getNextPotientialIndexInDirection(index, Navigator.directions.FORWARD);
 75             },
 76 
 77             /**
 78              * @returns {Number} the index of the previous focusable widget
 79              */
 80             previousIndex: function () {
 81                 var currentIndex = this.currentIndex();
 82                 return this.indexBefore(currentIndex);
 83             },
 84 
 85             /**
 86              * @param index
 87              * @returns {Number} the first focussable index before that supplied
 88              */
 89             indexBefore: function (index) {
 90                 return this._getNextPotientialIndexInDirection(index, Navigator.directions.BACKWARD);
 91             },
 92 
 93             /**
 94              * @returns {Number} the number of widgets in the container under control
 95              */
 96             indexCount: function () {
 97                 return this._container.getChildWidgetCount();
 98             },
 99 
100             /**
101              * Activates the widget at the provided index within the container under control.
102              * If a valid (0 <= index < length of container) index, corresponding to a focusable widget is provided and the index
103              * is not the currently active index, the widget indexed is activated.
104              * If the index corresponds to an unfocussable widget or an invalid index the function has no effect.
105              * @param {Number} index A 0 base index into the container being navigated
106              */
107             setIndex: function (index) {
108                 if (this._isValidIndex(index) && !this._indexedWidgetCantBeFocussed(index)) {
109                     this._fireItemChangeEvent(index, BeforeSelectedItemChangeEvent);
110                     this._setActiveIndexOnContainer(index);
111                     this._fireItemChangeEvent(index, SelectedItemChangeEvent);
112                 }
113             },
114 
115             /**
116              * Sets the container the navigator is managing
117              */
118             setContainer: function (container) {
119                 this._container = container;
120             },
121 
122             _getActiveIndex: function () {
123                 var activeWidget, activeIndex;
124                 activeWidget = this._container.getActiveChildWidget();
125                 activeIndex = this._container.getIndexOfChildWidget(activeWidget);
126                 if (activeIndex !== -1) {
127                     return activeIndex;
128                 } else {
129                     return null;
130                 }
131             },
132 
133             _setActiveIndexOnContainer: function (activeIndex) {
134                 if (activeIndex !== this.currentIndex()) {
135                     this._container.setActiveChildIndex(activeIndex);
136                 }
137             },
138 
139             _indexedWidgetCantBeFocussed: function (index) {
140                 var widgets, focussable;
141                 widgets = this._container.getChildWidgets();
142                 focussable = widgets[index].isFocusable();
143                 return !focussable;
144             },
145 
146             _getIndexIncrementFunction: function (direction) {
147                 var nextFn;
148                 function forwardFn(index) { return index + 1; }
149                 function backFn(index) { return index - 1; }
150 
151                 switch (direction) {
152                 case Navigator.directions.FORWARD:
153                     nextFn = forwardFn;
154                     break;
155                 case Navigator.directions.BACKWARD:
156                     nextFn = backFn;
157                     break;
158                 }
159                 return nextFn;
160             },
161 
162             _getNextPotientialIndexInDirection: function (currentIndex, direction) {
163                 var potentialActiveIndex, indexIsValid, incrementIndex;
164                 incrementIndex = this._getIndexIncrementFunction(direction);
165 
166                 potentialActiveIndex = incrementIndex(currentIndex);
167                 indexIsValid = this._isValidIndex(potentialActiveIndex);
168 
169                 while (indexIsValid && this._indexedWidgetCantBeFocussed(potentialActiveIndex)) {
170                     potentialActiveIndex = incrementIndex(potentialActiveIndex);
171                     indexIsValid = this._isValidIndex(potentialActiveIndex);
172                 }
173 
174                 return potentialActiveIndex;
175             },
176 
177             _fireItemChangeEvent: function (index, EventClass) {
178                 var event, item, target;
179                 target = this._container;
180                 item = this._container.getChildWidgets()[index];
181                 event = new EventClass(target, item, index);
182                 target.bubbleEvent(event);
183             }
184         });
185 
186         Navigator.directions = {
187             FORWARD: 0,
188             BACKWARD: 1
189         };
190 
191         return Navigator;
192     }
193 );
