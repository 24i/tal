 30 require.def('antie/devices/wiiu',
 31     [
 32         'antie/devices/browserdevice',
 33         'antie/events/keyevent'
 34     ],
 35     function(BrowserDevice, KeyEvent) {
 36 
 37         return BrowserDevice.extend({
 38             init: function(config) {
 39                 this._super(config);
 40             },
 41             /**
 42              * Adds key event listeners to WiiU Gamepad and WiiRemote, Wii U Pro controller is
 43              * currently unsupported in NWF
 44              */
 45             addKeyEventListener: function() {
 46                 this._registerWiiUGamePadInputControls();
 47                 this._registerWiiRemoteInputControls();
 48             },
 49             /**
 50              * Attempts to register the key mappings for Wii U gamepad 1
 51              */
 52             _registerWiiUGamePadInputControls : function() {
 53                 var self = this;
 54 
 55                 var gamePad = nwf.input.WiiUGamePad.getController(); // defaults to GamePad channel 0, aka CONTROLLER_1
 56                 gamePad.buttons.addEventListener(nwf.events.ButtonControlEvent.PRESS, function(buttonEvent) {
 57                     self._onGamePadButtonPress(buttonEvent);
 58                 }, this);
 59                 gamePad.buttons.addEventListener(nwf.events.ButtonControlEvent.RELEASE, function(buttonEvent) {
 60                     self._onGamePadButtonRelease(buttonEvent);
 61                 }, this);
 62             },
 63             _onGamePadButtonPress : function(buttonEvent) {
 64                 var controllerButton = nwf.input.ControllerButton;
 65                 switch (buttonEvent.button) {
 66                     case controllerButton.GAMEPAD_A:
 67                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_ENTER));
 68                         break;
 69                     case controllerButton.GAMEPAD_B:
 70                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_BACK));
 71                         break;
 72                     case controllerButton.GAMEPAD_UP:
 73                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_UP));
 74                         break;
 75                     case controllerButton.GAMEPAD_DOWN:
 76                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_DOWN));
 77                         break;
 78 
 79                     case controllerButton.GAMEPAD_LEFT:
 80                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_LEFT));
 81                         break;
 82                     case controllerButton.GAMEPAD_RIGHT:
 83                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_RIGHT));
 84                         break;
 85                 }
 86             },
 87             _onGamePadButtonRelease: function(buttonEvent) {
 88                 var controllerButton = nwf.input.ControllerButton;
 89                 switch (buttonEvent.button) {
 90                     case controllerButton.GAMEPAD_A:
 91                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_ENTER));
 92                         break;
 93                     case controllerButton.GAMEPAD_B:
 94                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_BACK));
 95                         break;
 96                     case controllerButton.GAMEPAD_UP:
 97                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_UP));
 98                         break;
 99                     case controllerButton.GAMEPAD_DOWN:
100                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_DOWN));
101                         break;
102                     case controllerButton.GAMEPAD_LEFT:
103                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_LEFT));
104                         break;
105                     case controllerButton.GAMEPAD_RIGHT:
106                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_RIGHT));
107                         break;
108                 }
109             },
110             /**
111              * Attempts to register the keymappings for Wii Remote 1
112              * Does not account for the Nunchuk
113              */
114             _registerWiiRemoteInputControls : function() {
115                 var self = this;
116 
117                 // Get instance of controller 0, which is Player 1 (controller array is zero-based)
118                 var wiiRemote = nwf.input.WiiRemote.getController(nwf.input.WiiRemote.REMOTE_1);
119                 wiiRemote.buttons.addEventListener(nwf.events.ButtonControlEvent.PRESS, function(buttonEvent) {
120                     self._onWiiRemoteButtonPress(buttonEvent);
121                 }, this);
122 
123                 wiiRemote.buttons.addEventListener(nwf.events.ButtonControlEvent.RELEASE, function(buttonEvent) {
124                     self._onWiiRemoteButtonRelease(buttonEvent);
125                 }, this);
126             },
127             _onWiiRemoteButtonPress : function(buttonEvent) {
128                 var controllerButton = nwf.input.ControllerButton;
129                 switch (buttonEvent.button) {
130                     case controllerButton.WII_REMOTE_A:
131                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_ENTER));
132                         break;
133                     case controllerButton.WII_REMOTE_B:
134                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_BACK));
135                         break;
136                     case controllerButton.WII_REMOTE_UP:
137                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_UP));
138                         break;
139                     case controllerButton.WII_REMOTE_DOWN:
140                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_DOWN));
141                         break;
142                     case controllerButton.WII_REMOTE_LEFT:
143                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_LEFT));
144                         break;
145                     case controllerButton.WII_REMOTE_RIGHT:
146                         this._application.bubbleEvent(new KeyEvent("keydown", KeyEvent.VK_RIGHT));
147                         break;
148                 }
149             },
150             _onWiiRemoteButtonRelease : function(buttonEvent) {
151                 var controllerButton = nwf.input.ControllerButton;
152                 switch (buttonEvent.button) {
153                     case controllerButton.WII_REMOTE_A:
154                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_ENTER));
155                         break;
156                     case controllerButton.WII_REMOTE_B:
157                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_BACK));
158                         break;
159                     case controllerButton.WII_REMOTE_UP:
160                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_UP));
161                         break;
162                     case controllerButton.WII_REMOTE_DOWN:
163                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_DOWN));
164                         break;
165                     case controllerButton.WII_REMOTE_LEFT:
166                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_LEFT));
167                         break;
168                     case controllerButton.WII_REMOTE_RIGHT:
169                         this._application.bubbleEvent(new KeyEvent("keyup", KeyEvent.VK_RIGHT));
170                         break;
171                 }
172             }
173         });
174     }
175 );
