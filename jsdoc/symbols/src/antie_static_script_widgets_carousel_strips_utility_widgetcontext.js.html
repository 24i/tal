 24 require.def('antie/widgets/carousel/strips/utility/widgetcontext',
 25     [
 26         'antie/class'
 27     ],
 28     function (Class) {
 29         'use strict';
 30         var WidgetContext;
 31         WidgetContext = Class.extend({
 32             init: function (widget, parent, STATES) {
 33                 this._widget = widget;
 34                 this._parent = parent;
 35                 this.STATES = STATES;
 36                 this._state = new this.STATES.INIT(this);
 37             },
 38 
 39             /**
 40              * Appends output element to parent if not already child
 41              */
 42             append: function () {
 43                 this._state.append(this, this._parent, this._widget);
 44             },
 45 
 46             /**
 47              * Prepends output element to parent if not already child
 48              */
 49             prepend: function () {
 50                 this._state.prepend(this, this._parent, this._widget);
 51             },
 52 
 53             /**
 54              * Element no longer needs to be visible (e.g. remove from DOM)
 55              */
 56             detach: function () {
 57                 this._state.detach(this, this._widget);
 58             },
 59 
 60             /**
 61              * @returns {Boolean} true if widget currently takes up space in its parent, false otherwise
 62              * e.g. would return true if rendered, in the document and without display: none set
 63              * would return false if not rendered, not in the DOM or with display: none set
 64              */
 65             hasLength: function () {
 66                 return this._state.hasLength();
 67             },
 68 
 69             inView: function () {
 70                 return this._state.inView();
 71             },
 72 
 73             setState: function (stateName) {
 74                 this._state = new this.STATES[stateName](this);
 75             }
 76         });
 77 
 78         return WidgetContext;
 79     }
 80 );
