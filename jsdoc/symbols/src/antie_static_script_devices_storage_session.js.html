 27 require.def(
 28 	'antie/devices/storage/session',
 29 	['antie/storageprovider'],
 30 	function(StorageProvider) {
 31 		'use strict';
 32 
 33 		var namespaces = {};
 34 		var SessionStorage = StorageProvider.extend({});
 35 
 36 		SessionStorage.getNamespace = function(namespace) {
 37 			if(!namespaces[namespace]) {
 38 				namespaces[namespace] = new SessionStorage();
 39 			}
 40 			return namespaces[namespace];
 41 		};
 42 
 43 		return SessionStorage;
 44 	}
 45 );
