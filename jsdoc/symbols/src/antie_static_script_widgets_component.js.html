 27 require.def('antie/widgets/component',
 28     [
 29         'antie/widgets/container',
 30         'antie/runtimecontext'
 31     ],
 32     function(Container, RuntimeContext) {
 33         'use strict';
 34 
 35         /**
 36          * The Component widget class represents sections of UI that may be dynamically loaded.
 37          * @name antie.widgets.Component
 38          * @class
 39          * @extends antie.widgets.Container
 40          * @requires antie.RuntimeContext
 41          * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 42          */
 43         return Container.extend(/** @lends antie.widgets.Component.prototype */ {
 44             /**
 45              * @constructor
 46              * @ignore
 47              */
 48             init: function(id) {
 49                 this._super(id);
 50                 this.addClass('component');
 51                 this._isModal = false;
 52             },
 53             /**
 54              * Hide the component.
 55              */
 56             hide: function() {
 57                 if(this.parentWidget) {
 58                     this.parentWidget.hide();
 59                 }
 60             },
 61             /**
 62              * Returns any state information required (in addition to the initial arguments) that is required to restore this component.
 63              * @returns State information
 64              */
 65             getCurrentState: function() {
 66                 return null;
 67             },
 68 
 69             getIsModal: function() {
 70                 return this._isModal;
 71             },
 72 
 73             setIsModal: function(modal) {
 74                 this._isModal = modal;
 75             },
 76 
 77             getConfig: function() {
 78                 return RuntimeContext.getDevice().getConfig();
 79             },
 80 
 81             /**
 82              * Returns whether the widget is a Component.
 83              * @returns {Boolean} True if the widget is a Component.
 84              */
 85             isComponent: function() {
 86                 return true;
 87             }
 88         });
 89     }
 90 );
