 37 require.def('antie/events/event',
 38 	['antie/class'],
 39 	function(Class) {
 40 		var eventCount = 0;
 41 		var eventListeners = {};
 42 	
 43 		/**
 44 	 	 * Abstract base class for events.
 45 	 	 * @abstract
 46 	 	 * @extends antie.Class
 47 	 	 * @name antie.events.Event
 48 		 * @class
 49 	 	 * @param {String} type The event type (e.g. <code>keydown</code>, <code>databound</code>).
 50 		 */
 51 		var Event = Class.extend(/** @lends antie.events.Event.prototype */ {
 52 			/**
 53 			 * @constructor
 54 			 * @ignore
 55 			 */
 56 			init: function(type) {
 57 				this.type = type;
 58 				this._propagationStopped = false;
 59 				this._defaultPrevented = false;
 60 				eventCount++;
 61 			},
 62 			/**
 63 			 * Stop propagation of the event through the widget tree.
 64 			 */
 65 			stopPropagation: function() {
 66 				this._propagationStopped = true;
 67 				eventCount--;
 68 				if (!eventCount) {
 69 					this.fireEvent('emptyStack');
 70 				}
 71 			},
 72 			/**
 73 			 * Check to see if the propagation of this event has been stopped.
 74 			 * @returns Boolean true if the event has been stopped, otherwise Boolean false.
 75 			 */
 76 			isPropagationStopped: function() {
 77 				return this._propagationStopped;
 78 			},
 79 			/**
 80 			 * Prevent any default handler being called for this event.
 81 			 */
 82 			preventDefault: function() {
 83 				this._defaultPrevented = true;
 84 			},
 85 			/**
 86 			 * Check to see if the default handler has been prevented.
 87 			 * @returns Boolean true if the default handler has been prevented.
 88 			 */
 89 			isDefaultPrevented: function() {
 90 				return this._defaultPrevented;
 91 			},
 92 			/**
 93 			 * Add an event listener function to the event stack. Used for 'meta-events'.
 94 			 * @memberOf antie.events.Event
 95 			 * @static
 96 			 * @param {String} ev The event type to listen for (e.g. <code>emptyStack</code>)
 97 			 * @param {Function} func The handler to be called when the event is fired.
 98 			 */
 99 			addEventListener: function(ev, func) {
100 				var listeners = eventListeners[ev];
101 				if(!listeners) {
102 					listeners = eventListeners[ev] = {};
103 				}
104 				if(!listeners[func]) {
105 					listeners[func] = func;
106 				}
107 			},
108 			/**
109 			 * Removes an event listener function to the event stack. Used for 'meta-events'.
110 			 * @memberOf antie.events.Event
111 			 * @static
112 			 * @param {String} ev The event type that the listener is to be removed from (e.g. <code>emptyStack</code>)
113 			 * @param {Function} func The handler to be removed.
114 			 */
115 			removeEventListener: function(ev, func) {
116 				var listeners = eventListeners[ev];
117 				if(listeners && listeners[func]) {
118 					delete(listeners[func]);
119 				}
120 			},
121 			/**
122 			 * Fires an event on the event stack.
123 			 * Note: this does not bubble or propagate the event, the concept is meaningless in the context
124 			 * of the event stack.
125 			 * @memberOf antie.events.Event
126 			 * @static
127 			 * @param {String} ev The event to fire (e.g. <code>emptyStack</code>).
128 			 * @see antie.events.Event
129 			 */
130 			fireEvent: function(ev) {
131 				var listeners = eventListeners[ev];
132 				if(listeners) {
133 					for(var func in listeners) {
134 						listeners[func]();
135 					}
136 				}
137 			}
138 		});
139 
140 		return Event;
141 	}
142 );
