 27 require.def('antie/events/keyevent',
 28 	['antie/events/event'],
 29 	function(Event) {
 30 		'use strict';
 31 
 32 		/**
 33 		 * Class of events raised when key/remote control buttons are pressed.
 34 		 * Any keycodes contained in these events will have been mapped to
 35 		 * application specific codes (e.g. KeyEvent.VK_ENTER).
 36 		 * @name antie.events.KeyEvent
 37 		 * @class
 38 		 * @extends antie.events.Event
 39 		 * @param {String} type The event type (e.g. <code>keydown</code>, <code>keyup</code>).
 40 		 * @param {Integer} keyCode The normalised keyCode of the key that caused this event to be raised (e.g. <code>KeyEvent.VK_ENTER</code>, <code>KeyEvent.VK_UP</code>).
 41 		 */
 42 		var KeyEvent = Event.extend(/** @lends antie.events.KeyEvent.prototype */ {
 43 			/**
 44 			 * @constructor
 45 			 * @ignore
 46 			 */
 47 			init: function(type, keyCode) {
 48 				var index;
 49 				this.keyCode = keyCode;
 50 
 51 				// Map hardware alphanumeric key codes back into characters (and space)
 52 				if(keyCode >= KeyEvent.VK_0 && keyCode <= KeyEvent.VK_9) {
 53 					index = keyCode - KeyEvent.VK_0;
 54 					this.keyChar = String.fromCharCode("0".charCodeAt(0) + index);
 55 				} else if(keyCode == KeyEvent.VK_SPACE) {
 56 					this.keyChar = " ";
 57 				} else if(keyCode >= KeyEvent.VK_A && keyCode <= KeyEvent.VK_Z) {
 58 					index = keyCode - KeyEvent.VK_A;
 59 					this.keyChar = String.fromCharCode("A".charCodeAt(0) + index);
 60 				}
 61 
 62 				this._super(type);
 63 			}
 64 		});
 65 
 66 		/* Device classes should map device-specific keycodes to the following... */
 67 
 68 		/**
 69 		 * Virtual key code for the enter/select button.
 70 		 * Based on CEA-2014-A CE-HTML Annex F
 71 		 * @memberOf antie.events.KeyEvent
 72 		 * @name VK_ENTER
 73 		 * @constant
 74 		 * @static
 75 		 */
 76 		KeyEvent.VK_ENTER = 13;
 77 
 78 		/**
 79 		 * Virtual key code for the left cursor/arrow button.
 80 		 * Based on CEA-2014-A CE-HTML Annex F
 81 		 * @memberOf antie.events.KeyEvent
 82 		 * @name VK_LEFT
 83 		 * @constant
 84 		 * @static
 85 		 */
 86 		KeyEvent.VK_LEFT = 37;
 87 
 88 		/**
 89 		 * Virtual key code for the up cursor/arrow button.
 90 		 * Based on CEA-2014-A CE-HTML Annex F
 91 		 * @memberOf antie.events.KeyEvent
 92 		 * @name VK_UP
 93 		 * @constant
 94 		 * @static
 95 		 */
 96 		KeyEvent.VK_UP = 38;
 97 
 98 		/**
 99 		 * Virtual key code for the right cursor/arrow button.
100 		 * Based on CEA-2014-A CE-HTML Annex F
101 		 * @memberOf antie.events.KeyEvent
102 		 * @name VK_RIGHT
103 		 * @constant
104 		 * @static
105 		 */
106 		KeyEvent.VK_RIGHT = 39;
107 
108 		/**
109 		 * Virtual key code for the down cursor/arrow button.
110 		 * Based on CEA-2014-A CE-HTML Annex F
111 		 * @memberOf antie.events.KeyEvent
112 		 * @name VK_DOWN
113 		 * @constant
114 		 * @static

117

119

121

123 		KeyEvent.VK_1 = 49;
