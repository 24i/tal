 39 		/**
 40 		 * Abstract base class for Bigscreen applications.
 41 		 * @name antie.Application
 42 		 * @class
 43 		 * @abstract
 44 		 * @extends antie.Class
 45 		 * @requires antie.widgets.ComponentContainer
 46 		 * @requires antie.widgets.List
 47 		 * @param {DOMElement} rootElement DOMElement into which to render the application.
 48 		 * @param {String} styleBaseURL URL pointing to path stylesheets are relative to.
 49 		 * @param {String} imageBaseURL URL pointing to path images are relative to.
 50 		 * @param {function(Event)} onReadyHandler Function called when application is ready.
 51 		 * @param {Object} [configOverride] Optional config to override default.
 52 		 */
 53 		var Application = Class.extend(/** @lends antie.Application.prototype */ {
 54 			/**
 55 			 * @constructor
 56 			 * @ignore
 57 			 */
 58 
 59 			init: function(rootElement, styleBaseUrl, imageBaseUrl, onReadyHandler, configOverride) {
 60 				RuntimeContext.setCurrentApplication(this);
 61 
 62 				this._rootElement = rootElement;
 63 				this._rootWidget = null;
 64 				this._focussedWidget = null;
 65 				this._onReadyHandler = onReadyHandler;
 66 
 67 				var self = this;
 68 
 69 				var _configuration = configOverride || antie.framework.deviceConfiguration;
 70 				if(!this._device) {
 71 					Device.load(_configuration, {
 72 						onSuccess: deviceLoaded,
 73 						onError: function(err) {
 74 							console.error("Unable to load device", err);
 75 						}
 76 					});
 77 				} else {
 78 					deviceLoaded(this._device);
 79 				}
 80 				function deviceLoaded(device) {
 81 				    var i;
 82 					self._device = device;
 83 					device.setApplication(self);
 84 					device.addKeyEventListener();
 85 					var _layout = self.getBestFitLayout();
 86 					_layout.css = _layout.css || [];
 87 					if (_configuration.css){
 88 						for ( i = 0; i < _configuration.css.length; i++){
 89 							if  (  _configuration.css[i].width == _layout.width
 90 								&& _configuration.css[i].height == _layout.height){
 91 								_layout.css = _layout.css.concat(_configuration.css[i].files);
 92 							}
 93 						}
 94 					}
 95 					
 96 					require([_layout.module], function(layout) {
 97 						self.setLayout(layout, styleBaseUrl, imageBaseUrl, _layout.css, _layout.classes, [], function() {
 98 							self.run();
 99 							self.route(device.getCurrentRoute());
100 						});
101 					});
102 				}
103 			},
104 			/**
105 			 * Called once application startup is ready (i.e. config has been loaded).
106 			 */
107 			run: function() {
108 				// intentionally left blank
109 			},
110 			/**
111 			 * Must be called when the application startup is complete and application can accept user input.
112 			 */
113 			ready: function() {
114 				if(this._onReadyHandler) {
115 					var self = this;
116 					// Run this after the current execution path is complete
117 					window.setTimeout(function() {
118 						self._onReadyHandler(self);
119 					}, 0);
120 				}
121 			},
122 			/**
123 			 * Gets the largest supported layout that fits within the available screen resolution.
124 			 * @returns An object literal describing which layout module to load.
125 			 */
126 			getBestFitLayout: function() {
127 			    var i;
128 				var _screenSize = this._device.getScreenSize();
129 				var _layouts = this._device.getConfig().layouts;
130 
131 				// sort the layouts by largest first
132 				_layouts.sort(function(a, b) {
133 					var	ad = (a.width*a.width) + (a.height*b.height),
134 						bd = (b.width*b.width) + (b.height*b.height);
135 					if(ad == bd) return 0;
136 					else if(ad < bd) return 1;
137 					else return -1;
138 				});
139 				var _module;
140 				for(i in _layouts) {
141 					if((_screenSize.height >= _layouts[i].height) && (_screenSize.width >= _layouts[i].width)) {
142 						return _layouts[i];
143 					}
144 				}
145 
146 				// No layouts fit
147 				// TODO: what do we do here?
148 				// TODO: for now we're just returning the smallest.
149 				return _layouts[_layouts.length-1];
150 			},
151 			/**
152 			 * Sets the current layout used by the application.
153 			 * @param {Object} layout An application-specific object literal describing layout-specific properties
154 			 * @param {String} styleBaseUrl Base URL of stylesheets for the application.
155 			 * @param {String} imageBaseUrl Base URL of images for the application.
156 			 * @param {Array} [additionalCSS] Additional stylesheet URLs to load.
157 			 * @param {Array} [additionalClasses] Additional classes to add to the document element.
158 			 * @param {Array} [additionalPreloadImages] Additional images to preload.
159 			 * @param {function()} [callback] Callback function to call when layout has been fully loaded.
160 			 */
161 			setLayout: function(layout, styleBaseUrl, imageBaseUrl, additionalCSS, additionalClasses, additionalPreloadImages, callback) {
162 			    var i;
163 				this._layout = layout;
164 				var tle = this._device.getTopLevelElement();
165                 
166 				var classes = layout.classes || [];
167 				if(additionalClasses) {
168 					classes = classes.concat(additionalClasses);
169 				}
170 				for(i = 0; i !== classes.length; i += 1) {
171 					this._device.addClassToElement(tle, classes[i]);
172 				}
173 
174 				var preloadImages = layout.preloadImages || [];
175 				if(additionalPreloadImages) {
176 					preloadImages = preloadImages.concat(additionalPreloadImages);
177 				}
178 				for(i = 0; i !== preloadImages.length; i += 1) {
179 					this._device.preloadImage(imageBaseUrl + preloadImages[i]);
180 				}
181 
182 				var css = layout.css || [];
183 				if(additionalCSS) {
184 					css = css.concat(additionalCSS);
185 				}
186 				if(callback) {
187 					var currentlyLoadingIndex = -1;
188 					var self = this;
189 					var cssLoadedCallback = function() {
190 						if(++currentlyLoadingIndex < css.length) {
191 							self._device.loadStyleSheet(styleBaseUrl + css[currentlyLoadingIndex], cssLoadedCallback);
192 						} else {
193 							callback();
194 						}
195 					}
196 					cssLoadedCallback();
197 				} else {
198 					for(i = 0; i !== css.length; i += 1) {
199 						this._device.loadStyleSheet(styleBaseUrl + css[i]);
200 					}
201 				}
202 			},
203 			/**
204 			 * Gets the current layout as set by {@see #setLayout}.
205 			 * @returns An application-specific object literal describing layout-specific properties.
206 			 */
207 			getLayout: function() {
208 				return this._layout;
209 			},
210 			/**
211 			 * Set the root widget of the application.
212 			 * @param {antie.widgets.Widget} widget The new root widget.
213 			 */
214 			setRootWidget: function(widget) {
215 				widget.addClass('rootwidget');
216 				if(widget instanceof List) {
217 					widget.setRenderMode(List.RENDER_MODE_CONTAINER);
218 				}
219 				this._rootWidget = widget;
220 				this._rootWidget._isFocussed = true;
221 				if(this._rootWidget.outputElement == null) {
222 					var device = this.getDevice();
223 					device.appendChildElement(this._rootElement, widget.render(device));
224 				}
225 				var self = this;
226 				if(self._onReadyHandler) {
227 					this._rootWidget.addEventListener('applicationready', function(e) {
228 						self._onReadyHandler(e);
229 					});
230 				}
231 			},
232 			/**
233 			 * Get the root widget of the application.
234 			 * @return The root widget of the application.
235 			 */
236 			getRootWidget: function() {
237 				return this._rootWidget;
238 			},
239 			/**
240 			 * Add a component container to the application.
241 			 * @param {String} id The ID of the new component container.
242 			 * @param {String} [modules] The requirejs module name of the component to pre-load.
243 			 * @param {Object} [args] An optional object to pass arguments into any pre-loaded component.
244 			 */
245 			addComponentContainer: function(id, module, args) {
246 				var container = new ComponentContainer(id);
247 				this._rootWidget.appendChildWidget(container);
248 
249 				if(module) {
250 					this.showComponent(id, module, args);
251 				}
252 
253 				return container;
254 			},
255 			/**
256 			 * Show a component in a component container. This resets any history for the container.
257 			 * @see #pushComponent
258 			 * @param {String} id The ID of the container into which to show the component.
259 			 * @param {String} modules The requirejs module name of the component to show.
260 			 * @param {Object} [args] An optional object to pass arguments to the component.
261 			 */
262 			showComponent: function(id, module, args) {

395             /**
396              * Launch a new application at the specified URL, passing in query string data and
397              * a route.
398              * @param {String}  url         The URL to launch (include protocol, host and port)
399              * @param {Object}  [data]      Parameters to pass in the query string. Property names are keys. Values are sent as strings. Use {undefined} as a value to send a valueless key.
400              * @param {Array}   [route]     Route for new application (a reference pointing to a new location within the application). @see getCurrentRoute(), @see setCurrentRoute()
401              * @param {Boolean} [overwrite] Set true to overwrite the query parameters of the current application location. Default behaviour is to merge the values passed in the 'data' param.
402              */
403             launchAppFromURL: function(url, data, route, overwrite) {
404                 var query = '';
405                 var hash = '';
406                 var key;
407                 var completeUrl = '';
408 
409                 // Get existing query data, or start a brand new object if applicable.
410                 var mergedData = overwrite ? {} : this.getCurrentAppURLParameters();
411 
412                 // Merge in the query data passed to this function, which takes precedence over anything existing.
413                 if (data) {
414                     for (key in data) {
415                         if (data.hasOwnProperty(key)) {
416                             mergedData[key] = data[key]; // No need to escape here...
417                         }
418                     }
419                 }
420                 // Construct a query string out of the merged data.
421                 for (key in mergedData) {
422                     if (mergedData.hasOwnProperty(key)) {
423                         // Keys with values get 'key=value' syntax, valueless keys just get 'key'
424                         if (mergedData[key] === undefined || mergedData[key] === null) {
425                             query += window.encodeURIComponent(key) + '&';
426                         }
427                         else {
428                             query += window.encodeURIComponent(key) + '=' + window.encodeURIComponent(mergedData[key]) + '&'; 
429                         }
430                     }
431                 }
432                 // Trim the last & off, add a leading ?
433                 if (query.length > 0) {
434                     query = '?' + query.slice(0, -1);
435                 }
436 
437                 // Construct the route string.
438                 if (route && route.length > 0) {
439                     hash = '#' + route.join('/');
440                 }
441                 
442                 completeUrl = this.getDevice().getHistorian().forward(url + query + hash);
443                 
444                 // Send the browser to the final URL
445                 this.getDevice().setWindowLocationUrl(completeUrl);
446             },
447             /**
448              * Return the URL of the current application, with no route or query string information -
449              * @see getCurrentRoute(), @see getCurrentAppURLParameters().
450              * @returns {String} URL of the current application, including protocol, host and port.
451              */
452             getCurrentAppURL: function() {
453                 var location = this.getDevice().getWindowLocation();
454                 return location.protocol + '//' + location.host + location.pathname;
455             },
456             /**
457              * Returns the query string of the current application as an object containing properties with
458              * string values.
459              * Keys without values are represented by properties with the value 'undefined'. Empty string values are
460              * represented in the object as a property with an empty string value.
461              * @returns {Object} Object containing properties held in the query string.
462              */
463             getCurrentAppURLParameters: function() {
464                 var location = this.getDevice().getWindowLocation();
465                 if (!location.search) {
466                     return {};
467                 }
468 
469                 // Split query string to array by & character
470                 var queryKeyValuePairs = location.search.replace(/^\?/, '').split('&');
471                 var queryKeyValuePair, value;
472                 var queryData = {};
473                 for (var i = 0; i < queryKeyValuePairs.length; i++) {
474                     // Assign each key/value to a property in queryData. undefined is set as the value if there is no value.
475                     queryKeyValuePair = queryKeyValuePairs[i].split('='); 
476                     value = queryKeyValuePair[1];
477                     if (queryKeyValuePair[1]) {
478                         value = window.decodeURIComponent(value);
479                     }
480                     queryData[window.decodeURIComponent(queryKeyValuePair[0])] = value;
481                 }
482 
483                 return queryData;
484             },
485 			/**
486 			 * Destroys the application, allowing you to run another. This is mainly for use when building
487 			 * unit or BDD tests.
488 			 */
489 			destroy: function () {
490 				RuntimeContext.clearCurrentApplication();
491 				ComponentContainer.destroy();
492 			},
493 			
494 			/**
495 			 * Navigates back to whatever launched the application (a parent TAL application, broadcast, or exit).
496 			 */
497 			back: function() {
498 			    var historian = this.getDevice().getHistorian();
499 			    if (historian.hasHistory()) {
500 			        this.getDevice().setWindowLocationUrl(historian.back());
501 			    }
502 			    else {
503 			        this.exit();
504 			    }
505 			},
506 
507             /**
508              * Returns a Boolean value to indicate whether the application can go back to a parent TAL application.
509              * @returns {Boolean} True if the application can return to a parent TAL application.
510              */
511             hasHistory: function() {
512                 return this.getDevice().getHistorian().hasHistory();
513             },
514 
515 			/**
516 			 * Exits the application by using the configured exit strategy for the device, even if there is a parent TAL
517 			 * application in the history stack. Will exit to broadcast if the first TAL application was launched from
518 			 * broadcast and a broadcast exit modifier is loaded.
519 			 */
520 			exit: function() {
521                 if (this.getDevice().getHistorian().hasBroadcastOrigin()) {
522                     this.getDevice().exitToBroadcast();
523                 } else {
524                     this.getDevice().exit();
525                 }
526 			}
