 27 require.def('antie/application',
 28 	[
 29 	 	'antie/class',
 30 	 	'antie/widgets/componentcontainer',
 31 		'antie/widgets/button',
 32 		'antie/widgets/list',
 33 		'antie/devices/device'
 34 	],
 35 	function(Class, ComponentContainer, Button, List, Device) {
 36 		/**
 37 		 * Contains a reference to the single instance of the antie.Application class.
 38 		 * @private
 39 		 */
 40 		var applicationObject;
 41 
 42 		/**
 43 		 * Abstract base class for Bigscreen applications.
 44 		 * @name antie.Application
 45 		 * @class
 46 		 * @abstract
 47 		 * @extends antie.Class
 48 		 * @requires antie.widgets.ComponentContainer
 49 		 * @requires antie.widgets.List
 50 		 * @param {DOMElement} rootElement DOMElement into which to render the application.
 51 		 * @param {String} styleBaseURL URL pointing to path stylesheets are relative to.
 52 		 * @param {String} imageBaseURL URL pointing to path images are relative to.
 53 		 * @param {function(Event)} onReadyHandler Function called when application is ready.
 54 		 * @param {Object} [configOverride] Optional config to override default.
 55 		 */
 56 		var Application = Class.extend(/** @lends antie.Application.prototype */ {
 57 			/**
 58 			 * @constructor
 59 			 * @ignore
 60 			 */
 61 			init: function(rootElement, styleBaseUrl, imageBaseUrl, onReadyHandler, configOverride) {
 62 				if(applicationObject) {
 63 					throw new Error("Application::init called for a second time. You can only have one application instance running at any time.");
 64 				}
 65 				this._rootElement = rootElement;
 66 				this._rootWidget = null;
 67 				this._focussedWidget = null;
 68 				this._onReadyHandler = onReadyHandler;
 69 
 70 				applicationObject = this;
 71 
 72 				var self = this;
 73 
 74 				var _configuration = configOverride || antie.framework.deviceConfiguration;
 75 				if(!this._device) {
 76 					Device.load(_configuration, {
 77 						onSuccess: deviceLoaded,
 78 						onError: function(err) {
 79 							console.error("Unable to load device", err);
 80 						}
 81 					});
 82 				} else {
 83 					deviceLoaded(this._device);
 84 				}
 85 				function deviceLoaded(device) {
 86 				    var i;
 87 					self._device = device;
 88 					device.setApplication(self);
 89 					device.addKeyEventListener();
 90 					var _layout = self.getBestFitLayout();
 91 					_layout.css = _layout.css || [];
 92 					if (_configuration.css){
 93 						for ( i = 0; i < _configuration.css.length; i++){
 94 							if  (  _configuration.css[i].width == _layout.width
 95 								&& _configuration.css[i].height == _layout.height){
 96 								_layout.css = _layout.css.concat(_configuration.css[i].files);
 97 							}
 98 						}
 99 					}
100 					
101 					require([_layout.module], function(layout) {
102 						self.setLayout(layout, styleBaseUrl, imageBaseUrl, _layout.css, _layout.classes, [], function() {
103 							self.run();
104 							self.route(device.getCurrentRoute());
105 						});
106 					});
107 				}
108 			},
109 			/**
110 			 * Called once application startup is ready (i.e. config has been loaded).
111 			 */
112 			run: function() {
113 				// intentionally left blank
114 			},
115 			/**
116 			 * Must be called when the application startup is complete and application can accept user input.
117 			 */
118 			ready: function() {
119 				if(this._onReadyHandler) {
120 					var self = this;
121 					// Run this after the current execution path is complete
122 					window.setTimeout(function() {
123 						self._onReadyHandler(self);
124 					}, 0);
125 				}
126 			},
127 			/**
128 			 * Gets the largest supported layout that fits within the available screen resolution.
129 			 * @returns An object literal describing which layout module to load.
130 			 */
131 			getBestFitLayout: function() {
132 			    var i;
133 				var _screenSize = this._device.getScreenSize();
134 				var _layouts = this._device.getConfig().layouts;
135 
136 				// sort the layouts by largest first
137 				_layouts.sort(function(a, b) {
138 					var	ad = (a.width*a.width) + (a.height*b.height),
139 						bd = (b.width*b.width) + (b.height*b.height);
140 					if(ad == bd) return 0;
141 					else if(ad < bd) return 1;
142 					else return -1;
143 				});
144 				var _module;
145 				for(i in _layouts) {
146 					if((_screenSize.height >= _layouts[i].height) && (_screenSize.width >= _layouts[i].width)) {
147 						return _layouts[i];
148 					}
149 				}
150 
151 				// No layouts fit
152 				// TODO: what do we do here?
153 				// TODO: for now we're just returning the smallest.
154 				return _layouts[_layouts.length-1];
155 			},
156 			/**
157 			 * Sets the current layout used by the application.
158 			 * @param {Object} layout An application-specific object literal describing layout-specific properties
159 			 * @param {String} styleBaseUrl Base URL of stylesheets for the application.
160 			 * @param {String} imageBaseUrl Base URL of images for the application.
161 			 * @param {Array} [additionalCSS] Additional stylesheet URLs to load.
162 			 * @param {Array} [additionalClasses] Additional classes to add to the document element.
163 			 * @param {Array} [additionalPreloadImages] Additional images to preload.
164 			 * @param {function()} [callback] Callback function to call when layout has been fully loaded.
165 			 */
166 			setLayout: function(layout, styleBaseUrl, imageBaseUrl, additionalCSS, additionalClasses, additionalPreloadImages, callback) {
167 			    var i;
168 				this._layout = layout;
169 				var tle = this._device.getTopLevelElement();
170                 
171 				var classes = layout.classes || [];
172 				if(additionalClasses) {
173 					classes = classes.concat(additionalClasses);
174 				}
175 				for(i = 0; i !== classes.length; i += 1) {
176 					this._device.addClassToElement(tle, classes[i]);
177 				}
178 
179 				var preloadImages = layout.preloadImages || [];
180 				if(additionalPreloadImages) {
181 					preloadImages = preloadImages.concat(additionalPreloadImages);
182 				}
183 				for(i = 0; i !== preloadImages.length; i += 1) {
184 					this._device.preloadImage(imageBaseUrl + preloadImages[i]);
185 				}
186 
187 				var css = layout.css || [];
188 				if(additionalCSS) {
189 					css = css.concat(additionalCSS);
190 				}
191 				if(callback) {
192 					var currentlyLoadingIndex = -1;
193 					var self = this;
194 					function cssLoadedCallback() {
195 						if(++currentlyLoadingIndex < css.length) {
196 							self._device.loadStyleSheet(styleBaseUrl + css[currentlyLoadingIndex], cssLoadedCallback);
197 						} else {
198 							callback();
199 						}
200 					}
201 					cssLoadedCallback();
202 				} else {
203 					console.log('no callback');
204 					for(i = 0; i !== css.length; i += 1) {
205 						this._device.loadStyleSheet(styleBaseUrl + css[i]);
206 					}
207 				}
208 			},
209 			/**
210 			 * Gets the current layout as set by {@see #setLayout}.
211 			 * @returns An application-specific object literal describing layout-specific properties.
212 			 */
213 			getLayout: function() {
214 				return this._layout;
215 			},
216 			/**
217 			 * Set the root widget of the application.
218 			 * @param {antie.widgets.Widget} widget The new root widget.
219 			 */
220 			setRootWidget: function(widget) {
221 				widget.addClass('rootwidget');
222 				if(widget instanceof List) {
223 					widget.setRenderMode(List.RENDER_MODE_CONTAINER);
224 				}
225 				this._rootWidget = widget;
226 				this._rootWidget._isFocussed = true;
227 				if(this._rootWidget.outputElement == null) {
228 					var device = this.getDevice();
229 					device.appendChildElement(this._rootElement, widget.render(device));
230 				}
231 				var self = this;
232 				if(self._onReadyHandler) {
233 					this._rootWidget.addEventListener('applicationready', function(e) {
234 						self._onReadyHandler(e);
235 					});
236 				}
237 			},
238 			/**
239 			 * Get the root widget of the application.
240 			 * @return The root widget of the application.
241 			 */
242 			getRootWidget: function() {
243 				return this._rootWidget;
244 			},
245 			/**
246 			 * Add a component container to the application.
247 			 * @param {String} id The ID of the new component container.
248 			 * @param {String} [modules] The requirejs module name of the component to pre-load.
249 			 * @param {Object} [args] An optional object to pass arguments into any pre-loaded component.
250 			 */
251 			addComponentContainer: function(id, module, args) {
252 				var container = new ComponentContainer(id);
253 				this._rootWidget.appendChildWidget(container);
254 
255 				if(module) {
256 					this.showComponent(id, module, args);
257 				}
258 
259 				return container;
260 			},
261 			/**
262 			 * Show a component in a component container. This resets any history for the container.
263 			 * @see #pushComponent
264 			 * @param {String} id The ID of the container into which to show the component.
265 			 * @param {String} modules The requirejs module name of the component to show.
266 			 * @param {Object} [args] An optional object to pass arguments to the component.
267 			 */
268 			showComponent: function(id, module, args) {
269 				this._rootWidget.getChildWidget(id).show(module, args);
270 			},
271 			/**
272 			 * Pushes a component into the history stack of a container (and shows it).
273 			 * @param {String} id The ID of the container into which to show the component.
274 			 * @param {String} modules The requirejs module name of the component to show.
275 			 * @param {Object} [args] An optional object to pass arguments to the component.
276 			 */
277 			pushComponent: function(id, module, args) {
278 				this._rootWidget.getChildWidget(id).pushComponent(module, args);
279 			},
280 			/**
281 			 * Pops a component from the history stack of a container (if a previous component exists)
282 			 * @param {String} id The ID of the container that contains the component to pop.
283 			 */
284 			popComponent: function(id) {
285 				this._rootWidget.getChildWidget(id).back();
286 			},
287 			/**
288 			 * Hides a component.
289 			 * @param {String} id The ID of the container that contains the component to hide.
290 			 */
291 			hideComponent: function(id) {
292 				this._rootWidget.getChildWidget(id).hide();
293 			},
294 			/**
295 			 * Gets a component by ID.
296 			 * @param {String} id The ID of the container to return.
297 			 */
298 			getComponent: function(id) {
299 				return this._rootWidget.getChildWidget(id);
300 			},
301 			/**
302 			 * Sets the active focus to the specified component.
303 			 * @param {String} id The ID of the container that contains the component to set focus on.
304 			 */
305 			setActiveComponent: function(id) {
306 				return this._rootWidget.setActiveChildWidget(this._rootWidget.getChildWidget(id));
307 			},
308 			/**
309 			 * Returns the Device object currently running this application.
310 			 * @returns The antie.devices.Device object currently running this application.
311 			 */
312 			getDevice: function() {
313 				return this._device;
314 			},
315 			/**
316 			 * Bubbles an event from the currently focussed widget up through the widget tree.
317 			 * @param {antie.events.Event} evt The event to bubble.
318 			 */
319 			bubbleEvent: function(evt) {
320 				if(this._focussedWidget) {
321 					this._focussedWidget.bubbleEvent(evt);
322 				}
323 			},
324 
325 			/**
326 			 * Broadcasts an event from the application level to every single
327 			 * object it contains.
328 			 * @param {antie.events.Event} evt The event to broadcast.
329 			 */
330 			broadcastEvent: function(evt) {
331 				if (evt.sentDown) { return; }
332 				evt.sentDown = 1;
333 				this._rootWidget.broadcastEvent(evt);
334 			},
335 			/**
336 			 * Returns the currently-focussed Button.
337 			 * @returns The antie.widgets.Button which currently has focus.
338 			 */
339 			getFocussedWidget: function() {
340 				return this._focussedWidget;
341 			},
342 			/**
343 			 * Set the currently focussed Button.
344 			 * @param {antie.widgets.Button} widget The antie.widgets.Button that has recieved focus.
345 			 * @private
346 			 */
347 			_setFocussedWidget: function(widget) {
348 				// Check to see the widget is a Button and itself has correct focus state before recording
349 				// it as the focussed widget.
350 				if((widget instanceof Button) && widget.isFocussed()) {
351 					this._focussedWidget = widget;
352 				}
353 			},
354 			/**
355 			 * Adds an event listener to the root widget. This is the last point of call for events bubbled
356 			 * up through the UI.
357 			 * @param {String} evt The event to handle.
358 			 * @param {function(Event)} handler The handler function to call when the event hits the root widget.
359 			 */
360 			addEventListener: function(evt, handler) {
361 				this._rootWidget.addEventListener(evt, handler);
362 			},
363 			/**
364 			 * Removes an event listener from the root widget.
365 			 * @param {String} evt The event to handle.
366 			 * @param {function(Event)} handler The handler function to remove.
367 			 */
368 			removeEventListener: function(evt, handler) {
369 				this._rootWidget.removeEventListener(evt, handler);
370 			},
371 			/**
372 			 * Gets the current route (a reference pointing to a location within the application).
373 			 * @returns The current route (location within the application).
374 			 */
375 			getCurrentRoute: function() {
376 				return this._device.getCurrentRoute();
377 			},
378 			/**
379 			 * Sets the current route (a reference pointing to a location within the application).
380 			 * @param {Array} route A route pointing to a location within the application.
381 			 */
382 			setCurrentRoute: function(route) {
383 				this._referer = this._device.getCurrentRoute();
384 				this._device.setCurrentRoute(route);
385 			},
386 			/**
387 			 * Gets the reference (e.g. URL) of the resource that launched the application or
388 			 * the previous location within the application.
389 			 * @returns A reference (e.g. URL) of the previous location.
390 			 */
391 			getReferer: function () {
392 				return this._referer || this._device.getReferrer();
393 			},
394 			/**
395 			 * Called after {@link #run} to launch the application at a specific location.
396 			 * @param {Array} route Location of application.
397 			 */
398 			route: function(route) {
399 				// intentionally left blank
400 			},
401 			/**
402 			 * Destroys the application, allowing you to run another. This is mainly for use when building
403 			 * unit or BDD tests.
404 			 */
405 			destroy: function () {
406 				applicationObject = undefined;
407 				ComponentContainer.destroy();
408 			}
409 		});
410 
411 		/**
412 		 * Returns the currently executing application.
413 		 * @name getCurrentApplication
414 		 * @memberOf antie.Application
415 		 * @static
416 		 * @function
417 		 */
418 		Application.getCurrentApplication = function() {
419 			return applicationObject;
420 		};
421 
422 		return Application;
423 	}
