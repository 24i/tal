 28 require.def(
 29     'antie/devices/anim/css3/optionstransitiondefinition',
 30     [
 31         'antie/devices/anim/css3/transitiondefinition',
 32         'antie/devices/anim/css3/easinglookup'
 33     ],
 34     function(TransitionDefinition, EasingLookup) {
 35         "use strict";
 36         return TransitionDefinition.extend(
 37             {
 38                 /*
 39                  * Transition definition from options.
 40                  * @param animation options object
 41                  * @param config animation configuration object
 42                  */
 43                 init: function(options, config) {
 44                     var units, property, timeEasing;
 45                     config = config || {};
 46                     this._super();
 47                     units = options.units || {};
 48                     for(property in options.to) {
 49                         if(options.to.hasOwnProperty(property)) {
 50                             timeEasing = options.easing || config.easing;
 51                             this.setProperty(property, {
 52                                duration: options.duration || config.duration,
 53                                delay: options.delay,
 54                                timingFn: timeEasing ? TransitionDefinition.easingLookup[timeEasing] : undefined
 55                             });
 56                         }
 57                     }
 58                 }
 59             }
 60         );
 61     }
 62 );
