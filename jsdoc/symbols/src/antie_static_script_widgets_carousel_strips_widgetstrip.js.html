 24 require.def('antie/widgets/carousel/strips/widgetstrip',
 25     [
 26         'antie/widgets/container'
 27     ],
 28     function (Container) {
 29         "use strict";
 30         /**
 31          * A container for the widgets displayed within a carousel
 32          * @name antie.widgets.carousel.strips.WidgetStrip
 33          * @class
 34          * @extends antie.widgets.Container
 35          * @param {String} id The unique ID of the widget.
 36          * @param {Object} orientation an object representing the strip's orientation.
 37          * One of antie.widgets.carousel.orientations.Horizontal or antie.widgets.carousel.orientations.Vertical
 38          */
 39         var WidgetStrip = Container.extend(/** @lends antie.widgets.carousel.strips.WidgetStrip.prototype */ {
 40             /**
 41              * @constructor
 42              * @ignore
 43              */
 44             init: function (id, orientation) {
 45                 this._super(id);
 46                 this.addClass(orientation.styleClass());
 47                 this._orientation = orientation;
 48                 this._lengths = [];
 49                 this.addClass('carouselwidgetstrip');
 50             },
 51 
 52             /**
 53              * Adds a widget to the end of the strip
 54              * @param {antie.widgets.Widget} widget The widget to append to the strip
 55              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
 56              * (Height for a vertical strip or width for horizontal.) If provided, this value will be used in
 57              * positioning calculations rather then a calculated value (can be useful when widgets change size)
 58              * Note length only currently working with non-wrapping strips.
 59              */
 60             append: function (widget, length) {
 61                 this._lengths.push(length);
 62                 return this.appendChildWidget(widget);
 63             },
 64 
 65             /**
 66              * Inserts widget within the strip
 67              * @param {Number} index A zero based index to begin insertion at, i.e. 0 prepends.
 68              * @param {antie.widgets.Widget} widget The widget to append to the strip
 69              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
 70              * (Height for a vertical strip or width for horizontal.) If provided, this value will be used in
 71              * positioning calculations rather then a calculated value (can be useful when widgets change size)
 72              * Note length only currently working with non-wrapping strips.
 73              */
 74             insert: function (index, widget, length) {
 75                 this._lengths.splice(index, 0, length);
 76                 return this.insertChildWidget(index, widget);
 77             },
 78 
 79             /**
 80              * Removes a widget from the strip
 81              * @param {antie.widgets.Widget} widget. Widget to remove from the strip
 82              * @param {Boolean} [retainElement=false] whether to keep the widget's output element in the DOM after removing widget
 83              */
 84             remove: function (widget, retainElement) {
 85                 var i, widgets;
 86                 widgets = this.widgets();
 87                 for (i = 0; i !== widgets.length; i += 1) {
 88                     if (widgets[i] === widget) {
 89                         this._lengths.splice(i, 1);
 90                     }
 91                 }
 92                 return this.removeChildWidget(widget, retainElement);
 93             },
 94 
 95             /**
 96              * Removes all widgets from the strip
 97              */
 98             removeAll: function () {
 99                 this._lengths = [];
100                 return this.removeChildWidgets();
101             },
102 
103             /**
104              * Get all the widgets in the strip.
105              * @returns {Array} The widgets currently in the strip
106              */
107             widgets: function () {
108                 return this.getChildWidgets();
109             },
110 
111             /**
112              * Get the distance in pixels to a widget at the supplied index
113              * @param {Number} index An index of a widget currently in the carousel
114              * @returns {Number} length in pixels along primary axis to primary edge of the provided index
115              * i.e. from the left edge of the strip to the left edge of the widget in a horizontal carousel
116              */
117             getLengthToIndex: function (index) {
118                 var suppliedLength;
119 
120                 suppliedLength = this._lengthToIndexUsingSuppliedValues(index);
121                 if (suppliedLength !== null) {
122                     return suppliedLength;
123                 } else {
124                     return this._lengthToIndexByCalculatingUsingElements(index);
125                 }
126             },
127 
128             /**
129              * Manually sets lengths of elements for movement calculations - useful for elements which change size while moving.
130              * @param lengths {Number} | {Array} If provided with a number all lengths will be set equal to this number
131              * If provided with an array, the lengths will be set with the corresponding widgets (so the first number will be used
132              * for the first widget's length, etc..)
133              */
134             setLengths: function (lengths) {
135                 var widgetCount, i;
136                 if (typeof lengths === 'number') {
137                     widgetCount = this.getChildWidgetCount();
138                     for (i = 0; i !== widgetCount; i += 1) {
139                         this._lengths[i] = lengths;
140                     }
141                 } else {
142                     this._lengths = lengths;
143                 }
144 
145             },
146 
147             /**
148              * Indicates whether the strip needs visible indices attaching before it is aligned
149              * @returns {Boolean} true if visible indices required, false if not.
150              */
151             needsVisibleIndices: function () {
152                 return false;
153             },
154 
155             /**
156              * Strip should ensure all widgets indexed in the array are attached to the parent
157              * @param {Array} indexArray
158              */
159             attachIndexedWidgets: function (indexArray) {
160 
161             },
162 
163             _lengthToIndexByCalculatingUsingElements: function (index) {
164                 var elements, widgets, endIndex, i;
165                 elements = [];
166                 widgets = this.getChildWidgets();
167                 endIndex = this._getValidatedIndex(widgets, index + 1);
168                 for (i = 0; i !== endIndex; i += 1) {
169                     elements.push(widgets[i].outputElement);
170                 }
171                 return this._getOffsetToLastElementInArray(elements);
172             },
173 
174             _lengthToIndexUsingSuppliedValues: function (index) {
175                 var length, missingLengths, i;
176                 length = 0;
177                 for (i = 0; i !== Math.max(0, index); i += 1) {
178                     if (this._lengths[i] === undefined) {
179                         missingLengths = true;
180                         break;
181                     } else {
182                         length += this._lengths[i];
183                     }
184                 }
185                 if (missingLengths) {
186                     return null;
187                 } else {
188                     return length;
189                 }
190             },
191 
192             /**
193              * Gets the length in pixels of the widget at the supplied index.
194              * @param {Number} index The index of a widget currently appended to the carousel. Supplied index must be valid (i.e. correspond to a wiget currently in the strip)
195              * @returns {Number} the length in pixels of the widget at the supplied index. Returns the length supplied at append or via setWidgetLength, if neither are specified attempts to calculate and return the length.
196              */
197             lengthOfWidgetAtIndex: function (index) {
198                 var widget;
199                 if (this._lengths[index] !== undefined) {
200                     return this._lengths[index];
201                 }
202                 widget = this.getChildWidgets()[index];
203                 return this._getWidgetLength(widget);
204             },
205 
206             /**
207              * Manually performs any processing required to put the carousel in a valid state after an append/insert
208              */
209             recalculate: function () {
210 
211             },
212 
213             /**
214              * Toggles autocalculation - Calculation in this context is any strip defined processing required after an append/insert to put the carousel in a valid state.
215              * @param {Boolean} on Turns autocalculation on (when true) or off (when false)
216              * Calculation is any strip defined processing required after an append/insert to put the carousel in a valid state
217              * Autocalculation is on by default when a carousel is created.
218              */
219             autoCalculate: function (on) {
220 
221             },
222 
223             _getValidatedIndex: function (array, index) {
224                 var endIndex;
225                 endIndex = index;
226                 if (index < 0) {
227                     endIndex = 0;
228                 }
229                 if (index > array.length) {
230                     endIndex = array.length;
231                 }
232                 return endIndex;
233             },
234 
235             _getOffsetToLastElementInArray: function (elementArray) {
236                 var length, lastIndex;
237                 length = 0;
238                 lastIndex = elementArray.length - 1;
239                 if (lastIndex >= 0) {
240                     length = this._getElementOffset(elementArray[elementArray.length - 1]);
241                 }
242                 return length;
243             },
244 
245             _getElementOffset: function (element) {
246                 var device;
247                 device = this._getDevice();
248                 return device.getElementOffset(element)[this._getEdge()];
249             },
250 
251             _getDevice: function () {
252                 return this.getCurrentApplication().getDevice();
253             },
254 
255             _getDimension: function () {
256                 return this._orientation.dimension();
257             },
258 
259             _getEdge: function () {
260                 return this._orientation.edge();
261             },
262 
263             _getWidgetLength: function (widget) {
264                 return this._getElementLength(widget.outputElement);
265             },
266 
267             _getElementLength: function (element) {
268                 var device;
269                 device = this._getDevice();
270                 return device.getElementSize(element)[this._getDimension()];
271             }
272         });
273 
274         return WidgetStrip;
275     }
276 );
