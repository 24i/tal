 27 require.def('antie/devices/broadcastsource/basetvsource',
 28     [
 29         'antie/class',
 30         'antie/devices/browserdevice'
 31     ],
 32     function (Class, Device) {
 33         'use strict';
 34 
 35         /**
 36          * Contains an abstract implementation of the antie base broadcast TV source.
 37          * @class
 38          * @name antie.devices.broadcastsource.BaseTVSource
 39          * @extends antie.Class
 40          * @protected
 41          */
 42 
 43         /**
 44          * The play states
 45          */
 46         var PLAY_STATE_UNKNOWN = -1;
 47         var BaseTvSource = Class.extend(/** @lends antie.devices.broadcastsource.BaseTVSource.prototype */ {
 48             /**
 49              * @constructor
 50              * @ignore
 51              */
 52             init: function () {
 53                 throw new Error("Abstract class constructor should not be called directly");
 54             },
 55             /**
 56              * Displays the currently tuned channel
 57              */
 58             showCurrentChannel: function () {
 59                 throw new Error("Device broadcast source does not override abstract method showCurrentChannel");
 60             },
 61             /**
 62              * Stops the currently playing programme and turns the screen black
 63              */
 64             stopCurrentChannel: function () {
 65                 throw new Error("Device broadcast source does not override abstract method stopCurrentChannel");
 66             },
 67             /**
 68              * Get the current channel name from broadcast and return as a string.
 69              * @returns A string with the name of the current channel.
 70              */
 71             getCurrentChannelName: function () {
 72                 throw new Error("Device broadcast source does not override abstract method getCurrentChannelName");
 73             },
 74             /**
 75              * Get the current channel information
 76              * @returns A Channel object representing the current channel.
 77              */
 78             getCurrentChannel : function () {
 79                 throw new Error("Device broadcast source does not override abstract method getCurrentChannel");
 80             },
 81             /**
 82              * Get the list of currently available channels.
 83              * @param params.onSuccess function called if the channel list is retrieved, passed a single argument which is a list of Channel objects.
 84              * @param params.onError function called if retrieving the channel list fails.
 85              */
 86             getChannelList : function (params) {
 87                 throw new Error("Device broadcast source does not override abstract method getChannelList");
 88             },
 89             /**
 90              * Sets the size and position of the visible broadcast source
 91              * @param top
 92              * @param left
 93              * @param width
 94              * @param height
 95              */
 96             setPosition : function(top, left, width, height) {
 97                 throw new Error("Device broadcast source does not override abstract method setPosition");
 98             },
 99             getPlayState : function() {
100                 return PLAY_STATE_UNKNOWN;
101             },
102             /**
103              * Requests the device switches a tuner to the channel specified as a DVB triplet
104              * @param params.onid The original network ID.
105              * @param params.tsid DVB or ISDB transport stream ID.
106              * @param params.sid  DVB or ISDB service ID, which must be within the range of 1 to 65535.
107              * @param params.onSuccess function to be called if the tuner was retuned successfully
108              * @param params.onError function to be called if the provided channel was unable to be tuned
109              */
110             setChannel : function(params) {
111                 throw new Error("Device broadcast source does not override abstract method setChannel");
112             },
113             /**
114              * Requests the device switches a tuner to the channel specified by the channel name.
115              * @param params.channelName String representation of the channel name as it appears in the EPG.
116              * @param params.onSuccess function to be called if the tuner was retuned successfully
117              * @param params.onError function to be called if the provided channel was unable to be tuned
118              */
119             setChannelByName : function(params) {
120                 throw new Error("Device broadcast source does not override abstract method setChannelByName");
121             },
122             /**
123              * Reverts the current screen settings and performs any clean up required before
124              * the user exits the application back to standard broadcast.
125              */
126             destroy : function() {
127                 throw new Error("Device broadcast source does not override abstract method destroy");
128             }
129         });
130 
131         Device.prototype.isBroadcastSourceSupported = function() {
132             return true;
133         };
134 
135         return BaseTvSource;
136     }
137 );
