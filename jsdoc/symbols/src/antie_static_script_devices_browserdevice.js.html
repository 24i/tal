 60             /**
 61              * Returns the index of an object in an array. Behaves as native JavaScript 1.5 Array.indexOf().
 62              * @param {Array} arr Array in which to search.
 63              * @param {Object} obj Object to search for.
 64              * @param {Integer} start Index from which to start searching.
 65              * @returns Index of object in array. -1 if object is not found.
 66              */
 67             arrayIndexOf: function(arr, obj, start) {
 68                 if (typeof(Array.prototype.indexOf) === 'function') {
 69                     return arr.indexOf(obj, start);
 70                 } else {
 71                     for (var i = (start || 0); i < arr.length; i++) {
 72                         if (arr[i] == obj) {
 73                             return i;
 74                         }
 75                     }
 76                     return -1;
 77                 }

169             /**
170              * Loads an external style sheet.
171              * @param {String} url The URL of the style sheet.
172              * @param {function(String)} [callback] Callback function when style has loaded/failed
173              * @returns The link element that will load the style sheet.
174              */
175             loadStyleSheet: function(url, callback) {
176                 var self = this;
177                 function supportsCssRules() {
178                     var style = self._createElement("style");
179                     style.type = "text/css";
180                     style.innerHTML = 'body {};';
181                     style.className = "added-by-antie";
182                     document.getElementsByTagName("head")[0].appendChild(style);
183                     try {
184                         style.sheet.cssRules;
185                         return true;
186                     } catch(ex) {
187                     } finally {
188                         style.parentNode.removeChild(style);
189                     }
190                     return false;
191                 }
192 
193                 if (callback && supportsCssRules()) {
194                     var style = this._createElement("style");
195                     style.type = "text/css";
196                     style.innerHTML = "@import url('" + url + "');";
197                     style.className = "added-by-antie";
198                     document.getElementsByTagName("head")[0].appendChild(style);
199 
200                     var interval = window.setInterval(function() {
201                         try {
202                             style.sheet.cssRules;
203                             window.clearInterval(interval);
204                         } catch(ex) {
205                             return;
206                         }
207                         callback(url);
208                     }, 200);
209                 } else {
210                     var link = this._createElement("link");
211                     link.type = "text/css";
212                     link.rel = "stylesheet";
213                     link.href = url;
214                     link.className = "added-by-antie";
215                     document.getElementsByTagName("head")[0].appendChild(link);
216 
217                     // Onload trickery from:
218                     // http://www.backalleycoder.com/2011/03/20/link-tag-css-stylesheet-load-event/
219                     if (callback) {
220                         var img = this._createElement("img");
221                         var done = function() {
222                             img.onerror = function() {};
223                             callback(url);
224                             img.parentNode.removeChild(img);
225                         };
226                         img.onerror = done;
227                         this.getTopLevelElement().appendChild(img);
228                         img.src = url;
229                     }
230                 }
231                 return style;
232             },

332             /**
333              * Adds global key event listener(s) to the user-agent.
334              * This must be added in a way that all key events within the user-agent
335              * cause self._application.bubbleEvent(...) to be called with a {@link KeyEvent}
336              * object with the mapped keyCode.
337              *
338              * @example
339              * document.onkeydown = function(e) {
340              *     self._application.bubbleEvent(new KeyEvent("keydown", keyMap[e.keyCode]));
341              * };
342              */
343             addKeyEventListener: function() {
344                 var self = this;
345                 var _keyMap = this.getKeyMap();
346                 var _pressed = {};
347 
348                 // We need to normalise these events on so that for every key pressed there's
349                 // one keydown event, followed by multiple keypress events whilst the key is
350                 // held down, followed by a single keyup event.
351 
352                 document.onkeydown = function(e) {
353                     e = e || window.event;
354                     var _keyCode = _keyMap[e.keyCode.toString()];
355                     if (_keyCode) {
356                         if (!_pressed[e.keyCode.toString()]) {
357                             self._application.bubbleEvent(new KeyEvent("keydown", _keyCode));
358                             _pressed[e.keyCode.toString()] = true;
359                         } else {
360                             self._application.bubbleEvent(new KeyEvent("keypress", _keyCode));
361                         }
362                         e.preventDefault();
363                     }
364                 };
365                 document.onkeyup = function(e) {
366                     e = e || window.event;
367                     var _keyCode = _keyMap[e.keyCode.toString()];
368                     if (_keyCode) {
369                         delete _pressed[e.keyCode.toString()];
370                         self._application.bubbleEvent(new KeyEvent("keyup", _keyCode));
371                         e.preventDefault();
372                     }
373                 };
374                 document.onkeypress = function(e) {
375                     e = e || window.event;
376                     var _keyCode = _keyMap[e.keyCode.toString()];
377                     if (_keyCode) {
378                         self._application.bubbleEvent(new KeyEvent("keypress", _keyCode));
379                         e.preventDefault();
380                     }
381                 };
382             },

462                 if (!(height = this._textSizeCache[cacheKey])) {

472                     this._measureTextElement.style.width = (typeof maxWidth === 'number') ? maxWidth + "px" : maxWidth;

596             /**
597              * Get an object giving access to the current URL, query string, hash etc.
598              * @returns {Object} Object containing, at a minimum, the properties:
599              * hash, host, href, pathname, protocol, search. These correspond to the properties
600              * in the window.location DOM API.
601              * Use getCurrentAppURL(), getCurrentAppURLParams() and getCurrentRoute() to get
602              * this information in a more generic way.
603              */
604             getWindowLocation: function() {
605                 var windowLocation, copyProps, prop, i, newLocation;
606                 windowLocation = this._windowLocation || window.location; // Allow stubbing for unit testing
607 
608                 // Has the device missed the route off the href? Fix this.
609                 if (windowLocation.hash && windowLocation.hash.length > 1 && windowLocation.href && windowLocation.href.lastIndexOf('#') === -1) {
610                     // Copy properties to new object, as modifying href on the original window.location triggers a navigation.
611                     newLocation = {};
612                     copyProps = ['assign', 'hash', 'host', 'href', 'pathname', 'protocol', 'search'];
613                     for (i = 0; i < copyProps.length; i++) {
614                         prop = copyProps[i];
615                         if (windowLocation.hasOwnProperty(prop)) {
616                             newLocation[prop] = windowLocation[prop];
617                         }
618                     }
619                     newLocation.href = newLocation.href + newLocation.hash;
620                 }
621 
622                 // Use copy of window.location if it was created, otherwise the original.
623                 return newLocation || windowLocation;
624             },
