 57             /**
 58              * Returns the index of an object in an array. Behaves as native JavaScript 1.5 Array.indexOf().
 59              * @param {Array} arr Array in which to search.
 60              * @param {Object} obj Object to search for.
 61              * @param {Integer} start Index from which to start searching.
 62              * @returns Index of object in array. -1 if object is not found.
 63              */
 64             arrayIndexOf: function(arr, obj, start) {
 65                 if (typeof(Array.prototype.indexOf) === 'function') {
 66                     return arr.indexOf(obj, start);
 67                 } else {
 68                     for (var i = (start || 0); i < arr.length; i++) {
 69                         if (arr[i] == obj) {
 70                             return i;
 71                         }
 72                     }
 73                     return -1;
 74                 }

 76             /**
 77              * Creates an element in the device's user-agent.
 78              * @private
 79              * @param {String} tagName The tag name of the element to create.
 80              * @param {String} [id] The id of the element to create.
 81              * @param {Array} [classNames] An array of class names to apply to the element.
 82              */
 83             _createElement: function(tagName, id, classNames) {
 84                 var el = document.createElement(tagName);
 85 
 86                 // don't add auto-generated IDs to the DOM
 87                 if (id && (id.substring(0, 1) != "#")) {
 88                     el.id = id;
 89                 }
 90                 if (classNames && (classNames.length > 0)) {
 91                     el.className = classNames.join(" ");
 92                 }
 93                 return el;
 94             },

166             /**
167              * Loads an external style sheet.
168              * @param {String} url The URL of the style sheet.
169              * @param {function(String)} [callback] Callback function when style has loaded/failed
170              * @returns The link element that will load the style sheet.
171              */
172             loadStyleSheet: function(url, callback) {
173                 var self = this;
174                 function supportsCssRules() {
175                     var style = self._createElement("style");
176                     style.type = "text/css";
177                     style.innerHTML = 'body {};';
178                     style.className = "added-by-antie";
179                     document.getElementsByTagName("head")[0].appendChild(style);
180                     try {
181                         style.sheet.cssRules;
182                         return true;
183                     } catch(ex) {
184                     } finally {
185                         style.parentNode.removeChild(style);
186                     }
187                     return false;
188                 }
189 
190                 if (callback && supportsCssRules()) {
191                     var style = this._createElement("style");
192                     style.type = "text/css";
193                     style.innerHTML = '@import url("' + url + '");';
194                     style.className = "added-by-antie";
195                     document.getElementsByTagName("head")[0].appendChild(style);
196 
197                     var interval = window.setInterval(function() {
198                         try {
199                             style.sheet.cssRules;
200                             window.clearInterval(interval);
201                         } catch(ex) {
202                             return;
203                         }
204                         callback(url);
205                     }, 200);
206                 } else {
207                     var link = this._createElement("link");
208                     link.type = "text/css";
209                     link.rel = "stylesheet";
210                     link.href = url;
211                     link.className = "added-by-antie";
212                     document.getElementsByTagName("head")[0].appendChild(link);
213 
214                     // Onload trickery from:
215                     // http://www.backalleycoder.com/2011/03/20/link-tag-css-stylesheet-load-event/
216                     if (callback) {
217                         var img = this._createElement("img");
218                         function done() {
219                             img.onerror = function() {};
220                             callback(url);
221                             img.parentNode.removeChild(img);
222                         };
223                         img.onerror = done;
224                         this.getTopLevelElement().appendChild(img);
225                         img.src = url;
226                     }
227                 }
228                 return style;
229             },

304             /**
305              * Removes a class from an element (and optionally descendants)
306              * @param {Element} el The element from which to remove the class.
307              * @param {String} className The class to remove.
308              * @param {Boolean} [deep] If true, and this element has the given class, remove the class from it's children recursively.
309              */
310             removeClassFromElement: function(el, className, deep) {
311                 if (new RegExp(" " + className + " ").test(" " + el.className + " ")) {
312                     el.className = trim((" " + el.className + " ").replace(" " + className + " ", " "));
313                 }
314                 if (deep) {
315                     for (var i = 0; i < el.childNodes.length; i++) {
316                         this.removeClassFromElement(el.childNodes[i], className, true);
317                     }
318                 }
319             },

329             /**
330              * Adds global key event listener(s) to the user-agent.
331              * This must be added in a way that all key events within the user-agent
332              * cause self._application.bubbleEvent(...) to be called with a {@link KeyEvent}
333              * object with the mapped keyCode.
334              *
335              * @example
336              * document.onkeydown = function(e) {
337              *     self._application.bubbleEvent(new KeyEvent("keydown", keyMap[e.keyCode]));
338              * };
339              */
340             addKeyEventListener: function() {
341                 var self = this;
342                 var _keyMap = this.getKeyMap();
343                 var _pressed = {};
344 
345                 // We need to normalise these events on so that for every key pressed there's
346                 // one keydown event, followed by multiple keypress events whilst the key is
347                 // held down, followed by a single keyup event.
348 
349                 document.onkeydown = function(e) {
350                     e = e || window.event;
351                     var _keyCode = _keyMap[e.keyCode.toString()];
352                     if (_keyCode) {
353                         if (!_pressed[e.keyCode.toString()]) {
354                             self._application.bubbleEvent(new KeyEvent("keydown", _keyCode));
355                             _pressed[e.keyCode.toString()] = true;
356                         } else {
357                             self._application.bubbleEvent(new KeyEvent("keypress", _keyCode));
358                         }
359                         e.preventDefault();
360                     }
361                 };
362                 document.onkeyup = function(e) {
363                     e = e || window.event;
364                     var _keyCode = _keyMap[e.keyCode.toString()];
365                     if (_keyCode) {
366                         delete _pressed[e.keyCode.toString()];
367                         self._application.bubbleEvent(new KeyEvent("keyup", _keyCode));
368                         e.preventDefault();
369                     }
370                 };
371                 document.onkeypress = function(e) {
372                     e = e || window.event;
373                     var _keyCode = _keyMap[e.keyCode.toString()];
374                     if (_keyCode) {
375                         self._application.bubbleEvent(new KeyEvent("keypress", _keyCode));
376                         e.preventDefault();
377                     }
378                 };
379             },

425             /**

443             /**
444              * Get the height (in pixels) of a given block of text (of a provided set of class names) when constrained to a fixed width.
445              *
446              * @deprecated This function does not always give accurate results. When measuring size, it only takes into account
447              * the classes on the text element being measured. It doesn't consider any CSS styles that may have been passed down
448              * through the DOM.
449              *
450              * @param {String} text The text to measure.
451              * @param {Integer} maxWidth The width the text is constrained to.
452              * @param {Array} classNames An array of class names which define the style of the text.
453              * @returns The height (in pixels) that is required to display this block of text.
454              */
455             getTextHeight: function(text, maxWidth, classNames) {
456                 /// TODO: is there a more efficient way of doing this?
457                 var cacheKey = maxWidth + ":" + classNames.join(" ") + ":" + text;
458                 var height;
459                 if (!(height = this._textSizeCache[cacheKey])) {
460                     if (!this._measureTextElement) {
461                         this._measureTextElement = this.createLabel("measure", null, "fW");
462                         this._measureTextElement.style.display = "block";
463                         this._measureTextElement.style.position = "absolute";
464                         this._measureTextElement.style.top = "-10000px";
465                         this._measureTextElement.style.left = "-10000px";
466                         this.appendChildElement(document.body, this._measureTextElement);
467                     }
468                     this._measureTextElement.className = classNames.join(" ");
469                     this._measureTextElement.style.width = (typeof maxWidth === 'number') ? maxWidth + "px" : maxWidth;
470                     this._measureTextElement.innerHTML = text;
471 
472                     height = this._textSizeCache[cacheKey] = this._measureTextElement.clientHeight;
473                 }
474                 return height;
475             },
476             /**
477              * Returns all direct children of an element which have the provided tagName.
478              * @param {Element} el The element who's children you wish to search.
479              * @param {String} tagName The tag name you are looking for.
480              * @returns An array of elements having the provided tag name.
481              */
482             getChildElementsByTagName: function(el, tagName) {
483                 var children = [];
484                 tagName = tagName.toLowerCase();
485                 for (var i = 0; i < el.childNodes.length; i++) {
486                     if(el.childNodes[i].tagName){
487                         if (el.childNodes[i].tagName.toLowerCase() == tagName) {
488                         children.push(el.childNodes[i]);
489                         }
490                     }
491                 }
492                 return children;
493             },

535              * @param {Element} el The element you wish to scroll.
536              * @param {Integer} [left] The x-coordinate you wish to center on. If null, the x-coordinate will not be altered.
537              * @param {Integer} [top] The y-coordinate you wish to center on. If null, the y-coordinate will not be altered.
538              * @param {Boolean} [skipAnim] By default the scroll will be animated, pass <code>true</code> here to prevent animation.
539              * @param {Function} [onComplete] Callback function to be called when the scroll has been completed.
540              * @returns A handle to any animation started by this movement. {@see #stopAnimation}

549             /**

551              * @returns An object with width and height properties.

556                     w = window.innerWidth;
557                     h = window.innerHeight;
558                 } else {
559                     var d = document.documentElement || document.body;
560                     h = d.clientHeight || d.offsetHeight;
561                     w = d.clientWidth || d.offsetWidth;
562                 }
563                 return {
564                     width: w,
565                     height: h
