 24 require.def('antie/widgets/carousel/keyhandlers/keyhandler',
 25     [
 26         'antie/class'
 27     ],
 28     function (Class) {
 29         "use strict";
 30         /**
 31          * The base KeyHandler class moves alignment of the carousel on LEFT and RIGHT key presses
 32          * when attached to a Carousel with Horizontal orientation, and moves alignment on UP and
 33          * DOWN key presses when attached to a Carousel with Vertical orientation.
 34          * @name antie.widgets.carousel.keyhandlers.KeyHandler
 35          * @class
 36          * @extends antie.widgets.Widget
 37          */
 38         return Class.extend(/** @lends antie.widgets.carousel.keyhandlers.KeyHandler.prototype */{
 39             /**
 40              * @constructor
 41              * @ignore
 42              */
 43             init: function () {
 44                 this._animationOptions = {};
 45             },
 46 
 47             /**
 48              * Adds listeners to the supplied carousel to provided behaviour when navigation keys are pressed
 49              * @param carousel
 50              */
 51             attach: function (carousel) {
 52                 this._carousel = carousel;
 53                 this._addKeyListeners();
 54                 this._addAlignmentListeners();
 55             },
 56 
 57             /**
 58              * Sets default animation options for key handled alignments
 59              * @param options {Object} Animation options object
 60              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
 61              * @param {Number} [options.duration] The duration of the alignment in ms
 62              * @param {String} [options.easing] The alignment easing function
 63              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 64              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
 65              */
 66             setAnimationOptions: function (options) {
 67                 this._animationOptions = options;
 68             },
 69 
 70             _addKeyListeners: function () {
 71                 var previousKey, nextKey, carousel, self;
 72                 self = this;
 73                 carousel = this._carousel;
 74                 previousKey = carousel.orientation().defaultKeys().PREVIOUS;
 75                 nextKey = carousel.orientation().defaultKeys().NEXT;
 76                 carousel.addEventListener('keydown', function (ev) {
 77                     switch (ev.keyCode) {
 78                     case previousKey:
 79                         if (carousel.previousIndex() !== null) {
 80                             carousel.completeAlignment();
 81                             carousel.alignPrevious(self._animationOptions);
 82                             ev.stopPropagation();
 83                         }
 84                         break;
 85                     case nextKey:
 86                         if (carousel.nextIndex() !== null) {
 87                             carousel.completeAlignment();
 88                             carousel.alignNext(self._animationOptions);
 89                             ev.stopPropagation();
 90                         }
 91                         break;
 92                     }
 93                 });
 94             },
 95 
 96             _addAlignmentListeners: function () {
 97 
 98             }
 99         });
100     }
101 );
