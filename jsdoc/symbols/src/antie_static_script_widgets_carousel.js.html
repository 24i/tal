 25 require.def('antie/widgets/carousel',
 26     [
 27         'antie/widgets/carousel/carouselcore',
 28         'antie/widgets/container'
 29     ],
 30     /**
 31      * Ordered list of widgets that can be navigated by moving the list or the selection point
 32      * Provides adapter layer to old widget methods
 33      * @name antie.widgets.Carousel
 34      * @class
 35      * @extends antie.widgets.carousel.CarouselCore
 36      * @param {string} id The id of the carousel, id_CarouselMask will be used as the id for the mask element
 37      * and id_WidgetStrip will be used as the id of the widget strip element
 38      * @param {Object} [orientation=antie.widgets.carousel.CarouselCore.orientations.VERTICAL] the orientation object of
 39      * the carousel. Vertical by default, for horizontal pass in antie.widgets.carousel.CarouselCore.orientations.HORIZONTAL
 40      */
 41     function (CarouselCore, Container) {
 42         "use strict";
 43         var Carousel = CarouselCore.extend(/** @lends antie.widgets.Carousel.prototype */ {
 44             /**
 45              * Appends a child widget to this widget.
 46              * @param {antie.widgets.Widget} widget The child widget to add.
 47              */
 48             appendChildWidget: function (widget) {
 49                 return this.append(widget);
 50             },
 51 
 52             /**
 53              * Inserts a child widget at the specified index.
 54              * @param {Integer} index The index where to insert the child widget.
 55              * @param {antie.widgets.Widget} widget The child widget to add.
 56              */
 57             insertChildWidget: function (index, widget) {
 58                 return this.insert(index, widget);
 59             },
 60 
 61             /**
 62              * Removes a specific child widget from this widget.
 63              * @param {antie.widgets.Widget} widget The child widget to remove.
 64              * @param {Boolean} [retainElement] Pass <code>true</code> to retain the child output element of the given widget
 65              */
 66             removeChildWidget: function (widget, retainElement) {
 67                 return this.remove(widget, retainElement);
 68             },
 69 
 70             /**
 71              * Remove all child widgets from this widget.
 72              */
 73             removeChildWidgets: function () {
 74                 return this.removeAll();
 75             },
 76 
 77             /**
 78              * Attempt to set focus to the given child widget.
 79              *
 80              * Note: You can only set focus to a focusable widget. A focusable widget is one that
 81              * contains an enabled antie.widgets.Button as either a direct or indirect child.
 82              *
 83              * Note: Widgets have 2 independent states: active and focussed. A focussed widget is
 84              * either the Button with focus, or any parent of that Button. An active widget is
 85              * one which is the active child of its parent Container. When the parent widget
 86              * receives focus, focus will be placed on the active child.
 87              *
 88              * Classes 'active' and 'focus' are appended to widgets with these states.
 89              *
 90              * @param {antie.widgets.Widget} widget The child widget to set focus to.
 91              * @returns Boolean true if the child widget was focusable, otherwise boolean false.
 92              */
 93             setActiveChildWidget: function (widget) {
 94                 if (widget === this._mask) {
 95                     return this._super(widget);
 96                 } else {
 97                     return this.setActiveWidget(widget);
 98                 }
 99             },
100 
101             /**
102              * Checks to see if a specific widget is a direct child of this widget.
103              * @param {antie.widgets.Widget} widget The widget to check to see if it is a direct child of this widget.
104              */
105             hasChildWidget: function (id) {
106                 if (id === this._mask.id) {
107                     return this._super(id);
108                 } else {
109                     return this._widgetStrip.hasChildWidget(id);
110                 }
111             },
112 
113             /**
114              * Get the current active widget.
115              * @returns The current active widget
116              */
117             getActiveChildWidget: function () {
118                 return this._widgetStrip.getActiveChildWidget();
119             },
120 
121             /**
122              * Get a child widget from its unique ID.
123              * @param {String} id The id of the child widget to return.
124              * @returns antie.widgets.Widget of the widget with the given ID, otherwise undefined if the child does not exist.
125              */
126             getChildWidget: function (id) {
127                 if (id === this._mask.id) {
128                     return this._mask;
129                 } else {
130                     return this._widgetStrip.getChildWidget(id);
131                 }
132             },
133 
134             /**
135              * Gets the number of direct child widgets.
136              * @returns The number of direct child widgets.
137              */
138             getChildWidgetCount: function () {
139                 return this._widgetStrip.getChildWidgetCount();
140             },
141 
142             /**
143              * Get an array of all this widget's children.
144              * @returns An array of all this widget's children.
145              */
146             getChildWidgets: function () {
147                 return this.items();
148             },
149 
150             /**
151              * Attempts to set focus to the child widget at the given index.
152              * @see #setActiveChildWidget
153              * @param {Integer} index Index of the child widget to set focus to.
154              * @returns Boolean true if the child widget was focusable, otherwise boolean false.
155              */
156             setActiveChildIndex: function (index) {
157                 return this.setActiveIndex(index);
158             },
159 
160             /**
161              * @see #setActiveChildWidget
162              * @returns {Integer} index Index of the child widget that is currently active.
163              */
164             getActiveChildIndex: function () {
165                 return this.getActiveIndex();
166             },
167 
168             /**
169              * Adds a CSS class to the widget strip if not already present.
170              * @param {String} className The class name to add.
171              */
172             addClass: function (className) {
173                 if (this._widgetStrip) {
174                     return this._widgetStrip.addClass(className);
175                 }
176             },
177 
178             /**
179              * Checks to see if the widget strip has a given CSS class.
180              * @param {String} className The class name to check.
181              * @returns Boolean true if the device has the className. Otherwise boolean false.
182              */
183             hasClass: function (className) {
184                 if (this._widgetStrip) {
185                     return this._widgetStrip.hasClass(className);
186                 } else {
187                     return false;
188                 }
189             },
190 
191             /**
192              * Removes a CSS class from the widget strip if present.
193              * @param {String} className The class name to remove.
194              */
195             removeClass: function (className) {
196                 if (this._widgetStrip) {
197                     return this._widgetStrip.removeClass(className);
198                 }
199             },
200 
201             /**
202              * Get an array of class names that this widget strip has.
203              * @returns An array of class names (Strings)
204              */
205             getClasses: function () {
206                 if (this._widgetStrip) {
207                     return this._widgetStrip.getClasses();
208                 } else {
209                     return [];
210                 }
211             },
212 
213             _directAppend: function (widget) {
214                 Container.prototype.appendChildWidget.call(this, widget);
215             }
216         });
217 
218         Carousel.orientations = CarouselCore.orientations;
219         return Carousel;
220 
221     }
222 );
