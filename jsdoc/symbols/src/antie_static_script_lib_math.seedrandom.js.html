  3 // seedrandom.js
  4 // Author: David Bau 12/25/2010
  5 //
  6 // Defines a method Math.seedrandom() that, when called, substitutes
  7 // an explicitly seeded RC4-based algorithm for Math.random().  Also
  8 // supports automatic seeding from local or network sources of entropy.
  9 //
 10 // Usage:
 11 //
 12 //   <script src=http://davidbau.com/encode/seedrandom-min.js></script>
 13 //
 14 //   Math.seedrandom('yipee'); Sets Math.random to a function that is
 15 //                             initialized using the given explicit seed.
 16 //
 17 //   Math.seedrandom();        Sets Math.random to a function that is
 18 //                             seeded using the current time, dom state,
 19 //                             and other accumulated local entropy.
 20 //                             The generated seed string is returned.
 21 //
 22 //   Math.seedrandom('yowza', true);
 23 //                             Seeds using the given explicit seed mixed
 24 //                             together with accumulated entropy.
 25 //
 26 //   <script src="http://bit.ly/srandom-512"></script>
 27 //                             Seeds using physical random bits downloaded
 28 //                             from random.org.
 29 //
 30 //   <script src="https://jsonlib.appspot.com/urandom?callback=Math.seedrandom">
 31 //   </script>                 Seeds using urandom bits from call.jsonlib.com,
 32 //                             which is faster than random.org.
 33 //
 34 // Examples:
 35 //
 36 //   Math.seedrandom("hello");            // Use "hello" as the seed.
 37 //   document.write(Math.random());       // Always 0.5463663768140734
 38 //   document.write(Math.random());       // Always 0.43973793770592234
 39 //   var rng1 = Math.random;              // Remember the current prng.
 40 //
 41 //   var autoseed = Math.seedrandom();    // New prng with an automatic seed.
 42 //   document.write(Math.random());       // Pretty much unpredictable.
 43 //
 44 //   Math.random = rng1;                  // Continue "hello" prng sequence.
 45 //   document.write(Math.random());       // Always 0.554769432473455
 46 //
 47 //   Math.seedrandom(autoseed);           // Restart at the previous seed.
 48 //   document.write(Math.random());       // Repeat the 'unpredictable' value.
 49 //
 50 // Notes:
 51 //
 52 // Each time seedrandom('arg') is called, entropy from the passed seed
 53 // is accumulated in a pool to help generate future seeds for the
 54 // zero-argument form of Math.seedrandom, so entropy can be injected over
 55 // time by calling seedrandom with explicit data repeatedly.
 56 //
 57 // On speed - This javascript implementation of Math.random() is about
 58 // 3-10x slower than the built-in Math.random() because it is not native
 59 // code, but this is typically fast enough anyway.  Seeding is more expensive,
 60 // especially if you use auto-seeding.  Some details (timings on Chrome 4):
 61 //
 62 // Our Math.random()            - avg less than 0.002 milliseconds per call
 63 // seedrandom('explicit')       - avg less than 0.5 milliseconds per call
 64 // seedrandom('explicit', true) - avg less than 2 milliseconds per call
 65 // seedrandom()                 - avg about 38 milliseconds per call

106 //
107 // seedrandom()
108 // This is the seedrandom function described above.
109 //
110 math['seedrandom'] = function seedrandom(seed, use_entropy) {
111   var key = [];
112   var arc4;
113 
114   // Flatten the seed string or build one from local entropy if needed.
115   seed = mixkey(flatten(
116     use_entropy ? [seed, pool] :
117     arguments.length ? seed :
118     [new Date().getTime(), pool, window], 3), key);
119 
120   // Use the seed to initialize an ARC4 generator.
121   arc4 = new ARC4(key);
122 
123   // Mix the randomness into accumulated entropy.
124   mixkey(arc4.S, pool);
125 
126   // Override Math.random
127 
128   // This function returns a random double in [0, 1) that contains
129   // randomness in every bit of the mantissa of the IEEE 754 value.
130 
131   math['random'] = function random() {  // Closure to return a random double:
132     var n = arc4.g(chunks);             // Start with a numerator n < 2 ^ 48
133     var d = startdenom;                 //   and denominator d = 2 ^ 48.
134     var x = 0;                          //   and no 'extra last byte'.
135     while (n < significance) {          // Fill up all significant digits by
136       n = (n + x) * width;              //   shifting numerator and
137       d *= width;                       //   denominator and generating a
138       x = arc4.g(1);                    //   new least-significant-byte.
139     }
140     while (n >= overflow) {             // To avoid rounding up, before adding
141       n /= 2;                           //   last byte, shift everything
142       d /= 2;                           //   right using integer math until
143       x >>>= 1;                         //   we have exactly the desired bits.
144     }
145     return (n + x) / d;                 // Form the number within [0, 1).
146   };
147 
148   // Return the seed that was used
149   return seed;
150 };
151 
152 //
153 // ARC4
154 //
155 // An ARC4 implementation.  The constructor takes a key in the form of
156 // an array of at most (width) integers that should be 0 <= x < (width).
157 //
158 // The g(count) method returns a pseudorandom integer that concatenates
159 // the next (count) outputs from ARC4.  Its return value is a number x
160 // that is in the range 0 <= x < (width ^ count).
161 //
162 /** @constructor */
163 function ARC4(key) {
164   var t, u, me = this, keylen = key.length;
165   var i = 0, j = me.i = me.j = me.m = 0;
166   me.S = [];
167   me.c = [];
168 
169   // The empty key [] is treated as [0].
170   if (!keylen) { key = [keylen++]; }
171 
172   // Set up S using the standard key scheduling algorithm.
173   while (i < width) { me.S[i] = i++; }
174   for (i = 0; i < width; i++) {
175     t = me.S[i];
176     j = lowbits(j + t + key[i % keylen]);
177     u = me.S[j];
178     me.S[i] = u;
179     me.S[j] = t;
180   }
181 
182   // The "g" method returns the next (count) outputs as one number.
183   me.g = function getnext(count) {
184     var s = me.S;
185     var i = lowbits(me.i + 1); var t = s[i];
186     var j = lowbits(me.j + t); var u = s[j];
187     s[i] = u;
188     s[j] = t;
189     var r = s[lowbits(t + u)];
190     while (--count) {
191       i = lowbits(i + 1); t = s[i];
192       j = lowbits(j + t); u = s[j];
193       s[i] = u;
194       s[j] = t;
195       r = r * width + s[lowbits(t + u)];
196     }
197     me.i = i;
198     me.j = j;
199     return r;
200   };
201   // For robust unpredictability discard an initial batch of values.
202   // See http://www.rsa.com/rsalabs/node.asp?id=2009
203   me.g(width);
204 }

206 //
207 // flatten()
208 // Converts an object tree to nested arrays of strings.
209 //
210 /** @param {Object=} result 
211   * @param {string=} prop */
212 function flatten(obj, depth, result, prop) {
213   result = [];
214   if (depth && typeof(obj) == 'object') {
215     for (prop in obj) {
216       if (prop.indexOf('S') < 5) {    // Avoid FF3 bug (local/sessionStorage)
217         try { result.push(flatten(obj[prop], depth - 1)); } catch (e) {}
218       }
219     }
220   }
221   return result.length ? result : '' + obj;
222 }

224 //
225 // mixkey()
226 // Mixes a string seed into a key that is an array of integers, and
227 // returns a shortened string seed that is equivalent to the result key.
228 //
229 /** @param {number=} smear 
230   * @param {number=} j */
231 function mixkey(seed, key, smear, j) {
232   seed += '';                         // Ensure the seed is a string
233   smear = 0;
234   for (j = 0; j < seed.length; j++) {
235     key[lowbits(j)] =
236       lowbits((smear ^= key[lowbits(j)] * 19) + seed.charCodeAt(j));
237   }
238   seed = '';
239   for (j in key) { seed += String.fromCharCode(key[j]); }
240   return seed;
241 }

243 //
244 // lowbits()
245 // A quick "n mod width" for width a power of 2.
246 //
247 function lowbits(n) { return n & (width - 1); }
