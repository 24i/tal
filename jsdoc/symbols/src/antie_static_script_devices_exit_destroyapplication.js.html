 31 require.def(
 32 	'antie/devices/exit/destroyapplication',
 33 	['antie/devices/browserdevice'],
 34 	function(Device) {
 35         'use strict';
 36 
 37 		// MIME type for the OIPF application manager, as defined by the spec (v1.2)
 38 		// Spec: http://www.oipf.tv/specifications/root/volume-5-declarative-application-environment-r112/download
 39     	var APPMANAGER_MIME_TYPE = 'application/oipfApplicationManager';
 40 		
 41         /**
 42 	     * Exits the application by invoking Application.destroyApplication()
 43 	     * after getting the OIPF application manager. In theory, according to
 44 	     * the spec, this has the same effect as calling window.close().
 45 	    */
 46         Device.prototype.exit = function() {
 47         	var factory = window.oipfObjectFactory;
 48 
 49         	// Check that the OIPF factory is implemented and supports the application manager
 50         	if(factory && factory.isObjectSupported(APPMANAGER_MIME_TYPE)) {
 51         		var appManager = factory.createApplicationManagerObject();
 52 
 53         		if(appManager) {
 54         			// Ask the application manager for an Application object for this app
 55         			var ownerApp = appManager.getOwnerApplication(window.document);
 56 
 57         			if(ownerApp) {
 58         				// Destroy this app and finish
 59         				ownerApp.destroyApplication();
 60         			}
 61         		}
 62             }
 63         }
 64     }
 65 );
