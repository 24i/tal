 24 require.def('antie/widgets/carousel/aligners/aligner',
 25     [
 26         'antie/class',
 27         "antie/events/beforealignevent",
 28         "antie/events/afteralignevent",
 29         "antie/widgets/carousel/aligners/alignmentqueue"
 30     ],
 31     function (Class, BeforeAlignEvent, AfterAlignEvent, AlignmentQueue) {
 32         "use strict";
 33         /**
 34          * Converts simple index based alignment instructions to combinations of
 35          * one or more pixel based alignments to be performed on the mask
 36          * @name antie.widgets.carousel.aligners.Aligner
 37          * @class
 38          * @extends antie.Class
 39          * @param {Object} mask The carousel's mask object
 40          */
 41         var Aligner;
 42         Aligner = Class.extend({
 43             init: function (mask) {
 44                 this._mask = mask;
 45                 this._queue = new AlignmentQueue(this._mask);
 46                 this._lastAlignIndex = null;
 47             },
 48 
 49             /**
 50              * Aligns the mask and widget strip to the next enabled widget after that currently aligned.
 51              * If no alignment has been performed previously it will align to the next enabled widget after that at index 0
 52              * If a wrapping strip and navigator are used the alignment will wrap to the start after the last widget is reached.
 53              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
 54              * @param {Object} navigator The carousel's current navigator
 55              * @param {Object} [options] An animation options object
 56              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
 57              * @param {Number} [options.duration] The duration of the alignment in ms
 58              * @param {String} [options.easing] The alignment easing function
 59              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 60              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
 61              */
 62             alignNext: function (navigator, options) {
 63                 this._align(navigator, Aligner.directions.FORWARD, options);
 64             },
 65 
 66             /**
 67              * Aligns the mask and widget strip to the next enabled widget before that currently aligned.
 68              * If no alignment has been performed previously it will align to the next enabled widget before that at index 0
 69              * If a wrapping strip and navigator are used the alignment will wrap to the end after the first widget is reached.
 70              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
 71              * @param {Object} navigator The carousel's current navigator
 72              * @param {Object} [options] An animation options object
 73              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
 74              * @param {Number} [options.duration] The duration of the alignment in ms
 75              * @param {String} [options.easing] The alignment easing function
 76              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 77              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
 78              */
 79             alignPrevious: function (navigator, options) {
 80                 this._align(navigator, Aligner.directions.BACKWARD, options);
 81             },
 82 
 83             /**
 84              * Aligns the mask and widget strip to the widget at the specified index
 85              * Will always move forward if the index is after that currently aligned and backwards if index is before
 86              * that currently aligned.
 87              * If an alignment is in progress, the new alignment will be queued to start after the current alignment completes.
 88              * @param {Number} index The index of the widget to align on.
 89              * @param {Object} [options] An animation options object
 90              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
 91              * @param {Number} [options.duration] The duration of the alignment in ms
 92              * @param {String} [options.easing] The alignment easing function
 93              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 94              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
 95              */
 96             alignToIndex: function (index, options) {
 97                 this._informMaskBeforeAlign(index);
 98                 this._moveNormally(index, options);
 99             },
100 
101             /**
102              * @returns {Number} The last index the carousel was asked to align to, or null if no alignments have completed.
103              */
104             indexOfLastAlignRequest: function () {
105                 return this._lastAlignIndex;
106             },
107 
108             /**
109              * Instantly completes any in-flight alignment animations, firing any callbacks that were provided.
110              * If several alignments have been queued, all will complete in order.
111              */
112             complete: function () {
113                 this._queue.complete();
114             },
115 
116             _align: function (navigator, direction, options) {
117                 var startIndex, targetIndex;
118 
119                 startIndex = this.indexOfLastAlignRequest();
120                 targetIndex = this._subsequentIndexInDirection(navigator, direction);
121 
122                 if (targetIndex !== null) {
123                     this._informMaskBeforeAlign(targetIndex);
124                     if (this._isWrap(startIndex, targetIndex, direction)) {
125                         this._wrap(startIndex, targetIndex, navigator, direction, options);
126                     } else {
127                         this._moveNormally(targetIndex, options);
128                     }
129                 }
130             },
131 
132             _subsequentIndexInDirection: function (navigator, direction) {
133                 var startPoint, lastAligned;
134                 lastAligned = this.indexOfLastAlignRequest();
135                 startPoint = (lastAligned === null) ? 0 : lastAligned;
136                 if (direction === Aligner.directions.FORWARD) {
137                     return navigator.indexAfter(startPoint);
138                 } else {
139                     return navigator.indexBefore(startPoint);
140                 }
141             },
142 
143             _isWrap: function (startIndex, targetIndex, direction) {
144                 if (direction === Aligner.directions.FORWARD && startIndex > targetIndex) {
145                     return true;
146                 }
147                 if (direction === Aligner.directions.BACKWARD && startIndex < targetIndex) {
148                     return true;
149                 }
150                 return false;
151             },
152 
153             _informMaskBeforeAlign: function (index) {
154                 this._mask.beforeAlignTo(this._lastAlignIndex, index);
155                 this._lastAlignIndex = index;
156             },
157 
158             _informMaskAfterAlign: function (index) {
159                 this._mask.afterAlignTo(index);
160             },
161 
162             _wrap: function (fromIndex, toIndex, navigator, direction, options) {
163                 if (this._fromIndexActive(fromIndex, navigator)) {
164                     this._visibleActiveItemWrap(fromIndex, toIndex, navigator, direction, options);
165                 } else {
166                     this._invisibleActiveItemWrap(fromIndex, toIndex, navigator, direction, options);
167                 }
168             },
169 
170             _fromIndexActive: function (fromIndex, navigator) {
171                 var activeIndex;
172                 activeIndex = navigator.currentIndex();
173                 return fromIndex === activeIndex;
174             },
175 
176             _invisibleActiveItemWrap: function (fromIndex, toIndex, navigator, direction, options) {
177                 var index = this._firstIndexToAlignForInvisibleActiveItemWrap(fromIndex, navigator, direction);
178                 this._queue.add(index, {skipAnim: true});
179                 this._queueFinalAlign(toIndex, options);
180                 this._queue.start();
181             },
182 
183             _firstIndexToAlignForInvisibleActiveItemWrap: function (fromIndex, navigator, direction) {
184                 var widgetCount;
185                 widgetCount = navigator.indexCount();
186                 if (direction === Aligner.directions.FORWARD) {
187                     return fromIndex - navigator.indexCount();
188                 } else {
189                     return widgetCount + fromIndex;
190                 }
191             },
192 
193             _visibleActiveItemWrap: function (fromIndex, toIndex, navigator, direction, options) {
194                 var firstAlignIndex;
195                 firstAlignIndex = this._firstIndexToAlignForVisibleActiveItemWrap(toIndex, navigator, direction);
196                 this._queue.add(firstAlignIndex, options);
197                 this._queueFinalAlign(toIndex, { skipAnim: true });
198                 this._queue.start();
199             },
200 
201             _firstIndexToAlignForVisibleActiveItemWrap: function (toIndex, navigator, direction) {
202                 if (direction === Aligner.directions.FORWARD) {
203                     return navigator.indexCount() + toIndex;
204                 } else {
205                     return -navigator.indexCount() + toIndex;
206                 }
207             },
208 
209             _moveNormally: function (toIndex, options) {
210                 this._queueFinalAlign(toIndex, options);
211                 this._queue.start();
212             },
213 
214             _queueFinalAlign: function (toIndex,  options) {
215                 var optionsWithCallback, self;
216 
217                 function OptionsClone() {}
218 
219                 function unwrappedComplete() {
220                     self._informMaskAfterAlign(toIndex);
221                 }
222 
223                 function wrappedComplete() {
224                     options.onComplete();
225                     self._informMaskAfterAlign(toIndex);
226                 }
227 
228                 self = this;
229                 OptionsClone.prototype = options;
230                 optionsWithCallback = new OptionsClone();
231                 if (options && options.onComplete) {
232                     optionsWithCallback.onComplete = wrappedComplete;
233                 } else {
234                     optionsWithCallback.onComplete = unwrappedComplete;
235                 }
236 
237                 this._queue.add(toIndex, optionsWithCallback);
238             }
239         });
240 
241         Aligner.directions = { FORWARD: 0, BACKWARD: 1 };
242 
243         return Aligner;
244     }
245 );
