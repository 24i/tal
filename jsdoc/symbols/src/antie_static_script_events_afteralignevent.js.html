 26 require.def('antie/events/afteralignevent',
 27     ['antie/events/event'],
 28     function (Event) {
 29         "use strict";
 30         /**
 31          * Class of events raised when a Mask is about to change alignment of its widget strip.
 32          * @name antie.events.AfterAlignEvent
 33          * @class
 34          * @extends antie.events.Event
 35          * @param {antie.widgets.carousel.Mask} target The mask that is about to align the strip
 36          * @param {Integer} to the index into the strip of the index alignment is moving to
 37          * @param {Integer} from the index into the strip of the index alignment is moving from
 38          */
 39         return Event.extend(/** @lends antie.events.AfterAlignEvent.prototype */ {
 40             /**
 41              * @constructor
 42              * @ignore
 43              */
 44             init: function (target, alignedIndex) {
 45                 this.target = target;
 46                 this.alignedIndex = alignedIndex;
 47                 this._super("afteralign");
 48             }
 49         });
 50     }
 51 );
