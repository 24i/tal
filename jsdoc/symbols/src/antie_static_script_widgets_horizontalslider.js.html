 27 require.def('antie/widgets/horizontalslider',
 28 	[
 29 	 	'antie/widgets/button',
 30 		'antie/events/focusevent',
 31 		'antie/events/blurevent',
 32 		'antie/events/keyevent',
 33 		'antie/events/selectevent',
 34 		'antie/events/sliderchangeevent'
 35 	],
 36 	function(Button, FocusEvent, BlurEvent, KeyEvent, SelectEvent, SliderChangeEvent) {
 37 		'use strict';
 38 
 39 		/**
 40 		 * The horizontal slider widget provides a method for users to pecificy a value between 0 and 1 using VK_LEFT and VK_RIGHT
 41 		 * @name antie.widgets.HorizontalSlider
 42 		 * @class
 43 		 * @extends antie.widgets.Container
 44 		 * @requires antie.events.FocusEvent
 45 		 * @requires antie.events.BlurEvent
 46 		 * @requires antie.events.KeyEvent
 47 		 * @requires antie.events.SelectEvent
 48 		 * @requires antie.events.SliderChangeEvent
 49 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 50 		 * @param {double} [initialValue] Initial value for the slider.
 51 		 * @param {double} [smallIncrement] Amount to increase/decrease the value by when VK_LEFT or VK_RIGHT is pressed.
 52 		 * @param {double} [largeIncrement] Amount to increase/decrease the value by when VK_LEFT or VK_RIGHT is held down.
 53 		 * @param {double} [largeIncrementAfter] Number of smallIncrements to perform until switching to largeIncrement when key is held down.
 54 		 */
 55 		return Button.extend(/** @lends antie.widgets.HorizontalSlider.prototype */ {
 56 			/**
 57 			 * @constructor
 58 			 * @ignore
 59 			 */
 60 			init: function(id, initialValue, smallIncrement, largeIncrement, largeIncrementAfter) {
 61 				this._value = initialValue ? initialValue : 0;
 62 				this._smallIncrement = smallIncrement ? smallIncrement : 0.01;
 63 				this._largeIncrement = largeIncrement ? largeIncrement : 0.05;
 64 				this._largeIncrementAfter = largeIncrementAfter ? largeIncrementAfter : 10;
 65 				this._lastLeft = -1;
 66 				this._keyPressTimer = null;
 67 
 68 				this._super(id);
 69 				this.addClass('horizontalslider');
 70 
 71 				this._currentIncrementCount = 0;
 72 				this._currentChanged = false;
 73 
 74 				var self = this;
 75 				this.addEventListener('keydown', function(e) {
 76 					self._onKeyDown(e);
 77 				});
 78 				this.addEventListener('keypress', function(e) {
 79 					self._onKeyPress(e);
 80 				});
 81 				this.addEventListener('keyup', function(e) {
 82 					self._onKeyUp(e);
 83 				});
 84 			},
 85 			/**
 86 			 * Renders the widget and any child widgets to device-specific output.
 87 			 * @param {antie.devices.Device} device The device to render to.
 88 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 89 			 */
 90 			render: function(device) {
 91 				this.outputElement = device.createContainer(this.id, this.getClasses());
 92 				this._button = device.createButton(this.id+'_slider');
 93 				this._left = device.createContainer(this.id+'_left');
 94 				device.addClassToElement(this._button, 'horizontalsliderhandle');
 95 				device.addClassToElement(this._left, 'horizontalsliderleft');
 96 				device.appendChildElement(this.outputElement, this._left);
 97 				device.appendChildElement(this.outputElement, this._button);
 98 				this._moveButton();
 99 				return this.outputElement;
100 			},
101 			/**
102 			 * Moves the inner button to show the current value.
103 			 * @private
104 			 */
105 			_moveButton: function() {
106 				if(this.outputElement) {
107 					var device = this.getCurrentApplication().getDevice();
108 					var elsize = device.getElementSize(this.outputElement);
109 					var left = Math.floor(this._value * elsize.width);
110 					if(left != this._lastLeft) {
111 						this._lastLeft = left;
112 
113 						if(this._value == 0) {
114 							this.addClass('start');
115 							this.removeClass('end');
116 						} else if(this._value == 1) {
117 							this.removeClass('start');
118 							this.addClass('end');
119 						} else {
120 							this.removeClass('start');
121 							this.removeClass('end');
122 						}
123 						device.moveElementTo({
124 							el: this._button, 
125 							to: {
126 								left: left
127 							},
128 							skipAnim: true
129 						});
130 						device.setElementSize(this._left, {width: left});
131 					}
132 				}
133 			},
134 			/**
135 			 * KeyDown handler. Allows quick presses of the left/right key to change the value.
136 			 * @private
137 			 */
138 			_onKeyDown: function(evt) {
139 				var self = this;
140 
141 				if(evt.keyCode != KeyEvent.VK_LEFT && evt.keyCode != KeyEvent.VK_RIGHT) {
142 					return;
143 				}
144 
145 				var changed = false;
146 				this._currentIncrementCount++;
147 				var inc = (this._currentIncrementCount > this._largeIncrementAfter) ? this._largeIncrement : this._smallIncrement;
148 
149 				if(evt.keyCode == KeyEvent.VK_LEFT && this._value > 0) {
150 					this._value -= inc;
151 					if(this._value < 0) {
152 						this._value = 0;
153 					}
154 					changed = true;
155 				} else if(evt.keyCode == KeyEvent.VK_RIGHT && this._value < 1) {
156 					this._value += inc;
157 					if(this._value > 1) {
158 						this._value = 1;
159 					}
160 					changed = true;
161 				}
162 				if(changed) {
163 					this._moveButton();
164 					this._currentChanged = true;
165 					this.fireEvent(new SliderChangeEvent("sliderchange", this, this._value));
166 				}
167 			},
168 
169 			/**
170 			 * KeyPress handler. Changes the value by either the small or large increment depending on how long the
171 			 * key is held down for. Fires the sliderchange event.
172 			 * @private
173 			 */
174 			_onKeyPress: function(evt) {
175 				if(evt.keyCode != KeyEvent.VK_LEFT && evt.keyCode != KeyEvent.VK_RIGHT) {
176 					return;
177 				}
178 
179 				// If we're tracking individual key presses, stop it now the key is being held down
180 				if(this._keyPressTimer) {
181 					window.clearTimeout(this._keyPressTimer);
182 					this._keyPressTimer = null;
183 				}
184 
185 				var changed = false;
186 				this._currentIncrementCount++;
187 				var inc = (this._currentIncrementCount > this._largeIncrementAfter) ? this._largeIncrement : this._smallIncrement;
188 
189 				if(evt.keyCode == KeyEvent.VK_LEFT && this._value > 0) {
190 					this._value -= inc;
191 					if(this._value < 0) {
192 						this._value = 0;
193 					}
194 					changed = true;
195 				} else if(evt.keyCode == KeyEvent.VK_RIGHT && this._value < 1) {
196 					this._value += inc;
197 					if(this._value > 1) {
198 						this._value = 1;
199 					}
200 					changed = true;
201 				}
202 				if(changed) {
203 					this._moveButton();
204 					this._currentChanged = true;
205 					this.fireEvent(new SliderChangeEvent("sliderchange", this, this._value));
206 				}
207 			},
208 			/**
209 			 * KeyUp handler. Resets the increment size and fires the sliderchangeend event.
210 			 * @private
211 			 */
212 			_onKeyUp: function(evt) {
213 				if(evt.keyCode != KeyEvent.VK_LEFT && evt.keyCode != KeyEvent.VK_RIGHT) {
214 					return;
215 				}
216 				// If the key is pressed again within 0.8s, keep changing the position
217 				if(this._keyPressTimer) {
218 					window.clearTimeout(this._keyPressTimer);
219 				}
220 				var self = this;
221 				this._keyPressTimer = window.setTimeout(function() {
222 						self._keyPressTimer = null;
223 						if(self._currentChanged) {
224 							self.fireEvent(new SliderChangeEvent("sliderchangeend", self, self._value));
225 							self._currentChanged = false;
226 							self._currentIncrementCount = 0;
227 							self._currentIncrement = self._smallIncrement;
228 						}
229 					}, 800);
230 			},
231 			/**
232 			 * Returns the current value shown by the progress indicator.
233 			 */
234 			getValue: function() {
235 				return this._value;
236 			},
237 			/**
238 			 * Sets the current value to be shown by the progress indicator.
239 			 * @param {Float} val The value to show (between 0.0 and 1.0 inclusive).
240 			 */
241 			setValue: function(val) {
242 				this._value = val;
243 				this._moveButton();
244 			},
245 			
246 			/**
247 			 * Sets the small increment
248 			 * @param {Float} val The increment value (between 0.0 and 1.0 inclusive).
249 			 */
250 			setSmallIncrement: function(val) {
251 				this._smallIncrement = val;
252 			}
253 		});
254 	}
255 );
