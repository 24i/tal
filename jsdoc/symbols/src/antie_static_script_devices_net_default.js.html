 27 require.def(
 28 	'antie/devices/net/default',
 29 	['antie/devices/browserdevice'],
 30 	function(Device) {
 31 		/**
 32 		 * Loads an external script that calls a specified callback function.
 33 		 * Used for loading data via JSON-P.
 34 		 * @param {String} url The URL of the script.
 35 		 * @param {RegExp} callbackFunctionRegExp Regular expression to replace matches with callback function name.
 36 		 * @param {Object} callbacks Object containing onSuccess and onLoad callback functions.
 37 		 * @param {Integer} timeout Timeout in milliseconds.
 38 		 * @param {String} [callbackSuffix] Suffix to append to end of callback function name.
 39 		 * @returns The script element that will load the script.
 40 		 */
 41 		Device.prototype.loadScript = function(url, callbackFunctionRegExp, callbacks, timeout, callbackSuffix) {
 42 			var self = this;
 43 			var script = null;
 44 			var funcName = "_antie_callback_" + (callbackSuffix || ((new Date() * 1) + "_" + Math.floor(Math.random() * 10000000)));
 45 			
 46 			if (window[funcName])
 47 				throw "A request with the name " + funcName + " is already in flight";
 48 			
 49 			var timeoutHandle = window.setTimeout(function() {
 50 				if (window[funcName]) {
 51 					if (script) {
 52 						self.removeElement(script);
 53 					}
 54 					delete window[funcName];
 55 					if (callbacks && callbacks.onError) {
 56 						callbacks.onError('timeout');
 57 					}
 58 				}
 59 			}, timeout || 5000);
 60 
 61 			window[funcName] = function(obj) {
 62 				if (timeout) {
 63 					window.clearTimeout(timeoutHandle);
 64 				}
 65 				if (callbacks && callbacks.onSuccess) {
 66 					callbacks.onSuccess(obj);
 67 				}
 68 				self.removeElement(script);
 69 				delete window[funcName];
 70 			};
 71 
 72 			script = this._createElement("script");
 73 			script.src = url.replace(callbackFunctionRegExp, funcName);
 74 			var head = document.getElementsByTagName("head")[0];
 75 			head.appendChild(script);
 76 			return script;
 77 		},
 78 		
 79 		/**
 80 		 * Loads a resource from a URL protected by device authentication.
 81 		 * @param {String} url The URL to load.
 82 		 * @param {Object} opts Object containing onLoad and onError callback functions.
 83 		 * @returns The request object used to load the resource.
 84 		 */
 85 		Device.prototype.loadAuthenticatedURL = function(url, opts) {
 86 			// Simple implementation - assuming XHR in browser can perform client-authenticated SSL requests
 87 			return this.loadURL(url, opts);
 88 		},
 89 		/**
 90 		 * Loads a resource from a URL.
 91 		 * @param {String} url The URL to load.
 92 		 * @param {Object} opts Object containing onLoad and onError callback functions.
 93 		 * @returns The request object used to load the resource.
 94 		 */
 95 		Device.prototype.loadURL = function(url, opts) {
 96 			var xhr = new XMLHttpRequest();
 97 			xhr.open(opts.method || 'GET', url, true);
 98 			xhr.onreadystatechange = function() {
 99 				if (this.readyState == 4) {
100 					this.onreadystatechange = null;
101 					if (this.status == 200) {
102 						if (opts.onLoad) {
103 							opts.onLoad(this.responseText);
104 						}
105 					} else {
106 						if (opts.onError) {
107 							opts.onError(this.responseText, this.status);
108 						}
109 					}
110 				}
111 			};
112 			if (opts && opts.headers) {
113 				for (var header in opts.headers) {
114 					xhr.setRequestHeader(header, opts.headers[header]);
115 				}
116 			}
117 
118 			try {
119 				xhr.send(opts.data || null);
120 			} catch(ex) {
121 				if (opts.onError) {
122 					opts.onError(ex);
123 				}
124 			}
125 			return xhr;
126 		},
127 		/**
128 		 * Performs a POST HTTP request to a URL on a different host/domain.
129 		 * @param {String} url The URL to post to.
130 		 * @param {Object} data Associative array of fields/values to post.
131 		 * @param {Object} opts Object containing onLoad and onError callback functions.
132 		 */
133 		Device.prototype.crossDomainPost = function(url, data, opts) {
134 			var iframe, form;
135             var postRequestHasBeenSent = false;
136             var blankPageToLoad = opts.blankUrl || "blank.html";
137             var timeoutHandle;
138 
139 			function iframeLoadTimeoutCallback() {
140 				iframe.onload = null;
141 				if(opts.onError) opts.onError("timeout");
142 			}
143 
144 			function iframeLoadedCallback() {
145 				var urlLoadedIntoInvisibleIFrame, errorGettingIFrameLocation;
146 				try {
147 					urlLoadedIntoInvisibleIFrame = iframe.contentWindow.location.href;
148 				} catch (exception) {
149                     errorGettingIFrameLocation = exception;
150 				}
151 
152 				if(errorGettingIFrameLocation || !urlLoadedIntoInvisibleIFrame) {
153 					// we didn't load the page - give the browser a second chance to load the iframe
154                     setTimeout(function() { iframe.src = blankPageToLoad+"#2"; }, 500);
155                     return;
156 				}
157 
158                 if (postRequestHasBeenSent === false) {
159                     postRequestHasBeenSent = true;
160 
161                     createForm();
162                     for(var name in data) {
163                         createField(name, data[name]);
164                     }
165                     form.submit();
166                 }
167                 else {
168                     if (timeoutHandle){
169                         clearTimeout(timeoutHandle);
170                     }
171 
172                     iframe.onload = null;
173                     try {
174                         var responseData = iframe.contentWindow.name;
175                         iframe.parentNode.removeChild(iframe);
176                         if (opts.onLoad) {
177                             opts.onLoad(responseData);
178                         }
179                     } catch (exception) {
180                         if(opts.onError) opts.onError(exception);
181                     }
182                 }
183             }
184 
185 			function createForm() {
186 				var doc = iframe.contentWindow.document;
187 				form = doc.createElement('form');
188 				form.method = 'POST';
189 				form.action = url;
190 				doc.body.appendChild(form);
191 			}
192 
193 			function createField(name, value) {
194 				var input = document.createElement('input');
195 				input.type = "hidden";
196 				input.name = name;
197 				input.value = value;
198 				form.appendChild(input);
199 			}
200 
201 			function createIframe() {
202 				iframe = document.createElement('iframe');
203 				iframe.style.width = "0";
204 				iframe.style.height = "0";
205 				iframe.src = blankPageToLoad+"#1";
206 				iframe.onload = iframeLoadedCallback;
207 				document.body.appendChild(iframe);
208 			}
209 
210 			timeoutHandle = setTimeout(iframeLoadTimeoutCallback, (opts.timeout || 10) * 1000); /* 10 second default */
211 			createIframe();
212 		},

214         /**
215          * Performs a cross domain GET for a decoded JSON object utilising CORS if supported by
216          * the device, falling back to a JSON-P call otherwise.
217          * @param {String} url The URL to load. A callback GET parameter will be appended if JSON-P is used.
218          * @param {Object} callbacks Object containing onSuccess and onError callbacks. onSuccess will be called
219          * with the decoded JSON object if the call is successful.
220          * @param {Object} [options] Options for the JSON-P fallback behaviour. All optional with sensible defaults.
221          * @param {Number} [options.timeout=5000] Timeout for the JSON-P call in ms. Default: 5000.
222          * @param {String} [options.id] Used in the callback function name for the JSON-P call. Default: a random string.
223          * @param {String} [options.callbackKey=callback] Key to use in query string when passing callback function name
224          * for JSON-P call. Default: callback
225          */
226         Device.prototype.executeCrossDomainGet = function(url, callbacks, options) {
227            var self, callbackKey, callbackQuery;
228            self = this;
229            options = options || {};
230            if (configSupportsCORS(this.getConfig())) {
231                this.loadURL(url, {
232                    onLoad: function(jsonResponse) {
233                        callbacks.onSuccess(self.decodeJson(jsonResponse));
234                    },
235                    onError: callbacks.onError
236                });
237            } else {
238                callbackKey = options.callbackKey || 'callback';
239                callbackQuery = '?' + callbackKey + '=%callback%';
240                if(url.indexOf('?') === -1) {
241                    url = url + callbackQuery;
242                } else {
243                    url = url.replace('?', callbackQuery + '&');
244                }
245                this.loadScript(url, /%callback%/, callbacks, options.timeout, options.id);
246            }
247         },
248 
249         /**
250          * Performs a cross domain POST HTTP using CORS or the content delivered as a single form field value depending on device capability
251          * @param {String} url The URL to post to.
252          * @param {Object} data JavaScript object to be JSON encoded and delivered as payload.
253          * @param {Object} opts Object containing onLoad and onError callback functions and a fieldName property to be
254          * used for the name of the form filed if the iframe hack is used
255          */
256         Device.prototype.executeCrossDomainPost = function(url, data, opts) {
257            var payload, modifiedOpts, formData;
258            payload = this.encodeJson(data);
259            if (configSupportsCORS(this.getConfig())) {
260                modifiedOpts = {
261                     onLoad: opts.onLoad,
262                     onError: opts.onError,
263                     headers: {
264                         "Content-Type": "application/json"
265                     },
266                     data: payload,
267                     method: "POST"
268                };
269                this.loadURL(url, modifiedOpts);
270            } else {
271                formData = {};
272                formData[opts.fieldName] = payload;
273                this.crossDomainPost(url, formData, {
274                    onLoad: opts.onLoad,
275                    onError: opts.onError
276                });
277            }
278         };
279 
280         function configSupportsCORS(config) {
281             return config && config.networking && config.networking.supportsCORS;
282         }
283 	}
