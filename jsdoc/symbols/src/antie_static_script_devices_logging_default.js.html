 28 //logs to the javascript console
 29 require.def(
 30 	'antie/devices/logging/default',
 31 	[
 32 		'module', 'antie/devices/device'
 33 	],
 34 	function( Module, Device ) 
 35 	{
 36 		var loggingMethods = {
 37 			/**
 38 			 * Sets the iterator pointer to the first item
 39 			 */
 40 			log: function() {
 41 				console.log.apply(console, arguments);
 42 			},
 43 			debug: function() {
 44 				console.debug.apply(console, arguments);
 45 			},						
 46 			info: function() {
 47 				console.info.apply(console, arguments);
 48 			},			
 49 			warn: function() {
 50 				console.warn.apply(console, arguments);
 51 			},		
 52 			error: function() {
 53 				console.error.apply(console, arguments);
 54 			}
 55 		};
 56 		
 57 		Device.addLoggingStrategy( Module.id, loggingMethods );
 58 	}
 59 );
