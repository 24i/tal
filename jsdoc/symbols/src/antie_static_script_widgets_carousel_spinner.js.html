 25 require.def('antie/widgets/carousel/spinner',
 26     [
 27         'antie/class'
 28     ],
 29     function (Class) {
 30         "use strict";
 31         /**
 32          * Manages communication with device for carousel animations
 33          * @name antie.widgets.carousel.Spinner
 34          * @class
 35          * @extends antie.Class
 36          * @param {antie.devices.Device} device The device abstraction object
 37          * @param {antie.widgets.carousel.Mask} mask The carousel mask to be controlled by the spinner
 38          * @param {Object} orientation The orientation object of the carousel
 39          */
 40         var Spinner;
 41         Spinner = Class.extend(/** @lends antie.widgets.carousel.Spinner.prototype */ {
 42             /**
 43              * @constructor
 44              * @ignore
 45              */
 46             init: function (device, mask, orientation) {
 47                 this._device = device;
 48                 this._mask = mask;
 49                 this._orientation = orientation;
 50                 this._animating = false;
 51                 this._currentAnimation = null;
 52             },
 53 
 54             /**
 55              * Moves the widget strip's left or top edge relative to the mask's top or left edge
 56              * by the specified number of pixels via the framework's animation methods.
 57              * Note that on a browser device the mask will need to have overflow set and the strip will need position: relative
 58              * for this to work.
 59              * @param {Number} relativePixels The target distance in pixels from the mask's primary edge to the primary edge of it's contents
 60              * @param {Object} animOptions @param {Object} [options] An animation options object
 61              * @param {Number} [animOptions.fps] The frames per second of the alignment, if using styletopleft animation
 62              * @param {Number} [animOptions.duration] The duration of the alignment in ms
 63              * @param {String} [animOptions.easing] The alignment easing function
 64              * @param {Boolean} [animOptions.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 65              * @param {Function} [animOptions.onComplete] A function which will be executed on completion of the alignment animation.
 66              */
 67             moveContentsTo: function (relativePixels, animOptions) {
 68                 var moveElementOptions;
 69                 moveElementOptions = this._getOptions(animOptions, relativePixels);
 70                 this.stopAnimation();
 71                 this._animating = true;
 72                 this._currentAnimation = this._device.moveElementTo(moveElementOptions);
 73             },
 74 
 75             /**
 76              * Completes any currently animating alignment, firing any associated callback.
 77              */
 78             stopAnimation: function () {
 79                 if (this._animating) {
 80                     this._device.stopAnimation(this._currentAnimation);
 81                     this._clearAnimating();
 82                 }
 83             },
 84 
 85             _getOptions: function (options, relativePixels) {
 86                 var destination, clonedOptions;
 87                 options = options || {};
 88                 destination = {};
 89                 destination[this._getEdge()] = relativePixels;
 90                 clonedOptions = {};
 91                 clonedOptions.el = this._mask.getWidgetStrip().outputElement;
 92                 clonedOptions.to = destination;
 93                 clonedOptions.from = options.from;
 94                 clonedOptions.duration = options.duration || 150;
 95                 clonedOptions.easing = options.easing || 'linear';
 96                 clonedOptions.fps = options.fps || '25';
 97                 clonedOptions.skipAnim = (options.skipAnim === undefined) ? true : options.skipAnim;
 98                 clonedOptions.onComplete = this._getWrappedOnComplete(options);
 99                 return clonedOptions;
100             },
101 
102             _getWrappedOnComplete: function (options) {
103                 var self;
104                 self = this;
105                 function wrappedComplete() {
106                     self._clearAnimating();
107                     if (options.onComplete && typeof options.onComplete === 'function') {
108                         options.onComplete();
109                     }
110 
111                 }
112                 return wrappedComplete;
113             },
114 
115             _clearAnimating: function () {
116                 this._animating = false;
117                 this._currentAnimation = null;
118             },
119 
120             _getEdge: function () {
121                 return this._orientation.edge();
122             }
123         });
124 
125         return Spinner;
126     }
127 );
