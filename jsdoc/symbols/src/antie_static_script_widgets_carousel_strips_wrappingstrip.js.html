 24 require.def('antie/widgets/carousel/strips/wrappingstrip',
 25     [
 26         'antie/widgets/carousel/strips/widgetstrip'
 27     ],
 28     function (WidgetStrip) {
 29         "use strict";
 30         /**
 31          * A container for the widgets displayed within a carousel
 32          * @name antie.widgets.carousel.strips.WrappingStrip
 33          * @class
 34          * @extends antie.widgets.carousel.strips.WidgetStrip
 35          * @param {String} id The unique ID of the widget.
 36          * @param {Object} orientation an object representing the strip's orientation.
 37          * One of antie.widgets.carousel.orientations.Horizontal or antie.widgets.carousel.orientations.Vertical
 38          */
 39         return WidgetStrip.extend(/** @lends antie.widgets.carousel.strips.WrappingStrip.prototype */{
 40             /**
 41              * @constructor
 42              * @ignore
 43              */
 44             init: function (id, orientation) {
 45                 this._super(id, orientation);
 46                 this._clones = [];
 47                 this._prependedClones = [];
 48                 this._appendedClones = [];
 49                 this._widgetElements = [];
 50                 this._elements = [];
 51                 this._elementIndexOffset = 0;
 52                 this.addClass('carouselwidgetstrip');
 53                 this._autoCalculate = true;
 54             },
 55 
 56             /**
 57              * Renders the widget and any child widgets to device-specific output.
 58              * @param {antie.devices.Device} device The device to render to.
 59              * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 60              */
 61             render: function (device) {
 62                 var i;
 63                 if (!this.outputElement) {
 64                     this.outputElement = device.createContainer(this.id, this.getClasses());
 65                 } else {
 66                     device.clearElement(this.outputElement);
 67                 }
 68                 for (i = 0; i !== this._elements.length; i++) {
 69                     device.appendChildElement(this.outputElement, this._elements[i]);
 70                 }
 71                 return this.outputElement;
 72             },
 73 
 74             /**
 75              * Adds a widget to the end of the strip.
 76              * Note will cause clone recalculation unless autoCalculate has been set to false
 77              * @param {antie.widgets.Widget} widget The widget to append to the strip
 78              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
 79              * (Height for a vertical strip or width for horizontal.) If providied, this value will be used in
 80              * positioning calculations rather then a calculated value (can be useful when widgets change size)
 81              * Note length only currently working with non-wrapping strips.
 82              */
 83             append: function (widget, length) {
 84                 this._super(widget, length);
 85                 this._recalculateIfAuto();
 86             },
 87 
 88             /**
 89              * Inserts widget within the strip
 90              * Note will cause clone recalculation unless autoCalculate has been set to false
 91              * @param {Number} index A zero based index to begin insertion at, i.e. 0 prepends.
 92              * @param {antie.widgets.Widget} widget The widget to append to the strip
 93              * @param {Number} [length] the length of the widget in pixels, measured along the primary axis.
 94              * (Height for a vertical strip or width for horizontal.) If provided, this value will be used in
 95              * positioning calculations rather then a calculated value (can be useful when widgets change size)
 96              * Note length only currently working with non-wrapping strips.
 97              */
 98             insert: function (index, widget, length) {
 99                 this._super(index, widget, length);
100                 this._recalculateIfAuto();
101             },
102 
103             /**
104              * Removes a widget from the strip
105              * @param {antie.widgets.Widget} widget. Widget to remove from the strip
106              */
107             remove: function (widget, retainElement) {
108                 this._super(widget, retainElement);
109                 this._recalculateIfAuto();
110             },
111 
112             /**
113              * @param index
114              * @returns {Number} length in pixels along primary axis to primary edge of the provided index
115              * i.e. from the left edge of the strip to the left edge of the widget in a horizontal carousel
116              */
117             getLengthToIndex: function (index) {
118                 var suppliedLength;
119                 suppliedLength = this._lengthToIndexUsingSuppliedValues(index);
120                 if (suppliedLength !== null) {
121                     return suppliedLength;
122                 } else {
123                     return this._getLengthOfElementArrayUpToIndex(this.getChildElements(), index + this._elementIndexOffset);
124                 }
125             },
126 
127             /**
128              * @returns {Array} elements an array of all elements appended to the strip (including clones)
129              */
130             getChildElements: function () {
131                 return this._elements;
132             },
133 
134             /**
135              * Removes any existing clones and appends sufficient clones to either end to allow for visual wrapping
136              */
137             recalculate: function () {
138                 this._removeClones();
139                 this._createClones(this._getMaskLength());
140             },
141 
142             /**
143              * Toggles auto calculation of clones when necessary (i.e. after an append or remove)
144              * @param {Boolean} on If on === true clones will be recalculated after append, remove or insert. If set false
145              * they will not be recalculated unless recalculate is called.
146              */
147             autoCalculate: function (on) {
148                 if (typeof on === 'boolean') {
149                     this._autoCalculate = on;
150                 }
151             },
152 
153             /**
154              * @param index
155              * @returns {Number} The primary length of the widget at the specified index. (width for horizontal, height for vertical)
156              */
157             lengthOfWidgetAtIndex: function (index) {
158                 var providedLength;
159                 providedLength = this._lengthOfWidgetAtIndexUsingSuppliedValues(index);
160                 if (typeof providedLength === 'number') {
161                     return providedLength;
162                 }
163                 return this._getElementLength(this._elements[index + this._elementIndexOffset]);
164             },
165 
166             _lengthToIndexUsingSuppliedValues: function (index) {
167                 return null;
168             },
169 
170             _lengthOfWidgetAtIndexUsingSuppliedValues: function (index) {
171                 return null;
172             },
173 
174             _refereshWidgetElements: function () {
175                 var widgets, i;
176                 this._widgetElements = [];
177                 widgets = this.getChildWidgets();
178                 for (i = 0; i !== widgets.length; i += 1) {
179                     this._widgetElements.push(widgets[i].outputElement || widgets[i].render(this._getDevice()));
180                 }
181             },
182 
183             _recalculateIfAuto: function () {
184                 if (this._autoCalculate === true) {
185                     this.recalculate();
186                 }
187             },
188 
189             _refreshElements: function () {
190                 this._refereshWidgetElements();
191                 this._elements = this._prependedClones.concat(this._widgetElements.concat(this._appendedClones));
192                 this._elementIndexOffset = this._getPrependedClones().length;
193             },
194 
195             _getLengthOfElementArrayUpToIndex: function (elementArray, index) {
196                 var elementsUpToIndex, endIndex, length;
197                 endIndex = this._getValidatedIndex(elementArray, index);
198                 elementsUpToIndex = elementArray.slice(0, endIndex + 1);
199                 return this._getOffsetToLastElementInArray(elementsUpToIndex);
200             },
201 
202             _firstFocusableIndex: function (widgets) {
203                 var i, focusableIndex;
204                 i = 0;
205                 focusableIndex = null;
206                 while (focusableIndex === null && i !== widgets.length) {
207                     focusableIndex = widgets[i].isFocusable() ? i : null;
208                     i += 1;
209                 }
210                 return focusableIndex;
211             },
212 
213             _createClones: function (maskLength) {
214                 var widgets, device, clonesOfFrontItems, clonesOfRearItems, i;
215                 device = this.getCurrentApplication().getDevice();
216                 widgets = this.getChildWidgets();
217 
218                 clonesOfFrontItems = this._cloneFrontItems(widgets, maskLength);
219                 for (i = 0; i !== clonesOfFrontItems.length; i += 1) {
220                     device.appendChildElement(this.outputElement, clonesOfFrontItems[i]);
221                 }
222 
223                 clonesOfRearItems = this._cloneRearItems(widgets, maskLength);
224                 for (i = 0; i !== clonesOfRearItems.length; i += 1) {
225                     device.prependChildElement(this.outputElement, clonesOfRearItems[i]);
226                 }
227                 this._prependedClones = clonesOfRearItems.reverse();
228                 this._appendedClones = clonesOfFrontItems;
229                 this._clones = clonesOfRearItems.concat(clonesOfFrontItems);
230                 this._refreshElements();
231             },
232 
233             _cloneFrontItems: function (widgets, maskLength) {
234                 var firstFocusableIndex, requiredLength, clones;
235                 clones = [];
236                 firstFocusableIndex = this._firstFocusableIndex(widgets);
237                 if (firstFocusableIndex !== null) {
238                     requiredLength = maskLength + this.getLengthToIndex(firstFocusableIndex);
239                     clones = this._cloneFromIndexToLength(widgets, 0, requiredLength);
240                 }
241                 return clones;
242             },
243 
244             _cloneRearItems: function (widgets, maskLength) {
245                 function shallowCloneArray(arr) {
246                     return arr.slice(0);
247                 }
248                 var reversedWidgets;
249                 reversedWidgets = shallowCloneArray(widgets);
250                 reversedWidgets.reverse();
251                 return this._cloneFrontItems(reversedWidgets, maskLength);
252             },
253 
254             _cloneFromIndexToLength: function (widgets, startIndex, length) {
255                 var clones, clonedLength, i, widget;
256                 clones = [];
257                 clonedLength = 0;
258                 i = startIndex;
259                 while (clonedLength < length) {
260                     widget = widgets[i];
261                     clones.push(this._cloneWidget(widget));
262                     clonedLength += this._getWidgetLength(widget);
263                     i += 1;
264                     if (i === widgets.length) {
265                         i = 0;
266                     }
267                 }
268                 return clones;
269             },
270 
271             _cloneWidget: function (widget) {
272                 return this._cloneElement(widget.outputElement);
273             },
274 
275             _cloneElement: function (element) {
276                 var device, clone;
277                 device = this._getDevice();
278                 clone = device.cloneElement(element, true, 'carouselclone', '_CarouselClone');
279                 this._removeStateStylesFromElement(clone);
280                 return clone;
281             },
282 
283             _removeStateStylesFromElement: function (element) {
284                 var device;
285                 device = this.getCurrentApplication().getDevice();
286                 device.removeClassFromElement(element, 'focus', true);
287                 device.removeClassFromElement(element, 'active', true);
288                 device.removeClassFromElement(element, 'buttonFocussed', true);
289             },
290 
291             _removeClones: function () {
292                 var i, clone, device;
293                 device = this.getCurrentApplication().getDevice();
294 
295                 for (i = 0; i !== this._clones.length; i += 1) {
296                     clone = this._clones[i];
297                     device.removeElement(clone);
298                 }
299                 this._clones = [];
300                 this._prependedClones = [];
301                 this._appendedClones = [];
302                 this._refreshElements();
303             },
304 
305             _getPrependedClones: function () {
306                 return this._prependedClones;
307             },
308 
309             _getAppendedClones: function () {
310                 return this._appendedClones;
311             },
312 
313             _getClones: function () {
314                 return this._clones;
315             },
316 
317             _getMaskLength: function () {
318                 return this.parentWidget.getLength();
319             }
320         });
321     }
322 );
