 53      * Calls a method on a plugin. The obj object should have two property,
 54      * name: the name of the method to call on the plugin
 55      * args: the arguments to pass to the plugin method.
 56      */
 57     function callPlugin(prefix, context, obj) {
 58         //Call the plugin, or load it.
 59         var plugin = s.plugins.defined[prefix], waiting;
 60         if (plugin) {
 61             plugin[obj.name].apply(null, obj.args);
 62         } else {
 63             //Put the call in the waiting call BEFORE requiring the module,
 64             //since the require could be synchronous in some environments,
 65             //like builds
 66             waiting = s.plugins.waiting[prefix] || (s.plugins.waiting[prefix] = []);
 67             waiting.push(obj);
 68 
 69             //Load the module
 70             req(["require/" + prefix], context.contextName);
 71         }
 72     }

 92     function configurePackageDir(packages, currentPackages, dir) {
 93         var i, location, pkgObj;
 94         for (i = 0; (pkgObj = currentPackages[i]); i++) {
 95             pkgObj = typeof pkgObj === "string" ? { name: pkgObj } : pkgObj;
 96             location = pkgObj.location;

 99             //or have a colon (indicates a protocol)
100             if (dir && (!location || (location.indexOf("/") !== 0 && location.indexOf(":") === -1))) {
101                 pkgObj.location = dir + "/" + (pkgObj.location || pkgObj.name);
102             }
103 
104             //Normalize package paths.
105             pkgObj.location = pkgObj.location || pkgObj.name;
106             pkgObj.lib = pkgObj.lib || "lib";
107             pkgObj.main = pkgObj.main || "main";
108 
109             packages[pkgObj.name] = pkgObj;
110         }

113     /**
114      * Determine if priority loading is done. If so clear the priorityWait

117         var priorityDone = true,
118             priorityWait = context.config.priorityWait,
119             priorityName, i;

134     /**
135      * Resumes tracing of dependencies and then checks if everything is loaded.
136      */
137     function resume(context) {
138         var args, i, paused = s.paused;
139         if (context.scriptCount <= 0) {
140             //Synchronous envs will push the number below zero with the
141             //decrement above, be sure to set it back to zero for good measure.
142             //require() calls that also do not end up loading scripts could
143             //push the number negative too.
144             context.scriptCount = 0;
145 
146             //Make sure any remaining defQueue items get properly processed.
147             while (defQueue.length) {
148                 args = defQueue.shift();
149                 if (args[0] === null) {
150                     req.onError(new Error('Mismatched anonymous require.def modules'));
151                 } else {
152                     callDefMain(args, context);
153                 }
154             }
155 
156             //Skip the resume if current context is in priority wait.
157             if (context.config.priorityWait && !isPriorityDone(context)) {
158                 return;
159             }
160 
161             if (paused.length) {
162                 //Reset paused since this loop will process current set.
163                 s.paused = [];
164 
165                 for (i = 0; (args = paused[i]); i++) {
166                     req.checkDeps.apply(req, args);
167                 }
168             }
169 
170             if (isWebWorker) {
171                 //In a web worker, since importScripts is synchronous,
172                 //it may think all dependencies are loaded, but still
173                 //in the middle of a list of dependency fetches, so
174                 //delay the checkLoaded in a timeout for the items to complete.
175                 //This is really hacky though, time for a rewrite.
176                 setTimeout(function () {
177                     req.checkLoaded(s.ctxName);
178                 }, 30);
179             } else {
180                 req.checkLoaded(s.ctxName);
181             }
182         }
183     }

185     /**
186      * Main entry point.
187      *
188      * If the only argument to require is a string, then the module that
189      * is represented by that string is fetched for the appropriate context.
190      *
191      * If the first argument is an array, then it will be treated as an array
192      * of dependency string names to fetch. An optional function callback can
193      * be specified to execute when all of those dependencies are available.
194      */
195     require = function (deps, callback, contextName, relModuleName) {
196         var context, config;
197         if (typeof deps === "string" && !isFunction(callback)) {
198             //Just return the module wanted. In this scenario, the
199             //second arg (if passed) is just the contextName.
200             return require.get(deps, callback, contextName, relModuleName);
201         }
202         // Dependencies first
203         if (!require.isArray(deps)) {
204             // deps is a config object
205             config = deps;
206             if (require.isArray(callback)) {
207                 // Adjust args if there are dependencies
208                 deps = callback;
209                 callback = contextName;
210                 contextName = relModuleName;
211                 relModuleName = arguments[4];
212             } else {
213                 deps = [];
214             }
215         }
216 
217         main(null, deps, callback, config, contextName, relModuleName);
218 
219         //If the require call does not trigger anything new to load,
220         //then resume the dependency processing. Context will be undefined
221         //on first run of require.
222         context = s.contexts[(contextName || (config && config.context) || s.ctxName)];
223         if (context && context.scriptCount === 0) {
224             resume(context);
225         }
226         //Returning undefined for Spidermonky strict checking in Komodo
227         return undefined;
228     };

256         var i, scripts, script, node = currentlyAddingScript;
257 
258         //Allow for anonymous functions
259         if (typeof name !== 'string') {
260             //Adjust args appropriately
261             contextName = callback;
262             callback = deps;
263             deps = name;
264             name = null;
265         }
266 
267         //This module may not have dependencies
268         if (!req.isArray(deps)) {
269             contextName = callback;
270             callback = deps;
271             deps = [];
272         }
273 
274         //If no name, and callback is a function, then figure out if it a
275         //CommonJS thing with dependencies.
276         if (!name && !deps.length && req.isFunction(callback)) {
277             //Remove comments from the callback string,
278             //look for require calls, and pull them into the dependencies.
279             callback
280                 .toString()
281                 .replace(commentRegExp, "")
282                 .replace(cjsRequireRegExp, function (match, dep) {

287             //could use exports, and such, so always add those as dependencies.
288             //This is a bit wasteful for RequireJS modules that do not need
289             //an exports or module object, but erring on side of safety.
290             //REQUIRES the function to expect the CommonJS variables in the
291             //order listed below.
292             deps = ["require", "exports", "module"].concat(deps);
293         }
294 
295         //If in IE 6-8 and hit an anonymous require.def call, do the interactive/

299             for (i = scripts.length - 1; i > -1 && (script = scripts[i]); i--) {

301                     node = script;
302                     break;
303                 }
304             }
305             if (!node) {
306                 req.onError(new Error("ERROR: No matching script interactive for " + callback));
307             }
308 
309             name = node.getAttribute("data-requiremodule");
310         }
311 
312         if (typeof name === 'string') {
313             //Do not try to auto-register a jquery later.
314             //Do this work here and in main, since for IE/useInteractive, this function
315             //is the earliest touch-point.
316             s.contexts[s.ctxName].jQueryDef = (name === "jquery");
317         }
318 
319         //Always save off evaluating the def call until the script onload handler.
320         //This allows multiple modules to be in a file without prematurely
321         //tracing dependencies, and allows for anonymous module support,
322         //where the module name is not known until the script onload event
323         //occurs.
324         defQueue.push([name, deps, callback, null, contextName]);
325     };

327     main = function (name, deps, callback, config, contextName, relModuleName) {
328         //Grab the context, or create a new one for the given context name.
329         var context, newContext, loaded, pluginPrefix,
330             canSetContext, prop, newLength, outDeps, mods, paths, index, i,
331             deferMods, deferModArgs, lastModArg, waitingName, packages,
332             packagePaths;
333 
334         contextName = contextName ? contextName : (config && config.context ? config.context : s.ctxName);
335         context = s.contexts[contextName];
336 
337         if (name) {
338                         // Pull off any plugin prefix.
339             index = name.indexOf("!");
340             if (index !== -1) {
341                 pluginPrefix = name.substring(0, index);
342                 name = name.substring(index + 1, name.length);
343             } else {
344                 //Could be that the plugin name should be auto-applied.
345                 //Used by i18n plugin to enable anonymous i18n modules, but
346                 //still associating the auto-generated name with the i18n plugin.
347                 pluginPrefix = context.defPlugin[name];
348             }
349 
350             
351             //If module already defined for context, or already waiting to be
352             //evaluated, leave.
353             waitingName = context.waiting[name];
354             if (context && (context.defined[name] || (waitingName && waitingName !== ap[name]))) {
355                 return;
356             }
357         }
358 
359         if (contextName !== s.ctxName) {
360             //If nothing is waiting on being loaded in the current context,
361             //then switch s.ctxName to current contextName.
362             loaded = (s.contexts[s.ctxName] && s.contexts[s.ctxName].loaded);
363             canSetContext = true;
364             if (loaded) {
365                 for (prop in loaded) {
366                     if (!(prop in empty)) {
367                         if (!loaded[prop]) {
368                             canSetContext = false;
369                             break;
370                         }
371                     }
372                 }
373             }
374             if (canSetContext) {
375                 s.ctxName = contextName;
376             }
377         }
378 
379         if (!context) {
380             newContext = {
381                 contextName: contextName,
382                 config: {
383                     waitSeconds: 7,
384                     baseUrl: s.baseUrl || "./",
385                     paths: {},
386                     packages: {}
387                 },
388                 waiting: [],
389                 specified: {
390                     "require": true,
391                     "exports": true,
392                     "module": true
393                 },
394                 loaded: {},
395                 scriptCount: 0,
396                 urlFetched: {},
397                 defPlugin: {},
398                 defined: {},
399                 modifiers: {}
400             };
401 
402                         if (s.plugins.newContext) {
403                 s.plugins.newContext(newContext);
404             }
405             
406             context = s.contexts[contextName] = newContext;
407         }
408 
409         //If have a config object, update the context's config object with
410         //the config values.
411         if (config) {
412             //Make sure the baseUrl ends in a slash.
413             if (config.baseUrl) {
414                 if (config.baseUrl.charAt(config.baseUrl.length - 1) !== "/") {
415                     config.baseUrl += "/";
416                 }
417             }
418 
419             //Save off the paths and packages since they require special processing,
420             //they are additive.
421             paths = context.config.paths;
422             packages = context.config.packages;
423 
424             //Mix in the config values, favoring the new values over
425             //existing ones in context.config.
426             req.mixin(context.config, config, true);
427 
428             //Adjust paths if necessary.
429             if (config.paths) {
430                 for (prop in config.paths) {
431                     if (!(prop in empty)) {
432                         paths[prop] = config.paths[prop];
433                     }
434                 }
435                 context.config.paths = paths;
436             }
437 
438             packagePaths = config.packagePaths;
439             if (packagePaths || config.packages) {
440                 //Convert packagePaths into a packages config.
441                 if (packagePaths) {
442                     for (prop in packagePaths) {
443                         if (!(prop in empty)) {
444                             configurePackageDir(packages, packagePaths[prop], prop);
445                         }
446                     }
447                 }
448 
449                 //Adjust packages if necessary.
450                 if (config.packages) {
451                     configurePackageDir(packages, config.packages);
452                 }
453 
454                 //Done with modifications, assing packages back to context config
455                 context.config.packages = packages;
456             }
457 
458             //If priority loading is in effect, trigger the loads now
459             if (config.priority) {
460                 //Create a separate config property that can be
461                 //easily tested for config priority completion.
462                 //Do this instead of wiping out the config.priority
463                 //in case it needs to be inspected for debug purposes later.
464                 req(config.priority);
465                 context.config.priorityWait = config.priority;
466             }
467 
468             //If a deps array or a config callback is specified, then call
469             //require with those args. This is useful when require is defined as a
470             //config object before require.js is loaded.
471             if (config.deps || config.callback) {
472                 req(config.deps || [], config.callback);
473             }
474 
475                         //Set up ready callback, if asked. Useful when require is defined as a
476             //config object before require.js is loaded.
477             if (config.ready) {
478                 req.ready(config.ready);
479             }
480             
481             //If it is just a config block, nothing else,
482             //then return.
483             if (!deps) {
484                 return;
485             }
486         }
487 
488         //Normalize dependency strings: need to determine if they have
489         //prefixes and to also normalize any relative paths. Replace the deps
490         //array of strings with an array of objects.
491         if (deps) {
492             outDeps = deps;
493             deps = [];
494             for (i = 0; i < outDeps.length; i++) {
495                 deps[i] = req.splitPrefix(outDeps[i], (name || relModuleName), context);
496             }
497         }
498 
499         //Store the module for later evaluation
500         newLength = context.waiting.push({
501             name: name,
502             deps: deps,
503             callback: callback
504         });
505 
506         if (name) {
507             //Store index of insertion for quick lookup
508             context.waiting[name] = newLength - 1;
509 
510             //Mark the module as specified so no need to fetch it again.
511             //Important to set specified here for the
512             //pause/resume case where there are multiple modules in a file.
513             context.specified[name] = true;
514 
515                         //Load any modifiers for the module.
516             mods = context.modifiers[name];
517             if (mods) {
518                 req(mods, contextName);
519                 deferMods = mods.__deferMods;
520                 if (deferMods) {
521                     for (i = 0; i < deferMods.length; i++) {
522                         deferModArgs = deferMods[i];
523 
524                         //Add the context name to the def call.
525                         lastModArg = deferModArgs[deferModArgs.length - 1];
526                         if (lastModArg === undefined) {
527                             deferModArgs[deferModArgs.length - 1] = contextName;
528                         } else if (typeof lastModArg === "string") {
529                             deferMods.push(contextName);
530                         }
531 
532                         require.def.apply(require, deferModArgs);
533                     }
534                 }
535             }
536                     }
537 
538         //If the callback is not an actual function, it means it already
539         //has the definition of the module as a literal value.
540         if (name && callback && !req.isFunction(callback)) {
541             context.defined[name] = callback;
542         }
543 
544         //If a pluginPrefix is available, call the plugin, or load it.
545                 if (pluginPrefix) {
546             callPlugin(pluginPrefix, context, {
547                 name: "require",
548                 args: [name, deps, callback, context]
549             });
550         }
551         
552         //Hold on to the module until a script load or other adapter has finished
553         //evaluating the whole file. This helps when a file has more than one
554         //module in it -- dependencies are not traced and fetched until the whole
555         //file is processed.
556         s.paused.push([pluginPrefix, name, deps, context]);
557 
558         //Set loaded here for modules that are also loaded
559         //as part of a layer, where onScriptLoad is not fired
560         //for those cases. Do this after the inline define and
561         //dependency tracing is done.
562         //Also check if auto-registry of jQuery needs to be skipped.
563         if (name) {
564             context.loaded[name] = true;
565             context.jQueryDef = (name === "jquery");
566         }
567     };

632     /**
633      * Registers a new plugin for require.
634      */
635     req.plugin = function (obj) {
636         var i, prop, call, prefix = obj.prefix, cbs = s.plugins.callbacks,
637             waiting = s.plugins.waiting[prefix], generics,
638             defined = s.plugins.defined, contexts = s.contexts, context;
639 
640         //Do not allow redefinition of a plugin, there may be internal
641         //state in the plugin that could be lost.
642         if (defined[prefix]) {
643             return req;
644         }
645 
646         //Save the plugin.
647         defined[prefix] = obj;
648 
649         //Set up plugin callbacks for methods that need to be generic to
650         //require, for lifecycle cases where it does not care about a particular
651         //plugin, but just that some plugin work needs to be done.
652         generics = ["newContext", "isWaiting", "orderDeps"];
653         for (i = 0; (prop = generics[i]); i++) {
654             if (!s.plugins[prop]) {
655                 makePluginCallback(prop, prop === "isWaiting");
656             }
657             cbs[prop].push(obj[prop]);
658         }
659 
660         //Call newContext for any contexts that were already created.
661         if (obj.newContext) {
662             for (prop in contexts) {
663                 if (!(prop in empty)) {
664                     context = contexts[prop];
665                     obj.newContext(context);
666                 }
667             }
668         }
669 
670         //If there are waiting requests for a plugin, execute them now.
671         if (waiting) {
672             for (i = 0; (call = waiting[i]); i++) {
673                 if (obj[call.name]) {
674                     obj[call.name].apply(null, call.args);
675                 }
676             }
677             delete s.plugins.waiting[prefix];
678         }
679 
680         return req;

683     /**
684      * As of jQuery 1.4.3, it supports a readyWait property that will hold off
685      * calling jQuery ready callbacks until all scripts are loaded. Be sure
686      * to track it if readyWait is available. Also, since jQuery 1.4.3 does
687      * not register as a module, need to do some global inference checking.
688      * Even if it does register as a module, not guaranteed to be the precise
689      * name of the global. If a jQuery is tracked for this context, then go
690      * ahead and register it as a module too, if not already in process.
691      */
692     function jQueryCheck(context, jqCandidate) {
693         if (!context.jQuery) {
694             var $ = jqCandidate || (typeof jQuery !== "undefined" ? jQuery : null);
695             if ($ && "readyWait" in $) {
696                 context.jQuery = $;
697 
698                 //Manually create a "jquery" module entry if not one already
699                 //or in process.
700                 if (!context.defined.jquery && !context.jQueryDef) {
701                     context.defined.jquery = $;
702                 }
703

964     /**
965      * Given a relative module name, like ./something, normalize it to
966      * a real name that can be mapped to a path.
967      * @param {String} name the relative name
968      * @param {String} baseName a real name that the name arg is relative
969      * to.
970      * @param {Object} context
971      * @returns {String} normalized name
972      */
973     req.normalizeName = function (name, baseName, context) {
974         //Adjust any relative paths.
975         var part;
976         if (name.charAt(0) === ".") {
977             //If have a base name, try to normalize against it,
978             //otherwise, assume it is a top-level require that will
979             //be relative to baseUrl in the end.
980             if (baseName) {
981                 if (context.config.packages[baseName]) {
982                     //If the baseName is a package name, then just treat it as one
983                     //name to concat the name with.
984                     baseName = [baseName];
985                 } else {
986                     //Convert baseName to array, and lop off the last part,
987                     //so that . matches that "directory" and not name of the baseName's
988                     //module. For instance, baseName of "one/two/three", maps to
989                     //"one/two/three.js", but we want the directory, "one/two" for
990                     //this normalization.
991                     baseName = baseName.split("/");
992                     baseName = baseName.slice(0, baseName.length - 1);
993                 }
994 
995                 name = baseName.concat(name.split("/"));
996                 for (i = 0; (part = name[i]); i++) {
997                     if (part === ".") {
998                         name.splice(i, 1);
999                         i -= 1;
1000                     } else if (part === "..") {
1001                         if (i === 1) {
1002                             //End of the line. Keep at least one non-dot
1003                             //path segment at the front so it can be mapped
1004                             //correctly to disk. Otherwise, there is likely
1005                             //no path mapping for '..'.
1006                             break;
1007                         } else if (i > 1) {
1008                             name.splice(i - 1, 2);
1009                             i -= 2;
1010                         }
1011                     }
1012                 }
1013                 name = name.join("/");
1014             }
1015         }
1016         return name;

1144                 waitInterval = context.config.waitSeconds * 1000,
1145                 //It is possible to disable the wait interval by using waitSeconds of 0.
1146                 expired = waitInterval && (context.startTime + waitInterval) < new Date().getTime(),
1147                 loaded, defined = context.defined,
1148                 modifiers = context.modifiers, waiting, noLoads = "",

1204             //the work below does not need to be done.

1254         context.isCheckLoaded = false;

1260             //added while doing the work above in checkLoaded, calling module

1266             //load. After defining the modules above, new require calls
1267             //could have been made.

1774  * This plugin handles i18n! prefixed modules. It does the following:
1775  *
1776  * 1) A regular module can have a dependency on an i18n bundle, but the regular
1777  * module does not want to specify what locale to load. So it just specifies
1778  * the top-level bundle, like "i18n!nls/colors".

1780  * This plugin will load the i18n bundle at nls/colors, see that it is a root/master
1781  * bundle since it does not have a locale in its name. It will then try to find
1782  * the best match locale available in that master bundle, then request all the
1783  * locale pieces for that best match locale. For instance, if the locale is "en-us",
1784  * then the plugin will ask for the "en-us", "en" and "root" bundles to be loaded
1785  * (but only if they are specified on the master bundle).
1786  *

1788  * locale pieces into each other, then finally sets the context.defined value
1789  * for the nls/colors bundle to be that mixed in locale.
1790  *
1791  * 2) A regular module specifies a specific locale to load. For instance,
1792  * i18n!nls/fr-fr/colors. In this case, the plugin needs to load the master bundle
1793  * first, at nls/colors, then figure out what the best match locale is for fr-fr,
1794  * since maybe only fr or just root is defined for that locale. Once that best
1795  * fit is found, all of its locale pieces need to have their bundles loaded.
1796  *
1797  * Once all the bundles for the locale pieces load, then it mixes in all those
