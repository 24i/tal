 24 require.def('antie/devices/parentalguidance/appdefaultpghandler',
 25     [
 26         'antie/devices/browserdevice',
 27         'antie/devices/parentalguidance/basepghandler'
 28     ],
 29     function (Device, BasePgHandler) {
 30         'use strict';
 31 
 32         var appDefaultPgHandler = BasePgHandler.extend({
 33 
 34             isChallengeActive: function() {
 35                 if (this._appHandler) {
 36                     return this._appHandler.isChallengeActive();
 37                 } else {
 38                     throw new Error('No default parental guidance handler is registered');
 39                 }
 40             },
 41             showChallenge: function(message, guidanceChallengeResponseCallBack) {
 42                 if (!this._appHandler) {
 43                     throw new Error('No default parental guidance handler is registered');
 44                 } else if (typeof(guidanceChallengeResponseCallBack.onGuidanceChallengeResponse) != "function") {
 45                     throw new Error('The guidanceChallengeResponseCallback object should contain an onGuidanceChallengeResponse' +
 46                         'function. The appHandler should call this function with a value from pgchallengeresponse.js as the first parameter');
 47                 } else {
 48                    return this._appHandler.showChallenge(message, guidanceChallengeResponseCallBack);
 49                 }
 50             },
 51             supportsMessage: function() {
 52               return true;
 53             },
 54             isConfigurable: function() {
 55                 return true;
 56             }
 57         });
 58 
 59         Device.prototype.parentalGuidanceHelper = new appDefaultPgHandler;
 60 
 61         /**
 62          *
 63          * @param {object} containing implementation of showChallenge() and isChallengeActive()
 64          */
 65         Device.prototype.registerAppPgHandler = function(appHandler) {
 66             this.parentalGuidanceHelper._appHandler = appHandler;
 67         };
 68 
 69     }
 70 );
