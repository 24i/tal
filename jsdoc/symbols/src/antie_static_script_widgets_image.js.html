 27 require.def('antie/widgets/image',
 28 	['antie/widgets/container'],
 29 	function(Container) {
 30 		/**
 31 		 * The Image widget displays an image. It supports lazy loading/unloading of images to conserve memory.
 32 		 * You can use CSS to set a background image on the container first and then when the image specified 
 33 		 * in setSrc is loaded up it will fill the <div> and obscure the background image.
 34 		 * @name antie.widgets.Image
 35 		 * @class
 36 		 * @extends antie.widgets.Container
 37 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 38 		 * @param {String} src The image source URL.
 39 		 * @param {Size} size The size of the image.
 40 		 */
 41 		var Image = Container.extend(/** @lends antie.widgets.Image.prototype */ {
 42 			/**
 43 			 * @constructor
 44 			 * @ignore
 45 			 */
 46 			init: function(id, src, size) {
 47 				this._super(id);
 48 				this._src = src;
 49 				this._size = size;
 50 				this._imageElement = null;
 51 				this._renderMode = Image.RENDER_MODE_CONTAINER;
 52 				this.addClass('image');
 53 			},
 54 			/**
 55 			 * Renders the widget and any child widgets to device-specific output.
 56 			 * @param {antie.devices.Device} device The device to render to.
 57 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 58 			 */
 59 			render: function(device) {
 60 				this._imageElement = device.createImage(this.id + "_img", null, this._src, this._size);
 61 				if(this._renderMode == Image.RENDER_MODE_CONTAINER) {
 62 					this.outputElement = this._super(device);
 63 					if(this._size) {
 64 						device.setElementSize(this.outputElement, this._size);
 65 					}
 66 					device.prependChildElement(this.outputElement, this._imageElement);
 67 				} else {
 68 					this.outputElement = this._imageElement;
 69 				}
 70 	
 71 				return this.outputElement;
 72 			},
 73 			/**
 74 			 * Sets the image source URL.
 75 			 * @param {String} src The new image source URL to display.
 76 			 */
 77 			setSrc: function(src) {
 78 				this._src = src;
 79 				if(this._imageElement) {
 80 					this._imageElement.src = src;
 81 				}
 82 			},
 83 			/**
 84 			 * Sets the image source URL.
 85 			 * @returns The current image source URL.
 86 			 */
 87 			getSrc: function() {
 88 				return this._src;
 89 			},
 90 			setRenderMode: function(mode) {
 91 				this._renderMode = mode;
 92 			},
 93 			getRenderMode: function() {
 94 				return this._renderMode;
 95 			}
 96 		});
 97 
 98 		Image.RENDER_MODE_IMG = 0;
 99 		Image.RENDER_MODE_CONTAINER = 1;
100 
101 		return Image;
102 	}
103 );
