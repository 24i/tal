 27 //Logs to via an XML HTTP Request ( XHR )
 28 require.def(
 29 	'antie/devices/logging/xhr',
 30 	[
 31 	 	'module',
 32 	 	'antie/application',
 33 		'antie/devices/device'
 34 	],
 35 	function( Module, Application, Device ) 
 36 	{
 37 		function zeroFill( number, width )
 38 		{
 39 		  width -= number.toString().length;
 40 		  if ( width > 0 )
 41 		  {
 42 		    return new Array( width + (/\./.test( number ) ? 2 : 1) ).join( '0' ) + number;
 43 		  }
 44 		  return number + ""; // always return a string
 45 		}
 46 		
 47 		function sendXHRLogMessage( tag, message )
 48 		{
 49 			var now = new Date();
 50 			var timeString = "" + zeroFill( now.getHours(), 2 ) + ":" + zeroFill( now.getMinutes(), 2 ) + ":" + zeroFill( now.getSeconds(), 2 );  
 51 			var messageObj = { command : "add", message : "[" + timeString + "][" + tag + "]" +  message };
 52 			xhrPost( "/tvpjsframeworklogging", {}, messageObj );
 53 		}
 54 		
 55 		var loggingMethods = {
 56 			/**
 57 			 * Sets the iterator pointer to the first item
 58 			 */
 59 			log: function() {
 60 					sendXHRLogMessage( "LOG", Array.prototype.join.call(arguments, '\n') )
 61 			},
 62 			debug: function() {
 63 					sendXHRLogMessage( "DEBUG", Array.prototype.join.call(arguments, '\n') )
 64 			},						
 65 			info: function() {
 66 					sendXHRLogMessage( "INFO", Array.prototype.join.call(arguments, '\n') )
 67 			},			
 68 			warn: function() {
 69 					sendXHRLogMessage( "WARN", Array.prototype.join.call(arguments, '\n') )
 70 			},		
 71 			error: function() {
 72 					sendXHRLogMessage( "ERROR", Array.prototype.join.call(arguments, '\n') )
 73 			}
 74 		};
 75 		
 76 		Device.addLoggingStrategy( Module.id, loggingMethods );
 77 
 78 		function xhrPost( url, opts, messageObject ) {
 79 			
 80 			var http = new XMLHttpRequest();
 81 					
 82 			var device 		= Application.getCurrentApplication().getDevice();
 83 			var jsonMessage = device.encodeJson( messageObject ); 
 84 			
 85 			http.open("POST", url, true);
 86 
 87 			//Send the proper header information along with the request
 88 			http.setRequestHeader("Content-type", "application/json; charset=utf-8");
 89 			
 90 			http.onreadystatechange = function() {//Call a function when the state changes.
 91 				if (this.readyState == 4) {
 92 					this.onreadystatechange = null;
 93 					if (this.status == 200) {
 94 						if (opts.onLoad) {
 95 							opts.onOkay(this.responseText);
 96 						}
 97 					} else {
 98 						if (opts.onError) {
 99 							opts.onError(this.responseText, this.status);
100 						}
101 					}
102 				}
103 			}
104 			http.send( jsonMessage );
105 		}
106 	}
107 );
