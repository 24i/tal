 27 require.def('antie/mediasource',
 28         [
 29 			'antie/class',
 30 			'antie/runtimecontext',
 31 			'antie/urlbuilder'
 32 		],
 33         function(Class, RuntimeContext, URLBuilder) {
 34             'use strict';
 35 
 36 			/**
 37 			 * Base MediaSource (of unknown type). Provides storage of source and type information about a media source.
 38 			 * @name antie.MediaSource
 39 			 * @class
 40 			 * @extends antie.Class
 41 			 * @param {String} src The URI of the media.
 42 			 * @param {String} type The type of the media.
 43 			 */
 44             var MediaSource = Class.extend(/** @lends antie.MediaSource.prototype */ {
 45 						/**
 46 						 * Create a new MediaSource.
 47 						 * @constructor
 48 						 * @ignore
 49 						 */
 50 						init: function(src, type) {
 51                             this.src = src;
 52                             this.type = type;
 53                         },
 54 						/**
 55 						* Check if this MediaSource is equal to another.
 56 						* @param {Object} mediaSource A MediaSource object to compare this one to.
 57 						* @returns Boolean true if the provided mediaSource is equal to this. Otherwise false.
 58 						*/
 59                         isEqual: function(mediaSource) {
 60                             return this.src === mediaSource.src && this.type === mediaSource.type;
 61                         },
 62 						/**
 63 						* Check to see if this media source refers to a live stream.
 64 						* @returns Boolean true if this is a live stream. Otherwise false.
 65 						*/
 66 						isLiveStream: function() {
 67 							return MediaSource.isLiveStream(this.type);
 68 						},
 69 						/**
 70 						* Get the URL of this stream, built from the known source, config-provided template, and tag replacement.
 71 						* @param {Object} [tags] A associative array of tags (%tag%) and values to repalce in the URL.
 72 						* @returns The URL of this source.
 73 						*/
 74 						getURL: function(tags) {
 75 							tags = tags || {};
 76 
 77 							var config = RuntimeContext.getCurrentApplication().getDevice().getConfig();
 78 							var streamingConfig = this.getMediaType() == MediaSource.MEDIA_TYPE_AUDIO ?
 79 									config.streaming.audio : config.streaming.video;
 80 
 81 							return new URLBuilder(streamingConfig.mediaURIFormat).getURL(this.src, tags);
 82 						},
 83 						/**
 84 						* Get the content type of this source.
 85 						* @returns The content type (MIME type) of this source.
 86 						*/
 87 						getContentType: function() {
 88 							return this.type;
 89 						},
 90 						/**
 91 						* Get the media type of this source. Either  <code>MediaType.MEDIA_TYPE_UNKNOWN</code>, <code>MediaType.MEDIA_TYPE_AUDIO</code> or <code>MediaType.MEDIA_TYPE_VIDEO</code>.
 92 						* @returns The media type of this content.
 93 						*/
 94 						getMediaType: function() {
 95 							return MediaSource.MEDIA_TYPE_UNKNOWN;
 96 						}
 97                     });
 98 
 99 			/**
100 			 * Unknown media type.
101 			 * @memberOf antie.MediaSource
102 			 * @name MEDIA_TYPE_UNKNOWN
103 			 * @constant
104 			 * @static
105 			 */
106 			MediaSource.MEDIA_TYPE_UNKNOWN = 0;
107 
108 			/**
109 			 * Video media type.
110 			 * @memberOf antie.MediaSource
111 			 * @name MEDIA_TYPE_VIDEO
112 			 * @constant
113 			 * @static
114 			 */
115 			MediaSource.MEDIA_TYPE_VIDEO = 1;
116 
117 			/**
118 			 * Audio media type.
119 			 * @memberOf antie.MediaSource
120 			 * @name MEDIA_TYPE_AUDIO
121 			 * @constant
122 			 * @static
123 			 */
124 			MediaSource.MEDIA_TYPE_AUDIO = 2;
125 
126 			/**
127 			 * Returns Given a content type, returns if it refers to a live stream.
128 			 * @name isLiveStream
129 			 * @memberOf antie.MediaSource
130 			 * @static
131 			 * @function
132 			 * @param {String} type The type of the media.
133 			 * @returns Boolean true if the given type refers to a live stream.
134 			 */
135 			MediaSource.isLiveStream = function(type) {
136 				return type.toLowerCase() === "application/vnd.apple.mpegurl";
137 			};
138 
139             return MediaSource;
140         });
