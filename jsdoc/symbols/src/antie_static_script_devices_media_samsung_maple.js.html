 41         function audioLevelCorrection(t) {
 42             return t * 40.0;
 43         }
 44         function invertAudioLevelCorrection(x) {
 45             return x / 40.0;
 46         }
 47         var SamsungPlayer = MediaInterface.extend({
 48             init: function(id, mediaType, eventHandlingCallback) {
 49                 this._super(id);
 50 
 51                 this._eventHandlingCallback = eventHandlingCallback;

 53                 this.playerPlugin = document.getElementById('playerPlugin');
 54                 this.audioPlugin = document.getElementById('audioPlugin');
 55                 this.tvmwPlugin = document.getElementById('pluginObjectTVMW');

 89                 // functions attached later.

101

104                 };

106 
107                 window.SamsungMapleOnNetworkDisconnected = function() {

109                 };

136                 };

138 
139                 window.SamsungMapleOnCurrentPlayTime = function (timeMs) {
140                     var seconds = timeMs / 1000.0;
141                     if ((self.mediaSource.isLiveStream() && self.videoPlayerState.ended == false) ||
142                         (seconds >= 0 && seconds < self.videoPlayerState.durationSeconds)) {
143                         self.videoPlayerState.currentTime = seconds;
144                         if (self.videoPlayerState.seeking) {
145                             self.videoPlayerState.seeking = false;
146                             self._eventHandlingCallback(new MediaEvent('seeked', self));
147                         }
148                         if (self.videoPlayerState.playing === false) {
149                             self._eventHandlingCallback(new MediaEvent('play', self));
150                             self._eventHandlingCallback(new MediaEvent('playing', self));
151                             self.videoPlayerState.playing = true;
152                         }
153                         else {
154                             // don't throw a timeupdate on the first event
155                             window.SamsungMapleOnTimeUpdate(seconds);
156                         }
157                     }
158                 };
159                 this.playerPlugin.OnCurrentPlayTime = 'SamsungMapleOnCurrentPlayTime';
160 
161                 window.SamsungMapleOnTimeUpdate = function(seconds) {
162                     self._eventHandlingCallback(new MediaEvent("timeupdate", self));
163                 };

276             // attribute double currentTime;
277             setCurrentTime: function(timeToSeekTo) {
278                 var offsetInSeconds = timeToSeekTo - this.videoPlayerState.currentTime;
279                 if (offsetInSeconds > 0) {
280                     var jumped = this.playerPlugin.JumpForward(offsetInSeconds);
281                     // Jump forward appears not to work consistently in the initial moments of a video playback.
282                     // if we are in the initial set up, lets try resuming instead
283                     // Samsung 2010 returns -1 for failure. Newer API returns false.
284                     if ((!jumped || jumped < 0) && this.videoPlayerState.currentTime < 1) {
285                         this.playerPlugin.Stop();
286                         this._resetVideoSize();
287                         this.playerPlugin.ResumePlay(this._getSamsungFormattedUrl, timeToSeekTo);
288                     }
289                 } else if (offsetInSeconds < 0) {
290                     this.playerPlugin.JumpBackward(Math.abs(offsetInSeconds));
291                 }
292                 this.videoPlayerState.seeking = true;
293                 this._eventHandlingCallback(new MediaEvent('seeking', this));
294                 this.videoPlayerState.currentTime = timeToSeekTo;
295             },

431             var audio = document.getElementById('audioPlugin');
432             return invertAudioLevelCorrection(audio.GetVolume());
433         };
434         /**
435          * Set the current volume.
436          * @param {Float} volume The new volume level (0.0 to 1.0).
437          */
438         Device.prototype.setVolume = function(volume) {
439             var audio = document.getElementById('audioPlugin');
440             if (volume > 1.0) {
441                 this.getLogger().warn("Samsung setVolume - Invalid volume specified (" + volume + " > 1.0). Clipped to 1.0");
442                 volume = 1.0;
443             } else if (volume < 0.0) {
444                 this.getLogger().warn("Samsung setVolume - Invalid volume specified (" + volume + " < 0.0). Clipped to 0.0");
445                 volume = 0;
446             }
447 
448             var currentVolume = audio.GetVolume();
449             var newVolume = audioLevelCorrection(volume);
450 
451             if ((newVolume > currentVolume) && (newVolume - currentVolume < 1.0)) {
452                 newVolume = currentVolume + 1;
453             } else if ((newVolume < currentVolume) && (currentVolume - newVolume < 1.0)) {
454                 newVolume = currentVolume - 1;
455             }
456             audio.SetVolume(newVolume);
457             return newVolume;
458         };
