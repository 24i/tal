 24 require.def('antie/widgets/carousel/binder',
 25     [
 26         'antie/class',
 27         'antie/iterator',
 28         'antie/events/databoundevent'
 29     ],
 30     function (Class, Iterator, DataBoundEvent) {
 31         "use strict";
 32         /**
 33          * Class for adding children to an existing widget based on a combination
 34          * of a data source and formatter.
 35          * @name antie.widgets.carousel.Binder
 36          * @class
 37          * @extends antie.Class
 38          */
 39         var Binder = Class.extend(/** @lends antie.widgets.carousel.Binder.prototype */ {
 40                 /**
 41                  * @constructor
 42                  * @ignore
 43                  */
 44                 init: function (formatter, dataSource) {
 45                     this._dataSource = dataSource;
 46                     this._formatter = formatter;
 47                 },
 48 
 49                 /**
 50                  * Creates new widgets which are then appended to
 51                  * the widget supplied. Continues until the end of the data returned
 52                  * by the source is reached.
 53                  * @param widget The parent of the widgets to be created.
 54                  */
 55                 appendAllTo: function (widget) {
 56                     this._bindAll(widget, this._appendItem);
 57                 },
 58 
 59                 _bindAll: function (widget, processItemFn, preBindFn, postBindFn) {
 60                     var callbacks, beforeBindEvent;
 61 
 62                     callbacks = this._getCallbacks(widget, processItemFn, postBindFn);
 63                     beforeBindEvent = new DataBoundEvent("beforedatabind", widget);
 64 
 65                     if (typeof preBindFn === 'function') {
 66                         preBindFn(beforeBindEvent);
 67                     }
 68 
 69                     widget.bubbleEvent(beforeBindEvent);
 70                     if (!(this._dataSource instanceof Array)) {
 71                         this._dataSource.load(callbacks);
 72                     } else {
 73                         callbacks.onSuccess(this._dataSource);
 74                     }
 75 
 76                 },
 77 
 78                 _getCallbacks: function (widget, processItemFn, postBindFn) {
 79                     var self = this;
 80                     return {
 81                         onSuccess: function (data) {
 82                             var it, boundItem, dataBoundEvent;
 83                             if (data instanceof Iterator) {
 84                                 it = data;
 85                             } else {
 86                                 it = new Iterator(data);
 87                             }
 88 
 89                             while (it.hasNext()) {
 90                                 boundItem = self._formatter.format(it);
 91                                 processItemFn(widget, boundItem);
 92                             }
 93 
 94                             dataBoundEvent = new DataBoundEvent("databound", widget, it);
 95                             if (typeof postBindFn === 'function') {
 96                                 postBindFn(dataBoundEvent);
 97                             }
 98 
 99                             widget.bubbleEvent(dataBoundEvent);
100                         },
101                         onError: function (error) {
102                             widget.bubbleEvent(new DataBoundEvent("databindingerror", widget, null, error));
103                         }
104                     };
105                 },
106 
107                 _appendItem: function (widget, item) {
108                     return widget.appendChildWidget(item);
109                 }
110             }
111         );
112         return Binder;
113     }
114 );
