 24 require.def('antie/widgets/carousel/aligners/alignmentqueue',
 25     [
 26         'antie/class'
 27     ],
 28     function (Class) {
 29         "use strict";
 30 
 31         function createAlignFunction(self, index, options) {
 32             return function () {
 33                 options = options || {};
 34                 var oldComplete = options.onComplete;
 35                 var newOptions = {
 36                     el: options.el,
 37                     to: options.to,
 38                     from: options.from,
 39                     duration: options.duration,
 40                     easing: options.easing,
 41                     skipAnim: options.skipAnim,
 42                     fps: options.fps,
 43                     onComplete: function () {
 44                         if (oldComplete) {
 45                             oldComplete();
 46                         }
 47                         self._next();
 48                     }
 49                 };
 50 
 51                 if (self._skip) {
 52                     newOptions.skipAnim = true;
 53                 }
 54 
 55                 self._mask.alignToIndex(index, newOptions);
 56             };
 57         }
 58 
 59         /**
 60          * A class to handle queue multiple alignments for execution in sequence
 61          * @class
 62          * @extends antie.widgets.Class
 63          * @param {Object} mask The carousel's mask object
 64          */
 65         return Class.extend({
 66             init: function (mask) {
 67                 this._mask = mask;
 68                 this._setSkip(false);
 69                 this._queue = [];
 70                 this._started = false;
 71             },
 72 
 73             /**
 74              * Queues an alignment operation
 75              * @param {Number} index The index of the widget to align on.
 76              * @param {Object} [options] An animation options object
 77              * @param {Number} [options.fps] The frames per second of the alignment, if using styletopleft animation
 78              * @param {Number} [options.duration] The duration of the alignment in ms
 79              * @param {String} [options.easing] The alignment easing function
 80              * @param {Boolean} [options.skipAnim] If set true, the alignment will complete instantly then fire any provided callback
 81              * @param {Function} [options.onComplete] A function which will be executed on completion of the alignment animation.
 82              */
 83             add: function (index, options) {
 84                 var self = this;
 85 
 86                 var alignFunction = createAlignFunction(self, index, options);
 87 
 88                 this._queue.push(alignFunction);
 89             },
 90 
 91             /**
 92              * Begins executing the alignment operations in the queue. If the
 93              * queue has already been started, but has not completed, this will
 94              * do nothing.
 95              */
 96             start: function () {
 97                 if (!this._started) {
 98                     this._runFirstInQueue();
 99                 }
100             },
101 
102             /**
103              * Completes all queued alignments in order, skipping any animation and
104              * firing any associated callbacks in sequence.
105              */
106             complete: function () {
107                 if (this._started) {
108                     this._setSkip(true);
109                     this._mask.stopAnimation();
110                 }
111             },
112 
113             _runFirstInQueue: function () {
114                 if (this._queue.length > 0) {
115                     this._started = true;
116                     this._queue[0]();
117                 } else {
118                     this._setSkip(false);
119                     this._started = false;
120                 }
121             },
122 
123             _next: function () {
124                 this._queue.shift();
125                 this._runFirstInQueue();
126             },
127 
128             _setSkip: function (skip) {
129                 this._skip = skip;
130             }
131         });
132     }
133 );
