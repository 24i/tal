 29 //logs to the jstestdriver console
 30 require.def(
 31 	'antie/devices/logging/jstestdriver',
 32 	[
 33 		'module', 'antie/devices/device'
 34 	],
 35 	function( Module, Device ) 
 36 	{
 37 		'use strict';
 38 
 39 		var enabledLevels = null;
 40 		var loggingMethods = {
 41 			/**
 42 			 * Sets the iterator pointer to the first item
 43 			 */
 44 			log: function() {
 45 					jstestdriver.console.log.apply(jstestdriver.console, arguments);
 46 			},
 47 			debug: function() {
 48 					jstestdriver.console.debug.apply(jstestdriver.console, arguments);
 49 			},						
 50 			info: function() {
 51 					jstestdriver.console.info.apply(jstestdriver.console, arguments);
 52 			},			
 53 			warn: function() {
 54 					jstestdriver.console.warn.apply(jstestdriver.console, arguments);
 55 			},		
 56 			error: function() {
 57 					jstestdriver.console.error.apply(jstestdriver.console, arguments);
 58 			}
 59 		};
 60 		
 61 		Device.addLoggingStrategy( Module.id, loggingMethods );
 62 	}
 63 );
