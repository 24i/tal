 29 require.def('antie/devices/broadcastsource/samsungtvsource',
 30     [
 31         'antie/devices/browserdevice',
 32         'antie/devices/broadcastsource/basetvsource',
 33         'antie/runtimecontext',
 34         'antie/devices/broadcastsource/channel',
 35         'antie/events/tunerunavailableevent',
 36         'antie/events/tunerstoppedevent',
 37         'antie/events/tunerpresentingevent'
 38     ],
 39     function (Device, BaseTvSource, RuntimeContext, Channel, TunerUnavailableEvent, TunerStoppedEvent, TunerPresentingEvent) {
 40         'use strict';
 41 
 42         /**
 43          * Contains a Samsung Maple implementation of the antie broadcast TV source.
 44          * @see http://www.samsungdforum.com/Guide/ref00014/PL_WINDOW_SOURCE.html
 45          * @class
 46          * @name antie.devices.broadcastsource.SamsungTVSource
 47          * @extends antie.devices.broadcastsource.BaseTVSource
 48          */
 49         var PL_WINDOW_SOURCE_TV = 0; // The TV source
 50         var PL_WINDOW_SOURCE_MEDIA = 43; // The media source
 51 
 52         var PL_TV_EVENT_NO_SIGNAL = 101;
 53         var PL_TV_EVENT_SOURCE_CHANGED = 114;
 54         var PL_TV_EVENT_TUNE_SUCCESS = 103;
 55 
 56         var WINDOW_PLUGIN_DOM_ELEMENT_ID = "pluginObjectWindow";
 57         var TV_PLUGIN_DOM_ELEMENT_ID = "pluginObjectTV";
 58 
 59         var SamsungSource = BaseTvSource.extend(/** @lends antie.devices.broadcastsource.SamsungTVSource.prototype */ {
 60             /**
 61              * @constructor
 62              * @ignore
 63              */
 64             init: function () {
 65                 this._samsungSefWindowPlugin = document.getElementById(WINDOW_PLUGIN_DOM_ELEMENT_ID);
 66 
 67                 if (!this._samsungSefWindowPlugin) {
 68                     throw new Error('Unable to initialise Samsung broadcast source');
 69                 }
 70 
 71                 var tvPlugin = document.getElementById(TV_PLUGIN_DOM_ELEMENT_ID);
 72 
 73                 var self = this;
 74 
 75                 tvPlugin.OnEvent = function(id) {
 76                     switch (parseInt(id)) {
 77                         case PL_TV_EVENT_NO_SIGNAL:
 78                             var unavailableEvent = new TunerUnavailableEvent();
 79                             RuntimeContext.getCurrentApplication().broadcastEvent(unavailableEvent);
 80                             break;
 81                         case PL_TV_EVENT_SOURCE_CHANGED:
 82                             var stoppedEvent = new TunerStoppedEvent();
 83                             RuntimeContext.getCurrentApplication().broadcastEvent(stoppedEvent);
 84                             break;
 85                         case PL_TV_EVENT_TUNE_SUCCESS:
 86                             var presentingEvent = new TunerPresentingEvent(self.getCurrentChannel());
 87                             RuntimeContext.getCurrentApplication().broadcastEvent(presentingEvent);
 88                             break;
 89                     }
 90                 };
 91 
 92                 tvPlugin.SetEvent(PL_TV_EVENT_TUNE_SUCCESS);
 93                 tvPlugin.SetEvent(PL_TV_EVENT_NO_SIGNAL);
 94                 tvPlugin.SetEvent(PL_TV_EVENT_SOURCE_CHANGED);
 95 
 96                 this._setBroadcastToFullScreen();
 97             },
 98             /**
 99              * Sets the current source to TV broadcast
100              * @see http://www.samsungdforum.com/Guide/ref00014/SetSource.html
101              */
102             showCurrentChannel: function () {
103                 this._samsungSefWindowPlugin.SetSource(PL_WINDOW_SOURCE_TV);
104             },
105             /**
106              * Sets the current source to the media source
107              * @see http://www.samsungdforum.com/Guide/ref00014/SetSource.html
108              */
109             stopCurrentChannel: function() {
110                 this._samsungSefWindowPlugin.SetSource(PL_WINDOW_SOURCE_MEDIA);
111             },
112             /**
113              * Gets the service name of the current channel
114              * @see http://www.samsungdforum.com/Guide/ref00014/getcurrentchannel_servicename.html
115              */
116             getCurrentChannelName: function () {
117                 var channelName = this._samsungSefWindowPlugin.GetCurrentChannel_Name();
118                 if (channelName < 0) {
119                     throw new Error('Channel name returned error code: ' + channelName);
120                 }
121                 return channelName;
122             },
123             getCurrentChannel : function() {
124                 try {
125                     var mapleChannel = webapis.tv.channel.getCurrentChannel();
126                     return this._createChannelFromMapleChannel(mapleChannel);
127 
128                 } catch (e) {
129                     return false;
130                 }
131             },
132             getChannelList: function (params) {
133 
134                 var self = this;
135 
136                 var createChannelList = function(mapleChannels) {
137                     var result = [];
138                     for (var i = 0; i < mapleChannels.length; i++) {
139                         result.push(self._createChannelFromMapleChannel(mapleChannels[i]));
140                     }
141                     params.onSuccess(result);
142                 };
143 
144                 try {
145                     var onFailedToRetrieveChannelList = function () {
146                         params.onError({
147                             name : "ChannelListError",
148                             message : "Channel list is not available"
149                         });
150                     };
151 
152                     webapis.tv.channel.getChannelList(createChannelList, onFailedToRetrieveChannelList,
153                                                         webapis.tv.channel.NAVIGATOR_MODE_ALL, 0, 1000000);
154 
155                 } catch (error) {
156 
157                     params.onError({
158                         name : "ChannelListError",
159                         message : "Channel list is empty or not available"
160                     });
161                 }
162             },
163             /**
164              * Sets the size and position of the visible broadcast source
165              * @param top
166              * @param left
167              * @param width
168              * @param height
169              */
170             setPosition : function(top, left, width, height) {
171                 this._samsungSefWindowPlugin.SetScreenRect(left, top, width, height);
172             },
173             /**
174              * Reverts the current screen settings and performs any clean up required before
175              * the user exits the application back to standard broadcast.
176              * @see http://www.samsungdforum.com/Guide/tec00103/index.html
177              */
178             destroy : function() {
179                 this.setPosition(0, -1, 0, 0);
180             },
181             setChannel : function(params) {
182                 this._tuneChannelByTriplet(params);
183             },
184             setChannelByName : function(params) {
185                 var currentChannel = this.getCurrentChannel();
186                 if (!currentChannel) {
187                     params.onError({
188                         name : "ChannelError",
189                         message: "Unable to determine current channel name"
190                     });
191 
192                 } else if (currentChannel.name === params.channelName) {
193                     this.showCurrentChannel();
194                     params.onSuccess();
195 
196                 } else {
197 
198                     this._tuneToChannelMatchingSpec({
199                             spec: { name: params.channelName },
200                             onError: params.onError,
201                             onSuccess: params.onSuccess
202                         }
203                     );
204                 }
205             },
206             _setBroadcastToFullScreen : function() {
207                 var currentLayout = RuntimeContext.getCurrentApplication().getLayout().requiredScreenSize;
208                 this.setPosition(0, 0, currentLayout.width, currentLayout.height);
209             },
210             /**
211              * @see http://img-developer.samsung.com/onlinedocs/samsung_webapi_guide/html/index.html
212              */
213             _tuneChannelByTriplet : function(params) {
214 
215                 this._tuneToChannelMatchingSpec({
216                     spec: {
217                         // FIXME: ONID is always reported as 65535, so we exclude it from the check in order
218                         // to prevent tuning information passed in (e.g. from MHEG) from causing the match
219                         // to fail.
220                         tsid: params.tsid,
221                         sid: params.sid
222                         },
223                     onError: params.onError,
224                     onSuccess: params.onSuccess
225                 });
226 
227             },
228             _tuneToChannelMatchingSpec : function (params) {
229 
230                 var self = this;
231 
232                 var onChannelListRetrieved = function (channels) {
233 
234                     var channel;
235 
236                     for (var i = 0; i < channels.length; i++) {
237                         if (self._channelMatchesSpec(channels[i], params.spec)) {
238                             channel = channels[i];
239                             break;
240                         }
241                     }
242 
243                     if (channel) {
244                         self._tuneToChannel({
245                             channel: channel,
246                             onError: params.onError,
247                             onSuccess: params.onSuccess
248                         });
249 
250                     } else {
251                         params.onError({
252                             name : "ChannelError",
253                             message : "Channel could not be found"
254                         });
255                     }
256                 };
257 
258                 this.getChannelList({
259                     onError: params.onError,
260                     onSuccess: onChannelListRetrieved
261                 });
262             },
263             _channelMatchesSpec: function(channel, spec) {
264                 var result = true;
265                 for (var prop in spec) {
266                     if (!spec.hasOwnProperty(prop)) {
267                         // Don't match on inherited properties (e.g. length)
268                         continue;
269                     }
270                     if (channel[prop] !== spec[prop]) {
271                         // Spec not matched
272                         result = false;
273                         break;
274                     }
275                 }
276                 return result;
277             },
278             /**
279              * @param params.channel Channel object
280              * @param params.onError Function to call with a string message on error.
281              * @param params.onSuccess Function to call on success.
282              * @private
283              */
284             _tuneToChannel: function(params) {
285 
286                 var channel = params.channel;
287 
288                 var newChannelArgs = {
289                     sourceID: channel.sourceId,
290                     programNumber: channel.sid,
291                     transportStreamID: channel.tsid,
292                     originalNetworkID: channel.onid,
293                     ptc: channel.ptc,
294                     major: channel.major,
295                     minor: channel.minor
296                 };
297 
298                 try {
299                     var tuneError = function () {
300                         params.onError({
301                             name : "ChangeChannelError",
302                             message : "Error tuning channel"
303                         });
304                     };
305 
306                     // Last argument is the Window ID.
307                     // See http://www.samsungdforum.com/Guide/ref00008/tvchannel/dtv_tvchannel_module.html
308                     webapis.tv.channel.tune(newChannelArgs, params.onSuccess, tuneError, 0);
309 
310                 } catch (e) {
311                     params.onError({
312                         name : "ChangeChannelError",
313                         message : "Error tuning channel"
314                     });
315                 }
316 
317             },
318             _createChannelFromMapleChannel: function (mapleChannel) {
319                 return new Channel({
320                     name: mapleChannel.channelName,
321                     onid: mapleChannel.originalNetworkID,
322                     tsid: mapleChannel.transportStreamID,
323                     sid: mapleChannel.programNumber,
324                     idType: undefined,
325                     ptc: mapleChannel.ptc,
326                     major: mapleChannel.major,
327                     minor: mapleChannel.minor,
328                     sourceId: mapleChannel.sourceID
329                 });
330             }
331         });
332 
333         /**
334          * Create a new widget giving control over broadcast television. Check whether
335          * the broadcast television API is available first with isBroadcastSourceSupported().
336          * @see antie.widgets.broadcastsource
337          * @returns {Object} Device-specific implementation of antie.widgets.broadcastsource
338          */
339         Device.prototype.createBroadcastSource = function() {
340             if (!this._broadcastSource) {
341                 this._broadcastSource = new SamsungSource();
342             }
343 
344             return this._broadcastSource;
345         };
346 
347         // Return for testing purposes only
348         return SamsungSource;
349     }
350 );
