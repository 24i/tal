 27 require.def('antie/devices/broadcastsource/samsungtvsource',
 28     [
 29         'antie/devices/browserdevice',
 30         'antie/devices/broadcastsource/basetvsource',
 31         'antie/application'
 32     ],
 33     function (Device, BaseTvSource, Application) {
 34         /**
 35          * Contains a Samsung Maple implementation of the antie broadcast TV source.
 36          * @see http://www.samsungdforum.com/Guide/ref00014/PL_WINDOW_SOURCE.html
 37          */
 38         var PL_WINDOW_SOURCE_TV = 0; // The TV source
 39         var PL_WINDOW_SOURCE_MEDIA = 43; // The media source
 40 
 41         var DOM_ELEMENT_ID = "pluginObjectWindow";
 42         var SamsungSource = BaseTvSource.extend(/** @lends antie.devices.broadcastsource.SamsungSource.prototype */ {
 43             /**
 44              * @constructor
 45              * @ignore
 46              */
 47             init: function () {
 48                 this._samsungSefWindowPlugin = document.getElementById(DOM_ELEMENT_ID);
 49 
 50                 if (!this._samsungSefWindowPlugin) {
 51                     throw new Error('Unable to initialise Samsung broadcast source');
 52                 }
 53 
 54                 this._setBroadcastToFullScreen();
 55             },
 56             /**
 57              * Sets the current source to TV broadcast
 58              * @see http://www.samsungdforum.com/Guide/ref00014/SetSource.html
 59              */
 60             showCurrentChannel: function () {
 61                 this._samsungSefWindowPlugin.SetSource(PL_WINDOW_SOURCE_TV);
 62             },
 63             /**
 64              * Sets the current source to the media source
 65              * @see http://www.samsungdforum.com/Guide/ref00014/SetSource.html
 66              */
 67             stopCurrentChannel: function() {
 68                 this._samsungSefWindowPlugin.SetSource(PL_WINDOW_SOURCE_MEDIA);
 69             },
 70             /**
 71              * Gets the service name of the current channel
 72              * @see http://www.samsungdforum.com/Guide/ref00014/getcurrentchannel_servicename.html
 73              */
 74             getCurrentChannelName: function () {
 75                 var channelName = this._samsungSefWindowPlugin.GetCurrentChannel_Name();
 76                 if (channelName < 0) {
 77                     throw new Error('Channel name returned error code: ' + channelName);
 78                 }
 79                 return channelName;
 80             },
 81             /**
 82              * Sets the size and position of the visible broadcast source
 83              * @param top
 84              * @param left
 85              * @param width
 86              * @param height
 87              */
 88             setPosition : function(top, left, width, height) {
 89                 this._samsungSefWindowPlugin.SetScreenRect(left, top, width, height);
 90             },
 91             /**
 92              * Reverts the current screen settings and performs any clean up required before
 93              * the user exits the application back to standard broadcast.
 94              * @see http://www.samsungdforum.com/Guide/tec00103/index.html
 95              */
 96             destroy : function() {
 97                 this.setPosition(0, -1, 0, 0);
 98             },
 99             _setBroadcastToFullScreen : function() {
100                 var currentLayout = Application.getCurrentApplication().getLayout().requiredScreenSize;
101                 this.setPosition(0, 0, currentLayout.width, currentLayout.height);
102             }
103         });
104 
105         /**
106          * Create a new widget giving control over broadcast television. Check whether
107          * the broadcast television API is available first with isBroadcastSourceSupported().
108          * @see antie.widgets.broadcastsource
109          * @returns {Object} Device-specific implementation of antie.widgets.broadcastsource
110          */
111         Device.prototype.createBroadcastSource = function() {
112             return new SamsungSource();
113         };
114     }
115 );
