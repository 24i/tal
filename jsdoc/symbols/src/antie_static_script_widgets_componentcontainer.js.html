 27 require.def('antie/widgets/componentcontainer',
 28     [
 29         'antie/widgets/container',
 30         'antie/events/componentevent'
 31     ],
 32     function (Container, ComponentEvent) {
 33         var _knownComponents = {};
 34         /**
 35          * The ComponentContainer widget class represents a container that Components (sections of UI) may be dynamically loaded into.
 36          * @name antie.widgets.ComponentContainer
 37          * @class
 38          * @extends antie.widgets.Container
 39          * @requires antie.events.ComponentEvent
 40          * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 41          */
 42         var ComponentContainer = Container.extend(/** @lends antie.widgets.ComponentContainer.prototype */ {
 43             /**
 44              * @constructor
 45              * @ignore
 46              */
 47             init: function (id) {
 48                 this._loadingIndex = 0;
 49                 this._loadingModule = null;
 50                 this._currentModule = null;
 51                 this._currentComponent = null;
 52                 this._currentArgs = null;
 53                 this._historyStack = [];
 54                 this._previousFocus = null;
 55                 this._super(id);
 56                 this.addClass('componentcontainer');
 57             },
 58             /**
 59              * Callback called when a requirejs containing a component has been loaded.
 60              * @private
 61              * @param {String} module The requirejs module name of the loaded component.
 62              * @param {Class} componentClass The subclass of antie.widgets.Component which has been loaded.
 63              * @param {Object} [args] The arguments passed to populate the component.
 64              */
 65             _loadComponentCallback: function (module, componentClass, args, keepHistory, state) {
 66                 if (this.getCurrentApplication() == null) {
 67                     // Application has been destroyed, abort
 68                     return;
 69                 }
 70 
 71                 var newComponent = new componentClass();
 72 
 73                 // Add the component to our table of known components.
 74                 _knownComponents[module] = newComponent;
 75 
 76                 // set the parent widget so the next event bubbles correctly through the tree
 77                 newComponent.parentWidget = this;
 78                 newComponent.bubbleEvent(new ComponentEvent('load', this, _knownComponents[module], args));
 79                 // clear the parent widget again
 80                 newComponent.parentWidget = null;
 81 
 82                 // Show the component.
 83                 this.show(module, args, keepHistory, state);
 84             },
 85             /**
 86              * Show a component within this container. If the specified component has not been loaded, load it.
 87              * @param {String} module The requirejs module name of the component to show.
 88              * @param {Object} [args] The arguments to populate the component.
 89              * @param {Boolean} [keepHistory] If true, the current component shown in this container is preserved in the history stack.
 90              * @param {Object} [state] Additional component-specific state information
 91              */
 92             show: function (module, args, keepHistory, state, fromBack, focus) {
 93                 this._loadingModule = module;
 94 
 95                 this._loadingIndex++;
 96                 var loadingIndex = this._loadingIndex;
 97 
 98                 var self;
 99                 if (_knownComponents[module]) {
100                     var device = this.getCurrentApplication().getDevice();
101 
102                     var _focus = this.getCurrentApplication().getFocussedWidget();
103                     if (this._currentComponent) {
104                         this.hide(null, args, keepHistory, state, fromBack);
105                     }
106 
107                     this._currentModule = module;
108                     this._currentComponent = _knownComponents[module];
109                     this._currentArgs = args;
110                     if (!fromBack) {
111                         this._previousFocus = _focus;
112                     }
113 
114                     if (!this._isFocussed) {
115                         // We don't have focus, so any of our children shouldn't
116                         // (_isFocussed state can be set to true if focussed widget is in a unloaded component)
117                         var p = this._currentComponent;
118                         while (p) {
119                             p.removeFocus();
120                             p = p._activeChildWidget;
121                         }
122                     }
123 
124                     // set the parent widget so the next event bubbles correctly through the tree
125                     this._currentComponent.parentWidget = this;
126 
127                     this._currentComponent.bubbleEvent(new ComponentEvent('beforerender', this, this._currentComponent, args, state, fromBack));
128 
129                     this._currentComponent.render(device);
130 
131                     // and clear it again
132                     this._currentComponent.parentWidget = null;
133 
134                     device.hideElement({
135                         el: this._currentComponent.outputElement,
136                         skipAnim: true
137                     });
138 
139                     this.appendChildWidget(this._currentComponent);
140 
141                     var evt = new ComponentEvent('beforeshow', this, this._currentComponent, args, state, fromBack);
142                     this._currentComponent.bubbleEvent(evt);
143 
144                     if (focus) {
145                         focus.focus();
146                     }
147 
148                     self = this;
149                     if (!evt.isDefaultPrevented()) {
150                         var config = device.getConfig();
151                         var animate = !config.widgets || !config.widgets.componentcontainer || (config.widgets.componentcontainer.fade !== false);
152                         device.showElement({
153                             el: this._currentComponent.outputElement,
154                             skipAnim: !animate
155                         });
156                     }
157 
158                     self._currentComponent.bubbleEvent(new ComponentEvent('aftershow', self, self._currentComponent, args, state, fromBack));
159 
160                     var focusRemoved = self.setActiveChildWidget(self._currentComponent);
161                     if (focusRemoved == false) {
162                         self._activeChildWidget = self._currentComponent;
163                         self.getCurrentApplication().getDevice().getLogger().warn('active component is not currently focusable', self._activeChildWidget);
164                     }
165                 } else {
166                     // hook into requirejs to load the component from the module and call us again
167                     self = this;
168                     require([module], function (componentClass) {
169                         // Check we've not navigated elsewhere whilst requirejs has been loading the module
170                         if (self._loadingModule === module && self._loadingIndex === loadingIndex) {
171                             self._loadComponentCallback(module, componentClass, args, keepHistory, state, fromBack, focus);
172                         }
173                     });
174                 }
175             },
176             /**
177              * Pushes a component into the history stack of the container (and shows it).
178              * @param {String} module The requirejs module name of the component to show.
179              * @param {Object} [args] An optional object to pass arguments to the component.
180              */
181             pushComponent: function (module, args) {
182                 this.show(module, args, true);
183             },
184             /**
185              * Returns the widget added to this container.
186              */
187             getContent: function () {
188                 return this._currentComponent;
189             },
190             /**
191              * Return this component container to the previous component in the history.
192              */
193             back: function () {
194                 var _focus = this._currentComponent.getIsModal() ? this._previousFocus : null;
195 
196                 var _lastComponent = this._historyStack.pop();
197                 if (_lastComponent) {
198                     this._previousFocus = _lastComponent.previousFocus;
199                     this.show(_lastComponent.module, _lastComponent.args, true, _lastComponent.state, true, _focus);
200                 } else {
201                     this.hide(null, null, false, null, false);
202                 }
203             },
204             /**
205              * Hide the component within this container.
206              */
207             hide: function (focusToComponent, args, keepHistory, state, fromBack) {
208                 var self = this;
209 
210                 if (this._currentComponent) {
211                     var evt = new ComponentEvent('beforehide', this, this._currentComponent, args, state, fromBack);
212                     this._currentComponent.bubbleEvent(evt);
213 
214                     var _state = keepHistory ? this._currentComponent.getCurrentState() : null;
215 
216                     // remove the child widget, but if default event is prevented, keep the output element in the DOM
217                     this.removeChildWidget(this._currentComponent, evt.isDefaultPrevented());
218 
219                     var _component = this._currentComponent;
220                     this._currentComponent = null;
221 
222                     // set the parent widget so the next event bubbles correctly through the tree
223                     _component.parentWidget = this;
224                     _component.bubbleEvent(new ComponentEvent('afterhide', this, _component, args, state, fromBack));
225                     // and clear it again
226                     _component.parentWidget = null;
227 
228                     if (keepHistory) {
229                         if (!fromBack) {
230                             this._historyStack.push({
231                                 module: this._currentModule,
232                                 args: this._currentArgs,
233                                 state: _state,
234                                 previousFocus: this._previousFocus
235                             });
236                         }
237                     } else {
238                         // Reset the history stack when a component is shown in this container without explicitly
239                         // enabling history.
240                         if (_component.getIsModal() && !fromBack) {
241                             if (this._historyStack.length > 0) {
242                                 this._historyStack[0].previousFocus.focus();
243                             } else if (this._previousFocus) {
244                                 this._previousFocus.focus();
245                             }
246                         }
247                         this._historyStack = [];
248                     }
249                 }
250                 if (this._isFocussed && focusToComponent) {
251                     this.parentWidget.setActiveChildWidget(this.parentWidget._childWidgets[focusToComponent]);
252                 }
253             },
254             getCurrentModule: function () {
255                 return this._currentModule;
256             },
257             getCurrentArguments: function () {
258                 return this._currentArgs;
259             }
260         });
261         ComponentContainer.destroy = function () {
262             for (var module in _knownComponents) {
263                 delete _knownComponents[module];
264             }
265         };
266         return ComponentContainer;
267     }
268 );
