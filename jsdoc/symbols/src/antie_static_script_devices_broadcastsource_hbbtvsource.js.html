 27 require.def('antie/devices/broadcastsource/hbbtvsource',
 28     [
 29         'antie/devices/browserdevice',
 30         'antie/devices/broadcastsource/basetvsource',
 31         'antie/application'
 32     ],
 33     function (Device, BaseTvSource, Application) {
 34         /**
 35          * Contains a HBBTV implementation of the antie broadcast TV source.
 36          */
 37         var DOM_ELEMENT_ID = 'broadcastVideoObject';
 38         var ID_DVB_T = 12;
 39         var HbbTVSource = BaseTvSource.extend(/** @lends antie.devices.broadcastsource.HbbTVSource.prototype */ {
 40             /**
 41              * @constructor
 42              * @ignore
 43              */
 44             init: function () {
 45                 this._broadcastVideoObject = document.getElementById(DOM_ELEMENT_ID);
 46 
 47                 if (!this._broadcastVideoObject) {
 48                     throw new Error('Unable to initialise HbbTV broadcast source');
 49                 }
 50 
 51                 // adding as instance rather then class var as module instantiated via method
 52                 this._playStates = {
 53                     UNREALIZED: 0,
 54                     CONNECTING: 1,
 55                     PRESENTING: 2,
 56                     STOPPED: 3
 57                 };
 58             },
 59             showCurrentChannel: function () {
 60                 // Check if exception is thrown by bindToCurrentChannel?
 61                 this._setBroadcastToFullScreen();
 62                 this._broadcastVideoObject.bindToCurrentChannel();
 63                 this._broadcastVideoObject.style.display = "block";
 64             },
 65             stopCurrentChannel: function () {
 66                 try {
 67                     if (this.getPlayState() === this._playStates.UNREALIZED) {
 68                         this._broadcastVideoObject.bindToCurrentChannel();
 69                     }
 70                 } catch(e) {
 71                     throw Error("Unable to bind to current channel");
 72                 }
 73 
 74                 this._broadcastVideoObject.stop();
 75                 this._broadcastVideoObject.style.display = "none";
 76             },
 77             getCurrentChannelName: function () {
 78                 var channelConfig = this._broadcastVideoObject.currentChannel;
 79                 // maybe check the hbbtv docs to check return type
 80                 if (channelConfig === null) {
 81                     throw new Error('Current channel name not available');
 82                 }
 83 
 84                 if (channelConfig.name === null) {
 85                     throw new Error('Current channel name is null');
 86                 }
 87 
 88                 if (typeof channelConfig.name === "undefined") {
 89                     throw new Error('Current channel name is not defined');
 90                 }
 91 
 92                 return channelConfig.name;
 93             },
 94             setPosition : function(top, left, width, height) {
 95                 this._broadcastVideoObject.style.top = top + "px";
 96                 this._broadcastVideoObject.style.left = left + "px";
 97                 this._broadcastVideoObject.style.width = width + "px";
 98                 this._broadcastVideoObject.style.height = height + "px";
 99             },
100             getPlayState : function() {
101                 return this._broadcastVideoObject.playState;
102             },
103             destroy : function() {
104                 // Not currently required for hbbtv
105             },
106             setChannel : function(params) {
107                 var self = this;
108                 var channelType;
109 
110                 // TODO: Clean this up
111                 // Attempt to get the channel type of the current channel
112                 // If this is not available, fall back to ID_DVB_T
113                 try {
114                     channelType = this._getChannelType();
115                 } catch(e) {
116                     channelType = ID_DVB_T;
117                 }
118                 var newChannel = this._broadcastVideoObject.createChannelObject(channelType, params.onid, params.tsid, params.sid);
119                 if (newChannel === null) {
120                     params.onError({
121                         name : "ChannelError",
122                         message : "Channel could not be found"
123                     });
124                     return;
125                 }
126 
127                 // Test the that the device can access the channelList, this will be required in the future
128                 try {
129                     var channelConfig = this._broadcastVideoObject.getChannelConfig();
130                     var channelList = channelConfig.channelList;
131                     var channelListCount = channelList.length;
132                 } catch(e) {
133                     params.onError({
134                         name : "ChannelListError",
135                         message : "Channel list is not available"
136                     });
137                 }
138 
139                 if (!channelListCount || channelListCount < 0) {
140                     params.onError({
141                         name : "ChannelListError",
142                         message : "Channel list is empty or not available"
143                     });
144                     return;
145                 }
146 
147                 var successEventListener = function(channel) {
148                     self._broadcastVideoObject.removeEventListener("ChannelChangeSucceeded", successEventListener);
149                     self._broadcastVideoObject.removeEventListener("ChannelChangeError", errorEventListener);
150                     params.onSuccess();
151                 };
152 
153                 var errorEventListener = function(channel, errorState) {
154                     self._broadcastVideoObject.removeEventListener("ChannelChangeSucceeded", successEventListener);
155                     self._broadcastVideoObject.removeEventListener("ChannelChangeError", errorEventListener);
156                     params.onError({
157                         name : "ChangeChannelError",
158                         message : "Error tuning channel"
159                     });
160                 };
161 
162                 this._broadcastVideoObject.addEventListener("ChannelChangeSucceeded", successEventListener);
163                 this._broadcastVideoObject.addEventListener("ChannelChangeError", errorEventListener);
164 
165                 this._broadcastVideoObject.setChannel(newChannel);
166             },
167             /**
168              * @Returns The type of identification for the channel, as indicated by one of the ID_* constants in the
169              * HBBTV specification
170              */
171             _getChannelType : function() {
172                 var channelType = this._broadcastVideoObject.currentChannel.idType;
173                 if (typeof channelType === "undefined") {
174                     channelType = ID_DVB_T;
175                 }
176                 return channelType;
177             },
178             /**
179              * Sets the size of the broadcast object to be the same as the required screen size identified by antie at
180              * application start up.
181              */
182             _setBroadcastToFullScreen : function() {
183                 var currentLayout = Application.getCurrentApplication().getLayout().requiredScreenSize;
184                 this.setPosition(0, 0, currentLayout.width, currentLayout.height);
185             }
186         });
187 
188         Device.prototype.isBroadcastSourceSupported = function() {
189             return this.getHistorian().hasBroadcastOrigin();
190         };
191 
192         /**
193          * Create a broadcastSource object on the Device to be
194          * accessed as a singleton to avoid the init being run
195          * multiple times
196          */
197         Device.prototype.createBroadcastSource = function() {
198             if (!this._broadcastSource) {
199                 this._broadcastSource = new HbbTVSource();
200             }
201 
202             return this._broadcastSource;
203         };
204 
205         // Return the HbbtvSource object for testing purposes
206         return HbbTVSource;
207     }
208 );
