 27 require.def('antie/devices/broadcastsource/hbbtvsource',
 28     [
 29         'antie/devices/browserdevice',
 30         'antie/devices/broadcastsource/basetvsource',
 31         'antie/runtimecontext',
 32         'antie/devices/broadcastsource/channel',
 33         'antie/events/tunerunavailableevent',
 34         'antie/events/tunerpresentingevent',
 35         'antie/events/tunerstoppedevent'
 36     ],
 37     function (Device, BaseTvSource, RuntimeContext, Channel, TunerUnavailableEvent, TunerPresentingEvent, TunerStoppedEvent ) {
 38         'use strict';
 39 
 40         /**
 41          * Contains a HBBTV implementation of the antie broadcast TV source.
 42          * @class
 43          * @name antie.devices.broadcastsource.HbbTVSource
 44          * @extends antie.devices.broadcastsource.BaseTVSource
 45          */
 46         var DOM_ELEMENT_ID = 'broadcastVideoObject';
 47         var ID_DVB_T = 12;
 48         var HbbTVSource = BaseTvSource.extend(/** @lends antie.devices.broadcastsource.HbbTVSource.prototype */ {
 49             /**
 50              * @constructor
 51              * @ignore
 52              */
 53             init: function () {
 54 
 55                 var self = this;
 56                 this._broadcastVideoObject = document.getElementById(DOM_ELEMENT_ID);
 57 
 58                 if (!this._broadcastVideoObject) {
 59                     throw new Error('Unable to initialise HbbTV broadcast source');
 60                 }
 61 
 62                 // adding as instance rather then class var as module instantiated via method
 63                 this._playStates = {
 64                     UNREALIZED: 0,
 65                     CONNECTING: 1,
 66                     PRESENTING: 2,
 67                     STOPPED: 3
 68                 };
 69 
 70                 this.playState = this._playStates.UNREALIZED;
 71 
 72                 this._broadcastVideoObject.addEventListener("PlayStateChange", function() {
 73 
 74                     var oldPlayState = self.playState;
 75                     // Note! The play state may change during the execution of this method; capture it so we
 76                     // have a consistent value for the duration of the event listener.
 77                     // See OIPF DAE specification section 7.14.9
 78                     var newPlayState = self.getPlayState();
 79 
 80                     if (oldPlayState === self._playStates.PRESENTING && newPlayState === self._playStates.UNREALIZED) {
 81                         RuntimeContext.getCurrentApplication().broadcastEvent(new TunerUnavailableEvent());
 82 
 83                     } else if (newPlayState === self._playStates.PRESENTING) {
 84                         RuntimeContext.getCurrentApplication().broadcastEvent(new TunerPresentingEvent(self.getCurrentChannel()));
 85 
 86                     } else if (newPlayState === self._playStates.STOPPED) {
 87                         RuntimeContext.getCurrentApplication().broadcastEvent(new TunerStoppedEvent());
 88                     }
 89 
 90                     self.playState = newPlayState;
 91 
 92                 });
 93             },
 94             showCurrentChannel: function () {
 95                 // Check if exception is thrown by bindToCurrentChannel?
 96                 this._setBroadcastToFullScreen();
 97                 this._broadcastVideoObject.bindToCurrentChannel();
 98                 this._broadcastVideoObject.style.visibility = "visible";
 99             },
100             stopCurrentChannel: function () {
101                 try {
102                     if (this.getPlayState() === this._playStates.UNREALIZED) {
103                         this._broadcastVideoObject.bindToCurrentChannel();
104                     }
105                 } catch(e) {
106                     throw new Error("Unable to bind to current channel");
107                 }
108 
109                 this._broadcastVideoObject.stop();
110                 this._broadcastVideoObject.style.visibility = "hidden";
111                 this.setPosition(0, 0, 0, 0);
112             },
113             getCurrentChannelName: function () {
114                 var channelConfig = this._broadcastVideoObject.currentChannel;
115                 // maybe check the hbbtv docs to check return type
116                 if (channelConfig === null) {
117                     throw new Error('Current channel name not available');
118                 }
119 
120                 if (channelConfig.name === null) {
121                     throw new Error('Current channel name is null');
122                 }
123 
124                 if (typeof channelConfig.name === "undefined") {
125                     throw new Error('Current channel name is not defined');
126                 }
127 
128                 return channelConfig.name;
129             },
130             getChannelList : function (params) {
131                 try {
132                     var result = this._getChannelList();
133                     params.onSuccess(result);
134 
135                 } catch (e) {
136                     params.onError(e);
137                 }
138             },
139             getCurrentChannel : function () {
140                 var result = false;
141 
142                 var currentChannel = this._broadcastVideoObject.currentChannel;
143 
144                 if (currentChannel) {
145                     result = new Channel({
146                         name : currentChannel.name,
147                         idType: currentChannel.idType,
148                         onid: currentChannel.onid,
149                         tsid: currentChannel.tsid,
150                         sid: currentChannel.sid
151                     });
152                 }
153 
154                 return result;
155             },
156             setChannelByName : function(params) {
157 
158                 var currentChannel = this.getCurrentChannel();
159 
160                 if (!currentChannel) {
161                     params.onError({
162                         name : "ChannelError",
163                         message: "Unable to determine current channel name"
164                     });
165 
166                 } else if (params.channelName === currentChannel.name) {
167                     this.showCurrentChannel();
168                     params.onSuccess();
169 
170                 } else {
171                     try {
172                         var channelList = this._getChannelList();
173                         var channel;
174 
175                         for (var i = 0; i < channelList.length; i++) {
176                             if (channelList[i].name === params.channelName) {
177                                 channel = channelList[i];
178                                 break;
179                             }
180                         }
181 
182                         if (!channel) {
183                             throw {
184                                 name : "ChannelError",
185                                 message: params.channelName + " not found in channel list"
186                             };
187                         }
188 
189                         this._tuneToChannelByTriplet(channel.idType, channel.onid, channel.tsid, channel.sid, params.onSuccess, params.onError);
190 
191                     } catch(e) {
192                         params.onError(e);
193                     }
194                 }
195             },
196             setPosition : function(top, left, width, height) {
197                 this._broadcastVideoObject.style.top = top + "px";
198                 this._broadcastVideoObject.style.left = left + "px";
199                 this._broadcastVideoObject.style.width = width + "px";
200                 this._broadcastVideoObject.style.height = height + "px";
201             },
202             getPlayState : function() {
203                 return this._broadcastVideoObject.playState;
204             },
205             destroy : function() {
206                 // Not currently required for hbbtv
207             },
208             setChannel : function(params) {
209                 var idType = this._getChannelIdType();
210                 this._tuneToChannelByTriplet(idType, params.onid, params.tsid, params.sid, params.onSuccess, params.onError);
211             },
212             _tuneToChannelByTriplet: function (idType, onid, tsid, sid, onSuccess, onError) {
213                 var newChannel = this._broadcastVideoObject.createChannelObject(idType, onid, tsid, sid);
214                 if (newChannel === null) {
215                     onError({
216                         name : "ChannelError",
217                         message : "Channel could not be found"
218                     });
219                 } else {
220                     this._tuneToChannelObject(newChannel, onSuccess, onError);
221                 }
222             },
223             _tuneToChannelObject: function (newChannel, onSuccess, onError) {
224                 var self = this;
225                 var successEventListener = function(/* channel */) {
226                     self._broadcastVideoObject.removeEventListener("ChannelChangeSucceeded", successEventListener);
227                     self._broadcastVideoObject.removeEventListener("ChannelChangeError", errorEventListener);
228                     onSuccess();
229                 };
230 
231                 var errorEventListener = function(/* channel, errorState */) {
232                     self._broadcastVideoObject.removeEventListener("ChannelChangeSucceeded", successEventListener);
233                     self._broadcastVideoObject.removeEventListener("ChannelChangeError", errorEventListener);
234                     onError({
235                         name : "ChangeChannelError",
236                         message : "Error tuning channel"
237                     });
238                 };
239 
240                 this._broadcastVideoObject.addEventListener("ChannelChangeSucceeded", successEventListener);
241                 this._broadcastVideoObject.addEventListener("ChannelChangeError", errorEventListener);
242 
243                 this._broadcastVideoObject.setChannel(newChannel);
244             },
245             /**
246              * @returns The type of the channel, as indicated by one of the ID_* constants in the HBBTV
247              *      specification. Defaults to ID_DVB_T in the case of error
248              *  @private
249              */
250             _getChannelIdType : function() {
251                 var idType;
252                 try {
253                     idType = this._broadcastVideoObject.currentChannel.idType;
254                     if (typeof idType === "undefined") {
255                         idType = ID_DVB_T;
256                     }
257                 } catch(e) {
258                     idType = ID_DVB_T;
259                 }
260 
261                 return idType;
262             },
263             /**
264              * Sets the size of the broadcast object to be the same as the required screen size identified by antie at
265              * application start up.
266              */
267             _setBroadcastToFullScreen : function() {
268                 var currentLayout = RuntimeContext.getCurrentApplication().getLayout().requiredScreenSize;
269                 this.setPosition(0, 0, currentLayout.width, currentLayout.height);
270             },
271             _getChannelList : function() {
272                 function getChannelFromChannelList (channelList, index) {
273                     if (typeof channelList.item === "function") {
274                         return channelList.item(index);
275                     } else {
276                         return channelList[index];
277                     }
278                 }
279 
280                 var channelConfig;
281 
282                 try {
283                     channelConfig = this._broadcastVideoObject.getChannelConfig();
284                 } catch (e) {
285                     throw {
286                         name : "ChannelListError",
287                         message : "Channel list is not available"
288                     };
289                 }
290 
291                 if (!channelConfig || !channelConfig.channelList || channelConfig.channelList.length === 0) {
292                     throw {
293                         name : "ChannelListError",
294                         message : "Channel list is empty or not available"
295                     };
296                 }
297 
298                 var result = [];
299                 for (var i = 0; i < channelConfig.channelList.length; i++) {
300                     var channel = getChannelFromChannelList(channelConfig.channelList, i);
301                     result.push(new Channel(
302                         {
303                             name: channel.name,
304                             idType: channel.idType,
305                             onid: channel.onid,
306                             sid: channel.sid,
307                             tsid: channel.tsid
308                         }
309                     ));
310                 }
311                 return result;
312             }
313         });
314 
315         Device.prototype.isBroadcastSourceSupported = function() {
316             return this.getHistorian().hasBroadcastOrigin();
317         };
318 
319         /**
320          * Create a broadcastSource object on the Device to be
321          * accessed as a singleton to avoid the init being run
322          * multiple times
323          */
324         Device.prototype.createBroadcastSource = function() {
325             if (!this._broadcastSource) {
326                 this._broadcastSource = new HbbTVSource();
327             }
328 
329             return this._broadcastSource;
330         };
331 
332         // Return the HbbtvSource object for testing purposes
333         return HbbTVSource;
334     }
335 );
