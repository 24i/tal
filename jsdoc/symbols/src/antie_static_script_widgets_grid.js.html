 27 require.def('antie/widgets/grid',
 28 	[
 29 	 	'antie/widgets/container',
 30 	 	'antie/events/keyevent',
 31 	 	'antie/events/selecteditemchangeevent'
 32 	],
 33 	function(Container, KeyEvent, SelectedItemChangeEvent) {
 34 		'use strict';
 35 
 36 		/**
 37 		 * The Grid widget class represents a grid of widgets that may be navigated between using up/down/left/right.
 38 		 * @name antie.widgets.Grid
 39 		 * @class
 40 		 * @extends antie.widgets.Container
 41 		 * @requires antie.events.KeyEvent
 42 		 * @requires antie.events.SelectedItemChangeEvent
 43 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 44 		 * @param {Integer} cols The number of columns in the grid.
 45 		 * @param {Integer} rows The number of rows in the grid.
 46 		 */
 47 		var Grid = Container.extend(/** @lends antie.widgets.Grid.prototype */ {
 48 			/**
 49 			 * @constructor
 50 			 * @ignore
 51 			 */
 52 			init: function(id, cols, rows) {
 53 				this._super(id);
 54 				this.addClass('grid');
 55 				this._cols = cols;
 56 				this._rows = rows;
 57 
 58 				this._selectedRow = 0;
 59 				this._selectedCol = 0;
 60 
 61 				for(var i=0; i<cols * rows; i++) this._childWidgetOrder.push(null);
 62 
 63 				var self = this;
 64 				this.addEventListener('keydown', function(e) { self._onKeyDown(e); });
 65 			},
 66 			/**
 67 			 * Get the widget positioned at the specified column and row.
 68 			 * @param {Integer} col The column the widget it in
 69 			 * @param {Integer} row The row the widget it in
 70 			 * @returns The widget in the specified column and row
 71 			 */
 72 			getWidgetAt: function(col, row) {
 73 				return this._childWidgetOrder[(this._cols * row) + col];
 74 			},
 75 			/**
 76 			 * Positions a widget at the specified column and row.
 77 			 * @param {Integer} col The column to position the widget in
 78 			 * @param {Integer} row The row to position the widget in
 79 			 * @returns The widget to position
 80 			 */
 81 			setWidgetAt: function(col, row, widget) {
 82 				if(!this.hasChildWidget(widget.id)) {
 83 					this._childWidgets[widget.id] = widget;
 84 					this._childWidgetOrder[(this._cols * row) + col] = widget;
 85 					widget.parentWidget = this;
 86 
 87 					// If there's no active child widget set, try and set it to this
 88 					// (Will only have an affect if it's focusable (i.e. contains a button))
 89 					if(this._activeChildWidget == null) {
 90 						this.setActiveChildWidget(widget);
 91 					}
 92 
 93 					if(this.outputElement && this._autoRenderChildren) {
 94 						var device = this.getCurrentApplication().getDevice();
 95 
 96 						if(!widget.outputElement) {
 97 							widget.render(device);
 98 						}
 99 
100 						device.appendChildElement(this.outputElement, widget.outputElement);
101 					}
102 				}
103 			},
104 			/**
105 			 * Renders the widget and any child widgets to device-specific output.
106 			 * @param {antie.devices.Device} device The device to render to.
107 			 * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
108 			 */
109 			render: function(device) {
110 				if(!this.outputElement) {
111 					this.outputElement = device.createContainer(this.id, this.getClasses());
112 				} else {
113 					device.clearElement(this.outputElement);
114 				}
115 
116 				var rowElement;
117 				for(var row=0; row<this._rows; row++) {
118 					rowElement = device.createContainer(this.id + "_row_" + row, ["row"]);
119 					for(var col=0; col<this._cols; col++) {
120 						var widget = this.getWidgetAt(col, row);
121 						if(widget) {
122 							if(col == 0) widget.addClass('firstcol');
123 							else if(col == this._cols - 1) widget.addClass('lastcol');
124 							device.appendChildElement(rowElement, this.getWidgetAt(col, row).render(device));
125 						} else {
126 							var classes = ['spacer'];
127 							if(col == 0) classes.push('firstcol');
128 							else if(col == this._cols - 1) classes.push('lastcol');
129 							device.appendChildElement(rowElement, device.createContainer(this.id + "_" + col + "_" + row, classes));
130 						}
131 					}
132 					device.appendChildElement(this.outputElement, rowElement);
133 				}
134 
135 				return this.outputElement;
136 			},
137 			/**
138 			 * Appends a child widget to this widget. Not supported for Grids.
139 			 * @param {antie.widgets.Widget} widget The child widget to add.
140 			 */
141 			appendChildWidget: function(widget) {
142 				throw new Error("Not supported");
143 			},
144 			/**
145 			 * Inserts a child widget at the specified index. Not supported for Grids.
146 			 * @param {Integer} index The index where to insert the child widget.
147 			 * @param {antie.widgets.Widget} widget The child widget to add.
148 			 */
149 			insertChildWidget: function(index, widget) {
150 				throw new Error("Not supported");
151 			},
152 			/**
153 			 * Removes a specific child widget from this widget. Not supported for Grids.
154 			 * @param {antie.widgets.Widget} widget The child widget to remove.
155 			 * @param {Boolean} [retainElement] Pass <code>true</code> to retain the child output element of the given widget
156 			 */
157 			removeChildWidget: function(widget, retainElement) {
158 				throw new Error("Not supported");
159 			},
160 			/**
161 			 * Attempt to set focus to the given child widget.
162 			 *
163 			 * Note: You can only set focus to a focusable widget. A focusable widget is one that
164 			 * contains an enabled antie.widgets.Button as either a direct or indirect child.
165 			 *
166 			 * Note: Widgets have 2 independant states: active and focussed. A focussed widget is
167 			 * either the Button with focus, or any parent of that Button. An active widget is
168 			 * one which is the active child of its parent Container. When the parent widget
169 			 * receives focus, focus will be placed on the active child.
170 			 *
171 			 * Classes 'active' and 'focus' are appended to widgets with these states.
172 			 * 
173 			 * @param {antie.widgets.Widget} widget The child widget to set focus to.
174 			 * @returns Boolean true if the child widget was focusable, otherwise boolean false.
175 			 */
176 			setActiveChildWidget: function(widget) {
177 				var changed = this._activeChildWidget != widget;
178 				if(this._super(widget)) {
179 					var selectedIndex = this.getIndexOfChildWidget(widget);
180 					this._selectedRow = Math.floor(selectedIndex / this._cols);
181 					this._selectedCol = Math.floor(selectedIndex % this._cols);
182 
183 					if(changed) {
184 						this.bubbleEvent(new SelectedItemChangeEvent(this, widget, selectedIndex));
185 					}
186 					return true;
187 				} else {
188 					return false;
189 				}
190 			},
191 			/**
192 			 * Key handler for grids. Processes KeyEvent.VK_UP, VK_DOWN, VK_LEFT and VK_RIGHT keys and stops propagation
193 			 * if the keypress is handled. Otherwise allows the event to be bubbled up to the parent widget to allow
194 			 * spatial navigation out of the list.
195 			 * @param {antie.events.KeyEvent} evt The key event.
196 			 */
197 			_onKeyDown: function(evt) {
198 				if(evt.keyCode != KeyEvent.VK_UP && evt.keyCode != KeyEvent.VK_DOWN
199 					&& evt.keyCode != KeyEvent.VK_LEFT && evt.keyCode != KeyEvent.VK_RIGHT) {
200 					return;
201 				}
202 
203 				var _newSelectedCol = this._selectedCol;
204 				var _newSelectedRow = this._selectedRow;
205 				var _newSelectedWidget = null;
206 				do {
207 					if(evt.keyCode == KeyEvent.VK_UP) {
208 						_newSelectedRow--;
209 					} else if(evt.keyCode == KeyEvent.VK_DOWN) {
210 						_newSelectedRow++;
211 					} else if(evt.keyCode == KeyEvent.VK_LEFT) {
212 						_newSelectedCol--;
213 					} else if(evt.keyCode == KeyEvent.VK_RIGHT) {
214 						_newSelectedCol++;
215 					}
216 					if(_newSelectedCol < 0 || _newSelectedCol >= this._cols) {
217 						break;
218 					}
219 					if(_newSelectedRow < 0 || _newSelectedRow >= this._rows) {
220 						break;
221 					}
222 
223 					var _newSelectedIndex = (_newSelectedRow * this._cols) + _newSelectedCol;
224 					var _widget = this._childWidgetOrder[_newSelectedIndex];
225 					if(_widget && _widget.isFocusable()) {
226 						_newSelectedWidget = _widget;
227 						break;
228 					}
229 				} while(true);
230 
231 				if(_newSelectedWidget) {
232 					this.setActiveChildWidget(_newSelectedWidget);
233 					evt.stopPropagation();
234 
235 				}
236 			},
237 
238 			/**
239 			 * Broadcasts an event from the application level to every single
240 			 * object it contains.
241 			 */
242 			broadcastEvent: function(evt) {
243 				this.fireEvent(evt);
244 				if(!evt.isPropagationStopped()) {
245 					for(var i=0; i<this._childWidgetOrder.length; i++) {
246 						// Grids are the only type of container that may contain
247 						// null entries in the _childWidgetOrder array
248 						if(this._childWidgetOrder[i]) {
249 							this._childWidgetOrder[i].broadcastEvent(evt);
250 						}
251 					}
252 				}
253 			}
254 		});
255 
256 		return Grid;
257 	}
258 );
