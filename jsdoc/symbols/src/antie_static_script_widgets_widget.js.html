 27 /**
 28  * User interface widgets.
 29  * @name antie.widgets.Widget
 30  * @namespace
 31  */
 32 
 33 require.def('antie/widgets/widget',
 34     [
 35         'antie/class',
 36         'require',
 37         'antie/application'
 38     ],
 39     function(Class, require, Application) {
 40         /**
 41          * Keep a count of generated IDs so we can ensure they're always unique
 42          * @private
 43          */
 44         var widgetUniqueIDIndex = 0;
 45 
 46         /**
 47          * The base widget class. A widget is a UI-component which can be rendered to device-specific output
 48          * via a {@link antie.devices.Device} object.
 49          * @name antie.widgets.Widget
 50          * @class
 51          * @abstract
 52          * @extends antie.Class
 53          * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 54          */
 55         return Class.extend(/** @lends antie.widgets.Widget.prototype */ {
 56             /**
 57              * @constructor
 58              * @ignore
 59              */
 60             init: function(id) {
 61                 this._classNames = {"widget":true};
 62                 this.parentWidget = null;
 63                 this.outputElement = null;
 64                 this._eventListeners = {};
 65                 this._dataItem = null; 	// Any data item bound to this widget
 66                 this._isFocussed = false;
 67 
 68                 function createUniqueID() {
 69                     return "#" + (new Date() * 1) + "_" + (widgetUniqueIDIndex++);
 70                 }
 71 
 72                 ;
 73 
 74                 // ensure all widgets have an ID
 75                 this.id = id ? id : createUniqueID();
 76             },
 77             /**
 78              * Renders the widget to device-specific output.
 79              * @param {antie.devices.Device} device The device to render to.
 80              * @returns A device-specific object that represents the widget as displayed on the device (in a browser, a DOMElement);
 81              */
 82             render: function(device) {
 83                 throw new Error("Widget::render called - the subclass for widget '" + this.id + "' must have not overridden the render method.");
 84             },
 85             /**
 86              * Adds a CSS class to the widget if not already present.
 87              * @param {String} className The class name to add.
 88              */
 89             addClass: function(className) {
 90                 if (!this._classNames[className]) {
 91                     this._classNames[className] = true;
 92                     if (this.outputElement) {
 93                         var device = this.getCurrentApplication().getDevice();
 94                         device.setElementClasses(this.outputElement, this.getClasses());
 95                     }
 96                 }
 97             },
 98             /**
 99              * Removes a CSS class from the widget if present.
100              * @param {String} className The class name to remove.
101              */
102             removeClass: function(className) {
103                 if (this._classNames[className]) {
104                     delete(this._classNames[className]);
105                     if (this.outputElement) {
106                         var device = this.getCurrentApplication().getDevice();
107                         device.setElementClasses(this.outputElement, this.getClasses());
108                     }
109                 }
110             },
111             /**
112              * Checks to see if the widget has a given CSS class.
113              * @param {String} className The class name to check.
114              * @returns Boolean true if the device has the className. Otherwise boolean false.
115              */
116             hasClass: function(className) {
117                 return (this._classNames[className] ? true : false);
118             },
119             /**
120              * Get an array of class names that this widget has.
121              * @returns An array of class names (Strings)
122              */
123             getClasses: function() {
124                 var _names = [];
125                 for (var i in this._classNames) {
126                     _names.push(i);
127                 }
128                 return _names;
129             },
130             /**
131              * Add an event listener function to this widget.
132              * @param {String} ev The event type to listen for (e.g. <code>keydown</code>)
133              * @param {Function} func The handler to be called when the event is fired.
134              * @see antie.events.Event
135              */
136             addEventListener: function(ev, func) {
137                 var listeners = this._eventListeners[ev];
138                 if (!listeners) {
139                     listeners = this._eventListeners[ev] = {};
140                 }
141                 if (!listeners[func]) {
142                     listeners[func] = func;
143                 }
144             },
145             /**
146              * Removes an event listener function to this widget.
147              * @param {String} ev The event type that the listener is to be removed from (e.g. <code>keydown</code>)
148              * @param {Function} func The handler to be removed.
149              * @see antie.events.Event
150              */
151             removeEventListener: function(ev, func) {
152                 var listeners = this._eventListeners[ev];
153                 if (listeners && listeners[func]) {
154                     delete(listeners[func]);
155                 }
156             },
157             /**
158              * Fires an event on this object, triggering any event listeners bound to this widget only.
159              * Note: this does not bubble or propagate the event to other widgets, for that functionality
160              * see {@link #bubbleEvent}.
161              * @param {antie.events.Event} ev The event to fire.
162              * @see antie.events.Event
163              */
164             fireEvent: function(ev) {
165                 var listeners = this._eventListeners[ev.type];
166                 if (listeners) {
167                     for (var func in listeners) {
168                         try {
169                             listeners[func](ev);
170                         } catch (exception) {
171                             var logger = this.getCurrentApplication().getDevice().getLogger();
172                             logger.error("Error in " + ev.type + " event listener on widget " + this.id + ": " + exception.message, exception, listeners[func]);
173                         }
174                     }
175                 }
176             },
177             /**
178              * Bubbles an event from object, triggering any event listeners bound to this widget and any
179              * parent widgets.
180              * To halt bubbling of the event, see {@link antie.events.Event#stopPropagation}.
181              * @param {antie.events.Event} ev The event to bubble.
182              * @see antie.events.Event
183              */
184             bubbleEvent: function(ev) {
185                 this.fireEvent(ev);
186                 if (!ev.isPropagationStopped()) {
187                     if (this.parentWidget != null) {
188                         this.parentWidget.bubbleEvent(ev);
189                     } else {
190                         ev.stopPropagation();
191                     }
192 
193                 }
194             },
195 
196             /**
197              * Broadcast an event from object, triggering any event listeners bound to this widget and any
198              * parent widgets.
199              * To halt bubbling of the event, see {@link antie.events.Event#stopPropagation}.
200              * @param {antie.events.Event} ev The event to bubble.
201              * @see antie.events.Event
202              */
203             broadcastEvent: function(ev) {
204                 this.fireEvent(ev);
205             },
206 
207             /**
208              * Checks to see if a widget is focussable, i.e. contains an enabled button.
209              * @see antie.widgets.Button
210              */
211             isFocusable: function() {
212                 // a widget can receive focus if any of it's children or children-of-children are Buttons
213                 // We're not a button and we have no children, so we're not.
214                 return false;
215             },
216             /**
217              * Gets a reference to the application responsible for creating the widget.
218              * @see antie.Application
219              */
220             getCurrentApplication: function() {
221                 try {
222                     if (!Application) Application = require('antie/application');
223                     return Application.getCurrentApplication();
224                 } catch (ex) {
225                     return null;
226                 }
227             },
228             /**
229              * Get any data item associated with this widget.
230              */
231             getDataItem: function() {
232                 return this._dataItem;
233             },
234             /**
235              * Associate a data item with this widget.
236              * @param {object} dataItem Object to associate with this widget.
237              */
238             setDataItem: function(dataItem) {
239                 this._dataItem = dataItem;
240             },
241             /**
242              * Returns the component this widget is a descendant of
243              */
244             getComponent: function() {
245                 var Component = require('antie/widgets/component');
246                 var widget = this;
247                 while (widget && !(widget instanceof Component)) {
248                     widget = widget.parentWidget;
249                 }
250                 return widget;
251             },
252             /**
253              * Remove focus state from this widget.
254              */
255             removeFocus: function() {
256                 this.removeClass('focus');
257                 this._isFocussed = false;
258             },
259             /**
260              * Get if this widget is in the current focus path.
261              * @returns Boolean true if this widget is in the focus path, otherwise false.
262              */
263             isFocussed: function() {
264                 return this._isFocussed;
265             },
266             /**
267              * Shows a widget. If animation is enabled the widget will be faded in.
268              * @param {Object}    options Details of the element to be shown, with optional parameters.
269              * @param {Boolean} [options.skipAnim] By default the showing of the element will be animated (faded in). Pass <code>true</code> here to prevent animation.
270              * @param {Function} [options.onComplete] Callback function to be called when the element has been shown.
271              * @param {Number}    [options.fps=25] Frames per second for fade animation.
272              * @param {Number}    [options.duration=840] Duration of fade animation, in milliseconds (ms).
273              * @param {String}    [options.easing=linear] Easing style for fade animation.
274              * @returns Boolean true if animation was called, otherwise false
275              */
276             show : function(options) {
277                 if (this.outputElement) {
278                     options.el = this.outputElement;
279                     var device = this.getCurrentApplication().getDevice();
280                     device.showElement(options);
281                 } else {
282                     throw new Error("Widget::show called - the current widget has not yet been rendered.");
283                 }
284             },
285             /**
286              * Hides a widget. If animation is enabled the widget will be faded out of view.
287              * @param {Object}    options Details of the element to be shown, with optional parameters.
288              * @param {Boolean} [options.skipAnim] By default the showing of the element will be animated (faded in). Pass <code>true</code> here to prevent animation.
289              * @param {Function} [options.onComplete] Callback function to be called when the element has been shown.
290              * @param {Number}    [options.fps=25] Frames per second for fade animation.
291              * @param {Number}    [options.duration=840] Duration of fade animation, in milliseconds (ms).
292              * @param {String}    [options.easing=linear] Easing style for fade animation.
293              * @returns Boolean true if animation was called, otherwise false
294              */
295             hide : function(options) {
296                 if (this.outputElement) {
297                     options.el = this.outputElement;
298                     var device = this.getCurrentApplication().getDevice();
299                     device.hideElement(options);
300                 } else {
301                     throw new Error("Widget::hide called - the current widget has not yet been rendered.");
302                 }
303             },
304             /**
305              * Moves a widget so that its top-left corner is at the given position.
306              * @param {Object}    options Details of the element to be shown, with optional parameters.
307              * @param {Boolean} [options.skipAnim] By default the showing of the element will be animated (faded in). Pass <code>true</code> here to prevent animation.
308              * @param {Function} [options.onComplete] Callback function to be called when the element has been shown.
309              * @param {Number}    [options.fps=25] Frames per second for fade animation.
310              * @param {Number}    [options.duration=840] Duration of fade animation, in milliseconds (ms).
311              * @param {String}    [options.easing=linear] Easing style for fade animation.
312              * @returns Boolean true if animation was called, otherwise false
313              */
314             moveTo : function(options) {
315                 if (this.outputElement) {
316                     options.el = this.outputElement;
317                     var device = this.getCurrentApplication().getDevice();
318                     device.moveElementTo(options);
319                 } else {
320                     throw new Error("Widget::moveTo called - the current widget has not yet been rendered.");
321                 }
322             }
323         });
324     }
325 );
