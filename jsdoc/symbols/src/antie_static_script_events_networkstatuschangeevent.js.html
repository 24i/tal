 27 require.def('antie/events/networkstatuschangeevent',
 28 	['antie/events/event'],
 29 	function(Event) {
 30 		'use strict';
 31 
 32 		/**
 33 		 * Class of events raised when the network state of a device changes (e.g. it goes offline).
 34 		 * @name antie.events.NetworkStatusChangeEvent
 35 		 * @class
 36 		 * @extends antie.events.Event
 37 		 * @param {Integer} networkStatus The new network status.
 38 		 */
 39 		var NetworkStatusChangeEvent = Event.extend(/** @lends antie.events.FocusEvent.prototype */ {
 40 			/**
 41 			 * @constructor
 42 			 * @ignore
 43 			 */
 44 			init: function(networkStatus) {
 45 				this.networkStatus = networkStatus;
 46 				this._super("networkstatuschange");
 47 			}
 48 		});
 49 
 50 		/**
 51 		 * Device is offline.
 52 		 * @memberOf antie.events.NetworkStatusChangeEvent
 53 		 * @name NETWORK_STATUS_OFFLINE
 54 		 * @constant
 55 		 * @static
 56 		 */
 57 		NetworkStatusChangeEvent.NETWORK_STATUS_OFFLINE = 0;
 58 
 59 		/**
 60 		 * Device is online.
 61 		 * @memberOf antie.events.NetworkStatusChangeEvent
 62 		 * @name NETWORK_STATUS_ONLINE
 63 		 * @constant
 64 		 * @static
 65 		 */
 66 		NetworkStatusChangeEvent.NETWORK_STATUS_ONLINE = 1;
 67 
 68 		return NetworkStatusChangeEvent;
 69 	}
 70 );
