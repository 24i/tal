 24 require.def('antie/widgets/carousel/strips/cullingstrip',
 25     [
 26         'antie/widgets/carousel/strips/widgetstrip',
 27         'antie/widgets/carousel/strips/utility/widgetcontext',
 28         'antie/widgets/carousel/strips/utility/states'
 29     ],
 30     function (WidgetStrip, WidgetContext, STATES) {
 31         'use strict';
 32         var CullingStrip;
 33         /**
 34          * A container for the widgets displayed within a carousel, in which widgets are only created when they
 35          * come into view, and where widgets are removed from the DOM when they go out of view
 36          * @name antie.widgets.carousel.strips.CullingStrip
 37          * @class
 38          * @extends antie.widgets.carousel.strips.WidgetStrip
 39          * @param {String} id The unique ID of the widget.
 40          * @param {Object} orientation an object representing the strip's orientation.
 41          * One of antie.widgets.carousel.orientations.Horizontal or antie.widgets.carousel.orientations.Vertical
 42          */
 43         CullingStrip = WidgetStrip.extend(/** @lends antie.widgets.carousel.strips.CullingStrip.prototype */{
 44             init: function (id, orientation) {
 45                 this._super(id, orientation);
 46                 this.setAutoRenderChildren(false);
 47                 this._widgetContexts = [];
 48             },
 49 
 50             append: function (widget, length) {
 51                 this._super(widget, length);
 52                 this._widgetContexts.push(this.createContext(widget, this));
 53             },
 54 
 55             render: function (device) {
 56                 var i, context;
 57                 if (!this.outputElement) {
 58                     this.outputElement = device.createContainer(this.id, this.getClasses());
 59                 } else {
 60                     device.clearElement(this.outputElement);
 61                 }
 62                 for (i = 0; i !== this._widgetContexts.length; i += 1) {
 63                     context = this._widgetContexts[i];
 64                     context.setState('INIT');
 65                 }
 66                 return this.outputElement;
 67             },
 68 
 69             insert: function (index, widget, length) {
 70                 this._super(index, widget, length);
 71                 this._widgetContexts.splice(index, 0, this.createContext(widget, this));
 72             },
 73 
 74             remove: function (widget) {
 75                 var i, widgets, returnValue;
 76                 widgets = this.widgets();
 77                 for (i = 0; i !== widgets.length; i += 1) {
 78                     if (widget === widgets[i]) {
 79                         this._widgetContexts.splice(i, 1);
 80                     }
 81                 }
 82                 returnValue = this._super(widget, false);
 83 
 84                 return returnValue;
 85             },
 86 
 87             removeAll: function () {
 88                 this._widgetContexts = [];
 89                 this._super();
 90             },
 91 
 92             needsVisibleIndices: function () {
 93                 return this._widgetContexts.length > 0;
 94             },
 95 
 96             attachIndexedWidgets: function (indexArray) {
 97                 var i, itemIndex, indexSet, firstIndexInView, preIndices, postIndices;
 98                 indexSet = {};
 99                 firstIndexInView = this._firstIndexInView();
100                 preIndices = [];
101                 postIndices = [];
102                 for (i = 0; i !== indexArray.length; i += 1) {
103                     itemIndex = indexArray[i];
104                     indexSet[itemIndex] = true;
105                     if (itemIndex < firstIndexInView) {
106                         preIndices.push(itemIndex);
107                     } else {
108                         postIndices.push(itemIndex);
109                     }
110                 }
111                 this._detatchWidgetsNotIndexed(indexSet);
112                 preIndices.reverse();
113                 for (i = 0; i !== preIndices.length; i += 1) {
114                     itemIndex = preIndices[i];
115                     this._widgetContexts[itemIndex].prepend();
116                 }
117                 for (i = 0; i !== postIndices.length; i += 1) {
118                     itemIndex = postIndices[i];
119                     this._widgetContexts[itemIndex].append();
120                 }
121             },
122 
123             /**
124              * Get the distance in pixels to a widget at the supplied index
125              * @param {Number} index An index of a widget currently in the carousel
126              * @returns {Number} length in pixels along primary axis to primary edge of the provided index
127              * i.e. from the left edge of the strip to the left edge of the widget in a horizontal carousel
128              * @throws {Error} if the indexed widget has not had a length set on append/insert or via setLengths
129              */
130             getLengthToIndex: function (index) {
131                 var firstAttached, i, length, totalLength;
132                 totalLength = 0;
133                 firstAttached = this._firstIndexWithLength();
134                 for (i = firstAttached; i < index; i += 1) {
135                     length = this._lengths[i];
136                     if (length === undefined) {
137                         this._throwNoLengthError();
138                     }
139                     totalLength += length;
140                 }
141                 return totalLength;
142             },
143 
144             /**
145              * Gets the length in pixels of the widget at the supplied index.
146              * @param {Number} index The index of a widget currently appended to the carousel. Supplied index must be valid (i.e. correspond to a widget currently in the strip with a provided length)
147              * @returns {Number} the length in pixels of the widget at the supplied index. Returns the length supplied at append or via setWidgetLength
148              * @throws {Error} if the indexed widget has not had a length set on append/insert or via setLengths
149              */
150             lengthOfWidgetAtIndex: function (index) {
151                 if (this._lengths[index] !== undefined) {
152                     return this._lengths[index];
153                 } else {
154                     this._throwNoLengthError();
155                 }
156             },
157 
158             createContext: function (widget, parent) {
159                 return new WidgetContext(widget, parent, STATES);
160             },
161 
162             _detatchWidgetsNotIndexed: function (indexSet) {
163                 var i;
164                 for (i = 0; i !== this._widgetContexts.length; i += 1) {
165                     if (!indexSet.hasOwnProperty(i)) {
166                         this._widgetContexts[i].detach();
167                     }
168                 }
169             },
170 
171             _firstIndexWithLength: function () {
172                 var i, attached, firstAttachedIndex;
173                 i = 0;
174                 attached = false;
175                 while (i < this._widgetContexts.length && attached === false) {
176                     attached = this._widgetContexts[i].hasLength();
177                     firstAttachedIndex = i;
178                     i += 1;
179                 }
180                 return firstAttachedIndex;
181             },
182 
183             _firstIndexInView: function () {
184                 var i, inView, firstInView;
185                 i = 0;
186                 inView = false;
187                 while (i < this._widgetContexts.length && inView === false) {
188                     inView = this._widgetContexts[i].inView();
189                     firstInView = i;
190                     i += 1;
191                 }
192                 return firstInView;
193             },
194 
195             _throwNoLengthError: function () {
196                 throw new Error("You must set widget lengths before aligning culling strip");
197             }
198         });
199         return CullingStrip;
200     }
201 );
