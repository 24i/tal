1 require.def('antie/widgets/carousel/strips/cullingstrip',
  2     [
  3         'antie/widgets/carousel/strips/widgetstrip',
  4         'antie/widgets/carousel/strips/utility/widgetcontext',
  5         'antie/widgets/carousel/strips/utility/states'
  6     ],
  7     function (WidgetStrip, WidgetContext, STATES) {
  8         'use strict';
  9         var CullingStrip;
 10         CullingStrip = WidgetStrip.extend({
 11             init: function (id, orientation) {
 12                 this._super(id, orientation);
 13                 this.setAutoRenderChildren(false);
 14                 this._widgetContexts = [];
 15             },
 16 
 17             append: function (widget, length) {
 18                 this._super(widget, length);
 19                 this._widgetContexts.push(this.createContext(widget, this));
 20             },
 21 
 22             render: function (device) {
 23                 var i, context;
 24                 if (!this.outputElement) {
 25                     this.outputElement = device.createContainer(this.id, this.getClasses());
 26                 } else {
 27                     device.clearElement(this.outputElement);
 28                 }
 29                 for (i = 0; i !== this._widgetContexts.length; i += 1) {
 30                     context = this._widgetContexts[i];
 31                     context.setState('INIT');
 32                 }
 33                 return this.outputElement;
 34             },
 35 
 36             insert: function (index, widget, length) {
 37                 this._super(index, widget, length);
 38                 this._widgetContexts.splice(index, 0, this.createContext(widget, this));
 39             },
 40 
 41             remove: function (widget) {
 42                 var i, widgets, returnValue;
 43                 widgets = this.widgets();
 44                 for (i = 0; i !== widgets.length; i += 1) {
 45                     if (widget === widgets[i]) {
 46                         this._widgetContexts.splice(i, 1);
 47                     }
 48                 }
 49                 returnValue = this._super(widget, false);
 50 
 51                 return returnValue;
 52             },
 53 
 54             removeAll: function () {
 55                 this._widgetContexts = [];
 56                 this._super();
 57             },
 58 
 59             needsVisibleIndices: function () {
 60                 return this._widgetContexts.length > 0;
 61             },
 62 
 63             attachIndexedWidgets: function (indexArray) {
 64                 var i, itemIndex, indexSet, firstIndexInView, preIndices, postIndices;
 65                 indexSet = {};
 66                 firstIndexInView = this._firstIndexInView();
 67                 preIndices = [];
 68                 postIndices = [];
 69                 for (i = 0; i !== indexArray.length; i += 1) {
 70                     itemIndex = indexArray[i];
 71                     indexSet[itemIndex] = true;
 72                     if (itemIndex < firstIndexInView) {
 73                         preIndices.push(itemIndex);
 74                     } else {
 75                         postIndices.push(itemIndex);
 76                     }
 77                 }
 78                 this._detatchWidgetsNotIndexed(indexSet);
 79                 preIndices.reverse();
 80                 for (i = 0; i !== preIndices.length; i += 1) {
 81                     itemIndex = preIndices[i];
 82                     this._widgetContexts[itemIndex].prepend();
 83                 }
 84                 for (i = 0; i !== postIndices.length; i += 1) {
 85                     itemIndex = postIndices[i];
 86                     this._widgetContexts[itemIndex].append();
 87                 }
 88             },
 89 
 90             getLengthToIndex: function (index) {
 91                 var firstAttached, i, length, totalLength;
 92                 totalLength = 0;
 93                 firstAttached = this._firstIndexWithLength();
 94                 for (i = firstAttached; i < index; i += 1) {
 95                     length = this._lengths[i];
 96                     if (length === undefined) {
 97                         this._throwNoLengthError();
 98                     }
 99                     totalLength += length;
100                 }
101                 return totalLength;
102             },
103 
104             lengthOfWidgetAtIndex: function (index) {
105                 if (this._lengths[index] !== undefined) {
106                     return this._lengths[index];
107                 } else {
108                     this._throwNoLengthError();
109                 }
110             },
111 
112             createContext: function (widget, parent) {
113                 return new WidgetContext(widget, parent, STATES);
114             },
115 
116             _detatchWidgetsNotIndexed: function (indexSet) {
117                 var i;
118                 for (i = 0; i !== this._widgetContexts.length; i += 1) {
119                     if (!indexSet.hasOwnProperty(i)) {
120                         this._widgetContexts[i].detach();
121                     }
122                 }
123             },
124 
125             _firstIndexWithLength: function () {
126                 var i, attached, firstAttachedIndex;
127                 i = 0;
128                 attached = false;
129                 while (i < this._widgetContexts.length && attached === false) {
130                     attached = this._widgetContexts[i].hasLength();
131                     firstAttachedIndex = i;
132                     i += 1;
133                 }
134                 return firstAttachedIndex;
135             },
136 
137             _firstIndexInView: function () {
138                 var i, inView, firstInView;
139                 i = 0;
140                 inView = false;
141                 while (i < this._widgetContexts.length && inView === false) {
142                     inView = this._widgetContexts[i].inView();
143                     firstInView = i;
144                     i += 1;
145                 }
146                 return firstInView;
147             },
148 
149             _throwNoLengthError: function () {
150                 throw new Error("You must set widget lengths before aligning culling strip");
151             }
152         });
153         return CullingStrip;
154     }
155 );
