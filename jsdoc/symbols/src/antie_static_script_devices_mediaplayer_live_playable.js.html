 28 require.def(
 29     "antie/devices/mediaplayer/live/playable",
 30     [
 31         "antie/class",
 32         "antie/runtimecontext",
 33         "antie/devices/device",
 34         "antie/devices/mediaplayer/mediaplayer"
 35     ],
 36     function (Class, RuntimeContext, Device, MediaPlayer) {
 37         "use strict";
 38 
 39         /**
 40          * Live player for devices that support playing live streams, but cannot seek within them.
 41          * Implements only a subset of functions in the underlying {antie.devices.mediaplayer.MediaPlayer}:
 42          * - beginPlayback (start playing from the live point, or wherever the device feels like)
 43          * - setSource, stop, reset, getState, getSource, getMimeType, addEventCallback, removeEventCallback,
 44          *   removeAllEventCallbacks
 45          * Does NOT implement the following functions:
 46          * - playFrom, pause, resume, getCurrentTime, getSeekableRange
 47          * See the documentation on {antie.devices.mediaplayer.MediaPlayer} for API details.
 48          * @name antie.devices.mediaplayer.live.Playable
 49          * @class
 50          * @extends antie.Class
 51          */
 52         var PlayableLivePlayer = Class.extend({
 53             init: function() {
 54                 this._mediaPlayer = RuntimeContext.getDevice().getMediaPlayer();
 55             },
 56 
 57             beginPlayback: function() {
 58                 this._mediaPlayer.beginPlayback();
 59             },
 60 
 61             setSource: function(mediaType, sourceUrl, mimeType) {
 62                 this._mediaPlayer.setSource(mediaType, sourceUrl, mimeType);
 63             },
 64 
 65             stop: function() {
 66                 this._mediaPlayer.stop();
 67             },
 68 
 69             reset: function() {
 70                 this._mediaPlayer.reset();
 71             },
 72 
 73             getState: function() {
 74                 return this._mediaPlayer.getState();
 75             },
 76 
 77             getSource: function() {
 78                 return this._mediaPlayer.getSource();
 79             },
 80 
 81             getMimeType: function() {
 82                 return this._mediaPlayer.getMimeType();
 83             },
 84 
 85             addEventCallback: function(thisArg, callback) {
 86                 this._mediaPlayer.addEventCallback(thisArg, callback);
 87             },
 88 
 89             removeEventCallback: function(thisArg, callback) {
 90                 this._mediaPlayer.removeEventCallback(thisArg, callback);
 91             },
 92 
 93             removeAllEventCallbacks: function() {
 94                 this._mediaPlayer.removeAllEventCallbacks();
 95             },
 96 
 97             getPlayerElement: function() {
 98                 return this._mediaPlayer.getPlayerElement();
 99             }
100         });
101 
102         var instance;
103 
104         Device.prototype.getLivePlayer = function () {
105             if(!instance) {
106                 instance = new PlayableLivePlayer();
107             }
108             return instance;
109         };
110 
111         Device.prototype.getLiveSupport = function () {
112             return MediaPlayer.LIVE_SUPPORT.PLAYABLE;
113         };
114     }
115 );
