 27 require.def('antie/widgets/media',
 28 	[
 29 		'antie/widgets/widget',
 30         'antie/devices/media/mediainterface',
 31 
 32 		// Include these so they're packaged with this file
 33 		// All subclasses (which will be dynamically loaded) should use these
 34 		'antie/events/mediaerrorevent',
 35 		'antie/events/mediasourceerrorevent'
 36 	],
 37 	function(Widget, MediaInterface) {
 38         'use strict';
 39 
 40 		/**
 41 		 * The Media widget is responsible for playing video and audio within an application.
 42 		 * It must be subclassed to provide support for different playback APIs.
 43 		 * Methods and events loosely match that of the HTML5 HTMLMediaElement:
 44 		 * @name antie.widgets.Media
 45 		 * @class
 46 		 * @abstract
 47 		 * @extends antie.widgets.Widget
 48 		 * @requires antie.events.MediaErrorEvent
 49 		 * @requires antie.events.MediaSourceErrorEvent
 50 		 * @param {String} [id] The unique ID of the widget. If excluded, a temporary internal ID will be used (but not included in any output).
 51 		 */
 52 		var Media = Widget.extend(/** @lends antie.widgets.Media.prototype */ {
 53 			/**
 54 			 * @constructor
 55 			 * @ignore
 56 			 */
 57 			init: function(id, mediaType) {
 58 				this._super(id);
 59 
 60                 var self = this;
 61 
 62                 var eventCallback = function (event) {
 63 
 64                     // Claim the event as our own before propagating it; event listeners will be added to this
 65                     // (media) widget, not to something hidden away within the MediaInterface.
 66                     event.target = self;
 67 
 68                     self.bubbleEvent(event);
 69                 };
 70 
 71                 this._mediaInterface = this.getCurrentApplication().getDevice().createMediaInterface(id, mediaType, eventCallback);
 72 
 73 				this.addClass('media');
 74 			},
 75             render: function(device) {
 76 	            this.outputElement = this._mediaInterface.render(device);
 77 	            return this.outputElement;
 78             },
 79             show: function(options) {
 80                 this._mediaInterface.show(options);
 81             },
 82             hide: function(options) {
 83                 this._mediaInterface.hide(options);
 84             },
 85             moveTo: function(options) {
 86                 this._mediaInterface.moveTo(options);
 87             },
 88             setWindow: function(left, top, width, height) {
 89                 this._mediaInterface.setWindow(left, top, width, height);
 90             },
 91             getError: function() {
 92                 return this._mediaInterface.getError();
 93             },
 94             setSources: function(sources, tags) {
 95                 this._mediaInterface.setSources(sources, tags);
 96             },
 97             getSources: function() {
 98                 return this._mediaInterface.getSources();
 99             },
100             getCurrentSource: function() {
101                 return this._mediaInterface.getCurrentSource();
102             },
103             getNetworkState: function() {
104                 return this._mediaInterface.getNetworkState();
105             },
106             getPreload: function() {
107                 return this._mediaInterface.getPreload();
108             },
109             setPreload: function(preload) {
110                 this._mediaInterface.setPreload(preload);
111             },
112             getBuffered: function() {
113                 return this._mediaInterface.getBuffered();
114             },
115             load: function() {
116                 this._mediaInterface.load();
117             },
118             canPlayType: function(type) {
119                 return this._mediaInterface.canPlayType(type);
120             },
121             getReadyState: function() {
122                 return this._mediaInterface.getReadyState();
123             },
124             getSeeking: function() {
125                 return this._mediaInterface.getSeeking();
126             },
127             setCurrentTime: function(currentTime) {
128                 this._mediaInterface.setCurrentTime(currentTime);
129             },
130             getCurrentTime: function() {
131                 return this._mediaInterface.getCurrentTime();
132             },
133             getInitialTime: function() {
134                 return this._mediaInterface.getInitialTime();
135             },
136             getDuration: function() {
137                 return this._mediaInterface.getDuration();
138             },
139             getStartOffsetTime: function() {
140                 return this._mediaInterface.getStartOffsetTime();
141             },
142             getPaused: function() {
143                 return this._mediaInterface.getPaused();
144             },
145             getDefaultPlaybackRate: function() {
146                 return this._mediaInterface.getDefaultPlaybackRate();
147             },
148             getPlaybackRate: function() {
149                 return this._mediaInterface.getPlaybackRate();
150             },
151             setPlaybackRate: function(playbackRate) {
152                 this._mediaInterface.setPlaybackRate(playbackRate);
153             },
154             getPlayed: function() {
155                 return this._mediaInterface.getPlayed();
156             },
157             getSeekable: function() {
158                 return this._mediaInterface.getSeekable();
159             },
160             getEnded: function() {
161                 return this._mediaInterface.getEnded();
162             },
163             getAutoPlay: function() {
164                 return this._mediaInterface.getAutoPlay();
165             },
166             setAutoPlay: function(autoplay) {
167                 this._mediaInterface.setAutoPlay(autoplay);
168             },
169             getLoop: function() {
170                 return this._mediaInterface.getLoop();
171             },
172             setLoop: function(loop) {
173                 this._mediaInterface.setLoop(loop);
174             },
175             play: function() {
176                 this._mediaInterface.play();
177             },
178             stop: function() {
179                 this._mediaInterface.stop();
180             },
181             pause: function() {
182                 this._mediaInterface.pause();
183             },
184             setNativeControls: function(controls) {
185                 this._mediaInterface.setNativeControls(controls);
186             },
187             getNativeControls: function() {
188                 return this._mediaInterface.getNativeControls();
189             },
190             setVolume: function(volume) {
191                 this.getCurrentApplication().getDevice().setVolume(volume);
192             },
193             getVolume: function() {
194                 return this.getCurrentApplication().getDevice().getVolume();
195             },
196             setMuted: function(muted) {
197                 this.getCurrentApplication().getDevice().setMuted(muted);
198             },
199             getMuted: function() {
200                 return this.getCurrentApplication().getDevice().getMuted();
201             },
202             destroy: function() {
203                 this._mediaInterface.destroy();
204             }
205         });
206 
207         Media.EMBED_MODE_EXTERNAL = MediaInterface.EMBED_MODE_EXTERNAL;
208         Media.EMBED_MODE_BACKGROUND = MediaInterface.EMBED_MODE_BACKGROUND;
209         Media.EMBED_MODE_EMBEDDED = MediaInterface.EMBED_MODE_EMBEDDED;
210 
211         Media.NETWORK_EMPTY = MediaInterface.NETWORK_EMPTY;
212         Media.NETWORK_IDLE = MediaInterface.NETWORK_IDLE;
213         Media.NETWORK_LOADING = MediaInterface.NETWORK_LOADING;
214         Media.NETWORK_NO_SOURCE = MediaInterface.NETWORK_NO_SOURCE;
215 
216         Media.MEDIA_ERR_UNKNOWN = MediaInterface.MEDIA_ERR_UNKNOWN;
217         Media.MEDIA_ERR_ABORTED = MediaInterface.MEDIA_ERR_ABORTED;
218         Media.MEDIA_ERR_NETWORK = MediaInterface.MEDIA_ERR_NETWORK;
219         Media.MEDIA_ERR_DECODE = MediaInterface.MEDIA_ERR_DECODE;
220         Media.MEDIA_ERR_SRC_NOT_SUPPORTED = MediaInterface.MEDIA_ERR_SRC_NOT_SUPPORTED;
221 
222         Media.HAVE_NOTHING = MediaInterface.HAVE_NOTHING;
223         Media.HAVE_METADATA = MediaInterface.HAVE_METADATA;
224         Media.HAVE_CURRENT_DATA = MediaInterface.HAVE_CURRENT_DATA;
225         Media.HAVE_FUTURE_DATA = MediaInterface.HAVE_FUTURE_DATA;
226         Media.HAVE_ENOUGH_DATA = MediaInterface.HAVE_ENOUGH_DATA;
227 
228         return Media;
229 	}
230 );
