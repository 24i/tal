 37 			}

 69 	 * Calculates the interpolated tween values of an Object based on the current time.
 70 	 * @param {Number} currentPosition The current position to evaluate the tween against.
 71 	 * @param {Object} params A configuration Object containing the values that this function requires.  The required properties in this Object are:
 72 	 *   @property {Object} originalState The original properties the Object is tweening from.
 73 	 *   @property {Object} to The destination properties the Object is tweening to.
 74 	 *   @property {Number} duration The length of the tween in milliseconds.
 75 	 *   @property {Number} timestamp The UNIX epoch time at which the tween began.
 76 	 *   @property {Function} easingFunc The function used to calculate the tween.  See the documentation for `Tweenable.prototype.formula` for more info on the appropriate function signature for this.
 77 	 * @param {Object} state A configuration object containing current state data of the tween.  Required properties:
 78 	 *   @property {Object} current The Object containing the current `Number` values of the tween.
 79 	 */
 80 	function tweenProps(currentPosition, params, state) {
 81 		var prop,
 82 			normalizedPosition;
 83 
 84 		normalizedPosition = (currentPosition - params.timestamp) / params.duration;
 85 
 86 		for (prop in state.current) {
 87 			if (state.current.hasOwnProperty(prop) && params.to.hasOwnProperty(prop)) {
 88 				state.current[prop] = tweenProp(params.originalState[prop], params.to[prop], params.easingFunc, normalizedPosition);
 89 			}
 90 		}
 91 
 92 		return state.current;
 93 	}

 95 	function tweenProp(from, to, easingFunc, position) {
 96 		return from + (to - from) * easingFunc(position);
 97 	}

133 	/**
134 	 * Handles the update logic for one step of a tween.
135 	 * @param {Object} params The configuration containing all of a tween's properties.  This requires all of the `params` @properties required for `tweenProps`, so see that.  It also requires:
136 	 *   @property {Object} owner The `Tweenable` instance that the tween this function is acting upon belongs to.
137 	 *   @property {Object} hook The Object containing all of the `hook`s that belong to `owner
138 	 * @param {Object} state The configuration Object containing all of the state properties for a `Tweenable` instance.  It requires all of the @properties listed for the `state` parameter of  `tweenProps`, so see that.  It also requires:
139 	 *   @property {Boolean} isAnimating Whether or not this tween as actually running.
140 	 *   @property {Number} loopId The property that the latest `setTimeout` invokation ID stored in.
141 	 */
142 	function timeoutHandler(params, state) {
143 		var currentTime;
144 
145 		function doStep(time) {
146 			applyFilter('beforeTween', params.owner, [state.current, params.originalState, params.to]);
147 			tweenProps(time, params, state);
148 			applyFilter('afterTween', params.owner, [state.current, params.originalState, params.to]);
149 
150 			if (params.hook.step) {
151 				invokeHook('step', params.hook, params.owner, [state.current]);
152 			}
153 
154 			params.step.call(state.current);
155 		}
156 
157 		currentTime = now();
158 
159 		if (currentTime < params.timestamp + params.duration && state.isAnimating) {
160 			// The tween is still running, schedule an update
161 			state.loopId = scheduleUpdate(function () {
162 				timeoutHandler(params, state);
163 			}, params.owner.fps);
164 
165 			doStep(currentTime);
166 		} else {
167 			// The duration of the tween has expired
168 			doStep(params.timestamp + params.duration);
169 			params.owner.stop(true);
170 		}
171 	}

217 	 * Start a tween.  This method can be called two ways.  The shorthand way:
218 	 *
219 	 *   tweenableInst.tween (from, to, [duration], [callback], [easing]);
220 	 *
221 	 * or the longhand way:
222 	 *
223 	 *   tweenableInst.tween ( {
224 	 *	 from:	   Object,
225 	 *	 to:		 Object,
226 	 *	 duration:   [Number],
227 	 *	 callback:   [Function],
228 	 *	 easing:	 [String],
229 	 *	 step:	   [Function]
230 	 *   });
231 	 *
232 	 * Regardless of how you invoke this method, the only required parameters are `from` and `to`.
233 	 *
234 	 * @param {Object} from The beginning state Object containing the properties to tween from.  NOTE:  The properties of this Object are modified over time (to reflect the values in `to`).
235 	 * @param {Object} to The target state Object containing the properties to tween to.
236 	 * @param {Number} duration The amount of time in milliseconds that the tween should run for.
237 	 * @param {Function} start The function to be invoked as soon as the this tween starts.  Mostly useful when used with the `queue` extension.
238 	 * @param {Function} callback The function to invoke as soon as this tween completes.  This function is given the tween's current state Object as the first parameter.
239 	 * @param {String} easing The name of the easing formula to use for this tween.  You can specify any easing fomula that was attached to `Tweenable.prototype.formula`.  If ommitted, the easing formula specified when the instance was created is used, or `linear` if that was omitted.
240 	 * @param {Function} step A function to call for each step of the tween.  A "step" is defined as one update cycle (frame) of the tween.  Many update cycles occur to create the illusion of motion, so this function will likely be called quite a bit.
241 	 */
242 	Tweenable.prototype.tween = function tween(from, to, duration, callback, easing) {
243 
244 		var self,
245 			params,
246 			state;
247 
248 		if (this._state.isAnimating) {
249 			return;
250 		}
251 
252 		self = this;
253 		params = this._tweenParams;
254 		state = this._state;
255 		this._state.loopId = 0;
256 		this._state.pausedAtTime = null;
257 
258 		// Normalize some internal values depending on how `tweenableInst.tween` was invoked
259 		if (to) {
260 			// Assume the shorthand syntax is being used.
261 			params.step = function () {
262 			};
263 			params.to = to || {};
264 			params.duration = duration || this.duration;
265 			params.callback = callback || function () {
266 			};
267 			params.easing = easing || this.easing;
268 			state.current = from || {};
269 		} else {
270 			// If the second argument is not present, assume the longhand syntax is being used.
271 			params.step = from.step || function () {
272 			};
273 			params.callback = from.callback || function () {
274 			};
275 			params.to = from.to || from.target || {};
276 			params.duration = from.duration || this.duration;
277             params.easing = from.easing || this.easing;
278             params.fps = from.fps || this.fps;
279             state.current = from.from || {};
280 		}
281 
282 		params.timestamp = now();
283 		params.easingFunc = this.formula[params.easing] || this.formula.linear;
284 
285 		// Ensure that there is always something to tween to.
286 		// Kinda dumb and wasteful, but makes this code way more flexible.
287 		weakCopy(state.current, params.to);
288 		weakCopy(params.to, state.current);
289 
290 		applyFilter('tweenCreated', params.owner, [state.current, params.originalState, params.to]);
291 		params.originalState = simpleCopy({}, state.current);
292 		state.isAnimating = true;
293 		this.resume();
294 
295 		if (from.start) {
296 			from.start();
297 		}
298 
299 		return this;
300 	};

654 		Tweenable.util.each(obj, function (obj, prop) {

659 			}

666 		Tweenable.util.each(tokenProps, function (collection, token) {
667 			// Extract the value from the string

669 		});
670 	}

673 		Tweenable.util.each(tokenProps, function (collection, token) {
