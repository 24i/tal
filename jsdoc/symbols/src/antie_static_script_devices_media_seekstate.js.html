 28 require.def('antie/devices/media/seekstate',
 29     [
 30         'antie/class',
 31         'antie/events/mediaevent'
 32     ],
 33 
 34     function(Class, MediaEvent){
 35         'use strict';
 36 
 37         var State = {
 38                         None        : 0,
 39                         Seeking     : 1,
 40                         Playing     : 2
 41                     };
 42 
 43     var SeekState = Class.extend({
 44 
 45         init :function(eventHandlingCallback) {
 46             this._eventHandlingCallback = eventHandlingCallback;
 47             this._currentTime = 0;
 48             this._state = State.None;
 49         },
 50 
 51         seekTo: function(time) {
 52             if (time === this._currentTime) {
 53                 return;
 54             }
 55 
 56             this._state = State.Seeking;
 57             this._currentTime = time;
 58 
 59             this._eventHandlingCallback(new MediaEvent("seeking"));
 60         },
 61 
 62         playing: function() {
 63             if (this._state === State.Seeking) {
 64                 this._eventHandlingCallback(new MediaEvent("seeked"));
 65             }
 66 
 67             this._state = State.Playing;
 68         }
 69 
 70     });
 71 
 72     return SeekState;
 73 
 74 }
 75 );
